74/// Definition of the cost schedule and other parameterization for the wasm vm.
75///
76/// Its [`Default`] implementation is the designated way to initialize this type. It uses
77/// the benchmarked information supplied by [`Config::WeightInfo`]. All of its fields are
78/// public and can therefore be modified. For example in order to change some of the limits
79/// and set a custom instruction weight version the following code could be used:
80/// ```rust
81/// use pallet_gear::{Schedule, Limits, InstructionWeights, Config};
82///
83/// fn create_schedule<T: Config>() -> Schedule<T> {
84///     Schedule {
85///         limits: Limits {
86///                 globals: 3,
87///                 locals: 3,
88///                 parameters: 3,
89///                 memory_pages: 16,
90///                 table_size: 3,
91///                 br_table_size: 3,
92///                 .. Default::default()
93///             },
94///         instruction_weights: InstructionWeights {
95///                 version: 5,
96///             .. Default::default()
97///         },
98///             .. Default::default()
99///     }
100/// }
101/// ```
102///
103/// # Note
104///
105/// Please make sure to bump the [`InstructionWeights::version`] whenever substantial
106/// changes are made to its values.
107#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]
108#[cfg_attr(feature = "std", serde(bound(serialize = "", deserialize = "")))]

143/// Describes the upper limits on various metrics.
144///
145/// # Note
146///
147/// The values in this struct should never be decreased. The reason is that decreasing those
148/// values will break existing programs which are above the new limits when a
149/// re-instrumentation is triggered.
150#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]
151#[derive(Clone, Encode, Decode, PartialEq, Eq, RuntimeDebug, TypeInfo)]
152pub struct Limits {
153    /// Maximum allowed stack height in number of elements.
154    ///
155    /// See <https://wiki.parity.io/WebAssembly-StackHeight> to find out
156    /// how the stack frame cost is calculated. Each element can be of one of the
157    /// wasm value types. This means the maximum size per element is 64bit.
158    ///
159    /// # Note
160    ///
161    /// It is safe to disable (pass `None`) the `stack_height` when the execution engine
162    /// is part of the runtime and hence there can be no indeterminism between different
163    /// client resident execution engines.
164    pub stack_height: Option<u32>,
165
166    /// Maximum number of globals a module is allowed to declare.
167    ///
168    /// Globals are not limited through the linear memory limit `memory_pages`.
169    pub globals: u32,
170
171    /// Maximum number of locals a function can have.
172    ///
173    /// As wasm engine initializes each of the local, we need to limit their number to confine
174    /// execution costs.
175    pub locals: u32,
176
177    /// Maximum numbers of parameters a function can have.
178    ///
179    /// Those need to be limited to prevent a potentially exploitable interaction with
180    /// the stack height instrumentation: The costs of executing the stack height
181    /// instrumentation for an indirectly called function scales linearly with the amount
182    /// of parameters of this function. Because the stack height instrumentation itself is
183    /// is not weight metered its costs must be static (via this limit) and included in
184    /// the costs of the instructions that cause them (call, call_indirect).
185    pub parameters: u32,
186
187    /// Maximum number of memory pages allowed for a program.
188    pub memory_pages: u16,
189
190    /// Maximum number of elements allowed in a table.
191    ///
192    /// Currently, the only type of element that is allowed in a table is funcref.
193    pub table_size: u32,
194
195    /// Maximum number of elements that can appear as immediate value to the br_table instruction.
196    pub br_table_size: u32,
197
198    /// The maximum length of a subject in bytes used for PRNG generation.
199    pub subject_len: u32,
200
201    /// The maximum nesting level of the call stack.
202    pub call_depth: u32,
203
204    /// The maximum size of a message payload in bytes.
205    pub payload_len: u32,
206
207    /// The maximum length of a program code in bytes. This limit applies to the instrumented
208    /// version of the code. Therefore `instantiate_with_code` can fail even when supplying
209    /// a wasm binary below this maximum size.
210    pub code_len: u32,
211
212    /// The maximum number of wasm data segments allowed for a program.
213    pub data_segments_amount: u32,
214}

216/// Describes the weight for all categories of supported wasm instructions.
217///
218/// There there is one field for each wasm instruction that describes the weight to
219/// execute one instruction of that name. There are a few exceptions:
220///
221/// 1. If there is a i64 and a i32 variant of an instruction we use the weight
222///    of the former for both.
223/// 2. The following instructions are free of charge because they merely structure the
224///    wasm module and cannot be spammed without making the module invalid (and rejected):
225///    End, Unreachable, Return, Else
226/// 3. The following instructions cannot be benchmarked because they are removed by any
227///    real world execution engine as a preprocessing step and therefore don't yield a
228///    meaningful benchmark result. However, in contrast to the instructions mentioned
229///    in 2. they can be spammed. We price them with the same weight as the "default"
230///    instruction (i64.const): Block, Loop, Nop
231/// 4. We price both i64.const and drop as InstructionWeights.i64const / 2. The reason
232///    for that is that we cannot benchmark either of them on its own but we need their
233///    individual values to derive (by subtraction) the weight of all other instructions
234///    that use them as supporting instructions. Supporting means mainly pushing arguments
235///    and dropping return values in order to maintain a valid module.

567/// Describes the weight for memory interaction.
568///
569/// Each weight with `lazy_pages_` prefix includes weight for storage read,
570/// because for each first page access we need to at least check whether page exists in storage.

572/// This weight is taken in account separately, when loading occurs.
573///
574/// Lazy-pages write accesses does not include cost for uploading page data to storage,
575/// because uploading happens after execution, so benchmarks do not include this cost.
576/// But they include cost for processing changed page data in runtime.
577#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]

585    pub lazy_pages_signal_write: Weight,

588    pub lazy_pages_signal_write_after_read: Weight,

808}

810#[inline]
811fn cost_i64const<T: Config>() -> u32 {
812    type W<T> = <T as Config>::WeightInfo;
813    // Since we cannot directly benchmark the weight of `i64.const` (or `i32.const`; we consider their weights to be the same for our purposes),
814    // we estimate it as the difference between the benchmarks `instr_call_const` and `instr_call`.
815    // The difference between these two benchmarks will give us the weight of `i64.const`, which for x86-64
816    // can be represented by a single `mov` instruction with the embedded const parameter
817    // (for e.x: `mov reg,0xDEADBEAFDEABEAF`).
818    //
819    // This approach may work, but the estimation is not very accurate.
820    // To reduce the impact of this inaccuracy on the assessment of other instructions
821    // (as we subtract the weight of `i64.const` from other instructions),
822    // we introduce a weight division coefficient called `I64CONST_WEIGHT_DIVIDER`.
823    // This helps bring the weight of `i64.const` closer to that of the x86-64 `mov` instruction's estimate.
824    const I64CONST_WEIGHT_DIVIDER: u32 = 2;
825
826    cost_instr_no_params_with_batch_size(W::<T>::instr_i64const) / I64CONST_WEIGHT_DIVIDER
827}

1257impl<T: Config> Default for MemoryWeights<T> {
1258    fn default() -> Self {
1259        // In benchmarks we calculate cost per wasm page,
1260        // so here we must convert it to cost per gear page.
1261        fn to_cost_per_gear_page(w: fn(u32) -> Weight) -> Weight {
1262            Weight::from_parts(
1263                cost(w).ref_time() / (WasmPage::SIZE / GearPage::SIZE) as u64,
1264                0,
1265            )
1266        }
1267
1268        const KB_SIZE: u64 = 1024;
1269
1270        // Memory access thru host function benchmark uses a syscall,
1271        // which accesses memory. So, we have to subtract corresponding syscall weight.
1272        fn host_func_access(w: fn(u32) -> Weight, syscall: fn(u32) -> Weight) -> Weight {
1273            let syscall_per_kb_weight = cost_batched(syscall).ref_time();
1274            let syscall_per_gear_page_weight =
1275                (syscall_per_kb_weight / KB_SIZE) * GearPage::SIZE as u64;
1276
1277            let ref_time = to_cost_per_gear_page(w)
1278                .ref_time()
1279                .saturating_sub(syscall_per_gear_page_weight);
1280
1281            Weight::from_parts(ref_time, 0)
1282        }
1283
1284        const KB_AMOUNT_IN_ONE_GEAR_PAGE: u64 = GearPage::SIZE as u64 / KB_SIZE;
1285        const {
1286            assert!(KB_AMOUNT_IN_ONE_GEAR_PAGE > 0);
1287            assert!((GearPage::SIZE as u64).is_multiple_of(KB_SIZE));
1288        }
1289
1290        type W<T> = <T as Config>::WeightInfo;
1291
1292        Self {
1293            lazy_pages_signal_read: to_cost_per_gear_page(W::<T>::lazy_pages_signal_read),
1294            lazy_pages_signal_write: to_cost_per_gear_page(W::<T>::lazy_pages_signal_write),
1295            lazy_pages_signal_write_after_read: to_cost_per_gear_page(
1296                W::<T>::lazy_pages_signal_write_after_read,
1297            ),
1298            lazy_pages_host_func_read: host_func_access(
1299                W::<T>::lazy_pages_host_func_read,
1300                W::<T>::gr_debug_per_kb,
1301            ),
1302            lazy_pages_host_func_write: host_func_access(
1303                W::<T>::lazy_pages_host_func_write,
1304                W::<T>::gr_read_per_kb,
1305            ),
1306            lazy_pages_host_func_write_after_read: host_func_access(
1307                W::<T>::lazy_pages_host_func_write_after_read,
1308                W::<T>::gr_read_per_kb,
1309            ),
1310            // As you can see from calculation: `load_page_data` doesn't include weight for db read.
1311            // This is correct situation, because this weight is already included in above
1312            // lazy-pages weights.
1313            load_page_data: to_cost_per_gear_page(W::<T>::lazy_pages_load_page_storage_data)
1314                .saturating_sub(to_cost_per_gear_page(W::<T>::lazy_pages_signal_read)),
1315            upload_page_data: cost(W::<T>::db_write_per_kb)
1316                .saturating_mul(KB_AMOUNT_IN_ONE_GEAR_PAGE)
1317                .saturating_add(T::DbWeight::get().writes(1)),
1318            mem_grow: cost_batched(W::<T>::mem_grow),
1319            mem_grow_per_page: cost_batched(W::<T>::mem_grow_per_page),
1320            // TODO: make it non-zero for para-chains (issue #2225)
1321            parachain_read_heuristic: Weight::zero(),
1322            _phantom: PhantomData,
1323        }
1324    }

1457struct ScheduleRules<'a, T: Config> {
1458    schedule: &'a Schedule<T>,
1459    params: Vec<u32>,
1460}
1461
1462impl<T: Config> Rules for ScheduleRules<'_, T> {
1463    fn instruction_cost(&self, instruction: &Instruction) -> Option<u32> {
1464        use Instruction::*;
1465
1466        let w = &self.schedule.instruction_weights;
1467        let max_params = self.schedule.limits.parameters;
1468
1469        Some(match instruction {
1470            // Returning None makes the gas instrumentation fail which we intend for
1471            // unsupported or unknown instructions.
1472            MemoryGrow { .. } => return None,
1473            //
1474            End | Unreachable | Return | Else | Block { .. } | Loop { .. } | Nop | Drop => 0,
1475            I32Const { .. } | I64Const { .. } => w.i64const,
1476            I32Load { .. }
1477            | I32Load8S { .. }
1478            | I32Load8U { .. }
1479            | I32Load16S { .. }
1480            | I32Load16U { .. } => w.i32load,
1481            I64Load { .. }
1482            | I64Load8S { .. }
1483            | I64Load8U { .. }
1484            | I64Load16S { .. }
1485            | I64Load16U { .. }
1486            | I64Load32S { .. }
1487            | I64Load32U { .. } => w.i64load,
1488            I32Store { .. } | I32Store8 { .. } | I32Store16 { .. } => w.i32store,
1489            I64Store { .. } | I64Store8 { .. } | I64Store16 { .. } | I64Store32 { .. } => {
1490                w.i64store
1491            }
1492            Select => w.select,
1493            If { .. } => w.r#if,
1494            Br { .. } => w.br,
1495            BrIf { .. } => w.br_if,
1496            Call { .. } => w.call,
1497            LocalGet { .. } => w.local_get,
1498            LocalSet { .. } => w.local_set,
1499            LocalTee { .. } => w.local_tee,
1500            GlobalGet { .. } => w.global_get,
1501            GlobalSet { .. } => w.global_set,
1502            MemorySize { .. } => w.memory_current,
1503            CallIndirect(idx) => {
1504                let params = self
1505                    .params
1506                    .get(*idx as usize)
1507                    .copied()
1508                    .unwrap_or(max_params);
1509                w.call_indirect
1510                    .saturating_add(w.call_indirect_per_param.saturating_mul(params))
1511            }
1512            BrTable(targets) => w
1513                .br_table
1514                .saturating_add(w.br_table_per_entry.saturating_mul(targets.len())),
1515            I32Clz => w.i32clz,
1516            I64Clz => w.i64clz,
1517            I32Ctz => w.i32ctz,
1518            I64Ctz => w.i64ctz,
1519            I32Popcnt => w.i32popcnt,
1520            I64Popcnt => w.i64popcnt,
1521            I32Eqz => w.i32eqz,
1522            I64Eqz => w.i64eqz,
1523            // TODO: rename fields
1524            I64ExtendI32S => w.i64extendsi32,
1525            I64ExtendI32U => w.i64extendui32,
1526            I32WrapI64 => w.i32wrapi64,
1527            I32Eq => w.i32eq,
1528            I64Eq => w.i64eq,
1529            I32Ne => w.i32ne,
1530            I64Ne => w.i64ne,
1531            I32LtS => w.i32lts,
1532            I64LtS => w.i64lts,
1533            I32LtU => w.i32ltu,
1534            I64LtU => w.i64ltu,
