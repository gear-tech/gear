41pub fn get_maximum_task_gas<T: Config>(task: &VaraScheduledTask<T::AccountId>) -> Gas {
42    use ScheduledTask::*;
43
44    match task {
45        PauseProgram(_) => 0,
46        RemoveCode(_) => todo!("#646"),
47        RemoveFromMailbox(_, _) => {
48            <T as Config>::WeightInfo::tasks_remove_from_mailbox().ref_time()
49        }
50        RemoveFromWaitlist(_, _) => {
51            <T as Config>::WeightInfo::tasks_remove_from_waitlist().ref_time()
52        }
53        RemovePausedProgram(_) => todo!("#646"),
54        WakeMessage(_, _) => cmp::max(
55            <T as Config>::WeightInfo::tasks_wake_message().ref_time(),
56            <T as Config>::WeightInfo::tasks_wake_message_no_wake().ref_time(),
57        ),
58        SendDispatch(_) => <T as Config>::WeightInfo::tasks_send_dispatch().ref_time(),
59        SendUserMessage { .. } => cmp::max(
60            <T as Config>::WeightInfo::tasks_send_user_message_to_mailbox().ref_time(),
61            <T as Config>::WeightInfo::tasks_send_user_message().ref_time(),
62        ),
63        RemoveGasReservation(_, _) => {
64            <T as Config>::WeightInfo::tasks_remove_gas_reservation().ref_time()
65        }
66        #[allow(deprecated)]
67        RemoveResumeSession(_) => 0,
68    }
69}

85    fn remove_from_mailbox(&mut self, user_id: T::AccountId, message_id: MessageId) -> Gas {
86        // Read reason.
87        let reason = UserMessageReadSystemReason::OutOfRent.into_reason();
88
89        let message = ReplyMessage::auto(message_id);
90
91        Pallet::<T>::create(user_id.clone(), message.id(), 0, true);
92
93        // Reading message from mailbox.
94        let mailboxed = Pallet::<T>::read_message(user_id.clone(), message_id, reason.clone())
95            .unwrap_or_else(|e| {
96                let err_msg = format!(
97                    "TaskHandler::remove_from_mailbox: failed reading message from mailbox. \
98                User - {user_id:?}, message - {message_id}, reason - {reason:?}. Got error: {e:?}."
99                );
100
101                log::error!("{err_msg}");
102                unreachable!("{err_msg}");
103            });
104
105        // Converting reply message into appropriate type for queueing.
106        let dispatch = message.into_stored_dispatch(
107            mailboxed.destination(),
108            mailboxed.source(),
109            mailboxed.id(),
110        );
111
112        // Queueing dispatch.
113        QueueOf::<T>::queue(dispatch).unwrap_or_else(|e| {
114            let err_msg = format!(
115                "TaskHandler::remove_from_mailbox: failed queuing message. \
116                Got error: {e:?}"
117            );
118
119            log::error!("{err_msg}");
120            unreachable!("{err_msg}");
121        });
122
123        let gas = <T as Config>::WeightInfo::tasks_remove_from_mailbox().ref_time();
124        log::trace!("Task gas: tasks_remove_from_mailbox = {gas}");
125
126        gas
127    }

129    fn remove_from_waitlist(&mut self, program_id: ActorId, message_id: MessageId) -> Gas {
130        // Wake reason.
131        let reason = MessageWokenSystemReason::OutOfRent.into_reason();
132
133        // Waking dispatch.
134        let waitlisted = Pallet::<T>::wake_dispatch(program_id, message_id, reason.clone())
135            .unwrap_or_else(|e| {
136                let err_msg = format!(
137                    "TaskHandler::remove_from_waitlist: failed waking dispatch. \
138                Program id - {program_id}, waking message - {message_id}, reason - {reason:?} \
139                Got error - {e:?}."
140                );
141
142                log::error!("{err_msg}");
143                unreachable!("{err_msg}");
144            });
145
146        self.send_signal(
147            message_id,
148            waitlisted.destination(),
149            SignalCode::RemovedFromWaitlist,
150        );
151
152        if !waitlisted.is_reply() && waitlisted.kind() != DispatchKind::Signal {
153            // Trap explanation.
154            let err = ErrorReplyReason::RemovedFromWaitlist;
155
156            // Trap reply payload.
157            let err_payload = err
158                .to_string()
159                .into_bytes()
160                .try_into()
161                .unwrap_or_else(|_| {
162                    let error_reply = err.to_string().into_bytes();
163                    let err_msg = format!(
164                        "TaskHandler::remove_from_waitlist: failed conversion of error reply into `Payload`. \
165                        Error reply bytes len - {len}, max payload len - {max_len}",
166                        len = error_reply.len(),
167                        max_len = Payload::max_len(),
168                    );
169
170                    log::error!("{err_msg}");
171                    unreachable!("{err_msg}");
172                });
173
174            let trap_reply = ReplyMessage::system(message_id, err_payload, 0, err);
175
176            // Generate trap reply.
177            if self.check_program_id(&waitlisted.source()) {
178                let trap_dispatch =
179                    trap_reply.into_stored_dispatch(program_id, waitlisted.source(), message_id);
180
181                // Creating `GasNode` for the reply.
182                Pallet::<T>::split(
183                    waitlisted.id(),
184                    trap_dispatch.id(),
185                    trap_dispatch.is_reply(),
186                );
187
188                // Queueing dispatch.
189                QueueOf::<T>::queue(trap_dispatch).unwrap_or_else(|e| {
190                    let err_msg = format!(
191                        "TaskHandler::remove_from_waitlist: failed queuing message. \
192                        Got error - {e:?}"
193                    );
194
195                    log::error!("{err_msg}");
196                    unreachable!("{err_msg}");
197                });
198            } else {
199                // Sending trap reply to user, by depositing event.
200                //
201                // There is no reason to use `Pallet::<T>::send_user_message( .. )`,
202                // because there is no need in reply in future, so no reason
203                // and funds to pay mailbox rent for it.
204                let trap_reply =
205                    trap_reply.into_stored(program_id, waitlisted.source(), message_id);
206                let trap_reply = trap_reply
207                    .try_into()
208                    .unwrap_or_else(|_| {
209                        // Signal message sent to user
210                        let err_msg = format!(
211                            "TaskHandler::remove_from_waitlist: failed conversion from stored into user message. \
212                            Message id - {message_id}, program id - {program_id}, destination - {dest}",
213                            dest = waitlisted.source()
214                        );
215
216                        log::error!("{err_msg}");
217                        unreachable!("{err_msg}")
218                    });
219
220                // Depositing appropriate event.
221                Pallet::<T>::deposit_event(Event::UserMessageSent {
222                    message: trap_reply,
223                    expiration: None,
224                });
225            }
226        }
227
228        // Consuming gas handler for waitlisted message.
229        Pallet::<T>::consume_and_retrieve(waitlisted.id());
230
231        if waitlisted.kind() == DispatchKind::Init {
232            let origin = waitlisted.source();
233            Self::process_failed_init(program_id, origin);
234        }
235
236        let gas = <T as Config>::WeightInfo::tasks_remove_from_waitlist().ref_time();
237        log::trace!("Task gas: tasks_remove_from_waitlist = {gas}");
238
239        gas
240    }

246    fn wake_message(&mut self, program_id: ActorId, message_id: MessageId) -> Gas {
247        match Pallet::<T>::wake_dispatch(
248            program_id,
249            message_id,
250            MessageWokenRuntimeReason::WakeCalled.into_reason(),
251        )
252        .ok()
253        {
254            Some(dispatch) => {
255                QueueOf::<T>::queue(dispatch).unwrap_or_else(|e| {
256                    let err_msg = format!(
257                        "TaskHandler::wake_message: failed queuing message. \
258                        Got error - {e:?}"
259                    );
260
261                    log::error!("{err_msg}");
262                    unreachable!("{err_msg}");
263                });
264
265                let gas = <T as Config>::WeightInfo::tasks_wake_message().ref_time();
266                log::trace!("Task gas: tasks_wake_message = {gas}");
267
268                gas
269            }
270            None => {
271                let gas = <T as Config>::WeightInfo::tasks_wake_message_no_wake().ref_time();
272                log::trace!("Task gas: tasks_wake_message_no_wake = {gas}");
273
274                gas
275            }
276        }
277    }

279    fn send_dispatch(&mut self, stashed_message_id: MessageId) -> Gas {
280        // No validation required. If program doesn't exist, then NotExecuted appears.
281
282        let (dispatch, hold_interval) = DispatchStashOf::<T>::take(stashed_message_id)
283            .unwrap_or_else(|| {
284                let err_msg = format!(
285                    "TaskHandler::send_dispatch: failed taking message from stash. Message id - {stashed_message_id}."
286                );
287
288                log::error!("{err_msg}");
289                unreachable!("{err_msg}");
290            });
291
292        // Charging locked gas for holding in dispatch stash.
293        Pallet::<T>::charge_for_hold(dispatch.id(), hold_interval, StorageType::DispatchStash);
294
295        QueueOf::<T>::queue(dispatch.into()).unwrap_or_else(|e| {
296            let err_msg = format!(
297                "TaskHandler::send_dispatch: failed queuing message. \
298                Got error - {e:?}"
299            );
300
301            log::error!("{err_msg}");
302            unreachable!("{err_msg}");
303        });
304
305        let gas = <T as Config>::WeightInfo::tasks_send_dispatch().ref_time();
306        log::trace!("Task gas: tasks_send_dispatch = {gas}");
307
308        gas
309    }

311    fn send_user_message(&mut self, stashed_message_id: MessageId, to_mailbox: bool) -> Gas {
312        // TODO: validate here destination and send error reply, if required.
313        // Atm despite the fact that program may exist, message goes into mailbox / event.
314        let (message, hold_interval) = DispatchStashOf::<T>::take(stashed_message_id)
315            .map(|(dispatch, interval)| (dispatch.into_parts().1, interval))
316            .unwrap_or_else(|| {
317                let err_msg = format!(
318                    "TaskHandler::send_user_message: failed taking message from stash. Message id - {stashed_message_id}."
319                );
320
321                log::error!("{err_msg}");
322                unreachable!("{err_msg}");
323            });
324
325        // Charge gas for message save.
326        Pallet::<T>::charge_for_hold(message.id(), hold_interval, StorageType::DispatchStash);
327
328        // Taking data for error log
329        let message_id = message.id();
330        let program_id = message.source();
331
332        // Cast message type.
333        let message = message.try_into().unwrap_or_else(|_| {
334            // Signal message sent to user
335            let err_msg = format!(
336                "TaskHandler::send_user_message: failed conversion from stored into user message. \
337                    Message id - {message_id}, program id - {program_id}.",
338            );
339
340            log::error!("{err_msg}");
341            unreachable!("{err_msg}");
342        });
343        Pallet::<T>::send_user_message_after_delay(message, to_mailbox);
344
345        if to_mailbox {
346            let gas = <T as Config>::WeightInfo::tasks_send_user_message_to_mailbox().ref_time();
347            log::trace!("Task gas: tasks_send_user_message_to_mailbox = {gas}");
348
349            gas
350        } else {
351            let gas = <T as Config>::WeightInfo::tasks_send_user_message().ref_time();
352            log::trace!("Task gas: tasks_send_user_message = {gas}");
353
354            gas
355        }
356    }
