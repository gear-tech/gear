34pub(crate) const RUNTIME_API_BLOCK_LIMITS_COUNT: u64 = 6;
35pub(crate) const ALLOWANCE_LIMIT_ERR: &str = "Calculation gas limit exceeded. Use your own RPC node with `--rpc-calculations-multiplier` parameter raised";

48    // Internal implementation of RPC call `gear_calculate_replyForHandle(..)`.
49    //
50    // The RPC call is used to figure out the reply that would be send
51    // on calling `Gear::send_message(..)` with following arguments.
52    pub(crate) fn calculate_reply_for_handle_impl(
53        origin: H256,
54        destination: ActorId,
55        payload: Vec<u8>,
56        gas_limit: u64,
57        value: u128,
58        allowance_multiplier: u64,
59    ) -> Result<ReplyInfo, String> {
60        // Enabling lazy-pages for this thread.
61        Self::enable_lazy_pages();
62
63        // Clearing queue.
64        QueueOf::<T>::clear();
65
66        // Calculating gas allowance for a whole operation,
67        // according to allowance multiplier.
68        let gas_allowance = allowance_multiplier.saturating_mul(BlockGasLimitOf::<T>::get());
69
70        // Updating gas allowance with calculated value.
71        Self::update_gas_allowance(gas_allowance);
72
73        // Casting types into runtime assoc-s.
74        let origin = origin.cast();
75        let value = value.unique_saturated_into();
76
77        // Preparing origin balance for extrinsic expenses.
78        let who = Self::prepare_origin_account(origin, gas_limit, value);
79
80        // Executing `send_message` call.
81        Self::send_message(who.into(), destination, payload, gas_limit, value, false)
82            .map_err(|e| Self::dispatch_err_to_string("send_message", e))?;
83
84        // Looking up queue head for message id sent above.
85        let (message_id, _) = Self::queue_head()?;
86
87        // Creating builtin dispatcher for queue processing.
88        let (builtin_dispatcher, _) = T::BuiltinDispatcherFactory::create();
89
90        // Creating new manager for queue processing.
91        let mut ext_manager = ExtManager::<T>::new(builtin_dispatcher);
92
93        // Queue processing loop.
94        //
95        // Running queue head message if exists.
96        while let Some((_, journal, _)) = Self::dequeue_head_and_run(&mut ext_manager, None)? {
97            // Looking through all notes in order to find required reply.
98            for note in &journal {
99                // Only paying attention on dispatch sends.
100                let JournalNote::SendDispatch { dispatch, .. } = note else {
101                    continue;
102                };
103
104                // Only paying attention if replies to `message_id`.
105                if let Some(code) = dispatch
106                    .reply_details()
107                    .map(ReplyDetails::into_parts)
108                    .and_then(|(replied_to, code)| replied_to.eq(&message_id).then_some(code))
109                {
110                    return Ok(ReplyInfo {
111                        payload: dispatch.payload_bytes().to_vec(),
112                        value: dispatch.value(),
113                        code,
114                    });
115                }
116            }
117
118            // Processing notes since reply wasn't found.
119            core_processor::handle_journal(journal, &mut ext_manager);
120
121            // If some message overcame block allowance, aborting processing.
122            if QueueProcessingOf::<T>::denied() {
123                return Err(ALLOWANCE_LIMIT_ERR.to_string());
124            }
125        }
126
127        // Ran out of messages in queue.
128        Err(String::from("Queue is empty, but reply wasn't found"))
129    }

131    // Internal implementation of RPC calls `gear_calculate*Entry*Gas(..)`.
132    //
133    // The RPC call is used to figure out required gas amount for successful
134    // execution of the message.
135    #[allow(clippy::too_many_arguments)]
136    pub(crate) fn calculate_gas_info_impl(
137        origin: H256,
138        kind: HandleKind,
139        initial_gas: u64,
140        payload: Vec<u8>,
141        value: u128,
142        allow_other_panics: bool,
143        allow_skip_zero_replies: bool,
144        allowance_multiplier: Option<u64>,
145    ) -> Result<GasInfo, String> {
146        // Enabling lazy-pages for this thread.
147        Self::enable_lazy_pages();
148
149        // Clearing queue.
150        QueueOf::<T>::clear();
151
152        // Calculating gas allowance for a whole operation,
153        // according to allowance multiplier.
154        let gas_allowance = allowance_multiplier
155            .unwrap_or(RUNTIME_API_BLOCK_LIMITS_COUNT)
156            .saturating_mul(BlockGasLimitOf::<T>::get());
157
158        // Updating gas allowance with calculated value.
159        Self::update_gas_allowance(gas_allowance);
160
161        // Casting types into runtime assoc-s.
162        let origin = origin.cast();
163        let value = value.unique_saturated_into();
164
165        // Preparing origin balance for extrinsic expenses.
166        let who = Self::prepare_origin_account(origin, initial_gas, value);
167
168        match kind {
169            // Executing `upload_program` call.
170            HandleKind::Init(code) => {
171                let salt = b"calculate_gas_salt".to_vec();
172
173                Self::upload_program(who.into(), code, salt, payload, initial_gas, value, false)
174                    .map_err(|e| Self::dispatch_err_to_string("upload_program", e))?;
175            }
176
177            // Executing `create_program` call.
178            HandleKind::InitByHash(code_id) => {
179                let salt = b"calculate_gas_salt".to_vec();
180
181                Self::create_program(
182                    who.into(),
183                    code_id,
184                    salt,
185                    payload,
186                    initial_gas,
187                    value,
188                    false,
189                )
190                .map_err(|e| Self::dispatch_err_to_string("create_program", e))?;
191            }
192
193            // Executing `send_message` call.
194            HandleKind::Handle(destination) => {
195                Self::send_message(who.into(), destination, payload, initial_gas, value, false)
196                    .map_err(|e| Self::dispatch_err_to_string("send_message", e))?;
197            }
198
199            // Executing `send_reply` call.
200            HandleKind::Reply(reply_to_id, _status_code) => {
201                Self::send_reply(who.into(), reply_to_id, payload, initial_gas, value, false)
202                    .map_err(|e| Self::dispatch_err_to_string("send_reply", e))?;
203            }
204
205            // Handle signal forbidden call.
206            HandleKind::Signal(_signal_from, _status_code) => {
207                return Err(String::from(
208                    "Gas calculation for `handle_signal` is not supported",
209                ));
210            }
211        };
212
213        // Looking up queue head for message id and destination sent above.
214        let (main_message_id, main_program_id) = Self::queue_head()?;
215
216        // Creating builtin dispatcher for queue processing.
217        let (builtin_dispatcher, _) = T::BuiltinDispatcherFactory::create();
218
219        // Creating new manager for queue processing.
220        let mut ext_manager = ExtManager::<T>::new(builtin_dispatcher);
221
222        // Creating forbidden funcs registry.
223        let forbidden_funcs = [SyscallName::GasAvailable];
224
225        // Getter for gas limit of the root message.
226        //
227        // For case when node is not consumed and has any (even zero) balance
228        // it means that it burned/sent all the funds and we must return it.
229        //
230        // For case when node is consumed and has zero balance it means that
231        // node moved its funds upstream to its ancestor. So this shouldn't
232        // be returned.
233        //
234        // For case when node is consumed and has non zero balance it means
235        // that it has gasless child that will consume gas further. So we
236        // handle this value as well.
237        let get_main_limit = || {
238            GasHandlerOf::<T>::get_limit(main_message_id)
239                .ok()
240                .or_else(|| {
241                    GasHandlerOf::<T>::get_limit_consumed(main_message_id)
242                        .ok()
243                        .filter(|limit| !limit.is_zero())
244                })
245        };
246
247        // Getter for identifying if message was in primary messages chain.
248        let from_main_chain = |msg_id| {
249            GasHandlerOf::<T>::get_origin_key(msg_id)
250                .map(|v| v == main_message_id.into())
251                .map_err(|_| Self::internal_err_string("Failed to get origin key"))
252        };
253
254        // Result to be returned.
255        let mut gas_info: GasInfo = Default::default();
256
257        // Queue processing loop.
258        //
259        // Running queue head message if exists.
260        while let Some((processed, journal, by_builtin)) =
261            Self::dequeue_head_and_run(&mut ext_manager, Some(forbidden_funcs.into()))?
262        {
263            // Defining if success reply was processed.
264            let success_reply = processed
265                .reply_details()
266                .map(|rd| rd.to_reply_code().is_success())
267                .unwrap_or(false);
268
269            // Extracting infallibly gas limit of processed message.
270            let gas_limit = processed.gas_limit().expect("Infallible");
271
272            // Defining if we skip checks for this message if allowed to.
273            let skip_if_allowed = !by_builtin && success_reply && gas_limit == 0;
274
275            // Looking through all notes in order to calculate gas properly.
276            for note in journal {
277                // Processing note.
278                core_processor::handle_journal(vec![note.clone()], &mut ext_manager);
279
280                // If some message overcame block allowance, aborting processing.
281                if QueueProcessingOf::<T>::denied() {
282                    return Err(ALLOWANCE_LIMIT_ERR.to_string());
283                }
284
285                // Querying gas limit of the main messages chain.
286                match get_main_limit() {
287                    // If some limit still exist, than checking the highest
288                    // diff from initial gas as calculated value.
289                    Some(remaining_gas) => {
290                        gas_info.min_limit = gas_info
291                            .min_limit
292                            .max(initial_gas.saturating_sub(remaining_gas));
293                    }
294
295                    // If limit no longer exists we need to check others for
296                    // infinite wait if they belong to main messages chain.
297                    None => {
298                        // Take into account that 'wait' syscall greedily
299                        // consumes all available gas.
300                        // Meanwhile, 'wait_for' and 'wait_up_to' should not
301                        // consume all available gas because of the limited
302                        // durations. If a duration is a big enough then it
303                        // won't matter how to calculate the limit:
304                        // it will be the same.
305                        if let JournalNote::WaitDispatch {
306                            waited_type: MessageWaitedType::Wait,
307                            ref dispatch,
308                            ..
309                        } = note
310                            && from_main_chain(dispatch.id())?
311                        {
312                            gas_info.min_limit = initial_gas;
313                        }
314                    }
315                }
316
317                // Parsing other types of the node for extra actions.
318                match note {
319                    // Checking sending for mailbox insertion (e.g. reserve).
320                    JournalNote::SendDispatch { dispatch, .. } => {
321                        // Extracting and casting destination to AccountId.
322                        let destination = dispatch.destination().cast();
323
324                        // Checking mailbox insertion and if newly created
325                        // dispatch is from main chain.
326                        //
327                        // NOTE: to pass `from_main_chain` call, message should
328                        // exist in system: at least in `Mailbox`.
329                        if MailboxOf::<T>::contains(&destination, &dispatch.id())
330                            && from_main_chain(dispatch.id())?
331                        {
332                            // Querying reserved balance for mailbox storing.
333                            //
334                            // NOTE: here goes extraction of the gas directly
335                            // from message, if gasless sent, than it's
336                            // queried from the storage tree.
337                            let gas_limit = dispatch
338                                .gas_limit()
339                                .or_else(|| GasHandlerOf::<T>::get_limit(dispatch.id()).ok())
340                                .ok_or_else(|| {
341                                    Self::internal_err_string(
342                                        "Failed to get gas limit after execution",
343                                    )
344                                })?;
345
346                            gas_info.reserved = gas_info.reserved.saturating_add(gas_limit);
347                        }
348                    }
349
350                    // Burning gas from main messages chain.
351                    JournalNote::GasBurned { amount, message_id } => {
352                        if from_main_chain(message_id)? {
353                            gas_info.burned = gas_info.burned.saturating_add(amount);
354                        }
355                    }
356
357                    // Checking execution for panic happened.
358                    JournalNote::MessageDispatched {
359                        outcome:
360                            CoreDispatchOutcome::MessageTrap { trap, .. }
361                            | CoreDispatchOutcome::InitFailure { reason: trap, .. },
362                        message_id,
363                        ..
364                    } if (message_id == main_message_id || !allow_other_panics)
365                        && !(skip_if_allowed && allow_skip_zero_replies) =>
366                    {
367                        return Err(format!("Program terminated with a trap: '{trap}'"));
368                    }
369
370                    _ => (),
371                }
372            }
373        }
374
375        // Defining if message is kept by waitlist.
376        gas_info.waited = WaitlistOf::<T>::contains(&main_program_id, &main_message_id);
377
378        // Returning result.
379        Ok(gas_info)
380    }

382    pub(crate) fn read_state_using_wasm_impl(
383        program_id: ActorId,
384        payload: Vec<u8>,
385        function: impl Into<String>,
386        wasm: Vec<u8>,
387        argument: Option<Vec<u8>>,
388        allowance_multiplier: Option<u64>,
389    ) -> Result<Vec<u8>, String> {
390        Self::enable_lazy_pages();
391
392        let schedule = T::Schedule::get();
393
394        if u32::try_from(wasm.len()).unwrap_or(u32::MAX) > schedule.limits.code_len {
395            return Err("Wasm too big".into());
396        }
397
398        let code = Code::try_new_mock_with_rules(
399            wasm,
400            |module| schedule.rules(module),
401            TryNewCodeConfig::new_no_exports_check(),
402        )
403        .map_err(|e| format!("Failed to construct program: {e:?}"))?;
404
405        if u32::try_from(code.instrumented_code().bytes().len()).unwrap_or(u32::MAX)
406            > schedule.limits.code_len
407        {
408            return Err("Wasm after instrumentation too big".into());
409        }
410
411        let code_and_id = CodeAndId::new(code);
412
413        let (_, instrumented_code, code_metadata) = code_and_id.into_parts().0.into_parts();
414
415        let payload_arg = payload;
416        let mut payload = argument.unwrap_or_default();
417        payload.append(&mut Self::read_state_impl(
418            program_id,
419            payload_arg,
420            allowance_multiplier,
421        )?);
422
423        let block_info = BlockInfo {
424            height: Self::block_number().unique_saturated_into(),
425            timestamp: <pallet_timestamp::Pallet<T>>::get().unique_saturated_into(),
426        };
427
428        let gas_allowance = allowance_multiplier
429            .unwrap_or(RUNTIME_API_BLOCK_LIMITS_COUNT)
430            .saturating_mul(BlockGasLimitOf::<T>::get());
431
432        Self::update_gas_allowance(gas_allowance);
433
434        core_processor::informational::execute_for_reply::<Ext, String>(
435            function.into(),
436            instrumented_code,
437            code_metadata,
438            None,
439            None,
440            payload,
441            gas_allowance,
442            block_info,
443        )
444    }

526    // Returns code and allocations of the given program id.
527    fn code_with_memory(program_id: ActorId) -> Result<CodeWithMemoryData, String> {
528        // Load active program from storage.
529        let program: ActiveProgram<_> = ProgramStorageOf::<T>::get_program(program_id)
530            .ok_or(String::from("Program not found"))?
531            .try_into()
532            .map_err(|e| format!("Get active program error: {e:?}"))?;
533
534        let code_id = program.code_id.cast();
535
536        let code_metadata = T::CodeStorage::get_code_metadata(code_id)
537            .ok_or_else(|| format!("Code '{code_id:?}' not found for program '{program_id:?}'"))?;
538
539        let schedule = T::Schedule::get();
540
541        // Check if the code needs to be reinstrumented.
542        let needs_reinstrumentation = match code_metadata.instrumentation_status() {
543            InstrumentationStatus::NotInstrumented => {
544                log::debug!(
545                    "Instrumented code doesn't exists for program '{program_id:?}' \
546                     we need to instrument it with instructions weights version {}",
547                    schedule.instruction_weights.version
548                );
549
550                true
551            }
552            InstrumentationStatus::Instrumented { version, .. } => {
553                version != schedule.instruction_weights.version
554            }
555            InstrumentationStatus::InstrumentationFailed { version } => {
556                if version == schedule.instruction_weights.version {
557                    return Err(format!(
558                        "Re-instrumentation already failed for program '{program_id:?}' \
559                        with instructions weights version {version}"
560                    ));
561                }
562
563                true
564            }
565        };
566
567        let instrumented_code_and_metadata = if needs_reinstrumentation {
568            Pallet::<T>::reinstrument_code(code_id, code_metadata, &schedule)
569                .map_err(|e| format!("Code {code_id:?} failed reinstrumentation: {e:?}"))?
570        } else {
571            let instrumented_code =
572                T::CodeStorage::get_instrumented_code(code_id).ok_or_else(|| {
573                    format!("Program '{program_id:?}' exists so must do code '{code_id:?}'")
574                })?;
575
576            InstrumentedCodeAndMetadata {
577                instrumented_code,
578                metadata: code_metadata,
579            }
580        };
581
582        let allocations = ProgramStorageOf::<T>::allocations(program_id).unwrap_or_default();
583
584        Ok(CodeWithMemoryData {
585            instrumented_code: instrumented_code_and_metadata.instrumented_code,
586            code_metadata: instrumented_code_and_metadata.metadata,
587            allocations,
588            memory_infix: program.memory_infix,
589        })
590    }

592    // Prepares account id to be able to execute some extrinsic in terms of funds.
593    fn prepare_origin_account(
594        origin: AccountIdOf<T>,
595        gas: u64,
596        value: BalanceOf<T>,
597    ) -> RawOrigin<AccountIdOf<T>> {
598        // Querying transferrable balance of the origin taking into account a possibility of
599        // a part of its `free` balance being `frozen`.
600        let origin_balance = <CurrencyOf<T> as fungible::Inspect<_>>::reducible_balance(
601            &origin,
602            Preservation::Expendable,
603            Fortitude::Polite,
604        );
605
606        // Calculating amount of value to be paid for gas.
607        let value_for_gas = <T as pallet_gear_bank::Config>::GasMultiplier::get().gas_to_value(gas);
608
609        // Required balance of the account.
610        let required_balance = CurrencyOf::<T>::minimum_balance()
611            .saturating_add(value_for_gas)
612            .saturating_add(value);
613
614        // Updating balance of the account.
615        let _ = CurrencyOf::<T>::deposit_creating(
616            &origin,
617            required_balance.saturating_sub(origin_balance),
618        );
619
620        // Returning origin account as signed origin.
621        RawOrigin::Signed(origin)
622    }

624    // Returns none if queue is empty, otherwise - processed message,
625    // resulting journal notes of the processing and bool, defining
626    // was it processed by builtin actor or not.
627    fn dequeue_head_and_run(
628        ext_manager: &mut ExtManager<T>,
629        forbidden_funcs: Option<BTreeSet<SyscallName>>,
630    ) -> Result<Option<(Dispatch, Vec<JournalNote>, bool)>, String> {
631        // Extracting queued dispatch.
632        let head =
633            QueueOf::<T>::dequeue().map_err(|_| Self::internal_err_string("Queue corrupted"))?;
634
635        let Some(dispatch) = head else {
636            return Ok(None);
637        };
638
639        // Extracting destination from dispatch.
640        let destination = dispatch.destination();
641
642        // Querying gas limit for dispatch.
643        let gas_limit = GasHandlerOf::<T>::get_limit(dispatch.id())
644            .map_err(|_| Self::internal_err_string("Failed to get gas limit"))?;
645
646        // Storing processed dispatch.
647        let processed = Dispatch::new(
648            dispatch.kind(),
649            Message::new(
650                dispatch.id(),
651                dispatch.source(),
652                dispatch.destination(),
653                dispatch
654                    .payload_bytes()
655                    .to_vec()
656                    .try_into()
657                    .expect("Infallible"),
658                Some(gas_limit),
659                dispatch.value(),
660                dispatch.details(),
661            ),
662        );
663
664        // Processing of the message, if destination is builtin actor.
665        let builtin_dispatcher = ext_manager.builtins();
666        if let Some(info) = builtin_dispatcher.lookup(&destination) {
667            let journal = builtin_dispatcher.run(info, dispatch, gas_limit);
668            return Ok(Some((processed, journal, true)));
669        }
670
671        let mut block_config = Self::block_config();
672
673        if let Some(forbidden_funcs) = forbidden_funcs {
674            block_config.forbidden_funcs = forbidden_funcs;
675        }
676
677        // Program's balance that it can spend during a message execution.
678        let disposable_balance = <CurrencyOf<T> as fungible::Inspect<_>>::reducible_balance(
679            &destination.cast(),
680            Preservation::Expendable,
681            Fortitude::Polite,
682        );
683
684        // Processing of the message, if destination is common program.
685        let journal = Self::run_queue_step(QueueStep {
686            block_config: &block_config,
687            gas_limit,
688            dispatch,
689            balance: disposable_balance.unique_saturated_into(),
690        });
691
692        Ok(Some((processed, journal, false)))
693    }

695    // Converts given dispatch error into dedicated runtime api string format.
696    fn dispatch_err_to_string(
697        extrinsic_name: &'static str,
698        e: DispatchErrorWithPostInfo<PostDispatchInfo>,
699    ) -> String {
700        // Extracting index of module returned error, if possible.
701        let error_module_idx = match e.error {
702            DispatchError::Module(ModuleError { index, .. }) => Some(index as usize),
703            _ => None,
704        };
705
706        // Converting dispatch error into string representation in default impl.
707        let error_message: &'static str = e.into();
708
709        // Creating result message.
710        let mut res = format!("Extrinsic `gear.{extrinsic_name}` failed: '{error_message}'");
711
712        // Extracting `pallet_gear` index from runtime to compare with dispatch error.
713        let Some(gear_module_idx) = PalletInfoOf::<T>::index::<Self>() else {
714            return Self::internal_err_string("No index found for `pallet_gear` in the runtime");
715        };
716
717        // Appending result message with pallet index returned error, if not this.
718        if let Some(module_idx) = error_module_idx.filter(|i| *i != gear_module_idx) {
719            res = format!("{res} (pallet index of the error: {module_idx}");
720        }
721
722        res
723    }

725    // Queries first element of the queue and extracts its message id and destination.
726    fn queue_head() -> Result<(MessageId, ActorId), String> {
727        QueueOf::<T>::iter()
728            .next()
729            .ok_or_else(|| Self::internal_err_string("Failed to get last message from the queue"))
730            .and_then(|queued| {
731                queued
732                    .map(|dispatch| (dispatch.id(), dispatch.destination()))
733                    .map_err(|_| Self::internal_err_string("Failed to extract queued dispatch"))
734            })
735    }
