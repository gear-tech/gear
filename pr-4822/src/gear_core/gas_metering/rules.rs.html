26/// This type provides the functionality of [`ConstantCostRules`].
27///
28/// This implementation of [`Rules`] will also check the WASM module for
29/// instructions that are not supported by Gear Protocol.
30pub struct CustomConstantCostRules {
31    constant_cost_rules: ConstantCostRules,
32}
33
34impl CustomConstantCostRules {
35    /// Create a new [`CustomConstantCostRules`].
36    ///
37    /// Uses `instruction_cost` for every instruction and `memory_grow_cost` to
38    /// dynamically meter the memory growth instruction.
39    pub fn new(instruction_cost: u32, memory_grow_cost: u32, call_per_local_cost: u32) -> Self {
40        Self {
41            constant_cost_rules: ConstantCostRules::new(
42                instruction_cost,
43                memory_grow_cost,
44                call_per_local_cost,
45            ),
46        }
47    }
48}
49
50impl Default for CustomConstantCostRules {
51    /// Uses instruction cost of `1` and disables memory growth instrumentation.
52    fn default() -> Self {
53        Self {
54            constant_cost_rules: ConstantCostRules::new(1, 0, 1),
55        }
56    }
57}
58
59impl Rules for CustomConstantCostRules {
60    fn instruction_cost(&self, instruction: &Instruction) -> Option<u32> {
61        if instruction.is_user_forbidden() {
62            return None;
63        }
64
65        self.constant_cost_rules.instruction_cost(instruction)
66    }
67
68    fn memory_grow_cost(&self) -> MemoryGrowCost {
69        self.constant_cost_rules.memory_grow_cost()
70    }
71
72    fn call_per_local_cost(&self) -> u32 {
73        self.constant_cost_rules.call_per_local_cost()
74    }
75}

77/// This type provides real gas cost of instructions on pallet-gear.
78pub struct ScheduleRules<'a> {
79    schedule: &'a Schedule,
80    params: Vec<u32>,
81}
82
83impl Schedule {
84    /// Returns real gas rules that are used by pallet gear.
85    pub fn rules(&self, module: &Module) -> impl Rules + use<'_> {
86        ScheduleRules {
87            schedule: self,
88            params: module
89                .type_section
90                .as_ref()
91                .iter()
92                .copied()
93                .flatten()
94                .map(|func| func.params().len() as u32)
95                .collect(),
96        }
97    }
98}
99
100impl Rules for ScheduleRules<'_> {
101    fn instruction_cost(&self, instruction: &Instruction) -> Option<u32> {
102        use Instruction::*;
103
104        let w = &self.schedule.instruction_weights;
105        let max_params = self.schedule.limits.parameters;
106
107        Some(match instruction {
108            // Returning None makes the gas instrumentation fail which we intend for
109            // unsupported or unknown instructions.
110            MemoryGrow { .. } => return None,
111            //
112            End | Unreachable | Return | Else | Block { .. } | Loop { .. } | Nop | Drop => 0,
113            I32Const { .. } | I64Const { .. } => w.i64const,
114            I32Load { .. }
115            | I32Load8S { .. }
116            | I32Load8U { .. }
117            | I32Load16S { .. }
118            | I32Load16U { .. } => w.i32load,
119            I64Load { .. }
120            | I64Load8S { .. }
121            | I64Load8U { .. }
122            | I64Load16S { .. }
123            | I64Load16U { .. }
124            | I64Load32S { .. }
125            | I64Load32U { .. } => w.i64load,
126            I32Store { .. } | I32Store8 { .. } | I32Store16 { .. } => w.i32store,
127            I64Store { .. } | I64Store8 { .. } | I64Store16 { .. } | I64Store32 { .. } => {
128                w.i64store
129            }
130            Select => w.select,
131            If { .. } => w.r#if,
132            Br { .. } => w.br,
133            BrIf { .. } => w.br_if,
134            Call { .. } => w.call,
135            LocalGet { .. } => w.local_get,
136            LocalSet { .. } => w.local_set,
137            LocalTee { .. } => w.local_tee,
138            GlobalGet { .. } => w.global_get,
139            GlobalSet { .. } => w.global_set,
140            MemorySize { .. } => w.memory_current,
141            CallIndirect(idx) => {
142                let params = self
143                    .params
144                    .get(*idx as usize)
145                    .copied()
146                    .unwrap_or(max_params);
147                w.call_indirect
148                    .saturating_add(w.call_indirect_per_param.saturating_mul(params))
149            }
150            BrTable(targets) => w
151                .br_table
152                .saturating_add(w.br_table_per_entry.saturating_mul(targets.len())),
153            I32Clz => w.i32clz,
154            I64Clz => w.i64clz,
155            I32Ctz => w.i32ctz,
156            I64Ctz => w.i64ctz,
157            I32Popcnt => w.i32popcnt,
158            I64Popcnt => w.i64popcnt,
159            I32Eqz => w.i32eqz,
160            I64Eqz => w.i64eqz,
161            // TODO: rename fields
162            I64ExtendI32S => w.i64extendsi32,
163            I64ExtendI32U => w.i64extendui32,
164            I32WrapI64 => w.i32wrapi64,
165            I32Eq => w.i32eq,
166            I64Eq => w.i64eq,
167            I32Ne => w.i32ne,
168            I64Ne => w.i64ne,
169            I32LtS => w.i32lts,
170            I64LtS => w.i64lts,
171            I32LtU => w.i32ltu,
172            I64LtU => w.i64ltu,
173            I32GtS => w.i32gts,
174            I64GtS => w.i64gts,
175            I32GtU => w.i32gtu,
176            I64GtU => w.i64gtu,
177            I32LeS => w.i32les,
178            I64LeS => w.i64les,
179            I32LeU => w.i32leu,
180            I64LeU => w.i64leu,
181            I32GeS => w.i32ges,
182            I64GeS => w.i64ges,
183            I32GeU => w.i32geu,
184            I64GeU => w.i64geu,
185            I32Add => w.i32add,
186            I64Add => w.i64add,
187            I32Sub => w.i32sub,
188            I64Sub => w.i64sub,
189            I32Mul => w.i32mul,
190            I64Mul => w.i64mul,
191            I32DivS => w.i32divs,
192            I64DivS => w.i64divs,
193            I32DivU => w.i32divu,
194            I64DivU => w.i64divu,
195            I32RemS => w.i32rems,
196            I64RemS => w.i64rems,
197            I32RemU => w.i32remu,
198            I64RemU => w.i64remu,
199            I32And => w.i32and,
200            I64And => w.i64and,
201            I32Or => w.i32or,
202            I64Or => w.i64or,
203            I32Xor => w.i32xor,
204            I64Xor => w.i64xor,
205            I32Shl => w.i32shl,
206            I64Shl => w.i64shl,
207            I32ShrS => w.i32shrs,
208            I64ShrS => w.i64shrs,
209            I32ShrU => w.i32shru,
210            I64ShrU => w.i64shru,
211            I32Rotl => w.i32rotl,
212            I64Rotl => w.i64rotl,
213            I32Rotr => w.i32rotr,
214            I64Rotr => w.i64rotr,
215            I32Extend8S => w.i32extend8s,
216            I32Extend16S => w.i32extend16s,
217            I64Extend8S => w.i64extend8s,
218            I64Extend16S => w.i64extend16s,
219            I64Extend32S => w.i64extend32s,
220        })
221    }
222
223    fn memory_grow_cost(&self) -> MemoryGrowCost {
224        MemoryGrowCost::Free
225    }
226
227    fn call_per_local_cost(&self) -> u32 {
228        self.schedule.instruction_weights.call_per_local
229    }
