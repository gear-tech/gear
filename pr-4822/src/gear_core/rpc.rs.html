19//! This module contains types commonly used as output in RPC calls.
20
21use alloc::vec::Vec;
22use gear_core_errors::ReplyCode;
23use parity_scale_codec::{Decode, Encode};
24use scale_info::TypeInfo;
25
26/// Pre-calculated gas consumption estimate for a message.
27///
28/// Intended to be used as a result in `calculateGasFor*` RPC calls.
29#[derive(Clone, Debug, Default, PartialEq, Eq, Encode, Decode, TypeInfo)]
30#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
31pub struct GasInfo {
32    /// The minimum amount of gas required for successful execution.
33    pub min_limit: u64,
34    /// The amount of gas that would be reserved.
35    pub reserved: u64,
36    /// The amount of gas that would be burned.
37    pub burned: u64,
38    /// The amount of gas that may be returned.
39    pub may_be_returned: u64,
40    /// Indicates whether the message was placed into the waitlist.
41    ///
42    /// This flag signifies that `min_limit` guarantees apply only to the first execution attempt.
43    pub waited: bool,
44}
45
46/// Pre-calculated reply information.
47///
48/// Intended to be used as a result in `calculateReplyFor*` RPC calls.
49#[derive(Clone, Debug, PartialEq, Eq, Encode, Decode, TypeInfo)]
50#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
51pub struct ReplyInfo {
52    /// Payload of the reply.
53    #[cfg_attr(feature = "std", serde(with = "impl_serde::serialize"))]
54    pub payload: Vec<u8>,
55    /// Value attached to the reply.
56    pub value: u128,
57    /// Reply code of the reply.
58    pub code: ReplyCode,
59}
