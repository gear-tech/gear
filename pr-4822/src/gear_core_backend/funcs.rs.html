103/// Trait is implemented for functions.

106/// `Args` is to make specialization based on function arguments
107/// `Ext` and `Res` are for syscall itself (`Syscall<Ext, Res>`)
108pub(crate) trait SyscallBuilder<Caller, Args: ?Sized, Res, Call>
109where
110    Call: Syscall<Caller, Res>,
111{
112    fn build(
113        self,
114        ctx: &mut MemoryCallerContext<Caller>,
115        args: &[Value],
116    ) -> Result<Call, HostError>;

119impl<Caller, Res, Call, Builder> SyscallBuilder<Caller, (), Res, Call> for Builder
120where
121    Builder: FnOnce() -> Call,
122    Call: Syscall<Caller, Res>,
123{
124    fn build(self, _: &mut MemoryCallerContext<Caller>, args: &[Value]) -> Result<Call, HostError> {
125        let _: [Value; 0] = args.try_into().map_err(|_| HostError)?;
126        Ok((self)())
127    }
128}

130impl<Caller, Res, Call, Builder> SyscallBuilder<Caller, [Value], Res, Call> for Builder
131where
132    Builder: for<'a> FnOnce(&'a [Value]) -> Call,
133    Call: Syscall<Caller, Res>,
134{
135    fn build(self, _: &mut MemoryCallerContext<Caller>, args: &[Value]) -> Result<Call, HostError> {
136        Ok((self)(args))
137    }
138}

140// implement [`SyscallBuilder`] for functions with different amount of arguments
141macro_rules! impl_syscall_builder {
142    ($($generic:ident),+) => {
143        #[allow(non_snake_case)]
144        impl<Caller, Ext, Res, Call, Builder, $($generic),+> SyscallBuilder<Caller, ($($generic,)+), Res, Call>
145            for Builder
146        where
147            Builder: FnOnce($($generic),+) -> Call,
148            Call: Syscall<Caller, Res>,
149            Caller: AsContextExt<State = HostState<Ext, BackendMemory<ExecutorMemory>>>,
150            Ext: BackendExternalities + 'static,
151            $( $generic: SyscallArg, )+
152        {
153            fn build(self, ctx: &mut MemoryCallerContext<Caller>, args: &[Value]) -> Result<Call, HostError> {
154                let ARGS_AMOUNT: usize = 0 $( + $generic::REQUIRED_ARGS )+;
155
156                if args.len() != ARGS_AMOUNT {
157                    return Err(HostError);
158                }
159
160                let mut registry: Option<MemoryAccessRegistry<Caller>> = None;
161
162                let mut index = 0;
163                $(
164                    let args_count = $generic::REQUIRED_ARGS;
165                    let args_slice = &args[index..index + args_count];
166                    let $generic = $generic::pre_process(&mut registry, args_slice)?;
167                    index += args_count;
168                )+
169
170                debug_assert_eq!(index, ARGS_AMOUNT);
171
172                if let Some(registry) = registry {
173                    let io = registry.pre_process(&mut ctx.caller_wrap);
174                    ctx.memory_wrap.set_io(io);
175                }
176
177                $(
178                    let $generic = $generic::post_process($generic, ctx);
179                )+
180
181                Ok(((self)($($generic),+)))
182            }
183        }
184    };
185}

195/// Fallible syscall context that parses `gas` and `err_ptr` arguments.
196#[derive(Copy, Clone)]
197struct FallibleSyscallContext {
198    res_ptr: u32,
199}
200
201impl SyscallContext for FallibleSyscallContext {
202    fn from_args(args: &[Value]) -> Result<(Self, Gas, &[Value]), HostError> {
203        let (gas, args) = args.split_first().ok_or(HostError)?;
204        let gas: Gas = SyscallValue(*gas).try_into()?;
205        let (res_ptr, args) = args.split_last().ok_or(HostError)?;
206        let res_ptr: u32 = SyscallValue(*res_ptr).try_into()?;
207        Ok((FallibleSyscallContext { res_ptr }, gas, args))
208    }
209}

211/// Fallible syscall that calls [`MemoryCallerContext::run_fallible`] underneath.
212#[derive(Copy, Clone)]
213struct FallibleSyscall<E, F> {
214    token: CostToken,
215    error: PhantomData<E>,
216    f: F,
217}
218
219impl<F> FallibleSyscall<(), F> {
220    fn new<E>(token: CostToken, f: F) -> FallibleSyscall<E, F> {
221        FallibleSyscall {
222            token,
223            error: PhantomData,
224            f,
225        }
226    }
227}
228
229impl<T, E, F, Caller, Ext> Syscall<Caller, ()> for FallibleSyscall<E, F>
230where
231    F: FnOnce(&mut MemoryCallerContext<Caller>) -> Result<T, RunFallibleError>,
232    E: From<Result<T, u32>>,
233    Caller: AsContextExt<State = HostState<Ext, BackendMemory<ExecutorMemory>>>,
234    Ext: BackendExternalities + 'static,
235{
236    type Context = FallibleSyscallContext;
237
238    fn execute(
239        self,
240        caller: &mut MemoryCallerContext<Caller>,
241        context: Self::Context,
242        syscall_name: SyscallName,
243    ) -> Result<(Gas, ()), HostError> {
244        let Self { token, f, .. } = self;
245        let FallibleSyscallContext { res_ptr } = context;
246        caller.check_func_forbiddenness(syscall_name)?;
247        caller.run_fallible::<T, _, E>(res_ptr, token, f)
248    }
249}

263/// Infallible syscall that calls [`MemoryCallerContext::run_any`] underneath
264#[derive(Copy, Clone)]
265struct InfallibleSyscall<F> {
266    token: CostToken,
267    f: F,
268}
269
270impl<F> InfallibleSyscall<F> {
271    fn new(token: CostToken, f: F) -> Self {
272        Self { token, f }
273    }
274}

276impl<T, F, Caller, Ext> Syscall<Caller, T> for InfallibleSyscall<F>
277where
278    F: FnOnce(&mut MemoryCallerContext<Caller>) -> Result<T, UndefinedTerminationReason>,
279    Caller: AsContextExt<State = HostState<Ext, BackendMemory<ExecutorMemory>>>,
280    Ext: BackendExternalities + 'static,
281{
282    type Context = InfallibleSyscallContext;
283
284    fn execute(
285        self,
286        caller: &mut MemoryCallerContext<Caller>,
287        _ctx: Self::Context,
288        syscall_name: SyscallName,
289    ) -> Result<(Gas, T), HostError> {
290        let Self { token, f } = self;
291        caller.check_func_forbiddenness(syscall_name)?;
292        caller.run_any::<T, _>(token, f)
293    }
294}

300impl<Caller, Ext> FuncsHandler<Caller>
301where
302    Caller: AsContextExt<State = HostState<Ext, BackendMemory<ExecutorMemory>>>,
303    Ext: BackendExternalities + 'static,

308    pub fn execute<Builder, Args, Res, Call>(
309        caller: &mut Caller,
310        args: &[Value],
311        builder: Builder,
312        syscall_name: SyscallName,
313    ) -> Result<WasmReturnValue, HostError>
314    where
315        Builder: SyscallBuilder<Caller, Args, Res, Call>,
316        Args: ?Sized,
317        Call: Syscall<Caller, Res>,
318        Res: Into<SyscallReturnValue>,
319    {
320        crate::log::trace_syscall::<Builder>(args);
321
322        let mut memory_caller_context = MemoryCallerContext::new(caller);
323
324        let (ctx, gas, args) = Call::Context::from_args(args)?;
325
326        memory_caller_context
327            .caller_wrap
328            .state_mut()
329            .ext
330            .decrease_current_counter_to(gas);
331
332        let syscall = builder.build(&mut memory_caller_context, args)?;
333
334        let (gas, value) = syscall.execute(&mut memory_caller_context, ctx, syscall_name)?;
335
336        let value = value.into();
337
338        Ok(WasmReturnValue {
339            gas: gas as i64,
340            inner: value.0,
341        })
342    }

658                        Ok(0)
659                    }
660                    Err(e) => {
661                        log::trace!("Free range {start:?}:{end:?} failed: {e}");
662                        Ok(1)
663                    }
664                }
665            },
666        )

1294    fn out_of_gas(ctx: &mut MemoryCallerContext<Caller>) -> UndefinedTerminationReason {
1295        let ext = ctx.caller_wrap.ext_mut();
1296        let current_counter = ext.current_counter_type();
1297        log::trace!(target: "syscalls", "system_break(OutOfGas): Current counter in global represents {current_counter:?}");
1298
1299        if current_counter == CounterType::GasAllowance {
1300            // We manually decrease it to 0 because global won't be affected
1301            // since it didn't pass comparison to argument of `gas_charge()`
1302            ext.decrease_current_counter_to(0);
1303        }
1304
1305        ActorTerminationReason::from(current_counter).into()
1306    }

1312    pub fn system_break(code: u32) -> impl Syscall<Caller> {
1313        InfallibleSyscall::new(
1314            CostToken::Null,
1315            move |ctx: &mut MemoryCallerContext<Caller>| {
1316                // At the instrumentation level, we can only use variants of the `SystemBreakCode`,
1317                // so we should never reach `unreachable!("{err_msg}")`.
1318                let termination_reason = SystemBreakCode::try_from(code)
1319                    .map(|system_break_code| match system_break_code {
1320                        SystemBreakCode::OutOfGas => Self::out_of_gas(ctx),
1321                        SystemBreakCode::StackLimitExceeded => Self::stack_limit_exceeded(),
1322                    })
1323                    .unwrap_or_else(|e| {
1324                        let err_msg = format!(
1325                            "system_break: Invalid system break code. \
1326                        System break code - {code}. \
1327                        Got error - {e}"
1328                        );
1329
1330                        log::error!("{err_msg}");
1331                        unreachable!("{err_msg}")
1332                    });
1333                Err(termination_reason)
1334            },
1335        )
1336    }
