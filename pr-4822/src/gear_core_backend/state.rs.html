19use crate::{
20    BackendExternalities,
21    error::{
22        ActorTerminationReason, TerminationReason, TrapExplanation, UndefinedTerminationReason,
23    },
24};
25use core::fmt::Debug;
26
27pub type HostState<Ext, Mem> = Option<State<Ext, Mem>>;
28
29pub struct State<Ext, Mem> {
30    pub ext: Ext,
31    pub memory: Mem,
32    pub termination_reason: UndefinedTerminationReason,
33}
34
35impl<Ext: BackendExternalities, Mem> State<Ext, Mem> {
36    /// Transforms [`Self`] into tuple of externalities, memory and
37    /// termination reason returned after the execution.
38    pub fn into_parts(self) -> (Ext, UndefinedTerminationReason) {
39        let State {
40            ext,
41            termination_reason,
42            ..
43        } = self;
44        (ext, termination_reason)
45    }
46
47    /// Terminates backend work after execution.
48    ///
49    /// The function handles `res`, which is the result of gear wasm
50    /// program entry point invocation, and the termination reason.
51    ///
52    /// If the `res` is `Ok`, then execution considered successful
53    /// and the termination reason will have the corresponding value.
54    ///
55    /// If the `res` is `Err`, then execution is considered to end
56    /// with an error and the actual termination reason, which stores
57    /// more precise information about the error, is returned.
58    ///
59    /// There's a case, when `res` is `Err`, but termination reason has
60    /// a value for the successful ending of the execution. This is the
61    /// case of calling `unreachable` panic in the program.
62    pub fn terminate<T: Debug, WasmCallErr: Debug>(
63        self,
64        res: Result<T, WasmCallErr>,
65        gas: u64,
66    ) -> (Ext, TerminationReason) {
67        log::trace!("Execution result = {res:?}");
68
69        let (mut ext, termination_reason) = self.into_parts();
70        let termination_reason = termination_reason.define(ext.current_counter_type());
71
72        ext.decrease_current_counter_to(gas);
73
74        let termination_reason = if res.is_err() {
75            if matches!(
76                termination_reason,
77                TerminationReason::Actor(ActorTerminationReason::Success)
78            ) {
79                ActorTerminationReason::Trap(TrapExplanation::Unknown).into()
80            } else {
81                termination_reason
82            }
83        } else if matches!(
84            termination_reason,
85            TerminationReason::Actor(ActorTerminationReason::Success)
86        ) {
87            termination_reason
88        } else {
89            let err_msg = "State::terminate: Termination reason is not success, but executor successfully ends execution";
90
91            log::error!("{err_msg}");
92            unreachable!("{err_msg}")
93        };
94
95        (ext, termination_reason)
96    }
97}
