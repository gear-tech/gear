19//! Module for messenger implementation.
20//!
21//! Messenger provides API for all available gear message storing.
22
23use crate::storage::{
24    Counted, CountedByKey, Counter, DequeueError, Interval, IterableByKeyMap, IterableMap, Mailbox,
25    MailboxError, MapStorage, Queue, Toggler, Waitlist, WaitlistError,
26};
27use core::fmt::Debug;
28
29/// Represents messenger's logic of centralized message processing behavior.
30pub trait Messenger {
31    /// Block number type of the messenger.
32    type BlockNumber;
33    /// Capacity of the messenger.
34    ///
35    /// This type defines length type of the queue, sent and
36    /// dequeued messages within same block amount type.
37    type Capacity;
38    /// Inner error type generated by gear's storage types.
39    type Error: Debug + DequeueError + MailboxError + WaitlistError;
40    /// Output error of each storage algorithm.
41    ///
42    /// Implements `From<Self::Error>` to be able to return
43    /// any required error type.
44    type OutputError: From<Self::Error> + Debug;
45
46    /// First key of the mailbox storage.
47    ///
48    /// Present to clarify compiler behavior over associated types.
49    type MailboxFirstKey;
50    /// Second key of the mailbox storage.
51    ///
52    /// Present to clarify compiler behavior over associated types.
53    type MailboxSecondKey;
54    /// Stored values type for `Self::Mailbox`.
55    ///
56    /// Present to clarify compiler behavior over associated types.
57    type MailboxedMessage;
58    /// Stored values type for `Self::Queue`.
59    ///
60    /// Present to clarify compiler behavior over associated types.
61    type QueuedDispatch;
62    /// Stored values type for `Self::DispatchStash`.
63    ///
64    /// Present to clarify compiler behavior over associated types.
65    type DelayedDispatch;
66    /// First key of the waitlist storage.
67    ///
68    /// Present to clarify compiler behavior over associated types.
69    type WaitlistFirstKey;
70    /// Second key of the waitlist storage.
71    ///
72    /// Present to clarify compiler behavior over associated types.
73    type WaitlistSecondKey;
74    /// Stored values type for `Self::Waitlist`.
75    ///
76    /// Present to clarify compiler behavior over associated types.
77    type WaitlistedMessage;
78    /// Key for value types for `Self::DispatchStash`.
79    ///
80    /// Present to clarify compiler behavior over associated types.
81    type DispatchStashKey;
82
83    /// Amount of messages sent from outside (from users)
84    ///    within the current block.
85    ///
86    /// Used as local counter for `MessageId` generation.
87    type Sent: Counter<Value = Self::Capacity>;
88
89    /// Amount of messages dequeued with the current block.
90    ///
91    /// Used for depositing informational event about how much messages
92    /// were took from queue in `process_queue` execution.
93    type Dequeued: Counter<Value = Self::Capacity>;
94
95    /// Allowance of queue processing.
96    ///
97    /// Used for checking could `process_queue` continue it's execution.
98    /// Execution finishes, once message requeued at the end of the queue,
99    /// because it alerts, that this execution exceed gas allowance of the
100    /// current block by gear's processing algorithm.
101    type QueueProcessing: Toggler;
102
103    /// Gear message queue.
104    ///
105    /// Message queue contains only messages addressed to programs.
106    /// Messages from queue process on idle of each block in `process_queue`,
107    /// function, except case of runtime upgrade - then processing skipped.
108    type Queue: Queue<Value = Self::QueuedDispatch, Error = Self::Error, OutputError = Self::OutputError>
109        + Counted<Length = Self::Capacity>
110        + IterableMap<Result<Self::QueuedDispatch, Self::OutputError>>;
111
112    /// Gear mailbox.
113    ///
114    /// Mailbox contains only messages addressed to user accounts.
115    /// Any address meant as user account if it's not program id.
116    ///
117    /// Only mailbox owner (user with message's destination address)
118    /// can claim value from the message, removing it afterward, or claim
119    /// and send reply on received message, if it still present (#642).
120    type Mailbox: Mailbox<
121            Key1 = Self::MailboxFirstKey,
122            Key2 = Self::MailboxSecondKey,
123            Value = Self::MailboxedMessage,
124            BlockNumber = Self::BlockNumber,
125            Error = Self::Error,
126            OutputError = Self::OutputError,
127        > + CountedByKey<Key = Self::MailboxFirstKey, Length = usize>
128        + IterableMap<(Self::MailboxedMessage, Interval<Self::BlockNumber>)>
129        + IterableByKeyMap<
130            (Self::MailboxedMessage, Interval<Self::BlockNumber>),
131            Key = Self::MailboxFirstKey,
132        >;
133
134    /// Gear waitlist.
135    ///
136    /// Waitlist contains messages, which execution should be
137    /// delayed for some logic.
138    ///
139    /// Message can be inserted into waitlist only in these cases:
140    /// 1. Destination program called `gr_wait` while was executing
141    ///    this message, so only this program can remove and
142    ///    requeue it by `gr_wake` call in any execution.
143    /// 2. The message sent to program, that hadn't finished its
144    ///    initialization, and will be automatically removed once
145    ///    result of initialization would be available.
146    /// 3. Restored after resuming paused programs. On pause we
147    ///    collect waitlist content addressed to the program,
148    ///    removing it afterwards. On resume, user should provide
149    ///    the same content to be able to unpause program, which
150    ///    gonna be added into waitlist again.
151    ///
152    /// More cases may be considered in future.
153    ///
154    /// Gear runtime also charges rent for holding in waitlist.
155    /// Note, that system can remove message from waitlist,
156    /// if it couldn't pay rent for holding there further.
157    /// For details, see `pallet-gear-scheduler`.
158    type Waitlist: Waitlist<
159            Key1 = Self::WaitlistFirstKey,
160            Key2 = Self::WaitlistSecondKey,
161            Value = Self::WaitlistedMessage,
162            BlockNumber = Self::BlockNumber,
163            Error = Self::Error,
164            OutputError = Self::OutputError,
165        > + CountedByKey<Key = Self::WaitlistFirstKey, Length = usize>
166        + IterableMap<(Self::WaitlistedMessage, Interval<Self::BlockNumber>)>
167        + IterableByKeyMap<
168            (Self::WaitlistedMessage, Interval<Self::BlockNumber>),
169            Key = Self::WaitlistFirstKey,
170        >;
171
172    type DispatchStash: MapStorage<
173            Key = Self::DispatchStashKey,
174            Value = (Self::DelayedDispatch, Interval<Self::BlockNumber>),
175        >;
176
177    /// Resets all related to messenger storages.
178    ///
179    /// It's a temporary production solution to avoid DB migrations
180    /// and would be available for test purposes only in the future.
181    fn reset() {
182        Self::Sent::reset();
183        Self::Dequeued::reset();
184        Self::QueueProcessing::allow();
185        Self::Queue::clear();
186        Self::Mailbox::clear();
187        Self::Waitlist::clear();
188        Self::DispatchStash::clear();
189    }
190}
