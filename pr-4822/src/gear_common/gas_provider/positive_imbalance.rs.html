21/// Opaque, move-only struct with private field to denote that value has been
22/// created without any equal and opposite accounting
23#[derive(RuntimeDebug, PartialEq, Eq)]
24pub struct PositiveImbalance<Balance: BalanceTrait>(Balance);
25
26impl<Balance: BalanceTrait> PositiveImbalance<Balance> {
27    /// Create a new positive imbalance from value amount.
28    pub fn new(amount: Balance) -> Self {
29        PositiveImbalance(amount)
30    }
31}
32
33impl<Balance: BalanceTrait> Imbalance for PositiveImbalance<Balance> {
34    type Balance = Balance;
35
36    fn peek(&self) -> Self::Balance {
37        self.0
38    }
39
40    fn apply_to(&self, amount: &mut Option<Balance>) -> Result<(), ImbalanceError> {
41        let amount_value = amount.unwrap_or_else(Zero::zero);
42        if let Some(amount_value) = amount_value.checked_add(&self.0) {
43            *amount = Some(amount_value);
44            Ok(())
45        } else {
46            Err(ImbalanceError)
47        }
48        // if self.0 > Balance::max_value() - amount_value {
49        //     return Err(ImbalanceError::Overflow);
50        // }
51        // *amount = Some(amount_value + self.0);
52        // Ok(())
53    }
54}

56#[cfg(test)]
57mod tests {
58    use super::*;
59
60    #[test]
61    fn amount_increases_when_not_overflowing_imbalance_applied() {
62        let imbalance = PositiveImbalance::<u32>::new(45);
63        let mut amount = Some(44);
64
65        let result = imbalance.apply_to(&mut amount);
66
67        assert_eq!(Ok(()), result);
68        assert_eq!(Some(89), amount);
69    }
70
71    #[test]
72    fn error_returned_when_overflowing_imbalance_applied() {
73        let imbalance = PositiveImbalance::<u32>::new(120);
74        let mut amount = Some(u32::MAX - 100);
75
76        let result = imbalance.apply_to(&mut amount);
77
78        assert_eq!(Err(ImbalanceError), result);
79        assert_eq!(Some(u32::MAX - 100), amount);
80    }
81}
