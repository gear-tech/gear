87    /// Transfer `value` to `destination`'s account.
88    ///
89    /// Sends the
90    /// [`pallet_balances::transfer`](https://crates.parity.io/pallet_balances/pallet/struct.Pallet.html#method.transfer)
91    /// extrinsic.
92    ///
93    /// This function returns a hash of the block with the transfer transaction.
94    pub async fn transfer_keep_alive(&self, destination: ActorId, value: u128) -> Result<H256> {
95        let destination: [u8; 32] = destination.into();
96
97        let tx = self.0.calls.transfer_keep_alive(destination, value).await?;
98
99        for event in tx.wait_for_success().await?.iter() {
100            if let Event::Balances(BalancesEvent::Transfer { .. }) =
101                event?.as_root_event::<Event>()?
102            {
103                return Ok(tx.block_hash());
104            }
105        }
106
107        // Sending zero value is a no-op, so now event occurs.
108        if value == 0 {
109            return Ok(tx.block_hash());
110        }
111
112        Err(Error::EventNotFound)
113    }

170    /// Create a new program from a previously uploaded code identified by
171    /// [`CodeId`](https://docs.gear.rs/gear_core/ids/struct.CodeId.html) and
172    /// initialize it with a byte slice `payload`.
173    ///
174    /// Sends the
175    /// [`pallet_gear::create_program`](https://docs.gear.rs/pallet_gear/pallet/struct.Pallet.html#method.create_program)
176    /// extrinsic.
177    ///
178    /// Parameters:
179    ///
180    /// - `code_id` is the code identifier that can be obtained by calling the
181    ///   [`upload_code`](Self::upload_code) function;
182    /// - `salt` is the arbitrary data needed to generate an address for a new
183    ///   program (control of salt uniqueness is entirely on the function
184    ///   callerâ€™s side);
185    /// - `payload` vector contains data to be processed in the `init` function
186    ///   of the newly deployed "child" program;
187    /// - `gas_limit` is the maximum gas amount allowed to spend for the program
188    ///   creation and initialization;
189    /// - `value` to be transferred to the program's account during
190    ///   initialization.
191    ///
192    /// This function returns a tuple with an init message identifier, newly
193    /// created program identifier, and a hash of the block with the message
194    /// enqueuing transaction.
195    ///
196    /// # See also
197    ///
198    /// - [`create_program`](Self::create_program) function initializes a newly
199    ///   created program with an encoded payload.
200    /// - [`create_program_bytes_batch`](Self::create_program_bytes_batch)
201    ///   function creates a batch of programs and initializes them.
202    /// - [`upload_code`](Self::upload_code) function uploads a code and returns
203    ///   its identifier.
204    /// - [`upload_program_bytes`](Self::upload_program_bytes) function uploads
205    ///   a new program and initialize it.
206    pub async fn create_program_bytes(
207        &self,
208        code_id: CodeId,
209        salt: impl AsRef<[u8]>,
210        payload: impl AsRef<[u8]>,
211        gas_limit: u64,
212        value: u128,
213    ) -> Result<(MessageId, ActorId, H256)> {
214        let salt = salt.as_ref().to_vec();
215        let payload = payload.as_ref().to_vec();
216
217        let tx = self
218            .0
219            .calls
220            .create_program(code_id, salt, payload, gas_limit, value)
221            .await?;
222
223        for event in tx.wait_for_success().await?.iter() {
224            if let Event::Gear(GearEvent::MessageQueued {
225                id,
226                destination,
227                entry: MessageEntry::Init,
228                ..
229            }) = event?.as_root_event::<Event>()?
230            {
231                return Ok((id.into(), destination.into(), tx.block_hash()));
232            }
233        }
234
235        Err(Error::EventNotFound)
236    }

238    /// Create a batch of programs.
239    ///
240    /// A batch is a set of programs to be created within one function call.
241    /// Every entry of the `args` iterator is a tuple of parameters used in the
242    /// [`create_program_bytes`](Self::create_program_bytes) function. It is
243    /// useful when deploying a multi-program dApp.
244    pub async fn create_program_bytes_batch(
245        &self,
246        args: impl IntoIterator<Item = (CodeId, impl AsRef<[u8]>, impl AsRef<[u8]>, u64, u128)>,
247    ) -> Result<(Vec<Result<(MessageId, ActorId)>>, H256)> {
248        let calls: Vec<_> = args
249            .into_iter()
250            .map(|(code_id, salt, payload, gas_limit, value)| {
251                RuntimeCall::Gear(GearCall::create_program {
252                    code_id: code_id.into(),
253                    salt: salt.as_ref().to_vec(),
254                    init_payload: payload.as_ref().to_vec(),
255                    gas_limit,
256                    value,
257                    keep_alive: false,
258                })
259            })
260            .collect();
261
262        let amount = calls.len();
263
264        let tx = self.0.calls.force_batch(calls).await?;
265        let mut res = Vec::with_capacity(amount);
266
267        for event in tx.wait_for_success().await?.iter() {
268            match event?.as_root_event::<Event>()? {
269                Event::Gear(GearEvent::MessageQueued {
270                    id,
271                    destination,
272                    entry: MessageEntry::Init,
273                    ..
274                }) => res.push(Ok((id.into(), destination.into()))),
275                Event::Utility(UtilityEvent::ItemFailed { error }) => {
276                    res.push(Err(self.0.api().decode_error(error).into()))
277                }
278                _ => (),
279            }
280        }
281
282        if res.len() == amount {
283            Ok((res, tx.block_hash()))
284        } else {
285            Err(Error::IncompleteBatchResult(res.len(), amount))
286        }
287    }

310    /// Migrates an active program identified by `src_program_id` onto another
311    /// node identified by `dest_node_api` and returns the migrated program
312    /// identifier. All source program data is taken at the time of
313    /// `src_block_hash` if it is specified or the most recent one.
314    pub async fn migrate_program(
315        &self,
316        src_program_id: ActorId,
317        src_block_hash: Option<H256>,
318        dest_node_api: &GearApi,
319    ) -> Result<ActorId> {
320        if dest_node_api.0.api().gprog(src_program_id).await.is_ok() {
321            return Err(Error::ProgramAlreadyExists(
322                src_program_id.as_ref().encode_hex(),
323            ));
324        }
325
326        let mut src_block_hash = src_block_hash;
327        if src_block_hash.is_none() {
328            src_block_hash = Some(self.last_block_hash().await?);
329        }
330
331        let dest_program_id = src_program_id;
332
333        // Collect data from the source program
334        let src_program_account_data = self
335            .account_data_at(src_program_id, src_block_hash)
336            .await
337            .or_else(|e| {
338            if let Error::GearSDK(GsdkError::StorageNotFound) = e {
339                Ok(AccountData {
340                    free: 0u128,
341                    reserved: 0,
342                    frozen: 0,
343                    flags: gsdk::metadata::runtime_types::pallet_balances::types::ExtraFlags(
344                        170141183460469231731687303715884105728,
345                    ),
346                })
347            } else {
348                Err(e)
349            }
350        })?;
351
352        let src_program_account_bank_data = self
353            .bank_data_at(src_program_id, src_block_hash)
354            .await
355            .or_else(|e| {
356                if let Error::GearSDK(GsdkError::StorageNotFound) = e {
357                    Ok(BankAccount { gas: 0, value: 0 })
358                } else {
359                    Err(e)
360                }
361            })?;
362
363        let bank_address = self.bank_address().await?;
364
365        let src_bank_account_data = self
366            .account_data_at(bank_address.clone(), src_block_hash)
367            .await
368            .or_else(|e| {
369                if let Error::GearSDK(GsdkError::StorageNotFound) = e {
370                    Ok(AccountData {
371                        free: 0u128,
372                        reserved: 0,
373                        frozen: 0,
374                        flags: gsdk::metadata::runtime_types::pallet_balances::types::ExtraFlags(
375                            170141183460469231731687303715884105728,
376                        ),
377                    })
378                } else {
379                    Err(e)
380                }
381            })?;
382
383        let mut src_program = self
384            .0
385            .api()
386            .gprog_at(src_program_id, src_block_hash)
387            .await?;
388
389        let src_program_pages = self
390            .0
391            .api()
392            .gpages_at(
393                src_program_id,
394                Some(src_program.memory_infix.0),
395                src_block_hash,
396            )
397            .await?;
398
399        let src_program_reserved_gas_node_ids: Vec<GearGasNodeId> = src_program
400            .gas_reservation_map
401            .iter()
402            .map(|gr| gr.0.into())
403            .collect();
404
405        let src_program_reserved_gas_nodes = self
406            .0
407            .api()
408            .gas_nodes_at(&src_program_reserved_gas_node_ids, src_block_hash)
409            .await?;
410
411        let mut src_program_reserved_gas_total = 0u64;
412        let mut accounts_with_reserved_funds = HashSet::new();
413        for gas_node in &src_program_reserved_gas_nodes {
414            if let GearGasNode::Reserved {
415                id, value, lock, ..
416            } = &gas_node.1
417            {
418                accounts_with_reserved_funds.insert(id);
419                src_program_reserved_gas_total += value + lock.0[LockId::Reservation as usize];
420            } else {
421                unreachable!("Unexpected gas node type");
422            }
423        }
424
425        let src_code_id = src_program.code_id.0.into();
426
427        let src_instrumented_code = self
428            .0
429            .api()
430            .instrumented_code_storage_at(src_code_id, src_block_hash)
431            .await?;
432
433        let src_code_metadata = self
434            .0
435            .api()
436            .code_metadata_storage_at(src_code_id, src_block_hash)
437            .await?;
438
439        // Apply data to the target program
440        dest_node_api
441            .force_set_balance(
442                dest_program_id.into_account_id(),
443                src_program_account_data.free,
444            )
445            .await?;
446
447        dest_node_api
448            .force_set_balance(bank_address, src_bank_account_data.free)
449            .await?;
450
451        dest_node_api
452            .0
453            .storage
454            .set_bank_account_storage(
455                src_program_id.into_account_id(),
456                src_program_account_bank_data,
457            )
458            .await?;
459
460        dest_node_api
461            .0
462            .storage
463            .set_instrumented_code_storage(src_code_id, &src_instrumented_code)
464            .await?;
465
466        dest_node_api
467            .0
468            .storage
469            .set_code_metadata_storage(src_code_id, &src_code_metadata)
470            .await?;
471
472        dest_node_api
473            .0
474            .storage
475            .set_gas_nodes(&src_program_reserved_gas_nodes)
476            .await?;
477
478        for account_with_reserved_funds in accounts_with_reserved_funds {
479            let src_account_bank_data = self
480                .bank_data_at(account_with_reserved_funds, src_block_hash)
481                .await
482                .or_else(|e| {
483                    if let Error::GearSDK(GsdkError::StorageNotFound) = e {
484                        Ok(BankAccount { gas: 0, value: 0 })
485                    } else {
486                        Err(e)
487                    }
488                })?;
489
490            let dest_account_data = dest_node_api
491                .account_data(account_with_reserved_funds)
492                .await
493                .or_else(|e| {
494                    if let Error::GearSDK(GsdkError::StorageNotFound) = e {
495                        Ok(AccountData {
496                            free: 0u128,
497                            reserved: 0,
498                            frozen: 0,
499                            flags:
500                                gsdk::metadata::runtime_types::pallet_balances::types::ExtraFlags(
501                                    170141183460469231731687303715884105728,
502                                ),
503                        })
504                    } else {
505                        Err(e)
506                    }
507                })?;
508            let dest_account_bank_data = self
509                .bank_data_at(account_with_reserved_funds, None)
510                .await
511                .or_else(|e| {
512                    if let Error::GearSDK(GsdkError::StorageNotFound) = e {
513                        Ok(BankAccount { gas: 0, value: 0 })
514                    } else {
515                        Err(e)
516                    }
517                })?;
518
519            dest_node_api
520                .force_set_balance(
521                    account_with_reserved_funds.into_account_id(),
522                    dest_account_data.free,
523                )
524                .await?;
525
526            dest_node_api
527                .0
528                .storage
529                .set_bank_account_storage(
530                    account_with_reserved_funds.into_account_id(),
531                    BankAccount {
532                        gas: src_account_bank_data
533                            .gas
534                            .saturating_add(dest_account_bank_data.gas),
535                        value: src_account_bank_data
536                            .value
537                            .saturating_add(dest_account_bank_data.value),
538                    },
539                )
540                .await?;
541        }
542
543        let dest_gas_total_issuance =
544            dest_node_api.0.api().total_issuance().await.or_else(|e| {
545                if let GsdkError::StorageNotFound = e {
546                    Ok(0)
547                } else {
548                    Err(e)
549                }
550            })?;
551
552        dest_node_api
553            .0
554            .storage
555            .set_total_issuance(
556                dest_gas_total_issuance.saturating_add(src_program_reserved_gas_total),
557            )
558            .await?;
559
560        dest_node_api
561            .0
562            .storage
563            .set_gpages(
564                dest_program_id,
565                src_program.memory_infix.0,
566                &src_program_pages,
567            )
568            .await?;
569
570        src_program.expiration_block = dest_node_api.last_block_number().await?;
571        dest_node_api
572            .0
573            .storage
574            .set_gprog(dest_program_id, src_program)
575            .await?;
576
577        Ok(dest_program_id)
578    }

580    /// Get all pages with their data for program at specified block.
581    pub async fn get_program_pages_data_at(
582        &self,
583        program_id: ActorId,
584        block_hash: Option<H256>,

593            );
594        }
595
596        Ok(res)

626    /// Save program (identified by `program_id`) memory dump to the file for
627    /// further restoring in gclient/gtest. Program memory dumped at the
628    /// time of `block_hash` if presented or the most recent block.
629    pub async fn save_program_memory_dump_at<P: AsRef<Path>>(
630        &self,
631        program_id: ActorId,
632        block_hash: Option<H256>,
633        file_path: P,
634    ) -> Result {
635        let program = self.0.api().gprog_at(program_id, block_hash).await?;
636
637        let program_pages = self
638            .0
639            .api()
640            .gpages_at(program_id, Some(program.memory_infix.0), block_hash)
641            .await?
642            .into_iter()
643            .map(|(page, data)| {
644                MemoryPageDump::new(
645                    GearPage::try_from(page)
646                        .unwrap_or_else(|_| panic!("Couldn't decode GearPage from u32: {page}")),
647                    PageBuf::decode(&mut &*data).expect("Couldn't decode PageBuf"),
648                )
649            })
650            .collect();
651
652        let program_account_data =
653            self.account_data_at(program_id, block_hash)
654                .await
655                .or_else(|e| {
656                    if let Error::GearSDK(GsdkError::StorageNotFound) = e {
657                        Ok(AccountData {
658                            free: 0u128,
659                            reserved: 0,
660                            frozen: 0,
661                            flags:
662                                gsdk::metadata::runtime_types::pallet_balances::types::ExtraFlags(
663                                    170141183460469231731687303715884105728,
664                                ),
665                        })
666                    } else {
667                        Err(e)
668                    }
669                })?;
670
671        ProgramMemoryDump {
672            pages: program_pages,
673            balance: program_account_data.free,
674            reserved_balance: program_account_data.reserved,
675        }
676        .save_to_file(file_path);
677
678        Ok(())
679    }

681    /// Replace entire program memory with one saved earlier in gclient/gtest
682    pub async fn replace_program_memory<P: AsRef<Path>>(
683        &self,
684        program_id: ActorId,
685        file_path: P,
686    ) -> Result {
687        let memory_dump = ProgramMemoryDump::load_from_file(file_path);
