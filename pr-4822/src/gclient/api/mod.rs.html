36/// The API instance contains methods to access the node.
37#[derive(Clone)]
38pub struct GearApi(Signer, Option<Arc<NodeInstance>>);
39
40impl GearApi {
41    /// Create api builder
42    pub fn builder() -> GearApiBuilder {
43        GearApiBuilder::default()
44    }
45
46    /// Create and init a new `GearApi` specified by its `address` on behalf of
47    /// the default `Alice` user.
48    pub async fn init(address: WSAddress) -> Result<Self> {
49        Self::init_with(address, "//Alice").await
50    }
51
52    /// Create and init a new `GearApi` specified by its `address` and `suri`.
53    ///
54    /// SURI is a Substrate URI that identifies a user by a mnemonic phrase or
55    /// provides default users from the keyring (e.g., "//Alice", "//Bob",
56    /// etc.). The password for URI should be specified in the same `suri`,
57    /// separated by the `':'` char.
58    pub async fn init_with(address: WSAddress, suri: impl AsRef<str>) -> Result<Self> {
59        Self::builder().suri(suri).build(address).await
60    }
61
62    /// Change SURI to the provided `suri` and return `Self`.
63    pub fn with(self, suri: impl AsRef<str>) -> Result<Self> {
64        let mut suri = suri.as_ref().splitn(2, ':');
65
66        Ok(Self(
67            self.0
68                .change(suri.next().expect("Infallible"), suri.next())?,
69            self.1,
70        ))
71    }
72
73    /// Create and init a new `GearApi` instance that will be used with the
74    /// local node working in developer mode (running with `--dev` argument).
75    pub async fn dev() -> Result<Self> {
76        Self::init(WSAddress::dev()).await
77    }
78
79    /// Create and init a new `GearApi` instance via spawning a new node process
80    /// in development mode using the `--dev` flag and listening on an a
81    /// random port number. The node process uses a binary specified by the
82    /// `path` param. Ideally, the binary should be downloaded by means of CI pipeline from <https://get.gear.rs>.
83    pub async fn dev_from_path(path: impl AsRef<OsStr>) -> Result<Self> {
84        let node = Node::from_path(path.as_ref())?.spawn()?;
85        let api = Self::init(node.address.into()).await?;
86        Ok(Self(api.0, Some(Arc::new(node))))
87    }
88
89    /// Create and init a new `GearApi` instance via spawning a new node process
90    /// in development mode using the `--chain=vara-dev`, `--validator`,
91    /// `--tmp` flags and listening on an a random port number. The node
92    /// process uses a binary specified by the `path` param. Ideally, the
93    /// binary should be downloaded by means of CI pipeline from <https://get.gear.rs>.
94    pub async fn vara_dev_from_path(path: impl AsRef<OsStr>) -> Result<Self> {
95        let node = Node::from_path(path.as_ref())?.arg("--validator").spawn()?;
96        let api = Self::init(node.address.into()).await?;
97        Ok(Self(api.0, Some(Arc::new(node))))
98    }
99
100    /// Print node logs.
101    pub fn print_node_logs(&mut self) {
102        if let Some(node) = self.1.as_mut() {
103            Arc::get_mut(node)
104                .expect("Unable to mutate `Node`")
105                .logs()
106                .expect("Unable to read logs");
107        }
108    }
109
110    /// Create and init a new `GearApi` instance that will be used with the
111    /// public Vara testnet node.
112    pub async fn vara_testnet() -> Result<Self> {
113        Self::init(WSAddress::vara_testnet()).await
114    }
115
116    /// Create and init a new `GearApi` instance that will be used with the
117    /// public Vara node.
118    pub async fn vara() -> Result<Self> {
119        Self::init(WSAddress::vara()).await
120    }
121
122    /// Create an [`EventListener`] to subscribe and handle continuously
123    /// incoming events.
124    pub async fn subscribe(&self) -> Result<EventListener> {
125        let events = self.0.api().subscribe_finalized_blocks().await?;
126        Ok(EventListener(events))
127    }
128
129    /// Set the number used once (`nonce`) that will be used while sending
130    /// extrinsics.
131    ///
132    /// If set, this nonce is added to the extrinsic and provides a unique
133    /// identifier of the transaction sent by the current account. The nonce
134    /// shows how many prior transactions have occurred from this account. This
135    /// helps protect against replay attacks or accidental double-submissions.
136    pub fn set_nonce(&mut self, nonce: u64) {
137        self.0.set_nonce(nonce)
138    }
139
140    /// Get the next number used once (`nonce`) from the node.
141    ///
142    /// Actually sends the `system_accountNextIndex` RPC to the node.
143    pub async fn rpc_nonce(&self) -> Result<u64> {
144        self.0
145            .api()
146            .tx()
147            .account_nonce(self.0.account_id())
148            .await
149            .map_err(Into::into)
150    }
151
152    /// Return the signer account address.
153    ///
154    /// # Examples
155    ///
156    /// ```
157    /// use gclient::{GearApi, Result};
158    /// use gsdk::ext::sp_runtime::AccountId32;
159    /// # use hex_literal::hex;
160    ///
161    /// #[tokio::test]
162    /// async fn account_test() -> Result<()> {
163    ///     let api = GearApi::dev().await?;
164    ///     let alice_id = hex!("d43593c715fdd31c61141abd04a99fd6822c8558854ccde39a5684e7a56da27d");
165    ///     assert_eq!(api.account_id(), &AccountId32::from(alice_id));
166    ///     Ok(())
167    /// }
168    /// ```
169    pub fn account_id(&self) -> &AccountId32 {
170        self.0.account_id()
171    }
172}

174impl From<Signer> for GearApi {
175    fn from(signer: Signer) -> Self {
176        Self(signer, None)
177    }
178}
179
180impl From<Api> for GearApi {
181    fn from(api: Api) -> Self {
182        Signer::new(api, "//Alice", None)
183            .expect("//Alice always works")
184            .into()
185    }
186}
187
188impl From<(Api, sr25519::Pair)> for GearApi {
189    fn from((api, signer): (Api, sr25519::Pair)) -> Self {
190        Signer::from((api, signer.into())).into()
191    }
192}
193
194impl From<GearApi> for Signer {
195    fn from(api: GearApi) -> Self {
196        api.0
197    }
198}

200/// Gear API builder
201pub struct GearApiBuilder {
202    /// gsdk api builder
203    inner: ApiBuilder,
204    /// suri for keypair
205    suri: String,
206}
207
208impl GearApiBuilder {
209    /// Set retries of rpc client
210    pub fn retries(mut self, retries: u8) -> Self {
211        self.inner = self.inner.retries(retries);
212        self
213    }
214
215    /// Set timeout of rpc client ( in milliseconds )
216    pub fn timeout(mut self, timeout: u64) -> Self {
217        self.inner = self.inner.timeout(timeout);
218        self
219    }
220
221    /// Set initial suri for keypiar
222    pub fn suri(mut self, suri: impl AsRef<str>) -> Self {
223        self.suri = suri.as_ref().into();
224        self
225    }
226
227    /// Build gear api
228    pub async fn build(self, address: WSAddress) -> Result<GearApi> {
229        let mut suri = self.suri.splitn(2, ':');
230        let api = self.inner.build(address.url().as_str()).await?;
231
232        Ok(GearApi(
233            api.signer(suri.next().expect("Infallible"), suri.next())?,
234            None,
235        ))
236    }
237}
238
239impl Default for GearApiBuilder {
240    fn default() -> Self {
241        Self {
242            suri: "//Alice".into(),
243            inner: ApiBuilder::default(),
244        }
245    }
