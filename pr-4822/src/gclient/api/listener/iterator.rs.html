29/// Event listener that allows catching and processing events propagated through
30/// the network.
31///
32/// # Examples
33///
34/// ```
35/// use gclient::GearApi;
36/// # use gclient::Result;
37///
38/// #[tokio::test]
39/// async fn listener_test() -> Result<()> {
40///     let api = GearApi::dev().await?;
41///     let mut listener = api.subscribe().await?;
42///     assert!(listener.blocks_running().await?);
43///     Ok(())
44/// }
45/// ```
46pub struct EventListener(pub(crate) Blocks);
47
48#[async_trait(?Send)]
49impl EventProcessor for EventListener {
50    fn not_waited() -> Error {
51        unreachable!()
52    }
53
54    async fn proc<T>(&mut self, predicate: impl Fn(Event) -> Option<T> + Copy) -> Result<T> {
55        while let Some(events) = self.0.next_events().await? {
56            if let Some(res) = self.proc_events_inner(events, predicate) {
57                return Ok(res);
58            }
59        }
60
61        Err(Self::not_waited())
62    }
63
64    async fn proc_many<T>(
65        &mut self,
66        predicate: impl Fn(Event) -> Option<T>,
67        validate: impl Fn(Vec<T>) -> (Vec<T>, bool),
68    ) -> Result<Vec<T>> {
69        let mut res = vec![];
70
71        while let Some(events) = self.0.next_events().await? {
72            for event in events.iter() {
73                if let Some(data) = predicate(event?.as_root_event::<Event>()?) {
74                    res.push(data);
75                }
76            }
77
78            let finished: bool;
79            (res, finished) = validate(res);
80
81            if finished {
82                break;
83            }
84        }
85
86        Ok(res)
87    }
88}
89
90impl EventListener {
91    /// Look through finalized blocks to find the
92    /// [`QueueNotProcessed`](https://docs.gear.rs/pallet_gear/pallet/enum.Event.html#variant.QueueNotProcessed)
93    /// event.
94    pub async fn queue_processing_reverted(&mut self) -> Result<H256> {
95        while let Some(events) = self.0.next_events().await? {
96            let events_bh = events.block_hash();
97
98            if let Some(res) = self.proc_events_inner(events, |e| {
99                matches!(e, Event::Gear(GearEvent::QueueNotProcessed)).then_some(events_bh)
100            }) {
101                return Ok(res);
102            }
103        }
104
105        Err(Self::not_waited())
106    }
107
108    /// Reads the next event from the stream and returns the respective block
109    /// hash.
110    pub async fn next_block_hash(&mut self) -> Result<H256> {
111        Ok(self
112            .0
113            .next_events()
114            .await?
115            .ok_or(Error::EventNotFound)?
116            .block_hash())
117    }
118
119    /// Check whether at least one new block has been produced after the
120    /// `previous` block.
121    pub async fn blocks_running_since(&mut self, previous: H256) -> Result<bool> {
122        let current = self.next_block_hash().await?;
123
124        Ok(current != previous)
125    }
126
127    /// Check whether new blocks are produced as expected.
128    pub async fn blocks_running(&mut self) -> Result<bool> {
129        let previous = self.next_block_hash().await?;
130
131        self.blocks_running_since(previous).await
132    }
133
134    fn proc_events_inner<T>(
135        &mut self,
136        events: BlockEvents,
137        predicate: impl Fn(Event) -> Option<T>,
138    ) -> Option<T> {
139        events
140            .iter()
141            .filter_map(|event| predicate(event.ok()?.as_root_event::<Event>().ok()?))
142            .next()
143    }
144}
