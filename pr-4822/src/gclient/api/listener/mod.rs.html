37/// Dispatch status returned after processing a message.
38#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
39pub enum DispatchStatus {
40    /// Successful execution.
41    Success,
42    /// Execution failed.
43    Failed,
44    /// Message has not been processed.
45    NotExecuted,
46}
47
48impl From<GenDispatchStatus> for DispatchStatus {
49    fn from(other: GenDispatchStatus) -> Self {
50        match other {
51            GenDispatchStatus::Success => Self::Success,
52            GenDispatchStatus::Failed => Self::Failed,
53            GenDispatchStatus::NotExecuted => Self::NotExecuted,
54        }
55    }
56}
57
58impl DispatchStatus {
59    /// Check whether `DispatchStatus` is
60    /// [`Success`](DispatchStatus::Success).
61    pub fn succeed(&self) -> bool {
62        matches!(self, DispatchStatus::Success)
63    }
64
65    /// Check whether `DispatchStatus` is [`Failed`](DispatchStatus::Failed).
66    pub fn failed(&self) -> bool {
67        matches!(self, DispatchStatus::Failed)
68    }
69
70    /// Check whether `DispatchStatus` is
71    /// [`NotExecuted`](DispatchStatus::NotExecuted).
72    pub fn not_executed(&self) -> bool {
73        matches!(self, DispatchStatus::NotExecuted)
74    }
75}

77/// Events processing trait.
78///
79/// This trait minimizes a boilerplate code when processing events by providing
80/// several default implementations.
81///
82/// See implementation example in [`EventListener`].
83#[async_trait(?Send)]
84pub trait EventProcessor {
85    /// This function is called if a received event has an unexpected type.
86    ///
87    /// # Examples
88    ///
89    /// Implement this function to return a not-found error:
90    ///
91    /// ```
92    /// # use gclient::Error;
93    /// fn not_waited() -> Error {
94    ///     Error::EventNotFoundInIterator
95    /// }
96    /// ```
97    fn not_waited() -> Error;
98
99    /// Event processing function.
100    ///
101    /// `predicate` contains specific processing logic depending on an event
102    /// type.
103    ///
104    /// Implementor is responsible for applying the `predicate` to every event
105    /// to be processed and collecting their results.
106    ///
107    /// This function is one of the central functions to be defined by the trait
108    /// implementor.
109    async fn proc<T>(&mut self, predicate: impl Fn(Event) -> Option<T> + Copy) -> Result<T>;
110
111    /// Multiple events processing function.
112    ///
113    ///  `predicate` contains specific processing logic depending on an event
114    /// type. `validator` checks whether an additional condition is met for the
115    /// batch of results.
116    async fn proc_many<T>(
117        &mut self,
118        predicate: impl Fn(Event) -> Option<T>,
119        validate: impl Fn(Vec<T>) -> (Vec<T>, bool),
120    ) -> Result<Vec<T>>;
121
122    /// Check whether the message identified by `message_id` has been processed.
123    async fn message_processed(&mut self, message_id: MessageId) -> Result<DispatchStatus> {
124        let message_id: GenMId = message_id.into();
125
126        self.proc(|e| {
127            if let Event::Gear(GearEvent::MessagesDispatched { statuses, .. }) = e {
128                statuses
129                    .into_iter()
130                    .find(|(mid, _)| mid == &message_id)
131                    .map(|(_, status)| status.into())
132            } else {
133                None
134            }
135        })
136        .await
137    }
138
139    /// Check whether the batch of messages identified by corresponding
140    /// `message_ids` has been processed.
141    async fn message_processed_batch(
142        &mut self,
143        message_ids: impl IntoIterator<Item = MessageId>,
144    ) -> Result<Vec<(MessageId, DispatchStatus)>> {
145        let message_ids: Vec<GenMId> = message_ids.into_iter().map(Into::into).collect();
146
147        Ok(self
148            .proc_many(
149                |e| {
150                    if let Event::Gear(GearEvent::MessagesDispatched { statuses, .. }) = e {
151                        let requested: Vec<_> = statuses
152                            .into_iter()
153                            .filter_map(|(mid, status)| {
154                                message_ids
155                                    .contains(&mid)
156                                    .then(|| (mid.into(), status.into()))
157                            })
158                            .collect();
159
160                        (!requested.is_empty()).then_some(requested)
161                    } else {
162                        None
163                    }
164                },
165                |v| {
166                    let count = v.iter().flatten().count() == message_ids.len();
167                    (v, count)
168                },
169            )
170            .await?
171            .into_iter()
172            .flatten()
173            .collect())
174    }
175
176    /// Get details of a reply to the message identified by `message_id`.
177    ///
178    /// If a reply has been received, this function returns its identifier
179    /// ([`MessageId`]), payload's bytes (in case of zero status code) or an
180    /// error message (otherwise), and an associated value.
181    async fn reply_bytes_on(
182        &mut self,
183        message_id: MessageId,
184    ) -> Result<(MessageId, Result<Vec<u8>, String>, u128)> {
185        let message_id: GenMId = message_id.into();
186
187        self.proc(|e| {
188            if let Event::Gear(GearEvent::UserMessageSent {
189                message:
190                    GenUserMessage {
191                        id,
192                        payload,
193                        value,
194                        details: Some(GenReplyDetails { to, code }),
195                        ..
196                    },
197                ..
198            }) = e
199            {
200                to.eq(&message_id).then(|| {
201                    let res = ReplyCode::from(code)
202                        .is_success()
203                        .then(|| payload.0.clone())
204                        .ok_or_else(|| String::from_utf8(payload.0).expect("Infallible"));
205
206                    (id.into(), res, value)
207                })
208            } else {
209                None
210            }
211        })
212        .await
213    }
214
215    /// Check whether the processing of a message identified by `message_id`
216    /// resulted in an error or has been successful.
217    ///
218    /// This function returns an error message in case of an error. It is needed
219    /// only due to the possibility of a message's
220    /// [`NotExecuted`](DispatchStatus::NotExecuted) state.
221    async fn err_or_succeed(&mut self, message_id: MessageId) -> Result<Option<String>> {
222        let message_id: GenMId = message_id.into();
223
224        self.proc(|e| match e {
225            Event::Gear(GearEvent::UserMessageSent {
226                message:
227                    GenUserMessage {
228                        payload,
229                        details: Some(GenReplyDetails { to, code }),
230                        ..
231                    },
232                ..
233            }) => {
234                if to == message_id && ReplyCode::from(code).is_success() {
235                    Some(Some(String::from_utf8(payload.0).expect("Infallible")))
236                } else {
237                    None
238                }
239            }
240            Event::Gear(GearEvent::MessagesDispatched { statuses, .. }) => match statuses
241                .into_iter()
242                .find(|(mid, _)| mid == &message_id)
243                .map(|(_, status)| status)
244            {
245                Some(GenDispatchStatus::Failed) | None => None,
246                _ => Some(None),
247            },
248            _ => None,
249        })
250        .await
251    }
252
253    /// Check whether processing batch of messages identified by corresponding
254    /// `message_ids` resulted in errors or has been successful.
255    ///
256    /// This function returns a vector of statuses with an associated message
257    /// identifier ([`MessageId`]). Each status can be an error message in case
258    /// of an error.
259    async fn err_or_succeed_batch(
260        &mut self,
261        message_ids: impl IntoIterator<Item = MessageId>,
262    ) -> Result<Vec<(MessageId, Option<String>)>> {
263        let message_ids: Vec<GenMId> = message_ids.into_iter().map(Into::into).collect();
264
265        Ok(self
266            .proc_many(
267                |e| match e {
268                    Event::Gear(GearEvent::UserMessageSent {
269                        message:
270                            GenUserMessage {
271                                payload,
272                                details: Some(GenReplyDetails { to, code }),
273                                ..
274                            },
275                        ..
276                    }) => {
277                        if message_ids.contains(&to) && ReplyCode::from(code).is_success() {
278                            Some(vec![(
279                                to.into(),
280                                Some(String::from_utf8(payload.0).expect("Infallible")),
281                            )])
282                        } else {
283                            None
284                        }
285                    }
286                    Event::Gear(GearEvent::MessagesDispatched { statuses, .. }) => {
287                        let requested: Vec<_> = statuses
288                            .into_iter()
289                            .filter_map(|(mid, status)| {
290                                if message_ids.contains(&mid)
291                                    && !matches!(status, GenDispatchStatus::Failed)
292                                {
293                                    Some((MessageId::from(mid), None))
294                                } else {
295                                    None
296                                }
297                            })
298                            .collect();
299
300                        (!requested.is_empty()).then_some(requested)
301                    }
302                    _ => None,
303                },
304                |v| {
305                    let count = v.iter().flatten().count() == message_ids.len();
306                    (v, count)
307                },
308            )
309            .await?
310            .into_iter()
311            .flatten()
312            .collect())
313    }
