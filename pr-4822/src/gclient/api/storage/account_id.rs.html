19use gear_core::ids::ActorId;
20use gsdk::ext::sp_runtime::AccountId32;
21
22/// A trait for convenient conversion into Substrate's AccountId32.
23pub trait IntoAccountId32 {
24    fn into_account_id(self) -> AccountId32;
25}
26
27impl IntoAccountId32 for AccountId32 {
28    fn into_account_id(self) -> AccountId32 {
29        self
30    }
31}
32
33impl IntoAccountId32 for &AccountId32 {
34    fn into_account_id(self) -> AccountId32 {
35        self.clone()
36    }
37}
38
39impl IntoAccountId32 for ActorId {
40    fn into_account_id(self) -> AccountId32 {
41        AccountId32::new(self.into_bytes())
42    }
43}
44
45impl IntoAccountId32 for &ActorId {
46    fn into_account_id(self) -> AccountId32 {
47        AccountId32::new(self.into_bytes())
48    }
49}
