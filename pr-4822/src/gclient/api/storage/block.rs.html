26impl GearApi {
27    /// Return the total gas limit per block (also known as a gas budget).
28    pub fn block_gas_limit(&self) -> Result<u64> {
29        self.0.api().gas_limit().map_err(Into::into)
30    }

32    /// The expected average block time at which BABE should be creating blocks.
33    ///
34    /// Since BABE is probabilistic it is not trivial to figure out what the
35    /// expected average block time should be based on the slot duration and the
36    /// security parameter `c` (where `1 - c` represents the probability of a
37    /// slot being empty).
38    pub fn expected_block_time(&self) -> Result<u64> {
39        self.0.api().expected_block_time().map_err(Into::into)
40    }

42    // Get block data
43    async fn get_block_at(&self, block_hash: Option<H256>) -> Result<GearBlock> {
44        let hash = if let Some(hash) = block_hash {
45            hash
46        } else {
47            self.0
48                .api()
49                .backend()
50                .latest_finalized_block_ref()
51                .await?
52                .hash()
53        };
54
55        self.0
56            .api()
57            .backend()
58            .block_header(hash)
59            .await?
60            .ok_or(Error::BlockDataNotFound)
61    }
62
63    /// Return a hash of the last block.
64    pub async fn last_block_hash(&self) -> Result<H256> {
65        Ok(self.get_block_at(None).await?.hash())
66    }
67
68    /// Return a number of the last block (also known as block height).
69    pub async fn last_block_number(&self) -> Result<u32> {
70        Ok(self.get_block_at(None).await?.number())
71    }
72
73    /// Return vector of events contained in the last block.
74    pub async fn last_events(&self) -> Result<Vec<RuntimeEvent>> {

78    /// Return a number of the specified block identified by the `block_hash`.

83    /// Get a hash of a block identified by its `block_number`.

115    /// Return vector of events contained in blocks since the block identified
116    /// by the `block_hash` but no more than in `max_depth` blocks.
117    pub async fn events_since(
118        &self,
119        block_hash: H256,
120        max_depth: usize,
121    ) -> Result<Vec<RuntimeEvent>> {
122        let mut block_hashes = Vec::with_capacity(max_depth);
123
124        let mut current = self.get_block_at(None).await?;
125        for _ in 0..max_depth {
126            let current_hash = current.hash();
127            block_hashes.push(current_hash);
128
129            if current_hash == block_hash {
130                break;
131            }
132
133            current = self.get_block_at(Some(current.parent_hash)).await?;
134        }
135
136        if block_hashes.contains(&block_hash) {
137            let mut events = vec![];
138
139            for hash in block_hashes.into_iter() {
140                events.append(self.events_at(hash).await?.as_mut());
141            }
142
143            Ok(events)
144        } else {
145            Err(Error::MaxDepthReached)
146        }
147    }

149    /// Check whether the message queue processing is stopped or not.
150    pub async fn queue_processing_enabled(&self) -> Result<bool> {
151        self.0.api().execute_inherent().await.map_err(Into::into)
152    }

154    /// Looks at two blocks from the stream and checks if the Gear block number
155    /// has grown from block to block or not.
156    pub async fn queue_processing_stalled(&self) -> Result<bool> {
157        let mut listener = self.subscribe().await?;
158
159        let current = listener.next_block_hash().await?;
160        let gear_current = self.0.api().gear_block_number(Some(current)).await?;
161
162        let mut next = current;
163        while next == current {
164            next = listener.next_block_hash().await?;
165        }
166        let gear_next = self.0.api().gear_block_number(Some(next)).await?;
167
168        Ok(gear_next <= gear_current)
169    }
