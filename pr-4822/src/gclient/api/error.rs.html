19use anyhow::Error as AError;
20use gsdk::Error as GearSDKError;
21use std::{io::Error as IOError, result::Result as StdResult};
22use subxt::error::{DispatchError, Error as SubxtError};
23
24/// `Result` type with a predefined error type ([`Error`]).
25pub type Result<T = (), E = Error> = StdResult<T, E>;
26
27/// Common error type.
28#[derive(Debug, thiserror::Error)]
29pub enum Error {
30    /// A wrapper around [`anyhow::Error`].
31    #[error(transparent)]
32    Anyhow(#[from] AError),
33    /// A wrapper around [`gsdk::Error`].
34    #[error(transparent)]
35    GearSDK(GearSDKError),
36    /// Occurs when attempting to iterate events without a subscription.
37    #[error("An attempt to iter events without subscription")]
38    EventsSubscriptionNotFound,
39    /// Occurs when events are stopped. Unreachable.
40    #[error("Events stopped (unreachable")]
41    EventsStopped,
42    /// A wrapper around [`subxt::error::Error`].
43    #[error(transparent)]
44    Subxt(Box<SubxtError>),
45    /// Subxt core error
46    #[error(transparent)]
47    SubxtCore(#[from] Box<subxt::ext::subxt_core::Error>),
48    /// Occurs when an event of the expected type cannot be found.
49    #[error("Expected event wasn't found")]
50    EventNotFound,
51    /// A wrapper around [`std::io::Error`].
52    #[error(transparent)]
53    IO(#[from] IOError),
54    /// Occurs when trying to upload a Wasm binary with the wrong file extension
55    /// (other than `.wasm`).
56    #[error("An attempt to upload invalid binary")]
57    WrongBinaryExtension,
58    /// Occurs when the balance reaches its maximum value.
59    #[error("Funds overcame u128::MAX")]
60    BalanceOverflow,
61    /// Occurs when a block identified by the specified hash cannot be found.
62    #[error("Block data not found")]
63    BlockDataNotFound,
64    /// Occurs when a hash of a block identified by the specified hash cannot be
65    /// found.
66    #[error("Block hash not found")]
67    BlockHashNotFound,
68    /// Occurs when the batch hasn't been fully processed.
69    #[error("Some of extrinsics wasn't processed within the batch")]
70    IncompleteBatchResult(usize, usize),
71    /// Occurs when a block cannot be found within the specified depth.
72    #[error("Max depth reached, but queried block wasn't found")]
73    MaxDepthReached,
74    /// Occurs when an event cannot be found in pre-queried events.
75    #[error("Event not found in pre-queried events")]
76    EventNotFoundInIterator,
77    /// Occurs when a storage entry with a specified address cannot be found.
78    #[error("Storage not found.")]
79    StorageNotFound,
80    /// Occurs when a timestamp record cannot be found in the storage.
81    #[error("Timestamp not found in storage.")]
82    TimestampNotFound,
83    /// A wrapper around [`parity_scale_codec::Error`].
84    #[error(transparent)]
85    Codec(#[from] parity_scale_codec::Error),
86    /// Occurs when decoding hex string failed.
87    #[error(transparent)]
88    Hex(#[from] hex::FromHexError),
89    /// Occurs when being migrated program already exists in destination node.
90    #[error("Program {0} already exists")]
91    ProgramAlreadyExists(String),
92    /// Occurs when parsing websocket domain failed.
93    #[error("Failed to parse WebSocket domain.")]
94    WSDomainInvalid,
95    /// Occurs when parsing domain url failed.
96    #[error(transparent)]
97    Url(#[from] url::ParseError),
98    /// A wrapper of module error [`gsdk::metadata::ModuleError`].
99    ///
100    /// # Example
101    ///
102    /// ```ignore
103    /// use gclient::{errors, Error, EventProcessor, GearApi};
104    ///
105    /// #[tokio::test]
106    /// async fn test_upload_failed() -> anyhow::Result<()> {
107    ///     let api = GearApi::dev_from_path("../target/release/gear").await?;
108    ///
109    ///     let err = api
110    ///         .upload_program(vec![], vec![], b"", u64::MAX, 0)
111    ///         .await
112    ///         .expect_err("Should fail");
113    ///
114    ///     assert!(matches!(
115    ///         err,
116    ///         Error::Module(errors::ModuleError::Gear(errors::Gear::GasLimitTooHigh))
117    ///     ));
118    ///
119    ///     Ok(())
120    /// }
121    /// ```
122    #[error(transparent)]
123    Module(gsdk::metadata::ModuleError),
124}
125
126impl From<SubxtError> for Error {
127    fn from(e: SubxtError) -> Self {
128        if let SubxtError::Runtime(DispatchError::Module(m)) = e {
129            return Error::Module(m.into());
130        }
131
132        Error::Subxt(Box::new(e))
133    }
134}
135
136impl From<GearSDKError> for Error {
137    fn from(e: GearSDKError) -> Self {
138        let e = if let GearSDKError::Subxt(e) = e {
139            if let SubxtError::Runtime(DispatchError::Module(m)) = *e {
140                return Error::Module(m.into());
141            }
142
143            GearSDKError::Subxt(e)
144        } else {
145            e
146        };
147
148        Error::GearSDK(e)
149    }
150}
151
152impl From<subxt::ext::subxt_core::Error> for Error {
153    fn from(value: subxt::ext::subxt_core::Error) -> Self {
154        Self::SubxtCore(Box::new(value))
155    }
156}
