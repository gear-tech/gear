19use crate::{Error, Result};
20use anyhow::anyhow;
21use std::{
22    fmt,
23    net::{AddrParseError, SocketAddrV4},
24};
25use url::Url;

27/// Full WebSocket address required to specify the node.
28#[derive(Clone, Eq, PartialEq, Ord, PartialOrd)]
29pub struct WSAddress {
30    // Host domain name or IP address.
31    //
32    // TODO: `String` here for saving lives, could be
33    // `Ipv4Address`(ip) + tls(wss?) after then.
34    domain: String,
35    port: Option<u16>,
36}
37
38impl WSAddress {
39    // Default substrate node port.
40    const DEFAULT_PORT: u16 = 9944;
41
42    // Local dev node.
43    const LOCALHOST: &'static str = "ws://127.0.0.1";
44
45    // Default secure WebSocket port.
46    const WSS_DEFAULT_PORT: u16 = 443;
47
48    // Vara testnet.
49    const VARA_TESTNET: &'static str = "wss://testnet.vara.network";
50
51    // Vara network.
52    const VARA: &'static str = "wss://rpc.vara.network";
53
54    /// Create a new `WSAddress` from a host `domain` and `port`.
55    ///
56    /// This method does not do any validation of `domain`,
57    /// see [`WSAddress::try_new`] if you need it.
58    pub fn new(domain: impl AsRef<str>, port: impl Into<Option<u16>>) -> Self {
59        Self {
60            domain: domain.as_ref().into(),
61            port: port.into(),
62        }
63    }
64
65    /// Try to create a new `WSAddress` from `domain` and `port`.
66    ///
67    /// Unlike the [`WSAddress::new`] method, this function checks
68    /// that the `domain` is valid.
69    pub fn try_new(domain: impl AsRef<str>, port: impl Into<Option<u16>>) -> Result<Self> {
70        let domain = domain.as_ref().to_string();
71        let port = port.into();
72
73        let url = Url::parse(domain.as_ref())?;
74
75        let valid_domain = matches!(url.scheme(), "ws" | "wss")
76            && !url.cannot_be_a_base()
77            && url.has_host()
78            && url.port().is_none()
79            && url.query().is_none()
80            && url.fragment().is_none();
81
82        if !valid_domain {
83            return Err(Error::WSDomainInvalid);
84        }
85
86        Ok(Self { domain, port })
87    }
88
89    /// Return the address of the local node working in developer mode (running
90    /// with `--dev` argument).
91    ///
92    /// # Examples
93    ///
94    /// ```
95    /// use gclient::WSAddress;
96    ///
97    /// let address = WSAddress::dev();
98    /// assert_eq!(address, WSAddress::new("ws://127.0.0.1", 9944));
99    /// ```
100    pub fn dev() -> Self {
101        Self::dev_with_port(Self::DEFAULT_PORT)
102    }
103
104    /// Return the address of the local node working in developer mode (running
105    /// with `--dev` argument).
106    ///
107    /// # Examples
108    ///
109    /// ```
110    /// use gclient::WSAddress;
111    ///
112    /// let address = WSAddress::dev_with_port(1234);
113    /// assert_eq!(address, WSAddress::new("ws://127.0.0.1", 1234));
114    /// ```
115    pub fn dev_with_port(port: u16) -> Self {
116        Self::new(Self::LOCALHOST, port)
117    }
118
119    /// Return the default address of the public Vara testnet node.
120    ///
121    /// # Examples
122    ///
123    /// ```
124    /// use gclient::WSAddress;
125    ///
126    /// let address = WSAddress::vara_testnet();
127    /// assert_eq!(address, WSAddress::new("wss://testnet.vara.network", 443));
128    /// ```
129    pub fn vara_testnet() -> Self {
130        Self::new(Self::VARA_TESTNET, Self::WSS_DEFAULT_PORT)
131    }
132
133    /// Return the default address of the public Vara node.
134    ///
135    /// # Examples
136    ///
137    /// ```
138    /// use gclient::WSAddress;
139    ///
140    /// let address = WSAddress::vara();
141    /// assert_eq!(address.url(), "wss://rpc.vara.network");
142    /// ```
143    pub fn vara() -> Self {
144        Self::new(Self::VARA, None)
145    }
146
147    /// Convert the address to the URL string.
148    ///
149    /// # Examples
150    ///
151    /// ```
152    /// use gclient::WSAddress;
153    ///
154    /// let address = WSAddress::new("wss://my-node.example.com", 443);
155    /// assert_eq!(address.url(), "wss://my-node.example.com:443");
156    /// ```
157    pub fn url(&self) -> String {
158        if let Some(port) = self.port {
159            format!("{}:{port}", self.domain)
160        } else {
161            self.domain.to_string()
162        }
163    }
164}
165
166impl fmt::Debug for WSAddress {
167    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
168        fmt::Display::fmt(self, f)
169    }
170}
171
172impl fmt::Display for WSAddress {
173    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
174        write!(f, "{}", self.url())
175    }
176}

178impl From<SocketAddrV4> for WSAddress {
179    fn from(addr: SocketAddrV4) -> Self {
180        let tls = addr.port() == 443;
181        let scheme_prefix = if tls { "wss" } else { "ws" }.to_string() + "://";
182
183        Self::new(scheme_prefix + &addr.ip().to_string(), addr.port())
184    }
185}
186
187impl TryInto<SocketAddrV4> for WSAddress {
188    type Error = Error;
189
190    fn try_into(self) -> Result<SocketAddrV4, Self::Error> {
191        let domain = self.domain.split("://").collect::<Vec<_>>();
192        let (ip, mb_port) = if domain.len() != 2 {
193            return Err(anyhow!("Invalid domain").into());
194        } else {
195            match domain[0] {
196                "ws" => (domain[1], 80),
197                "wss" => (domain[1], 443),
198                _ => return Err(anyhow!("Invalid scheme").into()),
199            }
200        };
201
202        Ok(format!("{}:{}", ip, self.port.unwrap_or(mb_port))
203            .parse()
204            .map_err(|e: AddrParseError| anyhow!(e))?)
205    }
206}
