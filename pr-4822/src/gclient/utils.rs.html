19use crate::{Error, Result};
20pub use gear_utils::now_micros;
21use std::{fs, path::Path};
22
23/// Return the full path to the optimized Wasm binary file with the `demo_name`
24/// name located in the `root_path` directory.
25///
26/// # Examples
27///
28/// ```
29/// let wasm_path = gclient::wasm_target(".", "demo_ping");
30/// assert_eq!(wasm_path, "./target/wasm32-gear/release/demo_ping.opt.wasm");
31/// ```
32pub fn wasm_target(root_path: impl AsRef<str>, demo_name: impl AsRef<str>) -> String {
33    format!(
34        "{}/target/wasm32-gear/release/{}.opt.wasm",
35        root_path.as_ref(),
36        demo_name.as_ref()
37    )
38}
39
40/// Read and return contents of a Wasm file specified by the `path`.
41pub fn code_from_os(path: impl AsRef<Path>) -> Result<Vec<u8>> {
42    path.as_ref()
43        .extension()
44        .filter(|&extension| extension.eq("wasm"))
45        .ok_or(Error::WrongBinaryExtension)?;
46
47    fs::read(fs::canonicalize(path)?).map_err(Into::into)
48}
49
50/// Convert hex string to byte array.
51pub fn hex_to_vec(string: impl AsRef<str>) -> Result<Vec<u8>> {
52    hex::decode(string.as_ref().trim_start_matches("0x")).map_err(Into::into)
53}
