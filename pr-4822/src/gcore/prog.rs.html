19//! API for creating programs from Gear programs.
20//!
21//! Once a Wasm code has been uploaded to the chain, the Gear program can create
22//! a new program from the code blob using its identifier.
23//!
24//! To create a new program, you are to provide the code identifier obtained
25//! after running the `upload_code` extrinsic, unique salt (arbitrary data
26//! needed to instantiate several programs from one code blob), and the init
27//! message that consists at least from a payload and value.
28
29use crate::{
30    ActorId, CodeId, MessageId,
31    errors::{Error, Result, SyscallError},
32};
33use gsys::{ErrorWithTwoHashes, HashWithValue};
34
35/// Create a new program and returns its address.
36///
37/// This function creates a program initialization message.
38///
39/// Parameters:
40/// - `code_id` is the code identifier of newly creating program that is
41///   represented as blake2b hash;
42/// - `salt` is the arbitrary data needed to generate an address for a new
43///   program (control of salt uniqueness is entirely on the program developer's
44///   side);
45/// - `payload` that can be used in the `init` function of the newly deployed
46///   "child" program;
47/// - `value` sent with the init message.
48///
49/// # Examples
50///
51/// Basically we can use "automatic" salt generation ("nonce"):
52///
53/// ```
54/// use gcore::{CodeId, prog};
55///
56/// static mut NONCE: i32 = 0;
57///
58/// fn increase() {
59///     unsafe { NONCE += 1 };
60/// }
61///
62/// fn get() -> i32 {
63///     unsafe { NONCE }
64/// }
65///
66/// #[unsafe(no_mangle)]
67/// extern "C" fn handle() {
68///     // We assume we already have a code identifier
69///     let submitted_code: CodeId =
70///         hex_literal::hex!("abf3746e72a6e8740bd9e12b879fbdd59e052cb390f116454e9116c22021ae4a")
71///             .into();
72///     let (message_id, new_program_id) =
73///         prog::create_program(submitted_code, &get().to_le_bytes(), b"", 0)
74///             .expect("Unable to create a program");
75///     increase();
76/// }
77/// ```
78///
79/// Another case for salt is to receive it as input:
80///
81/// ```
82/// use gcore::{msg, prog};
83/// # use gcore::CodeId;
84///
85/// #[unsafe(no_mangle)]
86/// extern "C" fn handle() {
87///     # let submitted_code: CodeId = hex_literal::hex!("abf3746e72a6e8740bd9e12b879fbdd59e052cb390f116454e9116c22021ae4a").into();
88///     // ...
89///     let mut salt = vec![0u8; msg::size()];
90///     msg::read(&mut salt).expect("Unable to read");
91///     let (message_id, new_program_id) = prog::create_program(submitted_code, &salt, b"", 0)
92///         .expect("Unable to create a program");
93/// }
94/// ```
95///
96/// Moreover, messages can be sent to a new program:
97///
98/// ```
99/// use gcore::{msg, prog};
100/// # use gcore::CodeId;
101///
102/// #[unsafe(no_mangle)]
103/// extern "C" fn handle() {
104///     # let submitted_code: CodeId = hex_literal::hex!("abf3746e72a6e8740bd9e12b879fbdd59e052cb390f116454e9116c22021ae4a").into();
105///     # let mut salt = vec![0u8; msg::size()];
106///     # msg::read(&mut salt).unwrap();
107///     // ...
108///     let (_, new_program_id) = prog::create_program(submitted_code, &salt, b"", 0)
109///         .expect("Unable to create a program");
110///     msg::send(new_program_id, b"payload for a new program", 0)
111///         .expect("Unable to send");
112/// }
113/// ```
114pub fn create_program(
115    code_id: CodeId,
116    salt: &[u8],
117    payload: &[u8],
118    value: u128,
119) -> Result<(MessageId, ActorId)> {
120    create_program_delayed(code_id, salt, payload, value, 0)
121}
122
123/// Same as [`create_program`], but with an explicit gas limit.
124#[cfg(not(feature = "ethexe"))]
125pub fn create_program_with_gas(
126    code_id: CodeId,
127    salt: &[u8],
128    payload: &[u8],
129    gas_limit: u64,
130    value: u128,
131) -> Result<(MessageId, ActorId)> {
132    create_program_with_gas_delayed(code_id, salt, payload, gas_limit, value, 0)
133}

135/// Same as [`create_program`], but creates a new program after the `delay`
136/// expressed in block count.
137pub fn create_program_delayed(
138    code_id: CodeId,
139    salt: &[u8],
140    payload: &[u8],
141    value: u128,
142    delay: u32,
143) -> Result<(MessageId, ActorId)> {
144    let cid_value = HashWithValue {
145        hash: code_id.into(),
146        value,
147    };
148
149    let mut res: ErrorWithTwoHashes = Default::default();
150
151    let salt_len = salt.len().try_into().map_err(|_| Error::SyscallUsage)?;
152
153    let payload_len = payload.len().try_into().map_err(|_| Error::SyscallUsage)?;
154
155    unsafe {
156        gsys::gr_create_program(
157            cid_value.as_ptr(),
158            salt.as_ptr(),
159            salt_len,
160            payload.as_ptr(),
161            payload_len,
162            delay,
163            res.as_mut_ptr(),
164        )
165    };
166    SyscallError(res.error_code).into_result()?;
167
168    Ok((res.hash1.into(), res.hash2.into()))
169}
