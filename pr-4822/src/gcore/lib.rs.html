19//! Lightweight library for use in Gear programs.
20//!
21//! This library should be used as a standard library when writing Gear
22//! programs. Compared to [`gstd`](https://docs.gear.rs/gstd/) crate, this
23//! library provides lower-level primitives that allow you to develop less
24//! expensive programs. Choose it if you are ready to write more code
25//! but get a more efficient Wasm.
26//!
27//! Note that you are to define panic and out-of-memory handlers, as the crate
28//! does not provide them by default.
29//!
30//! # Examples
31//!
32//! ```rust,ignore
33//! #![no_std]
34//! #![feature(alloc_error_handler)]
35//!
36//! extern crate galloc;
37//!
38//! use gcore::msg;
39//!
40//! #[unsafe(no_mangle)]
41//! extern "C" fn handle() {
42//!     let mut bytes = [0; 64];
43//!     msg::read(&mut bytes).expect("Unable to read");
44//!     if let Ok(payload) = core::str::from_utf8(&bytes) {
45//!         if payload == "PING" {
46//!             msg::reply(b"PONG", 0).expect("Unable to reply");
47//!         }
48//!     }
49//! }
50//!
51//! # #[cfg(target_arch = "wasm32")]
52//! #[alloc_error_handler]
53//! pub fn oom(_: core::alloc::Layout) -> ! {
54//!     core::arch::wasm32::unreachable()
55//! }
56//!
57//! # #[cfg(target_arch = "wasm32")]
58//! #[panic_handler]
59//! fn panic(_: &core::panic::PanicInfo) -> ! {
60//!     core::arch::wasm32::unreachable()
61//! }
62//! # fn main() {}
63//! ```
