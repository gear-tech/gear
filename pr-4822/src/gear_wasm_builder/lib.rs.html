39/// WASM building tool.
40pub struct WasmBuilder {
41    wasm_project: WasmProject,
42    cargo: CargoCommand,
43    excluded_features: Vec<&'static str>,
44}
45
46impl WasmBuilder {
47    /// Create a new `WasmBuilder`.
48    pub fn new() -> Self {
49        WasmBuilder::create(WasmProject::new())
50    }
51
52    fn create(wasm_project: WasmProject) -> Self {
53        WasmBuilder {
54            wasm_project,
55            cargo: CargoCommand::new(),
56            excluded_features: vec![],
57        }
58    }
59
60    /// Exclude features from the build.
61    pub fn exclude_features(mut self, features: impl Into<Vec<&'static str>>) -> Self {
62        self.excluded_features = features.into();
63        self
64    }
65
66    /// Add pre-processor for wasm file.
67    pub fn with_pre_processor(mut self, pre_processor: Box<dyn PreProcessor>) -> Self {
68        self.wasm_project.add_preprocessor(pre_processor);
69        self
70    }
71
72    /// Add check of recommended toolchain.
73    pub fn with_recommended_toolchain(mut self) -> Self {
74        self.cargo.set_check_recommended_toolchain(true);
75        self
76    }
77
78    /// Force the recommended toolchain to be used, but do not check whether the
79    /// current toolchain is recommended.
80    ///
81    /// NOTE: For internal use only, not recommended for production programs.
82    ///
83    /// An example usage can be found in `examples/out-of-memory/build.rs`.
84    #[doc(hidden)]
85    pub fn with_forced_recommended_toolchain(mut self) -> Self {
86        self.cargo.set_force_recommended_toolchain(true);
87        self
88    }
89
90    /// Build the program and produce an output WASM binary.
91    ///
92    /// Returns `None` if `__GEAR_WASM_BUILDER_NO_BUILD` flag is set.
93    /// Returns `Some(_)` with a tuple of paths to wasm & opt wasm file
94    /// if the build was successful.
95    pub fn build(self) -> Option<(PathBuf, PathBuf)> {
96        if env::var("__GEAR_WASM_BUILDER_NO_BUILD").is_ok() || is_intellij_sync() {
97            _ = self.wasm_project.provide_dummy_wasm_binary_if_not_exist();
98            return None;
99        }
100
101        match self.build_project() {
102            Err(e) => {
103                eprintln!("error: {e}");
104                e.chain()
105                    .skip(1)
106                    .for_each(|cause| eprintln!("|      {cause}"));
107                process::exit(1);
108            }
109            Ok(r) => r,
110        }
111    }
112
113    fn build_project(mut self) -> Result<Option<(PathBuf, PathBuf)>> {
114        self.wasm_project.generate()?;
115
116        self.cargo
117            .set_manifest_path(self.wasm_project.manifest_path());
118        self.cargo.set_target_dir(self.wasm_project.target_dir());
119        let profile = self.wasm_project.profile();
120        let profile = if profile == "debug" { "dev" } else { profile };
121        self.cargo.set_profile(profile.to_string());
122        self.cargo.set_features(&self.enabled_features()?);
123
124        self.cargo.run()?;
125        self.wasm_project.postprocess()
126    }
127
128    fn manifest_path(&self) -> Result<String> {
129        let manifest_path = env::var("CARGO_MANIFEST_DIR")?;
130        Ok(manifest_path)
131    }
132
133    /// Returns features enabled for the current build.
134    fn enabled_features(&self) -> Result<Vec<String>> {
135        let project_features = self.wasm_project.features();
136        let enabled_features_iter = env::vars().filter_map(|(key, _)| {
137            key.strip_prefix("CARGO_FEATURE_")
138                .map(|feature| feature.to_lowercase())
139        });
140        let mut matched_features = Vec::new();
141        let mut unmatched_features = Vec::new();
142        for enabled_feature in enabled_features_iter {
143            // Features coming via the CARGO_FEATURE_<feature> environment variable are in
144            // normilized form, i.e. all dashes are replaced with underscores.
145            let enabled_feature_regex =
146                Regex::new(&format!("^{}$", enabled_feature.replace('_', "[-_]")))?;
147            if self
148                .excluded_features
149                .iter()
150                .any(|excluded_feature| enabled_feature_regex.is_match(excluded_feature))
151            {
152                continue;
153            }
154            if let Some(project_feature) = project_features
155                .iter()
156                .find(|project_feature| enabled_feature_regex.is_match(project_feature))
157            {
158                matched_features.push(project_feature.clone());
159            } else {
160                unmatched_features.push(enabled_feature);
161            }
162        }
163
164        // It may turn out that crate with a build script is built as a dependency of
165        // another crate with build script in the same process (runtime -> pallet-gear
166        // -> examples). In that case, all the CARGO_FEATURE_<feature>
167        // environment variables are propagated down to the dependent crate
168        // which might not have the corresponding features at all.
169        // In such situation, we just warn about unmatched features for diagnostic
170        // purposes and ignore them as cargo itself checks initial set of
171        // features before they reach the build script.
172        if !unmatched_features.is_empty() && unmatched_features != ["default"] {
173            println!(
174                "cargo:warning=Package {}: features `{}` are not available and will be ignored",
175                self.manifest_path()?,
176                unmatched_features.join(", ")
177            );
178        }
179
180        // NOTE: Filter out feature `gcli`.
181        //
182        // dependency feature `gcli` could be captured here
183        // but it is not needed for the build.
184        //
185        // TODO: Filter dep features in this function (#3588)
186        Ok(matched_features
187            .into_iter()
188            .filter(|feature| feature != "gcli")
189            .collect())
190    }
191}

199fn is_intellij_sync() -> bool {
200    // Intellij Rust uses rustc wrapper during project sync
201    env::var("RUSTC_WRAPPER")
202        .unwrap_or_default()
203        .contains("intellij")
204}
205
206// The `std` feature is excluded by default because it is usually used for
207// building so called WASM wrapper which is a static library exposing the built
208// WASM.
209const FEATURES_TO_EXCLUDE_BY_DEFAULT: &[&str] = &["std"];
210
211/// Shorthand function to be used in `build.rs`.
212///
213/// See [WasmBuilder::build()].
214pub fn build() -> Option<(PathBuf, PathBuf)> {
215    WasmBuilder::new()
216        .exclude_features(FEATURES_TO_EXCLUDE_BY_DEFAULT.to_vec())
217        .build()
218}
219
220/// Shorthand function to be used in `build.rs`.
221///
222/// See [WasmBuilder::build()].
223pub fn recommended_nightly() -> Option<(PathBuf, PathBuf)> {
224    WasmBuilder::new()
225        .exclude_features(FEATURES_TO_EXCLUDE_BY_DEFAULT.to_vec())
226        .with_recommended_toolchain()
227        .build()
