1use anyhow::{Error, Result};
2use cargo_metadata::{DependencyKind, Metadata, Node, Package, PackageId};
3use itertools::Itertools;

6const DEFAULT_DENIED_DUPLICATE_CRATES: [&str; 1] = ["gstd"];
7/// Returns the list of crates name.
8fn denied_duplicate_crates() -> HashSet<&'static str> {
9    option_env!("__GEAR_WASM_BUILDER_DENIED_DUPLICATE_CRATES").map_or_else(
10        || DEFAULT_DENIED_DUPLICATE_CRATES.into(),
11        |v| v.split(',').collect(),
12    )
13}

15pub(crate) fn check(metadata: &Metadata, root_id: &PackageId) -> Result<()> {
16    let denied_duplicate_crates = denied_duplicate_crates();
17    let mut packages = metadata.packages.clone();
18    packages.sort_by(|a, b| a.name.cmp(&b.name));
19
20    let mut duplicates: Vec<String> = Vec::new();
21
22    if let Some(resolve) = &metadata.resolve {
23        for (name, group) in &packages
24            .iter()
25            .filter(|p| denied_duplicate_crates.contains(&p.name.as_str()))
26            .chunk_by(|p| &p.name)
27        {
28            let group: Vec<&Package> = group.collect();
29
30            if group.len() <= 1 {
31                continue;
32            }
33
34            if group
35                .iter()
36                .all(|p| is_normal_dep(&resolve.nodes, root_id, &p.id))
37            {
38                let mut versions: Vec<_> = group.into_iter().map(|p| &p.version).collect();
39                versions.sort();
40                let versions = versions.iter().join(", ");
41
42                duplicates.push(format!(
43                    "multiple versions for dependency `{name}`: {versions}"
44                ));
45            }
46        }
47    }
48
49    if duplicates.is_empty() {
50        Ok(())
51    } else {
52        Err(Error::msg(duplicates.join("\n")))
53    }
54}

56fn is_normal_dep(nodes: &[Node], local_id: &PackageId, dep_id: &PackageId) -> bool {
57    fn depends_on(node: &Node, dep_id: &PackageId) -> bool {
58        node.deps.iter().any(|dep| {
59            dep.pkg == *dep_id
60                && dep
61                    .dep_kinds
62                    .iter()
63                    .any(|info| matches!(info.kind, DependencyKind::Normal))
64        })
65    }
66
67    nodes
68        .iter()
69        .filter(|node| depends_on(node, dep_id))
70        .any(|node| node.id == *local_id || is_normal_dep(nodes, local_id, &node.id))
71}
