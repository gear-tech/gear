19use crate::{builder_error::BuilderError, multiple_crate_versions};
20use anyhow::{Context, Result, ensure};
21use cargo_metadata::{CrateType, Metadata, MetadataCommand, Package};
22use std::{collections::BTreeMap, path::Path};
23
24/// Helper to get a crate info extracted from the `Cargo.toml`.
25#[derive(Debug, Default)]
26pub struct CrateInfo {
27    /// Original name of the crate.
28    pub name: String,
29    /// Crate name converted to the snake case.
30    pub snake_case_name: String,
31    /// Crate version.
32    pub version: String,
33    /// Crate features.
34    pub features: BTreeMap<String, Vec<String>>,
35    /// Crate custom profiles
36    pub profiles: BTreeMap<String, toml::Value>,
37    /// Workspace patches
38    pub patch: BTreeMap<String, toml::Value>,
39}
40
41impl CrateInfo {
42    /// Create a new `CrateInfo` from a path to the `Cargo.toml`.
43    pub fn from_manifest(manifest_path: &Path) -> Result<Self> {
44        ensure!(
45            manifest_path.exists(),
46            BuilderError::ManifestPathInvalid(manifest_path.to_path_buf())
47        );
48
49        let mut meta_cmd = MetadataCommand::new();
50        let metadata = meta_cmd
51            .manifest_path(manifest_path)
52            // As we are being called inside a build-script, this env variable is set.
53            // However, this can lead to cross-compilation errors.
54            .env_remove("CARGO_ENCODED_RUSTFLAGS")
55            .exec()
56            .context("unable to invoke `cargo metadata`")?;
57
58        let root_package = Self::root_package(&metadata)
59            .ok_or_else(|| BuilderError::RootPackageNotFound.into())
60            .and_then(Self::check)?;
61
62        let manifest = cargo_toml::Manifest::from_path(metadata.workspace_root.join("Cargo.toml"))
63            .context("manifest parsing failed")?;
64        let profiles = manifest
65            .profile
66            .custom
67            .into_iter()
68            .map(|(k, v)| Ok((k, toml::Value::try_from(v)?)))
69            .collect::<Result<_>>()
70            .context("failed to convert profile to `toml::Value`")?;
71        let patch = manifest
72            .patch
73            .into_iter()
74            .map(|(k, v)| Ok((k, toml::Value::try_from(v)?)))
75            .collect::<Result<_>>()
76            .context("failed to convert patch to `toml::Value`")?;
77
78        multiple_crate_versions::check(&metadata, &root_package.id)?;
79
80        let name = root_package.name.clone().into_inner();
81        let snake_case_name = name.replace('-', "_");
82        let version = root_package.version.to_string();
83        let features = root_package.features.clone();
84
85        Ok(Self {
86            name,
87            snake_case_name,
88            version,
89            features,
90            profiles,
91            patch,
92        })
93    }
94
95    fn root_package(metadata: &Metadata) -> Option<&Package> {
96        let root_id = metadata.resolve.as_ref()?.root.as_ref()?;
97        metadata
98            .packages
99            .iter()
100            .find(|package| package.id == *root_id)
101    }
102
103    /// check package
104    ///
105    /// - if crate-type contains "lib" or "rlib":
106    fn check(pkg: &Package) -> Result<&Package> {
107        // cargo can't import executables (bin, cdylib etc), but libs
108        // only (rlib).
109        //
110        // if no `[lib]` table, the `crate_types` will be [ "lib" ]
111        // by default, we can not detect if this is "rlib" because it
112        // is the "compiler recommended" style of library.
113        //
114        // see also https://doc.rust-lang.org/reference/linkage.html
115        let validated_lib = |ty: &CrateType| matches!(ty, CrateType::Lib | CrateType::RLib);
116        let pkg_snake_case_name = pkg.name.replace('-', "_");
117
118        // Check for rustc version. See https://github.com/rust-lang/cargo/pull/12783
119        let compatible = rustc_version::version()?.lt(&rustc_version::Version::parse("1.79.0")?);
120
121        let _ = pkg
122            .targets
123            .iter()
124            .find(|target| {
125                if compatible {
126                    target.name.eq(&*pkg.name) && target.crate_types.iter().any(validated_lib)
127                } else {
128                    target.name.eq(&pkg_snake_case_name)
129                        && target.crate_types.iter().any(validated_lib)
130                }
131            })
132            .ok_or(BuilderError::CrateTypeInvalid)?;
133
134        Ok(pkg)
135    }
136}
