31/// Temporary project generated to build a WASM output.
32///
33/// This project is required due to the cargo locking during build.
34pub struct WasmProject {
35    original_dir: PathBuf,
36    out_dir: PathBuf,
37    target_dir: PathBuf,
38    wasm_target_dir: PathBuf,
39    file_base_name: Option<String>,
40    profile: String,
41    features: Option<Vec<String>>,
42    pre_processors: Vec<Box<dyn PreProcessor>>,
43}
44
45/// Pre-processor result.
46pub type PreProcessorResult<T> = Result<T>;
47
48/// Preprocessor target specifying the output file name.
49#[derive(Debug, Default, PartialEq, Eq)]
50pub enum PreProcessorTarget {
51    /// Use the default file name (i.e. overwrite the .wasm file).
52    #[default]
53    Default,
54    /// Use the given file name with pre-processor name at end.
55    ///
56    /// `Named("foo.wasm".into())` will be converted to
57    /// `"foo_{pre_processor_name}.wasm"`.
58    Named(String),
59}
60
61/// Pre-processor hook for wasm generation.
62pub trait PreProcessor {
63    /// Returns the name of the pre-processor. It must be some unique string.
64    fn name(&self) -> &'static str {
65        env!("CARGO_PKG_NAME")
66    }
67
68    /// Returns the result of the pre-processor.
69    fn pre_process(
70        &self,
71        original: PathBuf,
72    ) -> PreProcessorResult<Vec<(PreProcessorTarget, Vec<u8>)>>;
73}

75impl WasmProject {
76    /// Create a new `WasmProject`.
77    pub fn new() -> Self {
78        let original_dir: PathBuf = env::var("CARGO_MANIFEST_DIR")
79            .expect("`CARGO_MANIFEST_DIR` is always set in build scripts")
80            .into();
81
82        let out_dir: PathBuf = env::var("OUT_DIR")
83            .expect("`OUT_DIR` is always set in build scripts")
84            .into();
85
86        let profile = out_dir
87            .components()
88            .rev()
89            .take_while(|c| c.as_os_str() != "target")
90            .collect::<Vec<_>>()
91            .into_iter()
92            .rev()
93            .take_while(|c| c.as_os_str() != "build")
94            .last()
95            .expect("Path should have subdirs in the `target` dir")
96            .as_os_str()
97            .to_string_lossy()
98            .into();
99
100        let mut target_dir = out_dir
101            .ancestors()
102            .find(|path| path.ends_with(&profile))
103            .and_then(|path| path.parent())
104            .map(|p| p.to_owned())
105            .expect("Could not find target directory");
106
107        let mut wasm_target_dir = target_dir.clone();
108        wasm_target_dir.push("wasm32-gear");
109        wasm_target_dir.push(&profile);
110
111        target_dir.push("wasm-projects");
112        target_dir.push(&profile);
113
114        WasmProject {
115            original_dir,
116            out_dir,
117            target_dir,
118            wasm_target_dir,
119            file_base_name: None,
120            profile,
121            features: None,
122            pre_processors: vec![],
123        }
124    }

153    /// Generate a temporary cargo project that includes the original package as
154    /// a dependency.
155    pub fn generate(&mut self) -> Result<()> {
156        let original_manifest = self.original_dir.join("Cargo.toml");
157        let crate_info = CrateInfo::from_manifest(&original_manifest)?;
158        self.file_base_name = Some(crate_info.snake_case_name.clone());
159
160        let mut package = Table::new();
161        package.insert("name".into(), format!("{}-wasm", &crate_info.name).into());
162        package.insert("version".into(), crate_info.version.into());
163        package.insert("edition".into(), "2024".into());
164
165        let mut lib = Table::new();
166        lib.insert("name".into(), crate_info.snake_case_name.into());
167        lib.insert("crate-type".into(), vec!["cdylib".to_string()].into());
168
169        let mut dev_profile = Table::new();
170        dev_profile.insert("opt-level".into(), OPT_LEVEL.into());
171
172        let mut release_profile = Table::new();
173        release_profile.insert("lto".into(), "fat".into());
174        release_profile.insert("opt-level".into(), OPT_LEVEL.into());
175        release_profile.insert("codegen-units".into(), 1.into());
176
177        let mut production_profile = Table::new();
178        production_profile.insert("inherits".into(), "release".into());
179
180        let mut profile = crate_info.profiles;
181        profile.insert("dev".into(), dev_profile.clone().into());
182        profile.insert("release".into(), release_profile.into());
183        profile.insert("production".into(), production_profile.into());
184
185        let mut crate_package = Table::new();
186        crate_package.insert("package".into(), crate_info.name.into());
187        crate_package.insert(
188            "path".into(),
189            self.original_dir.display().to_string().into(),
190        );
191        crate_package.insert("default-features".into(), false.into());
192
193        let mut dependencies = Table::new();
194        dependencies.insert("orig-project".into(), crate_package.into());
195
196        let mut features = Table::new();
197        for feature in crate_info.features.keys() {
198            if feature != "default" {
199                features.insert(
200                    feature.into(),
201                    vec![format!("orig-project/{feature}")].into(),
202                );
203            }
204        }
205        self.features = Some(features.keys().cloned().collect());
206
207        let mut cargo_toml = Table::new();
208        cargo_toml.insert("package".into(), package.into());
209        cargo_toml.insert("lib".into(), lib.into());
210        cargo_toml.insert("dependencies".into(), dependencies.into());
211        cargo_toml.insert("profile".into(), profile.into());
212        cargo_toml.insert("features".into(), features.into());
213        cargo_toml.insert("workspace".into(), Table::new().into());
214        cargo_toml.insert("patch".into(), crate_info.patch.into());
215
216        smart_fs::write(self.manifest_path(), toml::to_string_pretty(&cargo_toml)?)
217            .context("Failed to write generated manifest path")?;
218
219        // Copy original `Cargo.lock` if any
220        let from_lock = self.original_dir.join("Cargo.lock");
221        let to_lock = self.out_dir.join("Cargo.lock");
222        drop(fs::copy(from_lock, to_lock));
223
224        let source_code =
225            "#![no_std]\n#[allow(unused_imports)]\npub use orig_project::*;\n".to_owned();
226
227        fs::create_dir_all(&self.wasm_target_dir).with_context(|| {
228            format!(
229                "Failed to create WASM target directory: {}",
230                self.wasm_target_dir.display()
231            )
232        })?;
233
234        let src_dir = self.out_dir.join("src");
235        fs::create_dir_all(&src_dir).context("Failed to create `src` directory")?;
236        smart_fs::write(src_dir.join("lib.rs"), source_code)?;
237
238        Ok(())
239    }

241    fn generate_bin_path(&self, file_base_name: &String) -> Result<()> {
242        let relative_path_to_wasm = pathdiff::diff_paths(&self.wasm_target_dir, &self.original_dir)
243            .with_context(|| {
244                format!(
245                    "wasm_target_dir={}; original_dir={}",
246                    self.wasm_target_dir.display(),
247                    self.original_dir.display()
248                )
249            })
250            .expect("Unable to calculate relative path")
251            .join(file_base_name);
252
253        smart_fs::write(
254            self.original_dir.join(".binpath"),
255            format!("{}", relative_path_to_wasm.display()),
256        )
257        .context("unable to write `.binpath`")?;
258
259        Ok(())
260    }

262    /// Generates output optimized wasm file, `.binpath` file for our tests
263    /// system and wasm binaries informational file.
264    /// Makes a copy of original wasm file in `self.wasm_target_dir`.
265    pub fn postprocess_opt<P: AsRef<Path>>(
266        &self,
267        original_wasm_path: P,
268        file_base_name: &str,
269    ) -> Result<PathBuf> {
270        let [original_copy_wasm_path, opt_wasm_path] = [".wasm", ".opt.wasm"]
271            .map(|ext| self.wasm_target_dir.join([file_base_name, ext].concat()));
272
273        // Copy original file to `self.wasm_target_dir`
274        smart_fs::copy_if_newer(&original_wasm_path, &original_copy_wasm_path)
275            .context("unable to copy WASM file")?;
276
277        // Optimize wasm using and `wasm-opt` and our optimizations.
278        if smart_fs::check_if_newer(&original_wasm_path, &opt_wasm_path)? {
279            let mut optimizer = Optimizer::new(&original_copy_wasm_path)?;
280            optimizer
281                .insert_stack_end_export()
282                .unwrap_or_else(|err| log::info!("Cannot insert stack end export: {err}"));
283            optimizer.strip_custom_sections();
284            optimizer.strip_exports();
285            optimizer.flush_to_file(&opt_wasm_path);
286
287            optimize::optimize_wasm(&opt_wasm_path, &opt_wasm_path, "4", true)
288                .map(|res| {
289                    log::info!(
290                        "Wasm-opt reduced wasm size: {} -> {}",
291                        res.original_size,
292                        res.optimized_size
293                    );
294                })
295                .unwrap_or_else(|err| {
296                    println!("cargo:warning=wasm-opt optimizations error: {err}");
297                });
298        }
299
300        // Create `wasm_binary.rs`
301        smart_fs::write(
302            self.out_dir.join("wasm_binary.rs"),
303            format!(
304                r#"#[allow(unused)]
305pub const WASM_BINARY: &[u8] = include_bytes!("{}");
306#[allow(unused)]
307pub const WASM_BINARY_OPT: &[u8] = include_bytes!("{}");"#,
308                display_path(original_copy_wasm_path.as_path()),
309                display_path(opt_wasm_path.as_path()),
310            ),
311        )
312        .context("unable to write `wasm_binary.rs`")?;
313        Ok(opt_wasm_path)
314    }

316    /// Post-processing after the WASM binary has been built.
317    ///
318    /// - Copy WASM binary from `OUT_DIR` to
319    ///   `target/wasm32-gear/<profile>`
320    /// - Generate optimized binary from the built program
321    /// - Generate `wasm_binary.rs` source file in `OUT_DIR`
322    pub fn postprocess(&self) -> Result<Option<(PathBuf, PathBuf)>> {
323        let file_base_name = self
324            .file_base_name
325            .as_ref()
326            .expect("Run `WasmProject::generate()` first");
327
328        let original_wasm_path = self.target_dir.join(format!(
329            "wasm32v1-none/{}/{file_base_name}.wasm",
330            self.profile
331        ));
332
333        fs::create_dir_all(&self.target_dir).context("Failed to create target directory")?;
334
335        self.generate_bin_path(file_base_name)?;
336
337        let mut wasm_files = vec![(original_wasm_path.clone(), file_base_name.clone())];
338
339        for pre_processor in &self.pre_processors {
340            let pre_processor_name = pre_processor.name().to_lowercase().replace('-', "_");
341            let pre_processor_output = pre_processor.pre_process(original_wasm_path.clone())?;
342
343            let default_targets = pre_processor_output
344                .iter()
345                .filter(|(target, _)| *target == PreProcessorTarget::Default)
346                .count();
347            if default_targets > 1 {
348                return Err(anyhow!(
349                    "Pre-processor \"{pre_processor_name}\" cannot have more than one default target."
350                ));
351            }
352
353            for (pre_processor_target, content) in pre_processor_output {
354                let (pre_processed_path, new_wasm_file) = match pre_processor_target {
355                    PreProcessorTarget::Default => (original_wasm_path.clone(), false),
356                    PreProcessorTarget::Named(filename) => {
357                        let path = Path::new(&filename);
358
359                        let file_stem = path
360                            .file_stem()
361                            .and_then(|s| s.to_str())
362                            .expect("Failed to get file stem");
363                        let extension = path.extension().and_then(|s| s.to_str());
364
365                        let filename = match extension {
366                            Some(extension) => {
367                                format!("{file_stem}_{pre_processor_name}.{extension}")
368                            }
369                            None => format!("{file_stem}_{pre_processor_name}"),
370                        };
371
372                        (
373                            self.wasm_target_dir.join(filename),
374                            extension == Some("wasm"),
375                        )
376                    }
377                };
378                fs::write(&pre_processed_path, content)?;
379
380                if new_wasm_file {
381                    let file_stem = pre_processed_path
382                        .file_stem()
383                        .and_then(|s| s.to_str().map(|s| s.to_string()))
384                        .expect("Failed to get file stem");
385                    wasm_files.push((pre_processed_path, file_stem));
386                }
387            }
388        }
389
390        // Tuple with PathBuf last wasm & opt.wasm
391        let mut wasm_paths: Option<(PathBuf, PathBuf)> = None;
392        for (wasm_path, file_base_name) in &wasm_files {
393            let wasm_opt = self.postprocess_opt(wasm_path, file_base_name)?;
394            wasm_paths = Some((wasm_path.clone(), wasm_opt));
395        }
396
397        for (wasm_path, _) in &wasm_files {
398            let code = fs::read(wasm_path)?;
399
400            validate_program(code)?;
401        }
402
403        if env::var("__GEAR_WASM_BUILDER_NO_FEATURES_TRACKING").is_err() {
404            self.force_rerun_on_next_run(&original_wasm_path)?;
405        }
406        Ok(wasm_paths)
407    }

409    // Force cargo to re-run the build script next time it is invoked.
410    // It is needed because feature set or toolchain can change.
411    fn force_rerun_on_next_run(&self, wasm_file_path: &Path) -> Result<()> {
412        let stamp_file_path = wasm_file_path.with_extension("stamp");
413        fs::write(&stamp_file_path, ChronoLocal::now().to_rfc3339())
414            .context("Failed to write stamp file")?;
415        println!("cargo:rerun-if-changed={}", stamp_file_path.display());
416        Ok(())
417    }

419    /// Provide a dummy WASM binary if there doesn't exist one.
420    pub fn provide_dummy_wasm_binary_if_not_exist(&self) -> PathBuf {
421        let wasm_binary_rs = self.out_dir.join("wasm_binary.rs");
422        if wasm_binary_rs.exists() {
423            return wasm_binary_rs;
424        }
425
426        let content = r#"#[allow(unused)]
427    pub const WASM_BINARY: &[u8] = &[];
428    #[allow(unused)]
429    pub const WASM_BINARY_OPT: &[u8] = &[];
430    "#;
431        let path = wasm_binary_rs.as_path();
432        fs::write(path, content)
433            .unwrap_or_else(|_| panic!("Writing `{}` should not fail!", display_path(path)));
434        wasm_binary_rs
435    }

438// Windows has path like `path\to\somewhere` which is incorrect for `include_*`
439// Rust's macros
440fn display_path<P: AsRef<Path>>(path: P) -> String {
441    path.as_ref().display().to_string().replace('\\', "/")
