19//! Filesystem functions that avoid file changes
20//! so we can avoid unnecessary build script launches.
21//! because cargo looks for `mtime` metadata file parameter
22
23use anyhow::Result;
24use std::{fs, io::ErrorKind, path::Path};
25
26pub(crate) fn copy_if_newer(from: impl AsRef<Path>, to: impl AsRef<Path>) -> Result<bool> {
27    let from = from.as_ref();
28    let to = to.as_ref();
29
30    if check_if_newer(from, to)? {
31        fs::copy(from, to)?;
32        Ok(true)
33    } else {
34        Ok(false)
35    }
36}
37
38pub(crate) fn check_if_newer(left: impl AsRef<Path>, right: impl AsRef<Path>) -> Result<bool> {
39    let right_metadata = fs::metadata(right);
40
41    if let Err(io_error) = right_metadata.as_ref()
42        && io_error.kind() == ErrorKind::NotFound
43    {
44        return Ok(true);
45    }
46
47    let right_metadata = right_metadata.unwrap();
48    let left_metadata = fs::metadata(left)?;
49
50    Ok(left_metadata.modified()? > right_metadata.modified()?)
51}
52
53fn check_changed(path: &Path, contents: &[u8]) -> Result<bool> {
54    // file does not exist
55    let Ok(metadata) = fs::metadata(path) else {
56        return Ok(true);
57    };
58
59    if metadata.len() != contents.len() as u64 {
60        return Ok(true);
61    }
62
63    let old_contents = fs::read(path)?;
64    Ok(old_contents != contents)
65}
66
67pub(crate) fn write<P: AsRef<Path>, C: AsRef<[u8]>>(path: P, contents: C) -> Result<()> {
68    let path = path.as_ref();
69    let contents = contents.as_ref();
70
71    if check_changed(path, contents)? {
72        fs::write(path, contents)?;
73    }
74
75    Ok(())
76}
