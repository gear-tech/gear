19//! # Gear Messenger Pallet
20//!
21//! The Gear Messenger Pallet provides functionality for handling messages.
22//!
23//! - [`Config`]
24//! - [`Pallet`]
25//!
26//! ## Overview
27//!
28//! The Gear Messenger Pallet's main aim is to separate message storages out
29//! of Gear's execution logic and provide soft functionality to manage them.
30//!
31//! The Gear Messenger Pallet provides functions for:
32//! - Counting amount of messages sent from outside (from extrinsics)
33//!   within the current block.
34//! - Counting amount of messages removed from queue to be processed
35//!   or skipped within the current block.
36//! - Managing continuation of queue processing within the current block.
37//! - Storing and managing message queue, it's pushing and popping algorithms.
38//! - Storing and managing mailbox, it's insertion and removal algorithms,
39//!   including the value claiming with Balances Pallet as `Currency`
40//!   `Config`'s associated type.
41//!
42//! ## Interface
43//!
44//! The Gear Messenger Pallet implements `gear_common::storage::Messenger` trait
45//! and shouldn't contain any other functionality, except this trait declares.
46//!
47//! ## Usage
48//!
49//! How to use the messaging functionality from the Gear Messenger Pallet:
50//!
51//! 1. Implement it's `Config` for your runtime with specified `Currency` type.
52//!
53//! ```ignore
54//! // `runtime/src/lib.rs`
55//! // ... //
56//!
57//! impl pallet_gear_messenger::Config for Runtime {
58//!     type Currency = .. ;
59//! }
60//!
61//! // ... //
62//! ```
63//!
64//! 2. Provide associated type for your pallet's `Config`, which implements
65//!    `gear_common::storage::Messenger` trait,
66//!    specifying associated types if needed.
67//!
68//! ```ignore
69//! // `some_pallet/src/lib.rs`
70//! // ... //
71//!
72//! use gear_common::storage::Messenger;
73//!
74//! #[pallet::config]
75//! pub trait Config: frame_system::Config {
76//!     // .. //
77//!
78//!     type Messenger: Messenger<Capacity = u32>;
79//!
80//!     // .. //
81//! }
82//! ```
83//!
84//! 3. Declare Gear Messenger Pallet in your `construct_runtime!` macro.
85//!
86//! ```ignore
87//! // `runtime/src/lib.rs`
88//! // ... //
89//!
90//! construct_runtime!(
91//!     pub enum Runtime
92//!         where // ... //
93//!     {
94//!         // ... //
95//!
96//!         GearMessenger: pallet_gear_messenger,
97//!
98//!         // ... //
99//!     }
100//! );
101//!
102//! // ... //
103//! ```
104//! `GearMessenger: pallet_gear_messenger`
105//!
106//! 4. Set `GearMessenger` as your pallet `Config`'s `Messenger type.
107//!
108//! ```ignore
109//! // `runtime/src/lib.rs`
110//! // ... //
111//!
112//! impl some_pallet::Config for Runtime {
113//!     // ... //
114//!
115//!     type Messenger = GearMessenger;
116//!
117//!     // ... //
118//! }
119//!
120//! // ... //
121//! ```
122//!
123//! 5. Work with Gear Messenger Pallet in your pallet with provided
124//!    associated type interface.
125//!
126//! ## Genesis config
127//!
128//! The Gear Messenger Pallet doesn't depend on the `GenesisConfig`.
129//!
130//! ## Assumptions
131//!
132//! * You should manually control storage load from queue and mailbox
133//!   length overflow (see Gear Payment Pallet).

198    pub enum Error<T> {
199        /// Occurs when given key already exists in queue.
200        QueueDuplicateKey,
201        /// Occurs when queue's element wasn't found in storage.
202        QueueElementNotFound,
203        /// Occurs when queue's head should contain value,
204        /// but it's empty for some reason.
205        QueueHeadShouldBeSet,
206        /// Occurs when queue's head should be empty,
207        /// but it contains value for some reason.
208        QueueHeadShouldNotBeSet,
209        /// Occurs when queue's tail element contains link
210        /// to the next element.
211        QueueTailHasNextKey,
212        /// Occurs when while searching queue's pre-tail,
213        /// element wasn't found.
214        QueueTailParentNotFound,
215        /// Occurs when queue's tail should contain value,
216        /// but it's empty for some reason.
217        QueueTailShouldBeSet,
218        /// Occurs when queue's tail should be empty,
219        /// but it contains value for some reason.
220        QueueTailShouldNotBeSet,
221        /// Occurs when given value already exists in mailbox.
222        MailboxDuplicateKey,
223        /// Occurs when mailbox's element wasn't found in storage.
224        MailboxElementNotFound,
225        /// Occurs when given value already exists in waitlist.
226        WaitlistDuplicateKey,
227        /// Occurs when waitlist's element wasn't found in storage.
228        WaitlistElementNotFound,
229    }

442    // Callback trait implementation.

444    // Pop front increases amount of messages dequeued.

446    where

449        fn call(_arg: &V) {

456    /// Callback function for success `push_front` action.
457    pub struct OnPushFront<T: crate::Config, V>(PhantomData<(T, V)>)
458    where
459        T::AccountId: Origin;
460
461    // Callback trait implementation.
462    //
463    // Push front means requeue in Gear Messenger Context,
464    // so the dequeued amount should be decreased and
465    // queue processing stopped.
466    impl<T: crate::Config, V> Callback<V> for OnPushFront<T, V>
467    where
468        T::AccountId: Origin,
469    {
470        fn call(_arg: &V) {
471            <Pallet<T> as Messenger>::Dequeued::decrease();
472            <Pallet<T> as Messenger>::QueueProcessing::deny();
473        }
474    }

645    // Gear Messenger Pallet hooks.
646    //
647    // The logic of the pallet provides block-dependent logic
648    // (amount of messages sent within the block, etc.), so we
649    // have to reset the state per each block initialization.
650    #[pallet::hooks]
651    impl<T: Config> Hooks<BlockNumberFor<T>> for Pallet<T>
652    where
653        T::AccountId: Origin,
654    {
655        /// Block initialization.
656        fn on_initialize(_bn: BlockNumberFor<T>) -> Weight {
657            // Amount of weight used for initialization.
658            let mut weight = Weight::zero();
659
660            // Clear amount of messages sent.
661            //
662            // Removes value from storage. Single DB write.
663            <Self as Messenger>::Sent::reset();
664            weight += T::DbWeight::get().writes(1);
665
666            // Clear amount of messages dequeued.
667            //
668            // Removes value from storage. Single DB write.
669            <Self as Messenger>::Dequeued::reset();
670            weight += T::DbWeight::get().writes(1);
671
672            // Allow queue processing in this block.
673            //
674            // Puts value in storage. Single DB write.
675            <Self as Messenger>::QueueProcessing::allow();
676            weight += T::DbWeight::get().writes(1);
677
678            weight
679        }
680    }
