40/// Execution error.
41#[repr(u32)]
42#[non_exhaustive]
43#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, Sequence, thiserror::Error)]
44pub enum ExecutionError {
45    /// An error occurs in attempt to charge more gas than available for operation.
46    #[error("Not enough gas for operation")]
47    NotEnoughGas = 100,
48
49    /// The error occurs when balance is less than required by operation.
50    #[error("Not enough value for operation")]
51    NotEnoughValue = 101,
52
53    /// Overflow in 'gr_read'
54    #[error("Length is overflowed to read payload")]
55    TooBigReadLen = 103,
56
57    /// Cannot take data in payload range
58    #[error("Cannot take data in payload range from message with size")]
59    ReadWrongRange = 104,
60
61    /// The error occurs when functions related to reply context, used without it.
62    #[error("Not running in reply context")]
63    NoReplyContext = 105,
64
65    /// The error occurs when functions related to signal context, used without it.
66    #[error("Not running in signal context")]
67    NoSignalContext = 106,
68
69    /// The error occurs when functions related to status code, used without required context.
70    #[error("No status code in reply/signal context")]
71    NoStatusCodeContext = 107,
72
73    /// An error occurs in attempt to send or push reply while reply function is banned.
74    #[error("Reply sending is only allowed in `init` and `handle` functions")]
75    IncorrectEntryForReply = 108,
76}

91/// Error using messages.
92#[repr(u32)]
93#[non_exhaustive]
94#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, Sequence, thiserror::Error)]
95pub enum MessageError {
96    /// Message has bigger then allowed one message size
97    #[error("Max message size exceed")]
98    MaxMessageSizeExceed = 300,
99
100    /// The error "Message limit exceeded" occurs when a program attempts to
101    /// send more than the maximum amount of messages allowed within a single
102    /// execution (current setting - 1024).
103    #[error("Message limit exceeded")]
104    OutgoingMessagesAmountLimitExceeded = 301,
105
106    /// The error occurs in case of attempt to send more than one replies.
107    #[error("Duplicate reply message")]
108    DuplicateReply = 302,
109
110    /// The error occurs in attempt to get the same message from the waitlist
111    /// again (which is waked already).
112    #[error("Duplicate waking message")]
113    DuplicateWaking = 303,
114
115    /// An attempt to commit or push a payload into an already formed message.
116    #[error("An attempt to commit or push a payload into an already formed message")]
117    LateAccess = 304,
118
119    /// The error occurs in case of not valid identifier specified.
120    #[error("Message with given handle is not found")]
121    OutOfBounds = 305,
122
123    /// The error occurs in attempt to initialize the same program twice within
124    /// a single execution.
125    #[error("Duplicated program initialization message")]
126    DuplicateInit = 306,
127
128    /// Everything less than existential deposit but greater than 0 is not considered as available balance and not saved in DB.
129    /// Value between 0 and existential deposit cannot be sent in message.
130    #[error("In case of non-zero message value must be greater than existential deposit")]
131    InsufficientValue = 307,
132
133    /// Everything less than mailbox threshold but greater than 0 is not considered as available gas limit and
134    /// not inserted in mailbox.
135    ///
136    /// Gas limit between 0 and mailbox threshold cannot be inserted in mailbox.
137    #[error("In case of non-zero message gas limit must be greater than mailbox threshold")]
138    InsufficientGasLimit = 308,
139
140    /// The error occurs when program tries to create reply deposit for message
141    /// that already been created within the execution.
142    #[error("Reply deposit already exists for given message")]
143    DuplicateReplyDeposit = 309,
144
145    /// The error occurs when program tries to create reply deposit for message
146    /// that wasn't sent within the execution or for reply.
147    #[error(
148        "Reply deposit could be only created for init or handle message sent within the execution"
149    )]
150    IncorrectMessageForReplyDeposit = 310,
151
152    /// The error occurs when program tries to send messages
153    /// with total size bigger than allowed.
154    #[error("Outgoing messages bytes limit exceeded")]
155    OutgoingMessagesBytesLimitExceeded = 311,
156
157    /// The error occurs when a wrong offset of the input buffer (currently executing message payload)
158    /// is provided.
159    #[error("Offset value for the input payload is out of it's size bounds")]
160    OutOfBoundsInputSliceOffset = 312,
161
162    /// The error occurs when a too big length value to form a slice (range) of the input buffer
163    /// (currently executing message payload) is provided.
164    #[error("Too big length value is set to form a slice (range) of the input buffer")]
165    OutOfBoundsInputSliceLength = 313,
166
167    // TODO: remove after delay refactoring is done
168    /// An error occurs in attempt to charge gas for dispatch stash hold.
169    #[error("Not enough gas to hold dispatch message")]
170    InsufficientGasForDelayedSending = 399,
171}

196/// An error occurred in API.
197#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, Sequence, thiserror::Error)]
198#[non_exhaustive]
199pub enum ExtError {
200    /// Execution error.
201    #[error("Execution error: {0}")]
202    Execution(#[from] ExecutionError),
203
204    /// Memory error.
205    #[error("Memory error: {0}")]
206    Memory(#[from] MemoryError),
207
208    /// Message error.
209    #[error("Message error: {0}")]
210    Message(#[from] MessageError),
211
212    /// Reservation error.
213    #[error("Reservation error: {0}")]
214    Reservation(#[from] ReservationError),
215
216    /// There is a new error variant old program don't support.
217    #[error("Unsupported error")]
218    Unsupported,
219}
220
221impl ExtError {
222    /// Convert error into code.
223    pub fn to_u32(self) -> u32 {
224        match self {
225            ExtError::Execution(err) => err as u32,
226            ExtError::Memory(err) => err as u32,
227            ExtError::Message(err) => err as u32,
228            ExtError::Reservation(err) => err as u32,
229            ExtError::Unsupported => u32::MAX,
230        }
231    }
232
233    /// Convert code into error.
234    pub fn from_u32(code: u32) -> Option<Self> {
235        match code {
236            100 => Some(ExecutionError::NotEnoughGas.into()),
237            101 => Some(ExecutionError::NotEnoughValue.into()),
238            103 => Some(ExecutionError::TooBigReadLen.into()),
239            104 => Some(ExecutionError::ReadWrongRange.into()),
240            105 => Some(ExecutionError::NoReplyContext.into()),
241            106 => Some(ExecutionError::NoSignalContext.into()),
242            107 => Some(ExecutionError::NoStatusCodeContext.into()),
243            108 => Some(ExecutionError::IncorrectEntryForReply.into()),
244            //
245            200 => Some(MemoryError::RuntimeAllocOutOfBounds.into()),
246            201 => Some(MemoryError::AccessOutOfBounds.into()),
247            //
248            300 => Some(MessageError::MaxMessageSizeExceed.into()),
249            301 => Some(MessageError::OutgoingMessagesAmountLimitExceeded.into()),
250            302 => Some(MessageError::DuplicateReply.into()),
251            303 => Some(MessageError::DuplicateWaking.into()),
252            304 => Some(MessageError::LateAccess.into()),
253            305 => Some(MessageError::OutOfBounds.into()),
254            306 => Some(MessageError::DuplicateInit.into()),
255            307 => Some(MessageError::InsufficientValue.into()),
256            308 => Some(MessageError::InsufficientGasLimit.into()),
257            309 => Some(MessageError::DuplicateReplyDeposit.into()),
258            310 => Some(MessageError::IncorrectMessageForReplyDeposit.into()),
259            311 => Some(MessageError::OutgoingMessagesBytesLimitExceeded.into()),
260            312 => Some(MessageError::OutOfBoundsInputSliceOffset.into()),
261            313 => Some(MessageError::OutOfBoundsInputSliceLength.into()),
262            399 => Some(MessageError::InsufficientGasForDelayedSending.into()),
263            //
264            500 => Some(ReservationError::InvalidReservationId.into()),
265            501 => Some(ReservationError::ReservationsLimitReached.into()),
266            502 => Some(ReservationError::ZeroReservationDuration.into()),
267            503 => Some(ReservationError::ZeroReservationAmount.into()),
268            504 => Some(ReservationError::ReservationBelowMailboxThreshold.into()),
269            //
270            0xffff /* SyscallUsage */ |
271            600 /* ProgramRent(ProgramRentError::MaximumBlockCountPaid) */ |
272            u32::MAX => Some(ExtError::Unsupported),
273            _ => None,
274        }
275    }
276}

278#[cfg(feature = "codec")]
279impl Encode for ExtError {
280    fn encode(&self) -> Vec<u8> {
281        ExtError::to_u32(*self).to_le_bytes().to_vec()
282    }
283}
284
285#[cfg(feature = "codec")]
286impl Decode for ExtError {
287    fn decode<I: Input>(input: &mut I) -> Result<Self, Error> {
288        let mut code = [0; 4];
289        input.read(&mut code)?;
290        let err =
291            ExtError::from_u32(u32::from_le_bytes(code)).ok_or("Failed to decode error code")?;
292        Ok(err)
293    }
294}

296#[cfg(test)]
297mod tests {
298    use super::*;
299    use alloc::collections::BTreeMap;
300
301    #[test]
302    fn error_codes_are_unique() {
303        let mut codes = BTreeMap::new();
304
305        for err in enum_iterator::all::<ExtError>() {
306            let code = err.to_u32();
307            if let Some(same_code_err) = codes.insert(code, err) {
308                panic!("{:?} has same code {:?} as {:?}", same_code_err, code, err);
309            }
310        }
311    }
312
313    #[test]
314    fn encode_decode() {
315        for err in enum_iterator::all::<ExtError>() {
316            let code = err.to_u32();
317            let decoded = ExtError::from_u32(code)
318                .unwrap_or_else(|| unreachable!("failed to decode error code: {}", code));
319            assert_eq!(err, decoded);
320        }
321    }
322
323    #[test]
324    fn error_code_no_specific_value() {
325        for err in enum_iterator::all::<ExtError>() {
326            let code = err.to_u32();
327            assert_ne!(code, 0); // success code
328        }
329    }
330
331    /// check forbidden error codes
332    ///
333    /// forbidden codes either:
334    /// 1. never actually used
335    /// 2. deprecated
336    ///
337    /// codes are forbidden to avoid collision in
338    /// old programs that built their logic on these error codes
339    /// if we accidentally re-use such codes
340    #[test]
341    fn error_codes_forbidden() {
342        let codes = [
343            0xffff, /* SyscallUsage */
344            600,    /* ProgramRent(ProgramRentError::MaximumBlockCountPaid) */
345        ];
346
347        // check forbidden code is `Unsupported` variant now
348        for code in codes {
349            let err = ExtError::from_u32(code);
350            assert_eq!(err, Some(ExtError::Unsupported));
351        }
352
353        // check forbidden code is never produced
354        for err in enum_iterator::all::<ExtError>() {
355            let code = err.to_u32();
356            assert!(!codes.contains(&code));
357        }
358    }
