19//! # Gear Gas Pallet
20//!
21//! The Gear Gas Pallet provides functionality for handling messages'
22//! execution resources.
23//!
24//! - [`Config`]
25//! - [`Pallet`]
26//!
27//! ## Overview
28//!
29//! The Gear Gas Pallet's main aim is to separate message's associated gas tree nodes storages out
30//!    of Gear's execution logic and provide soft functionality to manage them.
31//!
32//! The Gear Gas Pallet provides functions for:
33//! - Obtaining maximum gas amount available within one block of execution.
34//! - Managing number of remaining gas, i.e. gas allowance.
35//! - Managing gas tree: create, split, cut, etc new nodes determining
36//!   execution resources of messages.
37//!
38//! ## Interface
39//!
40//! The Gear Gas Pallet implements `gear_common::GasProvider` trait
41//! and shouldn't contain any other functionality, except this trait declares.
42//!
43//! ## Usage
44//!
45//! How to use the gas functionality from the Gear Gas Pallet:
46//!
47//! 1. Implement its `Config` for your runtime with specified `BlockGasLimit` type.
48//!
49//! ```ignore
50//! // `runtime/src/lib.rs`
51//! // ... //
52//!
53//! impl pallet_gear_gas::Config for Runtime {
54//!     type BlockGasLimit = .. ;
55//! }
56//!
57//! // ... //
58//! ```
59//!
60//! 2. Provide associated type for your pallet's `Config`, which implements
61//!    `gear_common::GasProvider` trait, specifying associated types if needed.
62//!
63//! ```ignore
64//! // `some_pallet/src/lib.rs`
65//! // ... //
66//!
67//! use gear_common::GasProvider;
68//!
69//! #[pallet::config]
70//! pub trait Config: frame_system::Config {
71//!     // .. //
72//!
73//!     type GasProvider: GasProvider<Balance = u64, ...>;
74//!
75//!     // .. //
76//! }
77//! ```
78//!
79//! 3. Declare Gear Gas Pallet in your `construct_runtime!` macro.
80//!
81//! ```ignore
82//! // `runtime/src/lib.rs`
83//! // ... //
84//!
85//! construct_runtime!(
86//!     pub enum Runtime
87//!         where // ... //
88//!     {
89//!         // ... //
90//!
91//!         GearGas: pallet_gear_gas,
92//!
93//!         // ... //
94//!     }
95//! );
96//!
97//! // ... //
98//! ```
99//! `GearGas: pallet_gear_gas,`
100//!
101//! 4. Set `GearGas` as your pallet `Config`'s `GasProvider` type.
102//!
103//! ```ignore
104//! // `runtime/src/lib.rs`
105//! // ... //
106//!
107//! impl some_pallet::Config for Runtime {
108//!     // ... //
109//!
110//!     type GasProvider = GearGas;
111//!
112//!     // ... //
113//! }
114//!
115//! // ... //
116//! ```
117//!
118//! 5. Work with Gear Gas Pallet in your pallet with provided
119//!    associated type interface.
120//!
121//! ## Genesis config
122//!
123//! The Gear Gas Pallet doesn't depend on the `GenesisConfig`.
124

180    // Gas pallet error.
181    #[pallet::error]
182    pub enum Error<T> {
183        Forbidden,
184        NodeAlreadyExists,
185        InsufficientBalance,
186        NodeNotFound,
187        NodeWasConsumed,
188
189        /// Errors stating that gas tree has been invalidated
190        ParentIsLost,
191        ParentHasNoChildren,
192
193        /// Output of `Tree::consume` procedure that wasn't expected.
194        ///
195        /// Outputs of consumption procedure are determined. The error is returned
196        /// when unexpected one occurred. That signals, that algorithm works wrong
197        /// and expected invariants are not correct.
198        UnexpectedConsumeOutput,
199
200        /// Node type that can't occur if algorithm work well
201        UnexpectedNodeType,
202
203        /// Value must have been caught, but was missed or blocked (for more info see `ValueNode::catch_value`).
204        ValueIsNotCaught,
205
206        /// Value must have been caught or moved upstream, but was blocked (for more info see `ValueNode::catch_value`).
207        ValueIsBlocked,
208
209        /// Value must have been blocked, but was either moved or caught (for more info see `ValueNode::catch_value`).
210        ValueIsNotBlocked,
211
212        /// `GasTree::consume` called on node, which has some balance locked.
213        ConsumedWithLock,
214
215        /// `GasTree::consume` called on node, which has some system reservation.
216        ConsumedWithSystemReservation,
217
218        /// `GasTree::create` called with some value amount leading to
219        /// the total value overflow.
220        TotalValueIsOverflowed,
221
222        /// Either `GasTree::consume` or `GasTree::spent` called on a node creating
223        /// negative imbalance which leads to the total value drop below 0.
224        TotalValueIsUnderflowed,
225    }

332    pub struct GasAllowance<T: Config>(PhantomData<T>);
333
334    impl<T: Config> common::storage::Limiter for GasAllowance<T> {
335        type Value = Balance;
336
337        fn get() -> Self::Value {
338            Allowance::<T>::get()
339        }
340
341        fn put(gas: Self::Value) {
342            Allowance::<T>::put(gas);
343        }
344
345        fn decrease(gas: Self::Value) {
346            Allowance::<T>::mutate(|v| *v = v.saturating_sub(gas));
347        }
348    }

350    impl<T: Config> GasProvider for Pallet<T> {
351        type ExternalOrigin = AccountIdOf<T>;
352        type NodeId = GasNodeId<MessageId, ReservationId>;
353        type Balance = Balance;
354        type Funds = Funds;
355        type InternalError = Error<T>;
356        type Error = DispatchError;
357
358        type GasTree = TreeImpl<
359            TotalIssuanceWrap<T>,
360            Self::InternalError,
361            Self::Error,
362            Self::ExternalOrigin,
363            Self::NodeId,
364            GasNodesWrap<T>,
365        >;
366    }

371        type Balance = Balance;
372
373        type GasAllowance = GasAllowance<T>;
374    }

376    #[pallet::hooks]
377    impl<T: Config> Hooks<BlockNumberFor<T>> for Pallet<T> {
378        /// Initialization
379        fn on_initialize(_bn: BlockNumberFor<T>) -> Weight {
380            // Reset block gas allowance
381            Allowance::<T>::put(BlockGasLimitOf::<T>::get());
382
383            T::DbWeight::get().writes(1)
384        }
385
386        /// Finalization
387        fn on_finalize(_bn: BlockNumberFor<T>) {}
388    }
