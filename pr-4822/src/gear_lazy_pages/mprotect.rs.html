19//! Wrappers around system memory protections.
20
21use crate::pages::{Page, PagesAmount, PagesAmountTrait, SizeManager, SizeNumber};
22use core::ops::RangeInclusive;
23use numerated::interval::Interval;
24use std::fmt::Debug;
25
26#[derive(Debug, derive_more::Display)]
27pub enum MprotectError {
28    #[display(
29        "Syscall mprotect error for interval {interval:#x?}, mask = {mask}, reason: {reason}"
30    )]
31    SyscallError {
32        interval: RangeInclusive<usize>,
33        mask: region::Protection,
34        reason: region::Error,
35    },
36    #[display("Interval size or page address overflow")]
37    Overflow,
38    #[display("Zero size is restricted for mprotect")]
39    ZeroSizeError,
40}
41
42/// Mprotect native memory interval [`addr`, `addr` + `size`].
43/// Protection mask is set according to protection arguments.
44unsafe fn sys_mprotect_interval(
45    addr: usize,
46    size: usize,
47    allow_read: bool,
48    allow_write: bool,
49    allow_exec: bool,
50) -> Result<(), MprotectError> {
51    if size == 0 {
52        return Err(MprotectError::ZeroSizeError);
53    }
54
55    let mut mask = region::Protection::NONE;
56    if allow_read {
57        mask |= region::Protection::READ;
58    }
59    if allow_write {
60        mask |= region::Protection::WRITE;
61    }
62    if allow_exec {
63        mask |= region::Protection::EXECUTE;
64    }
65    let res = unsafe { region::protect(addr as *mut (), size, mask) };
66    if let Err(reason) = res {
67        return Err(MprotectError::SyscallError {
68            interval: addr..=addr + size,
69            mask,
70            reason,
71        });
72    }
73    log::trace!("mprotect interval: {addr:#x}, size: {size:#x}, mask: {mask}");
74    Ok(())
75}
76
77/// Mprotect native memory interval [`addr`, `addr` + `size`].
78/// Protection mask is set according to protection arguments, `prot_exec` is set as false always.
79pub(crate) fn mprotect_interval(
80    addr: usize,
81    size: usize,
82    allow_read: bool,
83    allow_write: bool,
84) -> Result<(), MprotectError> {
85    unsafe { sys_mprotect_interval(addr, size, allow_read, allow_write, false) }
86}
87
88/// Mprotect all pages from `pages`.
89pub(crate) fn mprotect_pages<M: SizeManager, S: SizeNumber, I: Into<Interval<Page<S>>>>(
90    mem_addr: usize,
91    pages: impl Iterator<Item = I>,
92    size_ctx: &M,
93    allow_read: bool,
94    allow_write: bool,
95) -> Result<(), MprotectError> {
96    for interval in pages {
97        let interval: Interval<Page<S>> = interval.into();
98
99        let start = interval.start();
100
101        let addr = mem_addr
102            .checked_add(start.offset(size_ctx) as usize)
103            .ok_or(MprotectError::Overflow)?;
104
105        let size = interval
106            .raw_len()
107            .and_then(|raw| PagesAmount::<S>::new(size_ctx, raw))
108            .ok_or(MprotectError::Overflow)?
109            .offset(size_ctx);
110
111        unsafe {
112            sys_mprotect_interval(addr, size, allow_read, allow_write, false)?;
113        }
114    }
115    Ok(())
116}
