19//! `GlobalsAccessor` realizations for native and wasm runtimes.
20
21use crate::common::{Error, GlobalNames};
22use core::any::Any;
23use gear_core::{memory::HostPointer, str::LimitedStr};
24use gear_lazy_pages_common::{GlobalsAccessError, GlobalsAccessMod, GlobalsAccessor};
25use gear_sandbox_host::sandbox::SandboxInstance;
26use sp_wasm_interface_common::Value;
27
28#[derive(Debug, Clone, Copy)]
29pub(crate) enum GlobalNo {
30    Gas = 0,
31}
32
33#[derive(Debug)]
34pub(crate) struct GlobalsContext {
35    /// Global names
36    pub names: GlobalNames,
37    /// Raw pointer to the globals access provider
38    pub access_ptr: HostPointer,
39    /// Access mod, currently two: native or WASM runtime
40    pub access_mod: GlobalsAccessMod,
41}
42
43struct GlobalsAccessWasmRuntime<'a> {
44    pub instance: &'a mut SandboxInstance,
45}
46
47impl GlobalsAccessor for GlobalsAccessWasmRuntime<'_> {
48    fn get_i64(&mut self, name: &LimitedStr) -> Result<i64, GlobalsAccessError> {
49        // SAFETY: this is safe because this method is called only from signal handler context
50        unsafe {
51            self.instance
52                .signal_handler_get_global_val(name.as_str())
53                .and_then(|value| match value {
54                    Value::I64(value) => Some(value),
55                    _ => None,
56                })
57                .ok_or(GlobalsAccessError)
58        }
59    }
60
61    fn set_i64(&mut self, name: &LimitedStr, value: i64) -> Result<(), GlobalsAccessError> {
62        // SAFETY: this is safe because this method is called only from signal handler context
63        unsafe {
64            self.instance
65                .signal_handler_set_global_val(name.as_str(), Value::I64(value))
66                .ok()
67                .flatten()
68                .ok_or(GlobalsAccessError)?
69        }
70
71        Ok(())
72    }
73
74    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
75        unimplemented!("Has no use cases for this struct")
76    }
77}
78
79struct GlobalsAccessNativeRuntime<'a, 'b> {
80    pub inner_access_provider: &'a mut &'b mut dyn GlobalsAccessor,
81}
82
83impl GlobalsAccessor for GlobalsAccessNativeRuntime<'_, '_> {
84    fn get_i64(&mut self, name: &LimitedStr) -> Result<i64, GlobalsAccessError> {
85        self.inner_access_provider.get_i64(name)
86    }
87
88    fn set_i64(&mut self, name: &LimitedStr, value: i64) -> Result<(), GlobalsAccessError> {
89        self.inner_access_provider.set_i64(name, value)
90    }
91
92    fn as_any_mut(&mut self) -> &mut dyn Any {
93        unimplemented!("Has no use cases for this struct")
94    }
95}
96
97fn apply_for_global_internal(
98    mut globals_access_provider: impl GlobalsAccessor,
99    name: &str,
100    mut f: impl FnMut(u64) -> Result<Option<u64>, Error>,
101) -> Result<u64, Error> {
102    let name = LimitedStr::try_from(name).map_err(|_| Error::AccessGlobal(GlobalsAccessError))?;
103
104    let current_value = globals_access_provider.get_i64(&name)? as u64;
105    if let Some(new_value) = f(current_value)? {
106        globals_access_provider.set_i64(&name, new_value as i64)?;
107        Ok(new_value)
108    } else {
109        Ok(current_value)
110    }
111}
112
113pub(crate) unsafe fn apply_for_global(
114    globals_ctx: &GlobalsContext,
115    global_name: &str,
116    f: impl FnMut(u64) -> Result<Option<u64>, Error>,
117) -> Result<u64, Error> {
118    match globals_ctx.access_mod {
119        GlobalsAccessMod::WasmRuntime => {
120            let instance = unsafe { (globals_ctx.access_ptr as *mut SandboxInstance).as_mut() }
121                .ok_or(Error::HostInstancePointerIsInvalid)?;
122            apply_for_global_internal(GlobalsAccessWasmRuntime { instance }, global_name, f)
123        }
124        GlobalsAccessMod::NativeRuntime => {
125            let inner_access_provider =
126                unsafe { (globals_ctx.access_ptr as *mut &mut dyn GlobalsAccessor).as_mut() }
127                    .ok_or(Error::DynGlobalsAccessPointerIsInvalid)?;
128            apply_for_global_internal(
129                GlobalsAccessNativeRuntime {
130                    inner_access_provider,
131                },
132                global_name,
133                f,
134            )
135        }
136    }
