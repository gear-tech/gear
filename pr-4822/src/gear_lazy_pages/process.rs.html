29/// `process_lazy_pages` use struct which implements this trait,
30/// to process in custom logic two cases: host function call and signal.
31pub(crate) trait AccessHandler {
32    type Pages;
33    type Output;
34
35    /// Returns whether it is write access
36    fn is_write(&self) -> bool;
37
38    /// Returns whether gas exceeded status is allowed for current access.
39    fn check_status_is_gas_exceeded() -> Result<(), Error>;
40
41    /// Returns whether stack memory access can appear for the case.
42    fn check_stack_memory_access() -> Result<(), Error>;
43
44    /// Returns whether write accessed memory access is allowed for the case.
45    fn check_write_accessed_memory_access() -> Result<(), Error>;
46
47    /// Returns whether already accessed memory read access is allowed for the case.
48    fn check_read_from_accessed_memory() -> Result<(), Error>;
49
50    /// Charge for accessed gear page.
51    fn charge_for_page_access(
52        &mut self,
53        page: GearPage,
54        is_already_accessed: bool,
55    ) -> Result<Status, Error>;
56
57    /// Charge for one gear page data loading.
58    fn charge_for_page_data_loading(&mut self) -> Result<Status, Error>;
59
60    /// Get the biggest page from `pages`.
61    fn last_page(pages: &Self::Pages) -> Option<GearPage>;
62
63    /// Apply `f` for all `pages`.
64    fn process_pages(
65        pages: Self::Pages,
66        process_one: impl FnMut(GearPage) -> Result<(), Error>,
67    ) -> Result<(), Error>;
68
69    /// Drops and returns output.
70    fn into_output(self, ctx: &mut LazyPagesExecutionContext) -> Result<Self::Output, Error>;
71}
72
73/// Lazy-pages accesses processing main function.
74/// Acts differently for signals and host functions accesses,
75/// but main logic is the same:
76/// It removes read and write protections for page,
77/// then it loads wasm page data from storage to wasm page memory location.
78/// If native page size is bigger than gear page size, then this will be done
79/// for all gear pages from accessed native page.
80/// 1) Set new access pages protection accordingly it is read or write access.
81/// 2) If some page contains data in storage, then load this data and place it in
82///    program's wasm memory.
83/// 3) Charge gas for access and data loading.
84pub(crate) fn process_lazy_pages<H: AccessHandler>(
85    rt_ctx: &mut LazyPagesRuntimeContext,
86    exec_ctx: &mut LazyPagesExecutionContext,
87    mut handler: H,
88    pages: H::Pages,
89) -> Result<H::Output, Error> {
90    let wasm_mem_size = exec_ctx.wasm_mem_size.offset(rt_ctx);
91    unsafe {
92        if let Some(last_page) = H::last_page(&pages) {
93            // Check that all pages are inside wasm memory.
94            if last_page.end_offset(rt_ctx) as usize >= wasm_mem_size {
95                return Err(Error::OutOfWasmMemoryAccess);
96            }
97        } else {
98            // Accessed pages are empty - nothing to do.
99            return handler.into_output(exec_ctx);
100        }
101
102        if exec_ctx.status != Status::Normal {
103            H::check_status_is_gas_exceeded()?;
104            return handler.into_output(exec_ctx);
105        }
106
107        let stack_end = exec_ctx.stack_end;
108        let wasm_mem_addr = exec_ctx.wasm_mem_addr.ok_or(Error::WasmMemAddrIsNotSet)?;
109
110        // Returns `true` if new status is not `Normal`.
111        let update_status = |ctx: &mut LazyPagesExecutionContext, status| {
112            ctx.status = status;
113
114            // If new status is not [Status::Normal], then unprotect lazy-pages
115            // and continue work until the end of current wasm block. We don't care
116            // about future program execution correctness, because gas limit or allowance exceed.
117            match status {
118                Status::Normal => Ok(false),
119                Status::GasLimitExceeded => {
120                    log::trace!(
121                        "Gas limit or allowance exceed, so removes protection from all wasm memory \
122                        and continues execution until the end of current wasm block"
123                    );
124                    mprotect::mprotect_interval(wasm_mem_addr, wasm_mem_size, true, true)
125                        .map(|_| true)
126                }
127            }
128        };
129
130        let page_size = GearPage::size(rt_ctx) as usize;
131
132        let process_one = |page: GearPage| {
133            let page_offset = page.offset(rt_ctx);
134            let page_buffer_ptr = (wasm_mem_addr as *mut u8).add(page_offset as usize);
135
136            let protect_page = |prot_write| {
137                mprotect::mprotect_interval(page_buffer_ptr as usize, page_size, true, prot_write)
138            };
139
140            if page_offset < stack_end.offset(rt_ctx) {
141                // Nothing to do, page has r/w accesses and data is in correct state.
142                H::check_stack_memory_access()?;
143            } else if exec_ctx.is_write_accessed(page) {
144                // Nothing to do, page has r/w accesses and data is in correct state.
145                H::check_write_accessed_memory_access()?;
146            } else if exec_ctx.is_accessed(page) {
147                if handler.is_write() {
148                    // Charges for page write access
149                    let status = handler.charge_for_page_access(page, true)?;
150                    if update_status(exec_ctx, status)? {
151                        return Ok(());
152                    }
153
154                    // Sets read/write protection access for page and add page to write accessed
155                    protect_page(true)?;
156                    exec_ctx.set_write_accessed(page)?;
157                } else {
158                    // Nothing to do, page has read accesses and data is in correct state.
159                    H::check_read_from_accessed_memory()?;
160                }
161            } else {
162                // Charge for page access.
163                let status = handler.charge_for_page_access(page, false)?;
164                if update_status(exec_ctx, status)? {
165                    return Ok(());
166                }
167
168                let unprotected = if rt_ctx
169                    .page_has_data_in_storage(&mut exec_ctx.program_storage_prefix, page)
170                {
171                    // Charge for page data loading from storage.
172                    let status = handler.charge_for_page_data_loading()?;
173                    if update_status(exec_ctx, status)? {
174                        return Ok(());
175                    }
176
177                    // Set read/write access, in order to write page data to program memory.
178                    protect_page(true)?;
179
180                    // Load and write data to memory.
181                    let buffer_as_slice = slice::from_raw_parts_mut(page_buffer_ptr, page_size);
182                    if !rt_ctx.load_page_data_from_storage(
183                        &mut exec_ctx.program_storage_prefix,
184                        page,
185                        buffer_as_slice,
186                    )? {
187                        let err_msg = format!(
188                            "process_lazy_pages: `read` returns, that page has no data, but `exist` returns that there is one. \
189                            Page - {page:?}, program storage prefix - {:?}",
190                            &exec_ctx.program_storage_prefix,
191                        );
192
193                        log::error!("{err_msg}");
194                        unreachable!("{err_msg}")
195                    }
196                    true
197                } else {
198                    false
199                };
200
201                exec_ctx.set_accessed(page);
202                if handler.is_write() {
203                    if !unprotected {
204                        protect_page(true)?;
205                    }
206                    exec_ctx.set_write_accessed(page)?;
207                } else {
208                    // Set only read access for page.
209                    protect_page(false)?;
210                }
211            }
212
213            Ok(())
214        };
215
216        H::process_pages(pages, process_one)?;
217
218        handler.into_output(exec_ctx)
219    }
