31pub trait UserSignalHandler {
32    /// # Safety
33    ///
34    /// It's expected handler calls syscalls to protect memory
35    unsafe fn handle(info: ExceptionInfo) -> Result<(), Error>;
36}

38pub(crate) struct DefaultUserSignalHandler;
39
40impl UserSignalHandler for DefaultUserSignalHandler {
41    unsafe fn handle(info: ExceptionInfo) -> Result<(), Error> {
42        unsafe { user_signal_handler(info) }
43    }
44}

46#[derive(Debug)]
47pub struct ExceptionInfo {
48    /// Address where fault is occurred
49    pub fault_addr: *const (),
50    pub is_write: Option<bool>,
51}
52
53/// Before program execution some pages from wasm memory buffer have been protected.
54/// When wasm executer tries to access one of these pages,
55/// OS emits sigsegv or sigbus or EXCEPTION_ACCESS_VIOLATION.
56/// Using OS signal info, this function identifies memory page, which is accessed,
57/// and process access for this page. See more in `process::process_lazy_pages`.
58/// After processing is done, OS returns execution to the same machine
59/// instruction, which cause signal. Now memory which this instruction accesses
60/// is not protected and with correct data.
61unsafe fn user_signal_handler_internal(
62    rt_ctx: &mut LazyPagesRuntimeContext,
63    exec_ctx: &mut LazyPagesExecutionContext,
64    info: ExceptionInfo,
65) -> Result<(), Error> {
66    let native_addr = info.fault_addr as usize;
67    let is_write = info.is_write.ok_or(Error::ReadOrWriteIsUnknown)?;
68    let wasm_mem_addr = exec_ctx.wasm_mem_addr.ok_or(Error::WasmMemAddrIsNotSet)?;
69
70    if native_addr < wasm_mem_addr {
71        return Err(Error::OutOfWasmMemoryAccess);
72    }
73
74    let offset =
75        u32::try_from(native_addr - wasm_mem_addr).map_err(|_| Error::OutOfWasmMemoryAccess)?;
76    let page = GearPage::from_offset(rt_ctx, offset);
77
78    let gas_ctx = if let Some(globals_config) = exec_ctx.globals_context.as_ref() {
79        let gas_charger = GasCharger {
80            read_cost: exec_ctx.cost(CostNo::SignalRead),
81            write_cost: exec_ctx.cost(CostNo::SignalWrite),
82            write_after_read_cost: exec_ctx.cost(CostNo::SignalWriteAfterRead),
83            load_data_cost: exec_ctx.cost(CostNo::LoadPageDataFromStorage),
84        };
85
86        let gas_counter = unsafe {
87            globals::apply_for_global(
88                globals_config,
89                globals_config.names[GlobalNo::Gas as usize].as_str(),
90                |_| Ok(None),
91            )
92        }?;
93
94        Some((gas_counter, gas_charger))
95    } else {
96        None
97    };
98
99    let handler = SignalAccessHandler { is_write, gas_ctx };
100    process::process_lazy_pages(rt_ctx, exec_ctx, handler, page)
101}
102
103/// User signal handler. Logic can depends on lazy-pages version.
104/// See also "user_signal_handler_internal".
105pub(crate) unsafe fn user_signal_handler(info: ExceptionInfo) -> Result<(), Error> {
106    log::debug!("Interrupted, exception info = {info:?}");
107    LAZY_PAGES_CONTEXT.with(|ctx| {
108        let mut ctx = ctx.borrow_mut();
109        let (rt_ctx, exec_ctx) = ctx.contexts_mut()?;
110        unsafe { user_signal_handler_internal(rt_ctx, exec_ctx, info) }
111    })
112}
113
114struct SignalAccessHandler {
115    is_write: bool,
116    gas_ctx: Option<(u64, GasCharger)>,
117}
118
119impl AccessHandler for SignalAccessHandler {
120    type Pages = GearPage;
121    type Output = ();
122
123    fn is_write(&self) -> bool {
124        self.is_write
125    }
126
127    fn last_page(page: &Self::Pages) -> Option<GearPage> {
128        Some(*page)
129    }
130
131    fn check_status_is_gas_exceeded() -> Result<(), Error> {
132        // Because we unprotect all lazy-pages when status is `exceeded`, then
133        // we cannot receive signals from wasm memory until the end of execution.
134        Err(Error::SignalWhenStatusGasExceeded)
135    }
136
137    fn check_stack_memory_access() -> Result<(), Error> {
138        // Stack memory is always unprotected, so we cannot receive signal from it.
139        Err(Error::SignalFromStackMemory)
140    }
141
142    fn check_write_accessed_memory_access() -> Result<(), Error> {
143        // Write accessed memory is unprotected, so signal cannot be received from it.
144        Err(Error::SignalFromWriteAccessedPage)
145    }
146
147    fn check_read_from_accessed_memory() -> Result<(), Error> {
148        // Accessed memory is not read protected,
149        // so read memory access signal cannot be received from it.
150        Err(Error::ReadAccessSignalFromAccessedPage)
151    }
152
153    fn charge_for_page_access(
154        &mut self,
155        page: GearPage,
156        is_accessed: bool,
157    ) -> Result<Status, Error> {
158        let (gas_counter, gas_charger) = match self.gas_ctx.as_mut() {
159            Some(ctx) => ctx,
160            None => return Ok(Status::Normal),
161        };
162        gas_charger.charge_for_page_access(gas_counter, page, self.is_write, is_accessed)
163    }
164
165    fn charge_for_page_data_loading(&mut self) -> Result<Status, Error> {
166        let (gas_counter, gas_charger) = match self.gas_ctx.as_mut() {
167            Some(ctx) => ctx,
168            None => return Ok(Status::Normal),
169        };
170        Ok(gas_charger.charge_for_page_data_load(gas_counter))
171    }
172
173    fn process_pages(
174        page: GearPage,
175        mut process_one: impl FnMut(GearPage) -> Result<(), Error>,
176    ) -> Result<(), Error> {
177        process_one(page)
178    }
179
180    fn into_output(self, ctx: &mut LazyPagesExecutionContext) -> Result<Self::Output, Error> {
181        if let (Some((gas_counter, _)), Some(globals_config)) =
182            (self.gas_ctx, ctx.globals_context.as_ref())
183        {
184            let mut diff = 0;
185            unsafe {
186                globals::apply_for_global(
187                    globals_config,
188                    globals_config.names[GlobalNo::Gas as usize].as_str(),
189                    |current| {
190                        diff = current - gas_counter;
191                        Ok(Some(gas_counter))
192                    },
193                )?
194            };
195
196            // support old runtimes
197            if globals_config.names.len() == 2
198                && globals_config.names[1].as_str() == "gear_allowance"
199            {
200                unsafe {
201                    globals::apply_for_global(globals_config, "gear_allowance", |current| {
202                        Ok(Some(current.saturating_sub(diff)))
203                    })?
204                };
205            }
206        }
207        Ok(())
208    }
209}
