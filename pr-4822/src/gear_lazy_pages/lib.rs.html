19//! Lazy-pages support.
20//! In runtime data for program Wasm memory pages can be loaded in lazy manner.
21//! All pages, which is supposed to be lazy, must be mprotected before program execution.
22//! During execution data from storage is loaded for all pages, which has been accessed
23//! and which has data in storage.
24//! See also `process::process_lazy_pages`, `signal`, `host_func` for more information.
25//!
26//! Note: currently we restrict twice write signal from same page during one execution.
27//! It's not necessary behavior, but more simple and safe.

77#[derive(Debug, derive_more::Display, derive_more::From)]
78pub enum Error {
79    #[display("WASM memory native address {_0:#x} is not aligned to the native page size")]
80    WasmMemAddrIsNotAligned(usize),
81    Mprotect(MprotectError),
82    #[from(skip)]
83    #[display("Wasm memory end addr is out of usize: begin addr = {_0:#x}, size = {_1:#x}")]
84    WasmMemoryEndAddrOverflow(usize, usize),
85    #[display("Prefix of storage with memory pages was not set")]
86    MemoryPagesPrefixNotSet,
87    #[display("Memory size must be null when memory host addr is not set")]
88    MemorySizeIsNotNull,
89    #[display("Wasm mem size is too big")]
90    WasmMemSizeOverflow,
91    #[display("Stack end offset cannot be bigger than memory size")]
92    StackEndBiggerThanMemSize,
93    #[display("Stack end offset is too big")]
94    StackEndOverflow,
95    #[display("Wasm addr and size are not changed, so host func call is needless")]
96    NothingToChange,
97    #[display("Wasm memory addr must be set, when trying to change something in lazy pages")]
98    WasmMemAddrIsNotSet,
99    GlobalContext(ContextError),
100    #[from(skip)]
101    #[display("Wrong costs amount: get {_0}, must be {_1}")]
102    WrongCostsAmount(usize, usize),
103}
104
105fn check_memory_interval(addr: usize, size: usize) -> Result<(), Error> {
106    addr.checked_add(size)
107        .ok_or(Error::WasmMemoryEndAddrOverflow(addr, size))
108        .map(|_| ())
109}

111pub fn initialize_for_program(
112    wasm_mem_addr: Option<usize>,
113    wasm_mem_size: u32,
114    stack_end: Option<u32>,
115    program_key: Vec<u8>,
116    globals_config: Option<GlobalsAccessConfig>,
117    costs: Vec<u64>,
118) -> Result<(), Error> {
119    // Initialize new execution context
120    LAZY_PAGES_CONTEXT.with(|ctx| {
121        let mut ctx = ctx.borrow_mut();
122        let runtime_ctx = ctx.runtime_context_mut()?;
123
124        // Check wasm program memory host address
125        if let Some(addr) = wasm_mem_addr
126            && !addr.is_multiple_of(region::page::size())
127        {
128            return Err(Error::WasmMemAddrIsNotAligned(addr));
129        }
130
131        // Check stack_end is less or equal than wasm memory size
132        let stack_end = stack_end.unwrap_or_default();
133        if wasm_mem_size < stack_end {
134            return Err(Error::StackEndBiggerThanMemSize);
135        }
136
137        let wasm_mem_size =
138            WasmPagesAmount::new(runtime_ctx, wasm_mem_size).ok_or(Error::WasmMemSizeOverflow)?;
139        let wasm_mem_size_in_bytes = wasm_mem_size.offset(runtime_ctx);
140
141        // Check wasm program memory size
142        if let Some(addr) = wasm_mem_addr {
143            check_memory_interval(addr, wasm_mem_size_in_bytes)?;
144        } else if wasm_mem_size_in_bytes != 0 {
145            return Err(Error::MemorySizeIsNotNull);
146        }
147
148        let stack_end = WasmPage::new(runtime_ctx, stack_end).ok_or(Error::StackEndOverflow)?;
149
150        let costs: Costs = costs.try_into().map_err(|costs: Vec<u64>| {
151            Error::WrongCostsAmount(costs.len(), CostNo::Amount as usize)
152        })?;
153
154        let execution_ctx = LazyPagesExecutionContext {
155            costs,
156            wasm_mem_addr,
157            wasm_mem_size,
158            program_storage_prefix: PagePrefix::new_from_program_prefix(
159                [runtime_ctx.pages_storage_prefix.as_slice(), &program_key].concat(),
160            ),
161            accessed_pages: Default::default(),
162            write_accessed_pages: Default::default(),
163            stack_end,
164            globals_context: globals_config.map(|cfg| GlobalsContext {
165                names: runtime_ctx.global_names.clone(),
166                access_ptr: cfg.access_ptr,
167                access_mod: cfg.access_mod,
168            }),
169            status: Status::Normal,
170        };
171
172        // Set protection if wasm memory exist.
173        if let Some(addr) = wasm_mem_addr {
174            let stack_end_offset = execution_ctx.stack_end.offset(runtime_ctx) as usize;
175            // `+` and `-` are safe because we checked
176            // that `stack_end` is less or equal to `wasm_mem_size` and wasm end addr fits usize.
177            let addr = addr + stack_end_offset;
178            let size = wasm_mem_size_in_bytes - stack_end_offset;
179            if size != 0 {
180                mprotect::mprotect_interval(addr, size, false, false)?;
181            }
182        }
183
184        ctx.set_execution_context(execution_ctx);
185
186        log::trace!("Initialize lazy-pages for current program: {ctx:?}");
187
188        Ok(())
189    })
190}

192/// Protect lazy pages, after they had been unprotected.
193pub fn set_lazy_pages_protection() -> Result<(), Error> {
194    LAZY_PAGES_CONTEXT.with(|ctx| {
195        let ctx = ctx.borrow();
196        let (rt_ctx, exec_ctx) = ctx.contexts()?;
197        let mem_addr = exec_ctx.wasm_mem_addr.ok_or(Error::WasmMemAddrIsNotSet)?;
198
199        // Set r/w protection for all pages except stack pages and write accessed pages.
200        let start: GearPage = exec_ctx.stack_end.to_page(rt_ctx);
201        let end: GearPagesAmount = exec_ctx.wasm_mem_size.convert(rt_ctx);
202        let interval = start.to_end_interval(rt_ctx, end).unwrap_or_else(|| {
203            let err_msg = format!(
204                "set_lazy_pages_protection: `stack_end` must be less or equal to `wasm_mem_size`. \
205                Stack end start - {start:?}, wasm memory size - {end:?}",
206            );
207
208            log::error!("{err_msg}");
209            unreachable!("{err_msg}")
210        });
211        let pages = exec_ctx.write_accessed_pages.voids(interval);
212        mprotect::mprotect_pages(mem_addr, pages, rt_ctx, false, false)?;
213
214        // Set only write protection for already accessed, but not write accessed pages.
215        let pages = exec_ctx
216            .accessed_pages
217            .difference(&exec_ctx.write_accessed_pages);
218        mprotect::mprotect_pages(mem_addr, pages, rt_ctx, true, false)?;
219
220        // After that protections are:
221        // 1) Only execution protection for stack pages.
222        // 2) Only execution protection for write accessed pages.
223        // 3) Read and execution protection for accessed, but not write accessed pages.
224        // 4) r/w/e protections for all other WASM memory.
225
226        Ok(())
227    })
228}

230/// Unset lazy pages read/write protections.
231pub fn unset_lazy_pages_protection() -> Result<(), Error> {
232    LAZY_PAGES_CONTEXT.with(|ctx| {
233        let ctx = ctx.borrow();
234        let (rt_ctx, exec_ctx) = ctx.contexts()?;
235        let addr = exec_ctx.wasm_mem_addr.ok_or(Error::WasmMemAddrIsNotSet)?;
236        let size = exec_ctx.wasm_mem_size.offset(rt_ctx);
237        mprotect::mprotect_interval(addr, size, true, true)?;
238        Ok(())
239    })
240}

242/// Set current wasm memory begin addr in global context
243pub fn change_wasm_mem_addr_and_size(addr: Option<usize>, size: Option<u32>) -> Result<(), Error> {
244    if matches!((addr, size), (None, None)) {
245        return Err(Error::NothingToChange);
246    }
247
248    LAZY_PAGES_CONTEXT.with(|ctx| {
249        let mut ctx = ctx.borrow_mut();
250        let (rt_ctx, exec_ctx) = ctx.contexts_mut()?;
251
252        let addr = match addr {
253            Some(addr) => match addr % region::page::size() {
254                0 => addr,
255                _ => return Err(Error::WasmMemAddrIsNotAligned(addr)),
256            },
257
258            None => match exec_ctx.wasm_mem_addr {
259                Some(addr) => addr,
260                None => return Err(Error::WasmMemAddrIsNotSet),
261            },
262        };
263
264        let size = match size {
265            Some(raw) => WasmPagesAmount::new(rt_ctx, raw).ok_or(Error::WasmMemSizeOverflow)?,
266            None => exec_ctx.wasm_mem_size,
267        };
268
269        check_memory_interval(addr, size.offset(rt_ctx))?;
270
271        exec_ctx.wasm_mem_addr = Some(addr);
272        exec_ctx.wasm_mem_size = size;
273
274        Ok(())
275    })
276}

278/// Returns vec of lazy-pages which has been accessed
279pub fn write_accessed_pages() -> Result<Vec<u32>, Error> {
280    LAZY_PAGES_CONTEXT.with(|ctx| {
281        ctx.borrow()
282            .execution_context()
283            .map(|ctx| {
284                ctx.write_accessed_pages
285                    .iter()
286                    .flat_map(IntervalIterator::from)
287                    .map(|p| p.raw())
288                    .collect()
289            })
290            .map_err(Into::into)
291    })
292}

319/// Initialize lazy-pages once for process:
320/// 1) checks whether lazy-pages is supported in current environment
321/// 2) set signals handler
322///
323/// # Safety
324/// See [`sys::setup_signal_handler`]
325unsafe fn init_for_process<H: UserSignalHandler>() -> Result<(), InitError> {
326    use InitError::*;
327
328    #[cfg(target_vendor = "apple")]
329    {
330        // Support debugging under lldb on Darwin.
331        // When SIGBUS appears lldb will stuck on it forever, without this code.
332        // See also: https://github.com/mono/mono/commit/8e75f5a28e6537e56ad70bf870b86e22539c2fb7.
333
334        use mach::{
335            exception_types::*, kern_return::*, mach_types::*, port::*, thread_status::*, traps::*,
336        };
337
338        unsafe extern "C" {
339            // See https://web.mit.edu/darwin/src/modules/xnu/osfmk/man/task_set_exception_ports.html
340            fn task_set_exception_ports(
341                task: task_t,
342                exception_mask: exception_mask_t,
343                new_port: mach_port_t,
344                behavior: exception_behavior_t,
345                new_flavor: thread_state_flavor_t,
346            ) -> kern_return_t;
347        }
348
349        #[cfg(target_arch = "x86_64")]
350        static MACHINE_THREAD_STATE: i32 = x86_THREAD_STATE64;
351
352        // Took const value from https://opensource.apple.com/source/cctools/cctools-870/include/mach/arm/thread_status.h
353        // ```
354        // #define ARM_THREAD_STATE64		6
355        // ```
356        #[cfg(target_arch = "aarch64")]
357        static MACHINE_THREAD_STATE: i32 = 6;
358
359        unsafe {
360            task_set_exception_ports(
361                mach_task_self(),
362                EXC_MASK_BAD_ACCESS,
363                MACH_PORT_NULL,
364                EXCEPTION_STATE_IDENTITY as exception_behavior_t,
365                MACHINE_THREAD_STATE,
366            )
367        };
368    }
369
370    LAZY_PAGES_INITIALIZED.get_or_init(|| {
371        if let Err(err) = unsafe { sys::setup_signal_handler::<H>() } {
372            return Err(CanNotSetUpSignalHandler(err.to_string()));
373        }
374
375        log::trace!("Successfully initialize lazy-pages for process");
376
377        Ok(())
378    })
379}

386/// Initialize lazy-pages for current thread.
387pub fn init_with_handler<H: UserSignalHandler, S: LazyPagesStorage + 'static>(
388    _version: LazyPagesVersion,
389    ctx: LazyPagesInitContext,
390    pages_storage: S,
391) -> Result<(), InitError> {
392    use InitError::*;
393
394    let LazyPagesInitContext {
395        page_sizes,
396        global_names,
397        pages_storage_prefix,
398    } = ctx;
399
400    // Check that sizes are not zero
401    let page_sizes = page_sizes
402        .into_iter()
403        .map(TryInto::<NonZero<u32>>::try_into)
404        .collect::<Result<Vec<_>, _>>()
405        .map_err(|_| ZeroPageSize)?;
406
407    let page_sizes: PageSizes = match page_sizes.try_into() {
408        Ok(sizes) => sizes,
409        Err(sizes) => return Err(WrongSizesAmount(sizes.len(), SIZES_AMOUNT)),
410    };
411
412    // Check sizes suitability
413    let wasm_page_size = page_sizes[WasmSizeNo::SIZE_NO];
414    let gear_page_size = page_sizes[GearSizeNo::SIZE_NO];
415    let native_page_size = region::page::size();
416    if wasm_page_size < gear_page_size
417        || (gear_page_size.get() as usize) < native_page_size
418        || !u32::is_power_of_two(wasm_page_size.get())
419        || !u32::is_power_of_two(gear_page_size.get())
420        || !usize::is_power_of_two(native_page_size)
421    {
422        return Err(NotSuitablePageSizes);
423    }
424
425    // TODO: check globals from context issue #3057
426    // we only need to check the globals that are used to keep the state consistent in older runtimes.
427    if global_names[GlobalNo::Gas as usize].as_str() != "gear_gas" {
428        return Err(WrongGlobalNames(
429            "gear_gas".to_string(),
430            global_names[GlobalNo::Gas as usize].to_string(),
431        ));
432    }
433
434    // Set version even if it has been already set, because it can be changed after runtime upgrade.
435    LAZY_PAGES_CONTEXT.with(|ctx| {
436        ctx.borrow_mut()
437            .set_runtime_context(LazyPagesRuntimeContext {
438                page_sizes,
439                global_names,
440                pages_storage_prefix,
441                program_storage: Box::new(pages_storage),
442            })
443    });
444
445    // TODO: remove after usage of `wasmer::Store::set_trap_handler` for lazy-pages
446    // we capture executor signal handler first to call it later
447    // if our handler is not effective
448    wasmer_vm::init_traps();
449
450    unsafe { init_for_process::<H>()? }
451
452    unsafe { sys::init_for_thread().map_err(InitForThread)? }
453
454    Ok(())
455}
