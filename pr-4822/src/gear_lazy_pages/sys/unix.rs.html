32/// Signal handler which has been set before lazy-pages initialization.
33/// Currently use to support wasmer signal handler.
34/// Wasmer protects memory around wasm memory and for stack limits.
35/// It makes it only in `store` initialization when executor is created,
36/// see https://github.com/gear-tech/substrate/blob/gear-stable/client/executor/common/src/sandbox/wasmer_backend.rs
37/// and https://github.com/wasmerio/wasmer/blob/e6857d116134bdc9ab6a1dabc3544cf8e6aee22b/lib/vm/src/trap/traphandlers.rs#L548
38/// So, if we receive signal from unknown memory we should try to use old (wasmer) signal handler.
39static OLD_SIG_HANDLER: OnceLock<SigHandler> = OnceLock::new();
40
41cfg_if! {
42    if #[cfg(all(target_os = "linux", target_arch = "x86_64"))] {
43        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
44            let error_reg = nix::libc::REG_ERR as usize;
45            let error_code = unsafe { *ucontext }.uc_mcontext.gregs[error_reg];
46            // Use second bit from err reg. See https://git.io/JEQn3
47            Some(error_code & 0b10 == 0b10)
48        }
49    } else if #[cfg(all(target_os = "linux", target_arch = "aarch64"))] {
50        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
51            let esr = linux_aarch64::get_esr(&unsafe { &*ucontext }.uc_mcontext).expect("Failed to get ESR");
52            // Use the WNR bit to determine if it was a write access.
53            // See https://developer.arm.com/documentation/ddi0595/2021-03/AArch64-Registers/ESR-EL1--Exception-Syndrome-Register--EL1-?lang=en#fieldset_0-24_0_15-6_6
54            let is_wnr = (esr & 0b100_0000) != 0;
55            Some(is_wnr)
56        }
57    } else if #[cfg(all(target_os = "macos", target_arch = "x86_64"))] {
58        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
59            // See https://wiki.osdev.org/Exceptions
60            const WRITE_BIT_MASK: u32 = 0b10;
61            const TRAPNO: u16 = 0xe; // Page Fault
62
63            let mcontext = unsafe { *ucontext }.uc_mcontext;
64            let exception_state = unsafe { *mcontext }.__es;
65            let trapno = exception_state.__trapno;
66            let err = exception_state.__err;
67
68            assert_eq!(trapno, TRAPNO);
69
70            Some(err & WRITE_BIT_MASK == WRITE_BIT_MASK)
71        }
72    } else if #[cfg(all(target_os = "macos", target_arch = "aarch64"))] {
73        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
74            // See https://developer.arm.com/documentation/ddi0595/2021-03/AArch64-Registers/ESR-EL1--Exception-Syndrome-Register--EL1-
75            const WNR_BIT_MASK: u32 = 0b100_0000; // Write not Read bit
76            const EXCEPTION_CLASS_SHIFT: u32 = u32::BITS - 6;
77            const EXCEPTION_CLASS: u32 = 0b10_0100; // Data Abort from a lower Exception Level
78
79            let ucontext = unsafe { ucontext.as_mut() }?;
80            let mcontext = ucontext.uc_mcontext;
81            let exception_state = unsafe { *mcontext }.__es;
82            let esr = exception_state.__esr;
83
84            let exception_class = esr >> EXCEPTION_CLASS_SHIFT;
85            assert_eq!(exception_class, EXCEPTION_CLASS);
86
87            Some(esr & WNR_BIT_MASK == WNR_BIT_MASK)
88        }
89    } else {
90        compile_error!("lazy-pages are not supported on your system. Disable `lazy-pages` feature");
91    }
92}

94extern "C" fn handle_sigsegv<H>(sig: i32, info: *mut siginfo_t, ucontext: *mut c_void)
95where
96    H: UserSignalHandler,
97{
98    unsafe {
99        let addr = (*info).si_addr();
100        let is_write = ucontext_get_write(ucontext as *mut _);
101        let exc_info = ExceptionInfo {
102            fault_addr: addr as *mut _,
103            is_write,
104        };
105
106        if let Err(err) = H::handle(exc_info) {
107            let old_sig_handler_works = match err {
108                Error::OutOfWasmMemoryAccess | Error::WasmMemAddrIsNotSet => {
109                    old_sig_handler(sig, info, ucontext)
110                }
111                _ => false,
112            };
113            if !old_sig_handler_works {
114                panic!("Signal handler failed: {err}");
115            }
116        }
117    }
118}

122#[derive(Debug, Clone, Copy, derive_more::Display)]
123enum ThreadInitError {
124    #[display("Cannot get information about old signal stack: {_0}")]
125    OldStack(Errno),
126    #[display("Cannot mmap space for signal stack: {_0}")]
127    Mmap(Errno),
128    #[display("Cannot set new signal stack: {_0}")]
129    SigAltStack(Errno),
130}
131
132fn init_for_thread_internal() -> Result<(), ThreadInitError> {
133    use core::{mem, ptr};
134
135    // Should be enough for lazy-pages signal handler.
136    // Equal to libc::SIGSTKSZ on macos M1.
137    const SIGNAL_STACK_SIZE: usize = 0x20000;
138
139    enum StackInfo {
140        UseOldStack,
141        NewStack(*mut libc::c_void),
142    }
143
144    impl Drop for StackInfo {
145        fn drop(&mut self) {
146            if let StackInfo::NewStack(mmap_ptr) = self {
147                unsafe {
148                    // Deallocate the stack memory.
149                    if libc::munmap(*mmap_ptr, SIGNAL_STACK_SIZE) != 0 {
150                        log::error!(
151                            "Cannot deallocate signal stack memory during the thread shutdown: {}",
152                            errno::errno()
153                        );
154                    }
155                }
156            }
157        }
158    }
159
160    unsafe fn init_sigstack() -> Result<StackInfo, ThreadInitError> {
161        // Check whether old signal stack exist and suitable for lazy-pages signal handler.
162        let mut old_stack = unsafe { mem::zeroed() };
163        let res = unsafe { libc::sigaltstack(ptr::null(), &mut old_stack) };
164        if res != 0 {
165            return Err(ThreadInitError::OldStack(errno::errno()));
166        }
167        if old_stack.ss_flags & libc::SS_DISABLE == 0 && old_stack.ss_size >= SIGNAL_STACK_SIZE {
168            return Ok(StackInfo::UseOldStack);
169        }
170
171        // Alloc memory for new signal stack.
172        let ptr = unsafe {
173            libc::mmap(
174                ptr::null_mut(),
175                SIGNAL_STACK_SIZE,
176                libc::PROT_READ | libc::PROT_WRITE,
177                libc::MAP_PRIVATE | libc::MAP_ANON,
178                -1,
179                0,
180            )
181        };
182        if ptr == libc::MAP_FAILED {
183            return Err(ThreadInitError::Mmap(errno::errno()));
184        }
185
186        // Mark allocated memory as new signal stack.
187        let new_stack = libc::stack_t {
188            ss_sp: ptr,
189            ss_flags: 0,
190            ss_size: SIGNAL_STACK_SIZE,
191        };
192        let res = unsafe { libc::sigaltstack(&new_stack, ptr::null_mut()) };
193        if res != 0 {
194            return Err(ThreadInitError::SigAltStack(errno::errno()));
195        }
196
197        log::debug!("Set new signal stack: ptr = {ptr:?}, size = {SIGNAL_STACK_SIZE:#x}");
198
199        Ok(StackInfo::NewStack(ptr))
200    }
201
202    thread_local! {
203        static TLS: Result<StackInfo, ThreadInitError> = unsafe { init_sigstack() };
204    }
205
206    TLS.with(|tls| tls.as_ref().map(|_| ()).map_err(|err| *err))
207}

209pub(crate) unsafe fn init_for_thread() -> Result<(), String> {
210    init_for_thread_internal().map_err(|err| err.to_string())
211}

213pub(crate) unsafe fn setup_signal_handler<H>() -> io::Result<()>
214where
215    H: UserSignalHandler,
216{
217    let handler = signal::SigHandler::SigAction(handle_sigsegv::<H>);
218    // Set additional SA_ONSTACK and SA_NODEFER to avoid problems with wasmer executor.
219    // See comment from shorturl.at/KMO68 :
220    // ```
221    //  SA_ONSTACK allows us to handle signals on an alternate stack,
222    //  so that the handler can run in response to running out of
223    //  stack space on the main stack. Rust installs an alternate
224    //  stack with sigaltstack, so we rely on that.
225    //  SA_NODEFER allows us to reenter the signal handler if we
226    //  crash while handling the signal, and fall through to the
227    //  Breakpad handler by testing handlingSegFault.
228    // ```
229    let sig_action = signal::SigAction::new(
230        handler,
231        signal::SaFlags::SA_SIGINFO | signal::SaFlags::SA_ONSTACK | signal::SaFlags::SA_NODEFER,
232        signal::SigSet::empty(),
233    );
234
235    let signal = if cfg!(target_os = "macos") {
236        signal::SIGBUS
237    } else {
238        signal::SIGSEGV
239    };
240
241    let old_sigaction =
242        unsafe { signal::sigaction(signal, &sig_action) }.map_err(io::Error::from)?;
243    let handler = old_sigaction.handler();
244    let _ = OLD_SIG_HANDLER
245        .set(handler)
246        .map(|_| log::trace!("Save old signal handler: {handler:?}"));
247
248    Ok(())
249}
250
251unsafe fn old_sig_handler(sig: i32, info: *mut siginfo_t, ucontext: *mut c_void) -> bool {
252    if let Some(old_sig_handler) = OLD_SIG_HANDLER.get() {
253        match old_sig_handler {
254            SigHandler::SigDfl | SigHandler::SigIgn => false,
255            SigHandler::Handler(func) => {
256                func(sig);
257                true
258            }
259            SigHandler::SigAction(func) => {
260                func(sig, info, ucontext);
261                true
262            }
263        }
264    } else {
265        false
266    }
267}

269#[cfg(all(target_os = "linux", target_arch = "aarch64"))]
270mod linux_aarch64 {
271    use std::{ptr, slice};
272
273    const ESR_MAGIC: u32 = u32::from_be_bytes(*b"ESR\x01");
274
275    #[repr(packed)]
276    #[derive(Clone, Copy)]
277    struct Header {
278        magic: u32, // Magic number to identify the record type
279        size: u32,  // Size of the record in bytes
280    }
281
282    /// Scan through the 4 KiB __reserved buffer looking for an `esr_context` record.
283    /// Returns `Some(esr)` if we find a record whose magic == ESR_MAGIC, else `None`.
284    /// See: https://github.com/torvalds/linux/blob/7f9039c524a351c684149ecf1b3c5145a0dff2fe/arch/arm64/include/uapi/asm/sigcontext.h#L40
285    pub fn get_esr(mcontext: &nix::libc::mcontext_t) -> Option<usize> {
286        // SAFETY: See `mcontext_t` definition:
287        // ```C
288        //  struct sigcontext {
289        //      __u64 fault_address;
290        //      /* AArch64 registers */
291        //      __u64 regs[31];
292        //      __u64 sp;
293        //      __u64 pc;
294        //      __u64 pstate;
295        //      /* 4K reserved for FP/SIMD state and future expansion */
296        //      __u8 __reserved[4096] __attribute__((__aligned__(16)));
297        //  };
298        // ```
299        let reserved = unsafe {
300            let reserved_addr_unaligned = ptr::addr_of!(mcontext.pstate).add(1);
301            let reserved_addr =
302                reserved_addr_unaligned.add(reserved_addr_unaligned.align_offset(16)) as *const u8;
303            slice::from_raw_parts(reserved_addr, 4096)
304        };
305
306        let mut offset = 0usize;
307
308        while offset + 8 <= reserved.len() {
309            // Read header of the next context record:
310            let Header { magic, size } =
311                unsafe { (reserved.as_ptr().add(offset) as *const Header).read_unaligned() };
312            let size = size as usize;
313
314            // Sanity check: size must be at least 8 (header itself), and offset+size must not overflow 4096.
315            if size < 8 || offset + size > reserved.len() {
316                break;
317            }
318
319            if magic == ESR_MAGIC {
320                // The first 8 bytes are (magic, size). The next 8 bytes are the u64 ESR value.
321                if offset + 16 <= reserved.len() {
322                    let esr_bytes = reserved[offset + 8..offset + 16]
323                        .try_into()
324                        .expect("cannot fail");
325                    let esr = usize::from_ne_bytes(esr_bytes);
326                    return Some(esr);
327                } else {
328                    // Not enough room for a full u64 after the header: treat as “not found”.
329                    return None;
330                }
331            }
332
333            // Skip to next record
334            offset += size;
335        }
336
337        None
338    }
