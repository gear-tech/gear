19use std::fmt::{Debug, Display};
20use subxt::ext::codec::Decode;
21
22// TODO: refactor this after #2618
23macro_rules! export_module_error {
24    ($($path:ident)::* => $error:ident => $index:expr) => {
25        pub use crate::metadata::runtime_types::$($path)::*::pallet::Error as $error;
26    };
27    ($($($path:ident)::* => $error:ident => $index:expr),*) => {
28        $(
29            export_module_error!($($path)::* => $error => $index);
30        )*
31
32        /// Common error type for runtime modules.
33        #[derive(Debug)]
34        pub enum ModuleError {
35            $($error($error)),*,
36            Unknown {
37                pallet_index: u8,
38                error: [u8; 4],
39            }
40        }
41
42        impl Display for ModuleError {
43             fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
44                 write!(f, "{:?}", self)
45             }
46        }
47
48        impl std::error::Error for ModuleError {}
49
50        impl From<subxt::error::ModuleError> for ModuleError {
51            fn from(e: subxt::error::ModuleError) -> ModuleError {
52                let mut error = [0; 4];
53                error.copy_from_slice(&e.bytes()[1..]);
54
55                match e.pallet_index() {
56                     $($index => match $error::decode(&mut [e.error_index()].as_ref()) {
57                         Ok(e) => ModuleError::$error(e),
58                         Err(_) => ModuleError::Unknown {
59                             pallet_index: e.pallet_index(),
60                             error,
61                         },
62                     }),*,
63                     // `pallet_fellowship_referenda => 19`
64                     //
65                     // shares the same error with
66                     //
67                     // `pallet_fellowship_collective => 18`
68                     19 => {
69                         let mb_error = RanckedCollective::decode(&mut error.as_ref());
70                         match mb_error {
71                             Ok(e) => ModuleError::RanckedCollective(e),
72                             Err(_) => ModuleError::Unknown {
73                                 pallet_index: e.pallet_index(),
74                                 error,
75                             },
76                         }
77                     },
78                     _ => ModuleError::Unknown {
79                         pallet_index: e.pallet_index(),
80                         error,
81                     }
82                }
83            }
84        }
85    };
86}
