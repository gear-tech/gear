142macro_rules! impl_basic {
143    ($t:ty) => {
144        impl Clone for $t {
145            fn clone(&self) -> Self {
146                Self::decode(&mut self.encode().as_ref()).expect("Infallible")
147            }
148        }
149
150        impl PartialEq for $t {
151            fn eq(&self, other: &Self) -> bool {
152                self.encode().eq(&other.encode())
153            }
154        }
155    };
156    ($t:ty $(, $tt:ty) +) => {
157        impl_basic!{ $t }
158        $(impl_basic! { $tt }) +
159    };
160}

165    Reason<UserMessageReadRuntimeReason, UserMessageReadSystemReason>,
166    generated_core_errors::simple::ReplyCode, VoucherId

169impl From<RuntimeCall> for Value {
170    fn from(call: RuntimeCall) -> Value {
171        match call {
172            RuntimeCall::Gear(gear_call) => gear_call_to_scale_value(gear_call),
173            RuntimeCall::Sudo(sudo_call) => sudo_call_to_scale_value(sudo_call),
174            RuntimeCall::Balances(balances_call) => balances_call_to_scale_value(balances_call),
175            RuntimeCall::System(system_call) => system_call_to_scale_value(system_call),
176            _ => unimplemented!("other calls aren't supported for now."),
177        }
178    }
179}

181fn gear_call_to_scale_value(call: GearCall) -> Value {
182    let variant = match call {
183        GearCall::upload_code { code } => {
184            Value::named_variant("upload_code", [("code", Value::from_bytes(code))])
185        }
186        GearCall::upload_program {
187            code,
188            salt,
189            init_payload,
190            gas_limit,
191            value,
192            keep_alive,
193        } => Value::named_variant(
194            "upload_program",
195            [
196                ("code", Value::from_bytes(code)),
197                ("salt", Value::from_bytes(salt)),
198                ("init_payload", Value::from_bytes(init_payload)),
199                ("gas_limit", Value::u128(gas_limit as u128)),
200                ("value", Value::u128(value as u128)),
201                ("keep_alive", Value::bool(keep_alive)),
202            ],
203        ),
204        GearCall::create_program {
205            code_id,
206            salt,
207            init_payload,
208            gas_limit,
209            value,
210            keep_alive,
211        } => Value::named_variant(
212            "create_program",
213            [
214                ("code_id", Value::from_bytes(code_id.0)),
215                ("salt", Value::from_bytes(salt)),
216                ("init_payload", Value::from_bytes(init_payload)),
217                ("gas_limit", Value::u128(gas_limit as u128)),
218                ("value", Value::u128(value as u128)),
219                ("keep_alive", Value::bool(keep_alive)),
220            ],
221        ),
222        GearCall::send_message {
223            destination,
224            payload,
225            gas_limit,
226            value,
227            keep_alive,
228        } => Value::named_variant(
229            "send_message",
230            [
231                ("destination", Value::from_bytes(destination.0)),
232                ("payload", Value::from_bytes(payload)),
233                ("gas_limit", Value::u128(gas_limit as u128)),
234                ("value", Value::u128(value as u128)),
235                ("keep_alive", Value::bool(keep_alive)),
236            ],
237        ),
238        GearCall::send_reply {
239            reply_to_id,
240            payload,
241            gas_limit,
242            value,
243            keep_alive,
244        } => Value::named_variant(
245            "send_reply",
246            [
247                ("reply_to_id", Value::from_bytes(reply_to_id.0)),
248                ("payload", Value::from_bytes(payload)),
249                ("gas_limit", Value::u128(gas_limit as u128)),
250                ("value", Value::u128(value as u128)),
251                ("keep_alive", Value::bool(keep_alive)),
252            ],
253        ),
254        GearCall::claim_value { message_id } => Value::named_variant(
255            "claim_value",
256            [("message_id", Value::from_bytes(message_id.0))],
257        ),
258        _ => {
259            unimplemented!("calls that won't be used in batch call");
260        }
261    };
262
263    Value::unnamed_variant("Gear", [variant])
264}

266fn sudo_call_to_scale_value(call: SudoCall) -> Value {
267    let variant = match call {
268        SudoCall::sudo_unchecked_weight { call, weight } => Value::named_variant(
269            "sudo_unchecked_weight",
270            [
271                ("call", (*call).into()),
272                (
273                    "weight",
274                    Value::named_composite([
275                        ("ref_time", Value::u128(weight.ref_time as u128)),
276                        ("proof_size", Value::u128(weight.proof_size as u128)),
277                    ]),
278                ),
279            ],
280        ),
281        _ => unimplemented!("calls that won't be used in batch call"),
282    };
283
284    Value::unnamed_variant("Sudo", [variant])
285}

287fn balances_call_to_scale_value(call: BalancesCall) -> Value {
288    let variant = match call {
289        BalancesCall::force_set_balance { who, new_free } => {
290            let id = match who {
291                MultiAddress::Id(id) => id,
292                _ => unreachable!("internal error: unused multi-address variant occurred"),
293            };
294            Value::named_variant(
295                "force_set_balance",
296                [
297                    ("who", Value::unnamed_variant("Id", [Value::from_bytes(id)])),
298                    ("new_free", Value::u128(new_free)),
299                ],
300            )
301        }
302        _ => unreachable!("calls that won't be used in batch call"),
303    };
304
305    Value::unnamed_variant("Balances", [variant])
306}

308fn system_call_to_scale_value(call: SystemCall) -> Value {
309    let variant = match call {
310        SystemCall::set_storage { items } => {
311            let items_as_values: Vec<Value> = items
312                .iter()
313                .map(|i| {
314                    Value::unnamed_composite([Value::from_bytes(&i.0), Value::from_bytes(&i.1)])
315                })
316                .collect();
317            Value::named_variant(
318                "set_storage",
319                [("items", Value::unnamed_composite(items_as_values))],
320            )
321        }
322        SystemCall::set_code { code } => {
323            Value::named_variant("set_code", [("code", Value::from_bytes(code))])
324        }
325        SystemCall::set_code_without_checks { code } => Value::named_variant(
326            "set_code_without_checks",
327            [("code", Value::from_bytes(code))],
328        ),
329        _ => unreachable!("other calls aren't supported for now."),
330    };
331
332    Value::unnamed_variant("System", [variant])
333}

347impl Convert<subxt::utils::MultiAddress<subxt::utils::AccountId32, ()>>
348    for sp_runtime::MultiAddress<sp_runtime::AccountId32, ()>
349{
350    fn convert(self) -> subxt::utils::MultiAddress<subxt::utils::AccountId32, ()> {
351        match self {
352            sp_runtime::MultiAddress::Address20(id) => subxt::utils::MultiAddress::Address20(id),
353            sp_runtime::MultiAddress::Address32(id) => subxt::utils::MultiAddress::Address32(id),
354            sp_runtime::MultiAddress::Id(id) => subxt::utils::MultiAddress::Id(id.convert()),
355            sp_runtime::MultiAddress::Index(index) => subxt::utils::MultiAddress::Index(index),
356            sp_runtime::MultiAddress::Raw(raw) => subxt::utils::MultiAddress::Raw(raw),
357        }
358    }
359}

367fn prepaid_call_to_scale_value(call: PrepaidCall<u128>) -> Value {
368    match call {
369        PrepaidCall::SendMessage {
370            destination,
371            payload,
372            gas_limit,
373            value,
374            keep_alive,
375        } => Value::named_variant(
376            "SendMessage",
377            [
378                ("destination", Value::from_bytes(destination.0)),
379                ("payload", Value::from_bytes(payload)),
380                ("gas_limit", Value::u128(gas_limit as u128)),
381                ("value", Value::u128(value as u128)),
382                ("keep_alive", Value::bool(keep_alive)),
383            ],
384        ),
385        PrepaidCall::SendReply {
386            reply_to_id,
387            payload,
388            gas_limit,
389            value,
390            keep_alive,
391        } => Value::named_variant(
392            "SendReply",
393            [
394                ("reply_to_id", Value::from_bytes(reply_to_id.0)),
395                ("payload", Value::from_bytes(payload)),
396                ("gas_limit", Value::u128(gas_limit as u128)),
397                ("value", Value::u128(value as u128)),
398                ("keep_alive", Value::bool(keep_alive)),
399            ],
400        ),
401        PrepaidCall::UploadCode { code } => {
402            Value::named_variant("UploadCode", [("code", Value::from_bytes(code))])
403        }
404        PrepaidCall::DeclineVoucher => Value::unnamed_variant("DeclineVoucher", []),
405        _ => unreachable!("other prepaid calls aren't supported"),
406    }
407}
