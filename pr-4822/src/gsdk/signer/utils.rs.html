49    /// Logging balance spent
50    pub async fn log_balance_spent(&self, before: u128) -> Result<()> {
51        let signer_rpc = SignerRpc(Arc::new(self.clone()));
52        match signer_rpc.get_balance().await {
53            Ok(balance) => {
54                let after = before.saturating_sub(balance);
55                log::info!("\tBalance spent: {after}");
56            }
57            Err(e) => log::info!("\tAccount was removed from storage: {e}"),
58        }
59
60        Ok(())
61    }

63    /// Propagates log::info for given status.
64    pub(crate) fn log_status(status: &TxStatus) {
65        match status {
66            TxStatus::Validated => log::info!("\tStatus: Validated"),
67            TxStatus::Broadcasted { num_peers } => log::info!("\tStatus: Broadcast( {num_peers} )"),
68            TxStatus::NoLongerInBestBlock => log::info!("\tStatus: NoLongerInBestBlock"),
69            TxStatus::InBestBlock(b) => log::info!(
70                "\tStatus: InBestBlock( block hash: {}, extrinsic hash: {} )",
71                b.block_hash(),
72                b.extrinsic_hash()
73            ),
74            TxStatus::InFinalizedBlock(b) => log::info!(
75                "\tStatus: Finalized( block hash: {}, extrinsic hash: {} )",
76                b.block_hash(),
77                b.extrinsic_hash()
78            ),
79            TxStatus::Error { message: e } => log::error!("\tStatus: Error( {e:?} )"),
80            TxStatus::Dropped { message: e } => log::error!("\tStatus: Dropped( {e:?} )"),
81            TxStatus::Invalid { message: e } => log::error!("\tStatus: Invalid( {e:?} )"),
82        }
83    }

85    /// Listen transaction process and print logs.
86    pub async fn process(&self, tx: DynamicPayload) -> Result<TxInBlock> {
87        use subxt::tx::TxStatus::*;
88
89        let signer_rpc = SignerRpc(Arc::new(self.clone()));
90        let before = signer_rpc.get_balance().await?;
91
92        let mut process = self.sign_and_submit_then_watch(&tx).await?;
93        let (pallet, name) = (tx.pallet_name(), tx.call_name());
94        let extrinsic = format!("{pallet}::{name}").magenta().bold();
95
96        log::info!("Pending {extrinsic} ...");
97        let mut queue: Vec<BacktraceStatus> = Default::default();
98        let mut hash: Option<H256> = None;
99
100        while let Some(status) = process.next().await {
101            let status = status?;
102            Self::log_status(&status);
103
104            if let Some(h) = &hash {
105                self.backtrace
106                    .clone()
107                    .append(*h, BacktraceStatus::from(&status));
108            } else {
109                queue.push((&status).into());
110            }
111
112            match status {
113                Validated | Broadcasted { .. } | NoLongerInBestBlock => (),
114                InBestBlock(b) => {
115                    hash = Some(b.extrinsic_hash());
116                    self.backtrace.append(
117                        b.extrinsic_hash(),
118                        BacktraceStatus::InBestBlock {
119                            block_hash: b.block_hash(),
120                            extrinsic_hash: b.extrinsic_hash(),
121                        },
122                    );
123                }
124                InFinalizedBlock(b) => {
125                    log::info!("Submitted {extrinsic} !");
126                    log::info!("\tBlock Hash: {:?}", b.block_hash());
127                    log::info!("\tTransaction Hash: {:?}", b.extrinsic_hash());
128                    self.log_balance_spent(before).await?;
129                    return Ok(b);
130                }
131                _ => {
132                    self.log_balance_spent(before).await?;
133                    return Err(status.into());
134                }
135            }
136        }
137
138        Err(anyhow!("Transaction wasn't found").into())
139    }

141    /// Process sudo transaction.
142    pub async fn process_sudo(&self, tx: DynamicPayload) -> EventsResult {
143        let tx = self.process(tx).await?;
144        let events = tx.wait_for_success().await?;
145        for event in events.iter() {
146            let event = event?.as_root_event::<Event>()?;
147            if let Event::Sudo(SudoEvent::Sudid {
148                sudo_result: Err(err),
149            }) = event
150            {
151                return Err(self.api().decode_error(err).into());
152            }
153        }
154
155        Ok((tx.block_hash(), events))
156    }

158    /// Run transaction.
159    ///
160    /// This function allows us to execute any transactions in gear.
161    ///
162    /// # You may not need this.
163    ///
164    /// Read the docs of [`Signer`](`super::Signer`) to checkout the wrappred transactions,
165    /// we need this function only when we want to execute a transaction
166    /// which has not been wrapped in `gsdk`.
167    ///
168    /// # Example
169    ///
170    /// ```ignore
171    /// use gsdk::{
172    ///   Api,
173    ///   Signer,
174    ///   metadata::calls::BalancesCall,
175    ///   Value,
176    /// };
177    ///
178    /// let api = Api::new(None).await?;
179    /// let signer = Signer::new(api, "//Alice", None).await?;
180    ///
181    /// {
182    ///     let args = vec![
183    ///         Value::unnamed_variant("Id", [Value::from_bytes(dest.into())]),
184    ///         Value::u128(value),
185    ///     ];
186    ///     let in_block = signer.run_tx(BalancesCall::TransferKeepAlive, args).await?;
187    /// }
188    ///
189    /// // The code above euqals to:
190    ///
191    /// {
192    ///    let in_block = signer.calls.transfer_keep_alive(dest, value).await?;
193    /// }
194    ///
195    /// // ...
196    /// ```
197    pub async fn run_tx<Call: CallInfo>(
198        &self,
199        call: Call,
200        fields: impl Into<Composite<()>>,
201    ) -> Result<TxInBlock> {
202        let tx = subxt::dynamic::tx(Call::PALLET, call.call_name(), fields.into());
203
204        self.process(tx).await
205    }
206
207    /// Run transaction with sudo.
208    pub async fn sudo_run_tx<Call: CallInfo>(
209        &self,
210        call: Call,
211        fields: impl Into<Composite<()>>,
212    ) -> EventsResult {
213        let tx = subxt::dynamic::tx(Call::PALLET, call.call_name(), fields.into());
214
215        self.process_sudo(tx).await
216    }

218    /// `pallet_sudo::sudo`
219    pub async fn sudo(&self, call: RuntimeCall) -> EventsResult {
220        self.sudo_run_tx(SudoCall::Sudo, vec![Value::from(call)])
221            .await
222    }

224    /// Wrapper for submit and watch with nonce.
225    async fn sign_and_submit_then_watch(
226        &self,
227        tx: &DynamicPayload,
228    ) -> Result<TxProgressT, SubxtError> {
229        if let Some(nonce) = self.nonce {
230            self.api
231                .tx()
232                .create_signed(
233                    tx,
234                    &self.signer,
235                    PolkadotExtrinsicParamsBuilder::new().nonce(nonce).build(),
236                )
237                .await?
238                .submit_and_watch()
239                .await
240        } else {
241            self.api
242                .tx()
243                .sign_and_submit_then_watch_default(tx, &self.signer)
244                .await
245        }
246    }
