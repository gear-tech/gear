42/// Signer representation that provides access to gear API.
43/// Implements low-level methods such as [`run_tx`](`Inner::run_tx`)
44/// and [`force_batch`](`Signer.calls()::force_batch`).
45/// Other higher-level calls are provided by [`Signer::storage`],
46/// [`Signer::calls`], [`Signer::rpc`].
47#[derive(Clone)]
48pub struct Signer {
49    signer: Arc<Inner>,
50    /// Calls that get or set storage.
51    pub storage: SignerStorage,
52    /// Calls for interaction with on-chain programs.
53    pub calls: SignerCalls,
54    /// Calls to fetch data from node.
55    pub rpc: SignerRpc,
56}
57
58/// Implementation of low-level calls for [`Signer`].
59#[derive(Clone)]
60pub struct Inner {
61    api: Api,
62    /// Current signer.
63    signer: PairSigner<GearConfig, Pair>,
64    nonce: Option<u64>,
65    backtrace: Backtrace,
66}
67
68impl Signer {
69    /// Get backtrace of the signer.
70    pub fn backtrace(&self) -> Backtrace {
71        self.calls.0.backtrace.clone()
72    }
73
74    /// New signer api.
75    pub fn new(api: Api, suri: &str, passwd: Option<&str>) -> Result<Self> {
76        let signer = Inner {
77            api,
78            signer: PairSigner::new(
79                Pair::from_string(suri, passwd).map_err(|_| Error::InvalidSecret)?,
80            ),
81            nonce: None,
82            backtrace: Default::default(),
83        };
84
85        Ok(Self::from_inner(signer))
86    }
87
88    fn from_inner(signer: Inner) -> Self {
89        let signer = Arc::new(signer);
90
91        Self {
92            storage: SignerStorage(signer.clone()),
93            calls: SignerCalls(signer.clone()),
94            rpc: SignerRpc(signer.clone()),
95            signer,
96        }
97    }
98
99    fn replace_inner(&mut self, mut inner: Inner) {
100        let backtrace = self.backtrace();
101        inner.backtrace = backtrace;
102
103        let Signer {
104            signer,
105            storage,
106            calls,
107            rpc,
108        } = self;
109
110        *signer = Arc::new(inner);
111        *storage = SignerStorage(signer.clone());
112        *calls = SignerCalls(signer.clone());
113        *rpc = SignerRpc(signer.clone());
114    }
115
116    /// Change inner signer.
117    pub fn change(mut self, suri: &str, passwd: Option<&str>) -> Result<Self> {
118        let signer =
119            PairSigner::new(Pair::from_string(suri, passwd).map_err(|_| Error::InvalidSecret)?);
120
121        self.replace_inner(Inner {
122            signer,
123            ..self.signer.as_ref().clone()
124        });
125
126        Ok(self)
127    }
128
129    /// Set nonce of the signer
130    pub fn set_nonce(&mut self, nonce: u64) {
131        self.replace_inner(Inner {
132            nonce: Some(nonce),
133            ..self.signer.as_ref().clone()
134        });
135    }
136}
137
138impl Inner {
139    /// Get address of the current signer
140    pub fn address(&self) -> String {
141        self.account_id().to_ss58check()
142    }
143
144    /// Get address of the current signer
145    pub fn account_id(&self) -> &AccountId32 {
146        self.signer.account_id()
147    }
148
149    /// Get reference to inner unsigned api
150    pub fn api(&self) -> &Api {
151        &self.api
152    }
153
154    pub fn signer(&self) -> &PairSigner<GearConfig, Pair> {
155        &self.signer
156    }
157}
158
159impl From<(Api, PairSigner<GearConfig, Pair>)> for Signer {
160    fn from((api, signer): (Api, PairSigner<GearConfig, Pair>)) -> Self {
161        let signer = Inner {
162            api,
163            signer,
164            nonce: None,
165            backtrace: Backtrace::default(),
166        };
167
168        Self::from_inner(signer)
169    }
170}
171
172impl Deref for Signer {
173    type Target = Inner;
174
175    fn deref(&self) -> &Inner {
176        self.signer.as_ref()
177    }
178}
