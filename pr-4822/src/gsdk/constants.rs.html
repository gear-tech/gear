19//! Runtime constants query methods
20
21use crate::{Api, metadata::runtime_types::gear_common::GasMultiplier, result::Result};
22use parity_scale_codec::Decode;
23use sp_runtime::Percent;
24use subxt::utils::AccountId32;
25
26impl Api {
27    /// Query constant
28    fn query_constant<D: Decode>(&self, pallet: &'static str, constant: &'static str) -> Result<D> {
29        let addr = subxt::dynamic::constant(pallet, constant);
30        D::decode(&mut self.constants().at(&addr)?.encoded()).map_err(Into::into)
31    }
32}
33
34// pallet-babe
35impl Api {
36    /// Get expected block time.
37    pub fn expected_block_time(&self) -> Result<u64> {
38        self.query_constant("Babe", "ExpectedBlockTime")
39    }
40}
41
