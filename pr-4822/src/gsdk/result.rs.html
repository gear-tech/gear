22/// Transaction Errors
23#[derive(Debug, thiserror::Error)]
24pub enum TxError {
25    #[error("Transaction Error( {0} )")]
26    Error(String),
27    #[error("Transaction Invalid( {0} )")]
28    Invalid(String),
29    #[error("Transaction Dropped( {0} )")]
30    Dropped(String),
31    #[error("Not an error, this will never be reached.")]
32    None,
33}
34
35impl From<TxStatus> for Error {
36    fn from(status: TxStatus) -> Self {
37        match status {
38            TxStatus::Error { message } => TxError::Error(message),
39            TxStatus::Invalid { message } => TxError::Invalid(message),
40            TxStatus::Dropped { message } => TxError::Dropped(message),
41            unreachable => {
42                log::info!("Not an error tx status occurred {unreachable:?}");
43                TxError::None
44            }
45        }
46        .into()
47    }
48}

50/// Errors
51#[derive(Debug, thiserror::Error, derive_more::Unwrap)]
52pub enum Error {
53    #[error(transparent)]
54    Anyhow(#[from] anyhow::Error),
55    #[error(transparent)]
56    Base64Decode(#[from] base64::DecodeError),
57    #[error(transparent)]
58    Codec(#[from] parity_scale_codec::Error),
59    #[error("Code not found {0}")]
60    CodeNotFound(String),
61    #[error(transparent)]
62    Hex(#[from] hex::FromHexError),
63    #[error("Unable to get the name of the current executable binary")]
64    InvalidExecutable,
65    #[error("Password must be provided for logining with json file.")]
66    InvalidPassword,
67    #[error("Invalid public key")]
68    InvalidPublic,
69    #[error("Invalid secret key")]
70    InvalidSecret,
71    #[error("Invalid RPC params")]
72    InvalidRpcParams,
73    #[error(transparent)]
74    Io(#[from] std::io::Error),
75    #[error(transparent)]
76    SerdeJson(#[from] serde_json::Error),
77    #[error("The queried storage not found.")]
78    StorageNotFound,
79    #[error("The queried event not found.")]
80    EventNotFound,
81    #[error(transparent)]
82    Subxt(#[from] Box<subxt::Error>),
83    #[error(transparent)]
84    SubxtCore(#[from] Box<subxt::ext::subxt_core::Error>),
85    #[error(transparent)]
86    SubxtPublic(#[from] sp_core::crypto::PublicError),
87    #[error(transparent)]
88    SubxtMetadata(#[from] subxt::error::MetadataError),
89    #[error(transparent)]
90    ScaleValueEncode(#[from] Box<scale_value::scale::EncodeError>),
91    #[error(transparent)]
92    Tx(#[from] TxError),
93    #[error(transparent)]
94    SubxtRpc(#[from] jsonrpsee::core::ClientError),
95    #[error("Page {0} of Program {1} was not found in the storage.")]
96    PageNotFound(u32, String),
97    #[error("Program has been terminated.")]
98    ProgramTerminated,
99    #[error("Invalid rpc URL.")]
100    InvalidUrl,
101    #[error("Page {0} of Program {1} is invalid.")]
102    PageInvalid(u32, String),
103}
104
105impl From<subxt::Error> for Error {
106    fn from(value: subxt::Error) -> Self {
107        Self::Subxt(Box::new(value))
108    }
109}
110
111impl From<subxt::ext::subxt_core::Error> for Error {
112    fn from(value: subxt::ext::subxt_core::Error) -> Self {
113        Self::SubxtCore(Box::new(value))
114    }
115}
116
117impl From<scale_value::scale::EncodeError> for Error {
118    fn from(value: scale_value::scale::EncodeError) -> Self {
119        Self::ScaleValueEncode(Box::new(value))
120    }
121}
