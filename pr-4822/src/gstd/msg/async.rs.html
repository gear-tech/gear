36fn poll<F, R>(waiting_reply_to: MessageId, cx: &mut Context<'_>, f: F) -> Poll<Result<R>>
37where
38    F: Fn(Vec<u8>) -> Result<R>,
39{
40    let msg_id = crate::msg::id();
41
42    // check if message is timed out.
43    if let Some((expected, now)) = async_runtime::locks().is_timeout(msg_id, waiting_reply_to) {
44        // Remove lock after timeout.
45        async_runtime::locks().remove(msg_id, waiting_reply_to);
46
47        return Poll::Ready(Err(Error::Timeout(expected, now)));
48    }
49
50    match signals().poll(waiting_reply_to, cx) {
51        ReplyPoll::None => panic!(
52            "Somebody created a future with the MessageId that never ended in static replies!"
53        ),
54        ReplyPoll::Pending => Poll::Pending,
55        ReplyPoll::Some((payload, reply_code)) => {
56            // Remove lock after waking.
57            async_runtime::locks().remove(msg_id, waiting_reply_to);
58
59            match reply_code {
60                ReplyCode::Success(_) => Poll::Ready(f(payload)),
61                ReplyCode::Error(reason) => {
62                    Poll::Ready(Err(Error::ErrorReply(payload.into(), reason)))
63                }
64                ReplyCode::Unsupported => Poll::Ready(Err(Error::UnsupportedReply(payload))),
65            }
66        }
67    }
68}

70/// Same as [`MessageFuture`], but allows decoding the reply's payload instead
71/// of getting a byte vector.
72///
73/// Generic `T` type should implement the [`Decode`] trait.
74///
75/// # Examples
76///
77/// The following example explicitly annotates variable types for demonstration
78/// purposes only. Usually, annotating them is unnecessary because
79/// they can be inferred automatically.
80///
81/// ```ignored
82/// use gstd::{
83///     msg::{self, CodecMessageFuture},
84///     prelude::*,
85/// };
86/// # use gstd::ActorId;
87///
88/// #[derive(Decode)]
89/// #[codec(crate = gstd::codec)]
90/// struct Reply {
91///     field: String,
92/// }
93///
94/// #[gstd::async_main]
95/// async fn main() {
96///     # let dest = ActorId::zero();
97///     let future: CodecMessageFuture<Reply> =
98///         msg::send_bytes_for_reply_as(dest, b"PING", 0, 0).expect("Unable to send");
99///     let reply: Reply = future.await.expect("Unable to get a reply");
100///     let field: String = reply.field;
101/// }
102/// # fn main() {}
103/// ```
104pub struct CodecMessageFuture<T> {
105    /// A message identifier for an expected reply.
106    pub waiting_reply_to: MessageId,
107    /// Reply deposit that was allocated for this message. Checked in
108    /// handle_reply.
109    #[cfg_attr(feature = "ethexe", allow(unused))]
110    pub(crate) reply_deposit: u64,
111    /// Marker
112    ///
113    /// # Note
114    ///
115    /// Need to `pub` this field because we are constructing this
116    /// field in other files
117    pub(crate) _marker: PhantomData<T>,
118}
119
120impl_futures!(
121    CodecMessageFuture,
122    D,
123    D,
124    |fut, cx| => {
125        poll(fut.waiting_reply_to, cx, |reply| {
126            D::decode(&mut reply.as_ref()).map_err(Error::Decode)
127        })
128    }
129);

148/// struct InitReply {
149///     field: String,

157///             .expect("Unable to create a program");
158///     let (prog_id, reply): (ActorId, InitReply) = future.await.expect("Unable to get a reply");
159///     let field: String = reply.field;
160/// }
161/// # fn main() {}
162/// ```
163pub struct CodecCreateProgramFuture<T> {
164    /// A message identifier for an expected reply.
165    pub waiting_reply_to: MessageId,
166    /// An identifier of a newly created program.
167    pub program_id: ActorId,
168    /// Reply deposit that was allocated for this message. Checked in
169    /// handle_reply.
170    #[cfg_attr(feature = "ethexe", allow(unused))]
171    pub(crate) reply_deposit: u64,
172    /// Marker
173    ///
174    /// # Note
175    ///
176    /// Need to `pub` this field because we are constructing this
177    /// field in other files.
178    pub(crate) _marker: PhantomData<T>,
179}
180
181impl_futures!(
182    CodecCreateProgramFuture,
183    D,
184    (ActorId, D),
185    |fut, cx| => {
186        poll(fut.waiting_reply_to, cx, |reply| {
187            D::decode(&mut reply.as_ref())
188                .map(|payload| (fut.program_id, payload))
189                .map_err(Error::Decode)
190        })
191    }

194/// Future returned by async functions related to message sending that wait for
195/// a reply (see sending functions with `_for_reply` suffix, e.g.
196/// [`send_bytes_for_reply`](super::send_bytes_for_reply)).
197///
198/// To get the reply payload (in bytes), one should use `.await` syntax. After
199/// calling a corresponding async function, the program interrupts its execution
200/// until a reply arrives.
201///
202/// This future keeps the sent message identifier ([`MessageId`] to wake the
203/// program after a reply arrives.
204///
205/// # Examples
206///
207/// The following example explicitly annotates variable types for demonstration
208/// purposes only. Usually, annotating them is unnecessary because
209/// they can be inferred automatically.
210///
211/// ```ignored
212/// use gstd::msg::{self, MessageFuture};
213/// # use gstd::ActorId;
214///
215/// #[gstd::async_main]
216/// async fn main() {
217///     # let dest = ActorId::zero();
218///     let future: MessageFuture =
219///         msg::send_bytes_for_reply(dest, b"PING", 0, 0).expect("Unable to send");
220///     let reply: Vec<u8> = future.await.expect("Unable to get a reply");
221/// }
222/// # fn main() {}
223/// ```
224pub struct MessageFuture {
225    /// A message identifier for an expected reply.
226    ///
227    /// This identifier is generated by the corresponding send function (e.g.
228    /// [`send_bytes`](super::send_bytes)).
229    pub waiting_reply_to: MessageId,
230    /// Reply deposit that was allocated for this message. Checked in
231    /// handle_reply.
232    #[cfg_attr(feature = "ethexe", allow(unused))]
233    pub(crate) reply_deposit: u64,
234}
235
236impl_futures!(
237    MessageFuture,
238    Vec<u8>,
239    |fut, cx| => {
240        poll(fut.waiting_reply_to, cx, Ok)
241    }
242);

244/// Async functions that relate to creating programs wait for a reply from the
245/// program's init function. These functions have the suffix` _for_reply`, such
246/// as [`crate::prog::create_program_bytes_for_reply`].
247///
248/// To get the reply payload (in bytes), one should use `.await` syntax. After
249/// calling a corresponding async function, the program interrupts its execution
250/// until a reply arrives.
251///
252/// This future keeps the sent message identifier ([`MessageId`]) to wake the
253/// program after a reply arrives. Also, it keeps an identifier of a newly
254/// created program ([`ActorId`]).
255///
256/// # Examples
257///
258/// The following example explicitly annotates variable types for demonstration
259/// purposes only. Usually, annotating them is unnecessary because
260/// they can be inferred automatically.
261///
262/// ```ignored
263/// use gstd::{msg::CreateProgramFuture, prog, ActorId};
264/// # use gstd::CodeId;
265///
266/// #[gstd::async_main]
267/// async fn main() {
268///     # let code_id = CodeId::new([0; 32]);
269///     let future: CreateProgramFuture =
270///         prog::create_program_bytes_for_reply(code_id, b"salt", b"PING", 0, 0)
271///             .expect("Unable to create a program");
272///     let (prog_id, reply): (ActorId, Vec<u8>) = future.await.expect("Unable to get a reply");
273/// }
274/// # fn main() {}
275/// ```
276pub struct CreateProgramFuture {
277    /// A message identifier for an expected reply.
278    pub waiting_reply_to: MessageId,
279    /// An identifier of a newly created program.
280    pub program_id: ActorId,
281    /// Reply deposit that was allocated for this message. Checked in
282    /// handle_reply.
283    #[cfg_attr(feature = "ethexe", allow(unused))]
284    pub(crate) reply_deposit: u64,
285}
286

288    CreateProgramFuture,
289    (ActorId, Vec<u8>),
