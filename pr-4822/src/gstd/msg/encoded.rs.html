33/// Get a payload of the message that is currently being processed.
34///
35/// This function returns the decoded message's payload of a custom type.
36///
37/// Generic `D` type should implement the [`Decode`] trait.
38///
39/// # Examples
40///
41/// ```
42/// use gstd::{msg, prelude::*};
43///
44/// #[derive(Decode)]
45/// #[codec(crate = gstd::codec)]
46/// struct Input {
47///     field: String,
48/// }
49///
50/// #[unsafe(no_mangle)]
51/// extern "C" fn handle() {
52///     let payload: Input = msg::load().expect("Unable to load");
53///     msg::reply(payload.field, 0).expect("Unable to reply");
54/// }
55/// ```
56///
57/// # See also
58///
59/// - [`load_bytes`](super::load_bytes) function returns a payload as a byte
60///   vector.
61pub fn load<D: Decode>() -> crate::errors::Result<D> {
62    super::with_read_on_stack_or_heap(
63        |read_result: Result<&mut [u8]>| -> crate::errors::Result<D> {
64            let mut buffer = read_result? as &[u8];
65            D::decode(&mut buffer).map_err(Error::Decode)
66        },
67    )
68}

70/// Send a new message as a reply to the message being
71/// processed.
72///
73/// Some programs can reply to other programs, e.g., check another program's
74/// state and use it as a parameter for its business logic.
75///
76/// This function allows sending such replies, which are similar to standard
77/// messages in terms of payload and different only in how the message
78/// processing is handled by a dedicated program function called `handle_reply`.
79///
80/// The first argument is the encodable payload. The second argument is the
81/// value to be transferred from the current program account to the reply
82/// message target account.
83///
84/// Reply message transactions will be posted after processing is finished,
85/// similar to the standard message-sending function (e.g. [`send`]).
86///
87/// # Examples
88///
89/// ```
90/// use gstd::{msg, prelude::*};
91///
92/// #[derive(Encode)]
93/// #[codec(crate = gstd::codec)]
94/// struct Reply {
95///     a: i32,
96///     b: Option<bool>,
97/// }
98///
99/// #[unsafe(no_mangle)]
100/// extern "C" fn handle() {
101///     let payload = Reply {
102///         a: 42,
103///         b: Some(true),
104///     };
105///
106///     msg::reply(payload, 0).expect("Unable to reply");
107/// }
108/// ```
109///
110/// # See also
111///
112/// - [`reply_bytes`](super::reply_bytes) function sends a reply with an encoded
113///   payload.
114/// - [`reply_push`](super::reply_push), [`reply_commit`](super::reply_commit)
115///   functions allow forming a reply message in parts.
116/// - [`send`] function sends a new message to the program or user.
117pub fn reply<E: Encode>(payload: E, value: u128) -> Result<MessageId> {
118    with_optimized_encode(payload, |buffer| super::reply_bytes(buffer, value))
119}

121/// Same as [`reply`], but it spends gas from a reservation instead of
122/// borrowing gas from the gas limit provided with the incoming message.
123///
124/// The first argument is the reservation identifier [`ReservationId`] obtained
125/// by calling the corresponding API. The second argument is the encodable
126/// payload. The last argument is the value to be transferred from the current
127/// program account to the reply message target account.
128///
129/// # Examples
130///
131/// ```
132/// use gstd::{ReservationId, msg, prelude::*};
133///
134/// #[derive(Encode)]
135/// #[codec(crate = gstd::codec)]
136/// struct Reply {
137///     a: i32,
138///     b: Option<bool>,
139/// }
140///
141/// #[unsafe(no_mangle)]
142/// extern "C" fn handle() {
143///     let reservation_id = ReservationId::reserve(5_000_000, 100).expect("Unable to reserve");
144///     let payload = Reply {
145///         a: 42,
146///         b: Some(true),
147///     };
148///
149///     msg::reply_from_reservation(reservation_id, payload, 0).unwrap();
150/// }
151/// ```
152///
153/// # See also
154///
155/// - [`send_from_reservation`] function sends a new message to the program or
156///   user by using gas from a reservation.
157#[cfg(not(feature = "ethexe"))]
158pub fn reply_from_reservation<E: Encode>(
159    id: ReservationId,
160    payload: E,
161    value: u128,
162) -> Result<MessageId> {
163    with_optimized_encode(payload, |buffer| {
164        super::reply_bytes_from_reservation(id, buffer, value)
165    })
166}

199/// Same as [`reply`] but uses the input buffer as a payload source.
200///
201/// The first argument is the value to be transferred from the current program
202/// account to the reply message target account. The second argument is the
203/// index range that defines the input buffer's piece to be pushed back to the
204/// output.
205///
206/// # Examples
207///
208/// Send half of the incoming payload back to the sender as a reply.
209///
210/// ```
211/// use gstd::msg;
212///
213/// #[unsafe(no_mangle)]
214/// extern "C" fn handle() {
215///     msg::reply_input(0, 0..msg::size() / 2).expect("Unable to reply");
216/// }
217/// ```
218///
219/// # See also
220///
221/// - [`super::reply_push_input`] function pushes a payload part to the current
222///   reply message using the input buffer as a payload source.
223/// - [`MessageHandle::push_input`](super::MessageHandle::push_input) function
224///   allows using the input buffer as a payload source for an outcoming
225///   message.
226pub fn reply_input(value: u128, range: impl RangeBounds<usize>) -> Result<MessageId> {
227    let (offset, len) = utils::decay_range(range);
228
229    gcore::msg::reply_input(value, offset, len)
230}

322/// Send a new message to the program or user.
323///
324/// Gear allows programs to communicate with each other and users via messages.
325/// For example, the [`send`] function allows sending such messages.
326///
327/// The first argument is the address of the target account ([`ActorId`]). The
328/// second argument is the encodable payload. The last argument is the value to
329/// be transferred from the current program account to the message target
330/// account.
331///
332/// Send transaction will be posted after processing is finished, similar to the
333/// reply message [`reply`].
334///
335/// # Examples
336///
337/// Send a message to the arbitrary address:
338///
339/// ```
340/// use gstd::{ActorId, msg, prelude::*};
341///
342/// #[derive(Encode)]
343/// #[codec(crate = gstd::codec)]
344/// struct Output {
345///     a: i32,
346///     b: Option<bool>,
347/// }
348///
349/// #[unsafe(no_mangle)]
350/// extern "C" fn handle() {
351///     let payload = Output {
352///         a: 42,
353///         b: Some(true),
354///     };
355///
356///     // Receiver id is collected from bytes from 0 to 31
357///     let id: [u8; 32] = core::array::from_fn(|i| i as u8);
358///     msg::send(ActorId::new(id), payload, 0).expect("Unable to send");
359/// }
360/// ```
361///
362/// # See also
363///
364/// - [`reply`] function sends a new message as a reply to the message that is
365///   currently being processed.
366/// - [`MessageHandle::init`](super::MessageHandle::init),
367///   [`MessageHandle::push`](super::MessageHandle::push), and
368///   [`MessageHandle::commit`](super::MessageHandle::commit) functions allow
369///   forming a message to send in parts.
370#[wait_for_reply]
371pub fn send<E: Encode>(program: ActorId, payload: E, value: u128) -> Result<MessageId> {
372    with_optimized_encode(payload, |buffer| super::send_bytes(program, buffer, value))
373}

417/// Same as [`send`], but it spends gas from a reservation instead of borrowing
418/// it from the gas limit provided with the incoming message.
419///
420/// The first argument is the reservation identifier [`ReservationId`] obtained
421/// by calling the corresponding API. The second argument is the address of the
422/// target account ([`ActorId`]). The third argument is the encodable payload.
423/// Finally, the last argument is the value to be transferred from the current
424/// program account to the message target account.
425///
426/// # Examples
427///
428/// Send a message to the sender's address:
429///
430/// ```
431/// use gstd::{ReservationId, msg, prelude::*};
432///
433/// #[derive(Encode)]

437///     b: Option<bool>,
438/// }
439///
440/// #[unsafe(no_mangle)]
441/// extern "C" fn handle() {
442///     let payload = Output {
443///         a: 42,
444///         b: Some(true),
445///     };
446///     // Reserve 5 million of gas for 100 blocks
447///     let reservation_id = ReservationId::reserve(5_000_000, 100).expect("Unable to reserve");
448///     // Receiver id is the message source
449///     let actor_id = msg::source();
450///
451///     msg::send_from_reservation(reservation_id, actor_id, payload, 0).expect("Unable to send");
452/// }
453/// ```
454///
455/// # See also

458///   by using gas from a reservation.
