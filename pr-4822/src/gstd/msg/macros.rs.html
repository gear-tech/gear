19macro_rules! impl_futures {
20    ($f:ident, $r:ty, |$fut:ident, $cx:ident| => { $p:expr }) => {
21        impl_futures!($f, $r, ($fut, $cx), $p, );
22    };
23    ($f:ident, $g: tt, $r:ty, |$fut:ident, $cx:ident| => { $p:expr }) => {
24        impl_futures!($f, $r, ($fut, $cx), $p, $g);
25    };
26    ($f:ident, $r:ty, ($fut:ident, $cx:ident), $p:expr, $($g:tt)?) => {
27        impl $( <$g: Decode> )? FusedFuture for $f $( < $g > )? {
28            fn is_terminated(&self) -> bool {
29                !signals().waits_for(self.waiting_reply_to)
30            }
31        }
32
33        impl $( <$g: Decode> )? Future for $f $( < $g > )? {
34            type Output = Result<$r>;
35
36            fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
37                let $fut = &mut self;
38                let $cx = cx;
39
40                $p
41            }
42        }
43
44        impl $( <$g: Decode> )? $f $( < $g > )? {
45            /// Postpone handling for a maximum amount of blocks that could be paid, that
46            /// doesn't exceed a given duration.
47            pub fn up_to(self, duration: Option<u32>) -> Result<Self> {
48                async_runtime::locks().lock(
49                    crate::msg::id(),
50                    self.waiting_reply_to,
51                    Lock::up_to(duration.unwrap_or(Config::wait_up_to()))?,
52                );
53
54                Ok(self)
55            }
56
57            /// Postpone handling for a given specific amount of blocks.
58            pub fn exactly(self, duration: Option<u32>) -> Result<Self> {
59                async_runtime::locks().lock(
60                    crate::msg::id(),
61                    self.waiting_reply_to,
62                    Lock::exactly(duration.unwrap_or(Config::wait_for()))?,
63                );
64
65                Ok(self)
66            }
67
68            /// Execute a function when the reply is received.
69            ///
70            /// This callback will be executed in reply context and consume reply gas, so
71            /// adequate `reply_deposit` should be supplied in `*_for_reply` call
72            /// that comes before this. Note that the hook will still be executed on reply
73            /// even after original future resolves in timeout.
74            ///
75            /// # Examples
76            ///
77            /// Send message to echo program and wait for reply.
78            ///
79            /// ```
80            /// use gstd::{ActorId, msg, debug};
81            ///
82            /// #[gstd::async_main]
83            /// async fn main() {
84            ///     let dest = ActorId::from(1); // Replace with correct actor id
85            ///
86            ///     msg::send_bytes_for_reply(dest, b"PING", 0, 1_000_000)
87            ///         .expect("Unable to send")
88            ///         .handle_reply(|| {
89            ///             debug!("reply code: {:?}", msg::reply_code());
90            ///
91            ///             if msg::load_bytes().unwrap_or_default() == b"PONG" {
92            ///                debug!("successfully received pong");
93            ///             }
94            ///         })
95            ///         .expect("Error setting reply hook")
96            ///         .await
97            ///         .expect("Received error");
98            /// }
99            /// # fn main() {}
100            /// ```
101            ///
102            /// # Panics
103            ///
104            /// Panics if this is called second time.
105            #[cfg(not(feature = "ethexe"))]
106            pub fn handle_reply<F: FnOnce() + 'static>(self, f: F) -> Result<Self> {
107                if self.reply_deposit == 0 {
108                    return Err(Error::Gstd(crate::errors::UsageError::ZeroReplyDeposit));
109                }
110                async_runtime::reply_hooks().register(self.waiting_reply_to.clone(), f);
111
112                Ok(self)
113            }
114        }
115    };
116}
117
118pub(super) use impl_futures;
