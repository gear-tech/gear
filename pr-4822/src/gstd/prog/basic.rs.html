19use crate::{ActorId, CodeId, MessageId};
20use gcore::errors::Result;
21use gstd_codegen::wait_create_program_for_reply;
22
23/// Create a new program from the already existing on-chain code identified by
24/// [`CodeId`].
25///
26/// The function returns an initial message identifier and a newly created
27/// program identifier.
28///
29/// The first argument is the code identifier (see [`CodeId`] for details). The
30/// second argument is an arbitrary byte sequence (also known as `salt`) that
31/// allows the creation of multiple programs from the same code. The third and
32/// last arguments are the initialization message's payload and value to be
33/// transferred to the newly created program.
34///
35/// # Examples
36///
37/// Create a new program from the provided code identifier:
38///
39/// ```
40/// use gstd::{CodeId, msg, prog};
41///
42/// #[unsafe(no_mangle)]
43/// extern "C" fn handle() {
44///     let code_id: CodeId = msg::load().expect("Unable to load");
45///     let (init_message_id, new_program_id) =
46///         prog::create_program_bytes(code_id, "salt", b"INIT", 0)
47///             .expect("Unable to create a program");
48///     msg::send_bytes(new_program_id, b"PING", 0).expect("Unable to send");
49/// }
50/// ```
51#[wait_create_program_for_reply]
52pub fn create_program_bytes(
53    code_id: CodeId,
54    salt: impl AsRef<[u8]>,
55    payload: impl AsRef<[u8]>,
56    value: u128,
57) -> Result<(MessageId, ActorId)> {
58    create_program_bytes_delayed(code_id, salt, payload, value, 0)
59}
60
61/// Same as [`create_program_bytes`], but creates a new program after the
62/// `delay` expressed in block count.
63///
64/// # Examples
65///
66/// Create a new program from the provided code identifier after 100 blocks and
67/// send a message to it after 200 blocks:
68///
69/// ```
70/// use gstd::{CodeId, msg, prog};
71///
72/// #[unsafe(no_mangle)]
73/// extern "C" fn handle() {
74///     let code_id: CodeId = msg::load().expect("Unable to load");
75///     let (init_message_id, new_program_id) =
76///         prog::create_program_bytes_delayed(code_id, "salt", b"INIT", 0, 100)
77///             .expect("Unable to create a program");
78///     msg::send_bytes_delayed(new_program_id, b"PING", 0, 200).expect("Unable to send");
79/// }
80/// ```
81pub fn create_program_bytes_delayed(
82    code_id: CodeId,
83    salt: impl AsRef<[u8]>,
84    payload: impl AsRef<[u8]>,
85    value: u128,
86    delay: u32,
87) -> Result<(MessageId, ActorId)> {
88    gcore::prog::create_program_delayed(code_id, salt.as_ref(), payload.as_ref(), value, delay)
89}
90
91/// Same as [`create_program_bytes`], but with an explicit gas limit.
92#[cfg(not(feature = "ethexe"))]
93#[wait_create_program_for_reply]
94pub fn create_program_bytes_with_gas(

96    salt: impl AsRef<[u8]>,
97    payload: impl AsRef<[u8]>,
98    gas_limit: u64,

100) -> Result<(MessageId, ActorId)> {

102}

119        gas_limit,
120        value,
121        delay,
122    )
