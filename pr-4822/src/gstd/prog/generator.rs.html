19//! Program generation module
20
21use crate::{ActorId, CodeId, MessageId, prog, util::with_optimized_encode};
22use gcore::errors::Result;
23use gstd_codegen::wait_create_program_for_reply;
24use scale_info::scale::{Encode, alloc::vec::Vec};
25
26/// Helper to create programs without setting the salt manually.
27pub struct ProgramGenerator(u64);
28
29// The only existing instance since there is no public ways to construct it.
30static mut PROGRAM_GENERATOR: ProgramGenerator = ProgramGenerator(0);
31
32impl ProgramGenerator {
33    // Prefix for not crossing with the user salt.
34    const UNIQUE_KEY: [u8; 14] = *b"salt_generator";
35
36    /// Return the salt needed to create a new program.
37    ///
38    /// Salt is an arbitrary byte sequence unique after every function call.
39    ///
40    /// # Examples
41    ///
42    /// ```
43    /// use gstd::prog::ProgramGenerator;
44    ///
45    /// #[unsafe(no_mangle)]
46    /// extern "C" fn handle() {
47    ///     let salt = ProgramGenerator::get_salt();
48    /// }
49    /// ```
50    pub fn get_salt() -> Vec<u8> {
51        // Provide salt uniqueness across all programs from other messages.
52        let message_id = crate::msg::id();
53
54        let creator_nonce;
55        unsafe {
56            creator_nonce = PROGRAM_GENERATOR.0.to_be_bytes();
57            PROGRAM_GENERATOR.0 = PROGRAM_GENERATOR.0.saturating_add(1);
58        }
59
60        [&Self::UNIQUE_KEY, message_id.as_ref(), &creator_nonce].concat()
61    }
62
63    /// Create a new program from the already existing on-chain code identified
64    /// by [`CodeId`].
65    ///
66    /// The function returns an initial message identifier and a newly created
67    /// program identifier.
68    ///
69    /// The first argument is the code identifier (see [`CodeId`] for details).
70    /// The second and third arguments are the initialization message's payload
71    /// and the value to be transferred to the new program.
72    ///
73    /// # Examples
74    ///
75    /// Create a new program from the provided code identifier:
76    ///
77    /// ```
78    /// use gstd::{CodeId, msg, prog::ProgramGenerator};
79    ///
80    /// #[unsafe(no_mangle)]
81    /// extern "C" fn handle() {
82    ///     let code_id: CodeId = msg::load().expect("Unable to load");
83    ///     let (init_message_id, new_program_id) =
84    ///         ProgramGenerator::create_program_bytes(code_id, b"INIT", 0)
85    ///             .expect("Unable to create a program");
86    ///     msg::send_bytes(new_program_id, b"PING", 0).expect("Unable to send");
87    /// }
88    /// ```
89    #[wait_create_program_for_reply(Self)]
90    pub fn create_program_bytes(
91        code_id: CodeId,
92        payload: impl AsRef<[u8]>,
93        value: u128,
94    ) -> Result<(MessageId, ActorId)> {
95        Self::create_program_bytes_delayed(code_id, payload, value, 0)
96    }
97
98    /// Same as [`create_program_bytes`](Self::create_program_bytes), but
99    /// creates a new program after the `delay` expressed in block count.
100    pub fn create_program_bytes_delayed(
101        code_id: CodeId,
102        payload: impl AsRef<[u8]>,
103        value: u128,
104        delay: u32,
105    ) -> Result<(MessageId, ActorId)> {
106        prog::create_program_bytes_delayed(code_id, Self::get_salt(), payload, value, delay)
107    }
108
109    /// Same as [`create_program_bytes`](Self::create_program_bytes), but with
110    /// an explicit gas limit.
111    #[cfg(not(feature = "ethexe"))]
112    #[wait_create_program_for_reply(Self)]
113    pub fn create_program_bytes_with_gas(
114        code_id: CodeId,
115        payload: impl AsRef<[u8]>,
116        gas_limit: u64,
117        value: u128,
118    ) -> Result<(MessageId, ActorId)> {
119        Self::create_program_bytes_with_gas_delayed(code_id, payload, gas_limit, value, 0)
120    }
121
122    /// Same as [`create_program_bytes_with_gas`](Self::create_program_bytes_with_gas), but
123    /// creates a new program after the `delay` expressed in block count.
124    #[cfg(not(feature = "ethexe"))]
125    pub fn create_program_bytes_with_gas_delayed(
126        code_id: CodeId,
127        payload: impl AsRef<[u8]>,
128        gas_limit: u64,
129        value: u128,
130        delay: u32,
131    ) -> Result<(MessageId, ActorId)> {
132        prog::create_program_bytes_with_gas_delayed(
133            code_id,
134            Self::get_salt(),
135            payload,
136            gas_limit,
137            value,
138            delay,
139        )
140    }
141
142    /// Same as [`create_program_bytes`](Self::create_program_bytes), but allows
143    /// initialize program with the encodable payload.
144    #[wait_create_program_for_reply(Self)]
145    pub fn create_program<E: Encode>(
146        code_id: CodeId,
147        payload: E,
148        value: u128,
149    ) -> Result<(MessageId, ActorId)> {
150        with_optimized_encode(payload, |buffer| {
151            Self::create_program_bytes(code_id, buffer, value)
152        })
153    }
154
155    /// Same as [`create_program`](Self::create_program), but creates a new
156    /// program after the `delay` expressed in block count.
157    pub fn create_program_delayed<E: Encode>(
158        code_id: CodeId,
159        payload: E,
160        value: u128,
161        delay: u32,
162    ) -> Result<(MessageId, ActorId)> {
163        with_optimized_encode(payload, |buffer| {
164            Self::create_program_bytes_delayed(code_id, buffer, value, delay)
165        })
166    }
167
168    /// Same as [`create_program`](Self::create_program), but with an explicit
169    /// gas limit.
170    #[cfg(not(feature = "ethexe"))]
171    #[wait_create_program_for_reply(Self)]
172    pub fn create_program_with_gas<E: Encode>(
173        code_id: CodeId,
174        payload: E,
175        gas_limit: u64,
176        value: u128,
177    ) -> Result<(MessageId, ActorId)> {
178        with_optimized_encode(payload, |buffer| {
179            Self::create_program_bytes_with_gas(code_id, buffer, gas_limit, value)
180        })
181    }
182
183    /// Same as [`create_program_with_gas`](Self::create_program_with_gas), but
184    /// creates a new program after the `delay` expressed in block count.
185    #[cfg(not(feature = "ethexe"))]
186    pub fn create_program_with_gas_delayed<E: Encode>(
187        code_id: CodeId,
188        payload: E,
189        gas_limit: u64,
190        value: u128,
191        delay: u32,
192    ) -> Result<(MessageId, ActorId)> {
193        with_optimized_encode(payload, |buffer| {
194            Self::create_program_bytes_with_gas_delayed(code_id, buffer, gas_limit, value, delay)
195        })
196    }
