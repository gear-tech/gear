32/// An auxiliary function that reduces gas consumption during payload encoding.
33pub(crate) fn with_optimized_encode<T, E: Encode>(payload: E, f: impl FnOnce(&[u8]) -> T) -> T {
34    struct ExternalBufferOutput<'a> {
35        buffer: &'a mut [MaybeUninit<u8>],
36        offset: usize,
37    }
38
39    impl Output for ExternalBufferOutput<'_> {
40        fn write(&mut self, bytes: &[u8]) {
41            // SAFETY: same as
42            // `MaybeUninit::write_slice(&mut self.buffer[self.offset..end_offset], bytes)`.
43            // This code transmutes `bytes: &[T]` to `bytes: &[MaybeUninit<T>]`. These types
44            // can be safely transmuted since they have the same layout. Then `bytes:
45            // &[MaybeUninit<T>]` is written to uninitialized memory via `copy_from_slice`.
46            let end_offset = self.offset + bytes.len();
47            let this = unsafe { self.buffer.get_unchecked_mut(self.offset..end_offset) };
48            this.copy_from_slice(unsafe {
49                transmute::<&[u8], &[core::mem::MaybeUninit<u8>]>(bytes)
50            });
51            self.offset = end_offset;
52        }
53    }
54
55    gcore::stack_buffer::with_byte_buffer(payload.encoded_size(), |buffer| {
56        let mut output = ExternalBufferOutput { buffer, offset: 0 };
57        payload.encode_to(&mut output);
58        let ExternalBufferOutput { buffer, offset } = output;
59        // SAFETY: same as `MaybeUninit::slice_assume_init_ref(&buffer[..offset])`.
60        // `ExternalBufferOutput` writes data to uninitialized memory. So we can take
61        // slice `&buffer[..offset]` and say that it was initialized earlier
62        // because the buffer from `0` to `offset` was initialized.
63        let payload = unsafe { &*(&buffer[..offset] as *const _ as *const [u8]) };
64        f(payload)
65    })
66}

68/// Generate a registry from given meta types and encode it to hex.
69pub fn to_hex_registry(meta_types: Vec<MetaType>) -> String {
70    let mut registry = Registry::new();
71    registry.register_types(meta_types);
72
73    let registry: PortableRegistry = registry.into();
74    hex::encode(registry.encode())
75}

77/// Convert a given reference to a raw pointer.
78pub fn to_wasm_ptr<T: AsRef<[u8]>>(bytes: T) -> *mut [i32; 2] {
79    Box::into_raw(Box::new([
80        bytes.as_ref().as_ptr() as _,
81        bytes.as_ref().len() as _,
82    ]))
83}
84
85/// Convert a given vector to a raw pointer and prevent its deallocating.
86///
87/// It operates similarly to [`to_wasm_ptr`] except that it consumes the input
88/// and make it leak by calling [`core::mem::forget`].
89pub fn to_leak_ptr(bytes: impl Into<Vec<u8>>) -> *mut [i32; 2] {
90    let bytes = bytes.into();
91    let ptr = Box::into_raw(Box::new([bytes.as_ptr() as _, bytes.len() as _]));
92    core::mem::forget(bytes);
93    ptr
94}
