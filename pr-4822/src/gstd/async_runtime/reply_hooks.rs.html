19use crate::MessageId;
20use alloc::boxed::Box;
21use hashbrown::HashMap;
22
23pub(crate) struct HooksMap(HashMap<MessageId, Box<dyn FnOnce()>>);
24
25impl HooksMap {
26    pub fn new() -> Self {
27        Self(HashMap::new())
28    }
29
30    /// Register hook to be executed when a reply for message_id is received.
31    pub(crate) fn register<F: FnOnce() + 'static>(&mut self, mid: MessageId, f: F) {
32        if self.0.contains_key(&mid) {
33            panic!("handle_reply: reply hook for this message_id is already registered");
34        }
35        self.0.insert(mid, Box::new(f));
36    }
37
38    /// Execute hook for message_id (if registered)
39    pub(crate) fn execute_and_remove(&mut self, message_id: MessageId) {
40        if let Some(f) = self.0.remove(&message_id) {
41            f();
42        }
43    }
44
45    /// Clear hook for message_id without executing it.
46    pub(crate) fn remove(&mut self, message_id: MessageId) {
47        self.0.remove(&message_id);
48    }
49}
