19//! Module for signal-management and waking concrete message based on reply
20//! received.
21
22use crate::{MessageId, prelude::Vec};
23use core::task::{Context, Waker};
24use gear_core_errors::ReplyCode;
25use hashbrown::HashMap;
26
27pub type Payload = Vec<u8>;
28
29#[derive(Debug)]
30pub(crate) enum ReplyPoll {
31    None,
32    Pending,
33    Some((Payload, ReplyCode)),
34}
35
36struct WakeSignal {
37    message_id: MessageId,
38    payload: Option<(Payload, ReplyCode)>,
39    waker: Option<Waker>,
40}
41
42pub(crate) struct WakeSignals {
43    signals: HashMap<MessageId, WakeSignal>,
44}
45
46impl WakeSignals {
47    pub fn new() -> Self {
48        Self {
49            signals: HashMap::new(),
50        }
51    }
52
53    pub fn register_signal(&mut self, waiting_reply_to: MessageId) {
54        let message_id = crate::msg::id();
55
56        self.signals.insert(
57            waiting_reply_to,
58            WakeSignal {
59                message_id,
60                payload: None,
61                waker: None,
62            },
63        );
64
65        crate::async_runtime::locks().lock(message_id, waiting_reply_to, Default::default());
66    }
67
68    pub fn record_reply(&mut self) {
69        if let Some(signal) = self
70            .signals
71            .get_mut(&crate::msg::reply_to().expect("Shouldn't be called with incorrect context"))
72        {
73            signal.payload = Some((
74                crate::msg::load_bytes().expect("Failed to load bytes"),
75                crate::msg::reply_code().expect("Shouldn't be called with incorrect context"),
76            ));
77
78            if let Some(waker) = &signal.waker {
79                waker.wake_by_ref();
80            }
81
82            crate::exec::wake(signal.message_id).expect("Failed to wake the message")
83        } else {
84            crate::debug!(
85                "A message has received a reply though it wasn't to receive one, or a processed message has received a reply"
86            );
87        }
88    }
89
90    pub fn waits_for(&self, reply_to: MessageId) -> bool {
91        self.signals.contains_key(&reply_to)
92    }
93
94    pub fn poll(&mut self, reply_to: MessageId, cx: &mut Context<'_>) -> ReplyPoll {
95        match self.signals.remove(&reply_to) {
96            None => ReplyPoll::None,
97            Some(mut signal @ WakeSignal { payload: None, .. }) => {
98                signal.waker = Some(cx.waker().clone());
99                self.signals.insert(reply_to, signal);
100                ReplyPoll::Pending
101            }
102            Some(WakeSignal {
103                payload: Some(payload),
104                ..
105            }) => ReplyPoll::Some(payload),
106        }
107    }
108}
