19//! The `gstd` default prelude. Re-imports default `std` modules and traits.
20//! `std` can be safely replaced to `gstd` in the Rust programs.
21
22// Reexports from Rust's libraries
23
24#[cfg(not(feature = "ethexe"))]
25pub use crate::ReservationIdExt;
26pub use crate::{dbg, static_mut, static_ref};
27pub use ::alloc::{
28    borrow,
29    borrow::ToOwned,
30    boxed,
31    boxed::Box,
32    fmt, format, rc, str, string,
33    string::{String, ToString},
34    vec,
35    vec::Vec,
36};
37pub use core::{
38    any, array, ascii, assert_eq, assert_ne, cell, char, clone, cmp, convert, debug_assert,
39    debug_assert_eq, debug_assert_ne, default, future, hash, hint, iter, marker, matches, mem, num,
40    ops, option, panic, pin, prelude::rust_2021::*, primitive, ptr, result, slice, task, time,
41    todo, unimplemented, unreachable, write, writeln,
42};

44/// Collection types.
45///
46/// See [`alloc::collections`] & [`hashbrown`].
47///
48/// [`alloc::collections`]: ::alloc::collections
49pub mod collections {
50    pub use ::alloc::collections::*;
51    pub use ::hashbrown::{HashMap, HashSet, hash_map, hash_set};
52
53    /// Reexports from [`hashbrown`].
54    pub mod hashbrown {
55        pub use hashbrown::{Equivalent, TryReserveError};
56    }
57}
58/// Utilities related to FFI bindings.
59///
60/// See [`alloc::ffi`] & [`core::ffi`].
61///
62/// [`alloc::ffi`]: ::alloc::ffi
63pub mod ffi {
64    pub use ::alloc::ffi::{CString, FromVecWithNulError, IntoStringError, NulError};
65    pub use core::ffi::*;
66}
