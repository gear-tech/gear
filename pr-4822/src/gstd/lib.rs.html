19//! Standard library for use in Gear programs.
20//!
21//! This library should be used as a standard library when writing Gear
22//! programs. Compared to [`gcore`](https://docs.gear.rs/gcore/) crate,
23//! this library provides higher-level primitives that allow you to develop more
24//! complex dApps. Choose this library if you are ready to spend more gas but
25//! receive refined code.
26//!
27//! `gstd` crate provides many advanced tools for a developer, such as
28//! asynchronous programming primitives, arbitrary types encoding/decoding,
29//! providing convenient instruments for creating programs from programs, etc.
30//!
31//! # Minimum supported Rust version
32//! This crate requires **Rust >= 1.81** due to the implementation of the panic
33//! handler in the stable version.
34//!
35//! # Crate features
36#![cfg_attr(
37    feature = "document-features",
38    cfg_attr(doc, doc = ::document_features::document_features!())
39)]
40//! # Examples
41//!
42//! Decode input payload using a custom type:
43//!
44//! ```
45//! # const _: &'static str = stringify! {
46//! #![no_std]
47//! # };
48//!
49//! use gstd::{msg, prelude::*};
50//!
51//! #[derive(Decode, Encode, TypeInfo)]
52//! #[codec(crate = gstd::codec)]
53//! #[scale_info(crate = gstd::scale_info)]
54//! struct Payload {
55//!     question: String,
56//!     answer: u8,
57//! }
58//!
59//! #[unsafe(no_mangle)]
60//! extern "C" fn handle() {
61//!     let payload: Payload = msg::load().expect("Unable to decode payload");
62//!     if payload.question == "life-universe-everything" {
63//!         msg::reply(payload.answer, 0).expect("Unable to reply");
64//!     }
65//! }
66//! ```
67//!
68//! Asynchronous program example.
69//!
70//! It sends empty messages to three addresses and waits for at least two
71//! replies ("approvals") during initialization. When invoked, it handles only
72//! `PING` messages and sends empty messages to the three addresses, and waits
73//! for just one approval. If approval is obtained, the program replies with
74//! `PONG`.
75//!
76//! ```ignored
77//! # const _: &'static str = stringify! {
78//! #![no_std]
79//! # };
80//! use futures::future;
81//! use gstd::{msg, prelude::*, ActorId};
82//!
83//! static mut APPROVERS: [ActorId; 3] = [ActorId::zero(); 3];
84//!
85//! #[derive(Debug, Decode, TypeInfo)]
86//! #[codec(crate = gstd::codec)]
87//! #[scale_info(crate = gstd::scale_info)]
88//! pub struct Input {
89//!     pub approvers: [ActorId; 3],
90//! }
91//!
92//! #[gstd::async_init]
93//! async fn init() {
94//!     let payload: Input = msg::load().expect("Failed to decode input");
95//!     unsafe { APPROVERS = payload.approvers };
96//!
97//!     let mut requests: Vec<_> = unsafe { APPROVERS }
98//!         .iter()
99//!         .map(|addr| msg::send_bytes_for_reply(*addr, b"", 0, 0))
100//!         .collect::<Result<_, _>>()
101//!         .unwrap();
102//!
103//!     let mut threshold = 0;
104//!     while !requests.is_empty() {
105//!         let (.., remaining) = future::select_all(requests).await;
106//!         threshold += 1;
107//!         if threshold >= 2 {
108//!             break;
109//!         }
110//!         requests = remaining;
111//!     }
112//! }
113//!
114//! #[gstd::async_main]
115//! async fn main() {
116//!     let message = msg::load_bytes().expect("Failed to load payload bytes");
117//!     if message != b"PING" {
118//!         return;
119//!     }
120//!
121//!     let requests: Vec<_> = unsafe { APPROVERS }
122//!         .iter()
123//!         .map(|addr| msg::send_bytes_for_reply(*addr, b"", 0, 0))
124//!         .collect::<Result<_, _>>()
125//!         .unwrap();
126//!
127//!     _ = future::select_all(requests).await;
128//!     msg::reply(b"PONG", 0).expect("Unable to reply");
129//! }
130//! # fn main() {}
131//! ```

185    /// Panic
186    ///
187    /// Can be used to pass some data to error reply payload.
188    ///
189    /// Syscall this function uses is completely free in terms of gas usage.
190    ///
191    /// # Examples
192    ///
193    /// ```
194    /// use gstd::ext;
195    ///
196    /// #[unsafe(no_mangle)]
197    /// extern "C" fn handle() {
198    ///     ext::panic((1, "important data"));
199    /// }
200    /// ```
201    pub fn panic<T: Encode>(data: T) -> ! {
202        panic_bytes(data.encode())
203    }
204
205    /// Panic
206    ///
207    /// Can be used to pass some data to error reply payload.
208    ///
209    /// Syscall this function uses is completely free in terms of gas usage.
210    ///
211    /// # Examples
212    ///
213    /// ```
214    /// use gstd::ext;
215    ///
216    /// #[unsafe(no_mangle)]
217    /// extern "C" fn handle() {
218    ///     ext::panic_bytes([0, 1, 2, 3]);
219    /// }
220    /// ```
221    pub fn panic_bytes<T: AsRef<[u8]>>(data: T) -> ! {
222        gcore::ext::panic(data.as_ref())
223    }
224}
