19//! This module gives mechanism of waking for async lockers.
20
21use crate::{MessageId, collections::VecDeque};
22use core::cell::UnsafeCell;
23
24// Option<VecDeque> to make new `const fn`
25pub struct AccessQueue(UnsafeCell<Option<VecDeque<MessageId>>>);
26
27impl AccessQueue {
28    pub fn enqueue(&self, message_id: MessageId) {
29        let inner = unsafe { &mut *self.0.get() };
30
31        let vec_deque = inner.get_or_insert_with(VecDeque::new);
32        vec_deque.push_back(message_id);
33    }
34
35    pub fn dequeue(&self) -> Option<MessageId> {
36        let inner = unsafe { &mut *self.0.get() };
37
38        inner.as_mut().and_then(|v| v.pop_front())
39    }
40
41    pub fn contains(&self, message_id: &MessageId) -> bool {
42        let inner = unsafe { &*self.0.get() };
43
44        inner.as_ref().is_some_and(|v| v.contains(message_id))
45    }
46
47    pub fn len(&self) -> usize {
48        let inner = unsafe { &*self.0.get() };
49
50        inner.as_ref().map_or(0, |v| v.len())
51    }
52
53    pub fn first(&self) -> Option<&MessageId> {
54        let inner = unsafe { &*self.0.get() };
55
56        inner.as_ref().and_then(|v| v.front())
57    }
58
59    pub const fn new() -> Self {
60        AccessQueue(UnsafeCell::new(None))
61    }
62}
