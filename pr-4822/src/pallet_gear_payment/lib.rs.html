56/// A wrapper around the `pallet_transaction_payment::ChargeTransactionPayment`.
57/// Adjusts `DispatchInfo` to reflect custom fee add-ons.
58#[derive(Encode, Decode, Clone, Eq, PartialEq, TypeInfo)]
59pub struct CustomChargeTransactionPayment<T: Config>(ChargeTransactionPayment<T>);

64    CallOf<T>: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
65{
66    pub fn from(tip: BalanceOf<T>) -> Self {
67        Self(ChargeTransactionPayment::<T>::from(tip))

82// Follow pallet-transaction-payment implementation
83impl<T: Config> SignedExtension for CustomChargeTransactionPayment<T>
84where
85    T: TypeInfo,
86    BalanceOf<T>: Send + Sync + From<u64> + FixedPointOperand,
87    CallOf<T>: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
88{
89    const IDENTIFIER: &'static str = <ChargeTransactionPayment<T> as SignedExtension>::IDENTIFIER;
90    type AccountId = <ChargeTransactionPayment<T> as SignedExtension>::AccountId;
91    type Call = CallOf<T>;
92    type AdditionalSigned = <ChargeTransactionPayment<T> as SignedExtension>::AdditionalSigned;
93    type Pre = <ChargeTransactionPayment<T> as SignedExtension>::Pre;
94    fn additional_signed(&self) -> Result<Self::AdditionalSigned, TransactionValidityError> {
95        self.0.additional_signed()
96    }
97
98    fn validate(
99        &self,
100        who: &Self::AccountId,
101        call: &Self::Call,
102        info: &DispatchInfoOf<Self::Call>,
103        len: usize,
104    ) -> TransactionValidity {
105        // Override DispatchInfo struct for call variants exempted from weight fee multiplication
106        let info = Self::pre_dispatch_info(call, info);
107        let payer = Self::fee_payer_account(call, who);
108        self.0.validate(&payer, call, &info, len)
109    }
110
111    fn pre_dispatch(
112        self,
113        who: &Self::AccountId,
114        call: &Self::Call,
115        info: &DispatchInfoOf<Self::Call>,
116        len: usize,
117    ) -> Result<Self::Pre, TransactionValidityError> {
118        // Override DispatchInfo struct for call variants exempted from weight fee multiplication
119        let info = Self::pre_dispatch_info(call, info);
120        // Replace payer if delegated
121        let payer = Self::fee_payer_account(call, who);
122        self.0.pre_dispatch(&payer, call, &info, len)
123    }
124
125    fn post_dispatch(
126        maybe_pre: Option<Self::Pre>,
127        info: &DispatchInfoOf<Self::Call>,
128        post_info: &PostDispatchInfoOf<Self::Call>,
129        len: usize,
130        result: &sp_runtime::DispatchResult,
131    ) -> Result<(), TransactionValidityError> {
132        // There is no easy way to modify the original `DispatchInfo` struct similarly
133        // it's done in `pre_dispatch()` because a call is not supplied.
134        // However, we can just leave it as is and yet get the correct fee refund if any:
135        //   - if `None` is returned as the actual weight (i.e. worst case) nothing is supposed
136        //   to be refunded anyway and saturating subtraction guarantees we don't have overflow;
137        //   - if `post_info` has `Some(actual_weight)`, the minimum of it and `info.weight` will
138        //   be used to calculate the correct fee so it is just our responsibility to do
139        //   weight normalization before returning it from the extrinsic.
140        //
141        // TODO: still think of a more robust way to deal with fee refunds
142        <ChargeTransactionPayment<T> as SignedExtension>::post_dispatch(
143            maybe_pre, info, post_info, len, result,
144        )
145    }
146}

148impl<T: Config> CustomChargeTransactionPayment<T>
149where
150    CallOf<T>: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
151{
152    fn pre_dispatch_info<'a>(
153        call: &'a <T as frame_system::Config>::RuntimeCall,
154        info: &'a DispatchInfoOf<<T as frame_system::Config>::RuntimeCall>,
155    ) -> Cow<'a, DispatchInfoOf<<T as frame_system::Config>::RuntimeCall>> {
156        // If the call is not subject to fee multiplication, divide weight by fee multiplier.
157        // This action will effectively be cancelled out at the time the fee is calculated.
158        //
159        // TODO: consider reimplementing.
160        // This procedure does introduce a rounding error ùûÆ =  w - ‚é£w / m‚é¶‚ãÖm
161        // However, we argue that such error is negligible:
162        // - the rounding error can never exceed `m` (multiplier). Order of `w` (weight) is
163        // usually not less than 10^8 while the fee multiplier should not be greater than 10^3.
164        // Therefore the rounding error shouln't exceed 0.001% in the worst case.
165        // Note: this only applies to calls that do not affect message queue, that is are
166        // relatively rare. Still, a better solution can be found.
167        if !T::ExtraFeeCallFilter::contains(call) {
168            let multiplier = TransactionPayment::<T>::next_fee_multiplier();
169            if multiplier > Multiplier::saturating_from_integer(1) {
170                let mut info: DispatchInfo = *info;
171                info.weight = Weight::from_parts(
172                    multiplier
173                        .reciprocal() // take inverse
174                        .unwrap_or_else(Multiplier::max_value)
175                        .saturating_mul_int(info.weight.ref_time()),
176                    0,
177                );
178                Cow::Owned(info)
179            } else {
180                Cow::Borrowed(info)
181            }
182        } else {
183            Cow::Borrowed(info)
184        }
185    }
186
187    fn fee_payer_account<'a>(
188        call: &'a <T as frame_system::Config>::RuntimeCall,
189        who: &'a <T as frame_system::Config>::AccountId,
190    ) -> Cow<'a, <T as frame_system::Config>::AccountId> {
191        // Check if the extrinsic being called allows to charge fee payment to another account.
192        // The only such call at the moment is `GearVoucher::call`.
193        if let Some(acc) = T::DelegateFee::delegate_fee(call, who) {
194            Cow::Owned(acc)
195        } else {
196            Cow::Borrowed(who)
197        }
198    }
199}

201/// Custom fee multiplier which looks at the message queue size to increase weight fee
202pub struct GearFeeMultiplier<T, S>(sp_std::marker::PhantomData<(T, S)>);
203
204impl<T, S> Convert<Multiplier, Multiplier> for GearFeeMultiplier<T, S>
205where
206    T: Config,
207    S: Get<u128>,
208{
209    fn convert(_previous: Multiplier) -> Multiplier {
210        let len_step = S::get().max(1); // Avoiding division by 0.
211
212        let queue_len: u128 = QueueOf::<T>::len().saturated_into();
213        let multiplier = queue_len.saturating_div(len_step).saturating_add(1);
214        Multiplier::saturating_from_integer(multiplier)
215    }
216}

226    fn min() -> Multiplier {

229    fn target() -> Perquintill {
230        Default::default()
231    }
232    fn variability() -> Multiplier {
233        Default::default()
234    }
235}

237impl<T: Config> Pallet<T> {
238    /// Modification of the `pallet_transaction_payment::Pallet<T>::query_info()`
239    /// that is aware of the transaction fee customization based on a specific call
240    pub fn query_info<
241        Extrinsic: sp_runtime::traits::Extrinsic + GetDispatchInfo + ExtractCall<CallOf<T>>,
242    >(
243        unchecked_extrinsic: Extrinsic,
244        len: u32,
245    ) -> RuntimeDispatchInfo<BalanceOf<T>>
246    where
247        CallOf<T>: Dispatchable<Info = DispatchInfo>,
248        BalanceOf<T>: FixedPointOperand,
249    {
250        let DispatchInfo {
251            weight,
252            class,
253            pays_fee,
254        } = <Extrinsic as GetDispatchInfo>::get_dispatch_info(&unchecked_extrinsic);
255
256        let partial_fee = if unchecked_extrinsic.is_signed().unwrap_or(false) {
257            let call: CallOf<T> =
258                <Extrinsic as ExtractCall<CallOf<T>>>::extract_call(&unchecked_extrinsic);
259            // If call is exempted from weight multiplication pre-divide it with the fee multiplier
260            let adjusted_weight = if !T::ExtraFeeCallFilter::contains(&call) {
261                Weight::from_parts(
262                    TransactionPayment::<T>::next_fee_multiplier()
263                        .reciprocal()
264                        .unwrap_or_else(Multiplier::max_value)
265                        .saturating_mul_int(weight.ref_time()),
266                    0,
267                )
268            } else {
269                weight
270            };
271            TransactionPayment::<T>::compute_fee(
272                len,
273                &DispatchInfo {
274                    weight: adjusted_weight,
275                    class,
276                    pays_fee,
277                },
278                0u32.into(),
279            )
280        } else {
281            // Unsigned extrinsics have no partial fee.
282            0u32.into()
283        };
284
285        RuntimeDispatchInfo {
286            weight,
287            class,
288            partial_fee,
289        }
290    }
291
292    /// Modification of the `pallet_transaction_payment::Pallet<T>::query_fee_details()`
293    pub fn query_fee_details<
294        Extrinsic: sp_runtime::traits::Extrinsic + GetDispatchInfo + ExtractCall<CallOf<T>>,
295    >(
296        unchecked_extrinsic: Extrinsic,
297        len: u32,
298    ) -> FeeDetails<BalanceOf<T>>
299    where
300        CallOf<T>: Dispatchable<Info = DispatchInfo>,
301        BalanceOf<T>: FixedPointOperand,
302    {
303        let DispatchInfo {
304            weight,
305            class,
306            pays_fee,
307        } = <Extrinsic as GetDispatchInfo>::get_dispatch_info(&unchecked_extrinsic);
308
309        let tip = 0u32.into();
310
311        if unchecked_extrinsic.is_signed().unwrap_or(false) {
312            let call: CallOf<T> =
313                <Extrinsic as ExtractCall<CallOf<T>>>::extract_call(&unchecked_extrinsic);
314            let adjusted_weight = if !T::ExtraFeeCallFilter::contains(&call) {
315                Weight::from_parts(
316                    TransactionPayment::<T>::next_fee_multiplier()
317                        .reciprocal()
318                        .unwrap_or_else(Multiplier::max_value)
319                        .saturating_mul_int(weight.ref_time()),
320                    0,
321                )
322            } else {
323                weight
324            };
325            TransactionPayment::<T>::compute_fee_details(
326                len,
327                &DispatchInfo {
328                    weight: adjusted_weight,
329                    class,
330                    pays_fee,
331                },
332                tip,
333            )
334        } else {
335            // Unsigned extrinsics have no inclusion fee.
336            FeeDetails {
337                inclusion_fee: None,
338                tip,
339            }
340        }
341    }
342}
