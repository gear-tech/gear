65impl LazyPagesStorage for PagesStorage {
66    fn page_exists(&self, mut key: &[u8]) -> bool {
67        let PageKey {
68            program_id, page, ..
69        } = PageKey::decode_all(&mut key).expect("Invalid key");
70
71        ProgramsStorageManager::program_page(program_id, page).is_some()
72    }
73
74    fn load_page(&mut self, mut key: &[u8], buffer: &mut [u8]) -> Option<u32> {
75        let PageKey {
76            program_id, page, ..
77        } = PageKey::decode_all(&mut key).expect("Invalid key");
78
79        ProgramsStorageManager::program_page(program_id, page).map(|page_buf| {
80            buffer.copy_from_slice(page_buf.as_ref());
81            page_buf.len() as u32
82        })
83    }

160    /// Run next block.
161    ///
162    /// Block execution model is the following:
163    /// - increase the block number, update the timestamp
164    /// - process tasks from the task pool
165    /// - process messages in the queue.
166    ///
167    /// The system is always initialized with a 0 block number. Current block
168    /// number in the system is the number of the already executed block,
169    /// therefore block execution starts with a block info update (block
170    /// number increase, timestamp update). For example, if current block
171    /// number is 2, it means that messages and tasks on 2 were executed, so
172    /// the method goes to block number 3 and executes tasks and messages for
173    /// the updated block number.
174    ///
175    /// Task processing basically tries to execute the scheduled to the specific
176    /// block tasks:
177    /// - delayed sending
178    /// - waking message
179    /// - removing from the mailbox
180    /// - removing reservations
181    /// - removing stalled wait message.
182    ///
183    /// Messages processing executes messages until either queue becomes empty
184    /// or block gas allowance is fully consumed.
185    pub fn run_next_block(&self) -> BlockRunResult {
186        self.run_next_block_with_allowance(GAS_ALLOWANCE)
187    }
188
189    /// Runs blocks same as [`Self::run_next_block`], but with limited
190    /// allowance.
191    pub fn run_next_block_with_allowance(&self, allowance: Gas) -> BlockRunResult {
192        if allowance > GAS_ALLOWANCE {
193            usage_panic!(
194                "Provided allowance more than allowed limit of {GAS_ALLOWANCE}. \
195                Please, provide an allowance less than or equal to the limit."
196            );
197        }
198
199        self.0.borrow_mut().run_new_block(allowance)
200    }

202    /// Runs blocks same as [`Self::run_next_block`], but executes blocks to
203    /// block number `bn` including it.
204    pub fn run_to_block(&self, bn: u32) -> Vec<BlockRunResult> {
205        let mut manager = self.0.borrow_mut();
206
207        let mut current_block = manager.block_height();
208        if current_block > bn {
209            usage_panic!("Can't run blocks until bn {bn}, as current bn is {current_block}");
210        }
211
212        let mut ret = Vec::with_capacity((bn - current_block) as usize);
213        while current_block != bn {
214            let res = manager.run_new_block(GAS_ALLOWANCE);
215            ret.push(res);
216
217            current_block = manager.block_height();
218        }
219
220        ret
221    }

223    /// Runs `amount` of blocks only with processing task pool, without
224    /// processing the message queue.
225    pub fn run_scheduled_tasks(&self, amount: u32) -> Vec<BlockRunResult> {
226        let mut manager = self.0.borrow_mut();
227        let block_height = manager.block_height();
228
229        (block_height..block_height + amount)
230            .map(|_| {
231                let block_info = manager.blocks_manager.next_block();
232                let next_block_number = block_info.height;
233                manager.process_tasks(next_block_number);
234
235                let log = mem::take(&mut manager.log)
236                    .into_iter()
237                    .map(CoreLog::from)
238                    .collect();
239                BlockRunResult {
240                    block_info,
241                    gas_allowance_spent: GAS_ALLOWANCE - manager.gas_allowance,
242                    log,
243                    ..Default::default()
244                }
245            })
246            .collect()
247    }

421    /// Calculate reply that would be received when sending
422    /// message to initialized program with any of `Program::send*` methods.
423    pub fn calculate_reply_for_handle(
424        &self,
425        origin: impl Into<ProgramIdWrapper>,
426        destination: impl Into<ProgramIdWrapper>,
427        payload: impl Into<Vec<u8>>,
428        gas_limit: u64,
429        value: Value,
430    ) -> Result<ReplyInfo, String> {
431        let mut manager_mut = self.0.borrow_mut();
432
433        // Enter the overlay mode
434        manager_mut.enable_overlay();
435
436        // Clear the queue
437        manager_mut.dispatches.clear();
438
439        let origin = origin.into().0;
440        let destination = destination.into().0;
441        let payload = payload
442            .into()
443            .try_into()
444            .expect("failed to convert payload to limited payload");
445
446        // Prepare the message
447        let block_number = manager_mut.block_height() + 1;
448        let message = Message::new(
449            MessageId::generate_from_user(
450                block_number,
451                origin,
452                manager_mut.fetch_inc_message_nonce() as u128,
453            ),
454            origin,
455            destination,
456            payload,
457            Some(gas_limit),
458            value,
459            None,
460        );
461
462        if !ProgramsStorageManager::is_active_program(destination) {
463            usage_panic!("Actor with {destination} id is not active");
464        }
465
466        let dispatch = Dispatch::new(DispatchKind::Handle, message);
467
468        // Validate and route the dispatch
469        let message_id = manager_mut.validate_and_route_dispatch(dispatch);
470
471        // Run queue for reply to the `message_id`.
472        let block_config = manager_mut.block_config();
473
474        while let Some(dispatch) = manager_mut.dispatches.pop_front() {
475            // For testing purposes, we set the gas allowance to the maximum for each
476            // message
477            manager_mut.gas_allowance = GAS_ALLOWANCE;
478            // No need to check the flag after the execution, as we give infinite
479            // allowance for the reply calculation.
480            manager_mut.messages_processing_enabled = true;
481
482            // Process the dispatch and obtain the journal.
483            let journal = manager_mut.process_dispatch(&block_config, dispatch);
484
485            // Search for the reply in the journal.
486            for note in &journal {
487                let JournalNote::SendDispatch { dispatch, .. } = note else {
488                    continue;
489                };
490
491                if let Some(code) = dispatch
492                    .reply_details()
493                    .map(ReplyDetails::into_parts)
494                    .and_then(|(replied_to, code)| replied_to.eq(&message_id).then_some(code))
495                {
496                    // Before any return from the function, overlay must be disabled.
497                    manager_mut.disable_overlay();
498
499                    return Ok(ReplyInfo {
500                        payload: dispatch.payload_bytes().to_vec(),
501                        value: dispatch.value(),
502                        code,
503                    });
504                }
505            }
506
507            // As long as no reply was found, we need to handle the journal.
508            core_processor::handle_journal(journal, &mut *manager_mut);
509        }
510
511        // Before any return from the function, overlay must be disabled.
512        manager_mut.disable_overlay();
513
514        Err(String::from("Queue is empty, but reply wasn't found"))
515    }
516}

518impl Drop for System {
519    fn drop(&mut self) {
520        // Uninitialize
521        SYSTEM_INITIALIZED.with_borrow_mut(|initialized| *initialized = false);
522        let manager = self.0.borrow();
523        manager.gas_tree.clear();
524        manager.mailbox.clear();
525        manager.task_pool.clear();
526        manager.waitlist.clear();
527        manager.blocks_manager.reset();
528        manager.bank.clear();
529        manager.nonce_manager.reset();
530        manager.dispatches.clear();
531        manager.dispatches_stash.clear();
532
533        // Clear programs and accounts storages
534        ProgramsStorageManager::clear();
535        Accounts::clear();
536    }
537}

702            .expect("Failed to calculate reply for handle");

704            reply_info.code,
705            ReplyCode::Success(SuccessReplyReason::Auto)
706        );
707
708        // Check that overlay didn't change the state

710            sys.balance_of(pid),
711            program_balance_before_overlay + storing_value
712        );
713        assert_eq!(
714            sys.balance_of(DEFAULT_USER_ALICE),
715            alice_expected_balance_after_msg1
716        );
717        let mailbox = sys.get_mailbox(DEFAULT_USER_ALICE);

719            .dest(DEFAULT_USER_ALICE)
720            .payload_bytes(b"send");

736            alice_expected_balance_after_msg1 - block_result.spent_value() + storing_value;
