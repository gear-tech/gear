22impl ExtManager {
23    /// Spends given amount of gas from given `MessageId` in `GasTree`.
24    ///
25    /// Represents logic of burning gas by transferring gas from
26    /// current `GasTree` owner to actual block producer.
27    pub(crate) fn spend_gas(&mut self, id: MessageId, amount: u64) {
28        if amount.is_zero() {
29            return;
30        }
31
32        self.gas_tree.spend(id, amount).unwrap_or_else(|e| {
33            let err_msg = format!(
34                "spend_gas: failed spending gas. Message id - {id}, amount - {amount}. Got error - {e:?}"
35            );
36
37            unreachable!("{err_msg}");
38        });
39
40        let (external, multiplier, _) = self.gas_tree.get_origin_node(id).unwrap_or_else(|e| {
41            let err_msg = format!(
42                "spend_gas: failed getting origin node for the current one. Message id - {id}, Got error - {e:?}"
43            );
44            unreachable!("{err_msg}");
45        });
46
47        self.bank.spend_gas(external.cast(), amount, multiplier)
48    }
49
50    pub(crate) fn spend_burned(&mut self, id: MessageId, amount: u64) {
51        self.gas_burned
52            .entry(id)
53            .and_modify(|v| *v = v.saturating_add(amount))
54            .or_insert(amount);
55        self.spend_gas(id, amount);
56    }

58    pub(crate) fn cost_by_storage_type(storage_type: StorageType) -> u64 {
59        // Cost per block based on the storage used for holding
60        let schedule = Schedule::default();
61        let RentWeights {
62            waitlist,
63            dispatch_stash,
64            reservation,
65            mailbox,
66            ..
67        } = schedule.rent_weights;
68        match storage_type {
69            StorageType::Code => todo!("#646"),
70            StorageType::Waitlist => waitlist.ref_time,
71            StorageType::Mailbox => mailbox.ref_time,
72            StorageType::DispatchStash => dispatch_stash.ref_time,
73            StorageType::Program => todo!("#646"),
74            StorageType::Reservation => reservation.ref_time,
75        }
76    }

78    pub(crate) fn consume_and_retrieve(&mut self, id: impl Origin) {
79        let id_origin = id.into_origin();
80        let outcome = self.gas_tree.consume(id_origin).unwrap_or_else(|e| {
81            let err_msg = format!(
82                "consume_and_retrieve: failed consuming the rest of gas. Got error - {e:?}"
83            );
84
85            unreachable!("{err_msg}")
86        });
87
88        if let Some((imbalance, multiplier, external)) = outcome {
89            let gas_left = imbalance.peek();
90            log::debug!(
91                "Consumed message {id_origin}. Unreserving {gas_left} (gas) from {external:?}",
92            );
93
94            if !gas_left.is_zero() {
95                self.bank
96                    .withdraw_gas(external.cast(), gas_left, multiplier);
97            }
98        }
99    }

101    pub(crate) fn charge_for_hold(
102        &mut self,
103        id: impl Origin,
104        hold_interval: Interval<BlockNumber>,
105        storage_type: StorageType,
106    ) {
107        let id: MessageId = id.cast();
108        let current = self.block_height();
109
110        // Deadline of the task.
111        let deadline = hold_interval.finish.saturating_add(RESERVE_FOR);
112
113        // The block number, which was the last paid for hold.
114        //
115        // Outdated tasks can end up being store for free - this case has to be
116        // controlled by a correct selection of the `ReserveFor` constant.
117        let paid_until = current.min(deadline);
118
119        // holding duration
120        let duration: u64 = paid_until.saturating_sub(hold_interval.start).into();
121
122        // Cost per block based on the storage used for holding
123        let cost = Self::cost_by_storage_type(storage_type);
124
125        let amount = storage_type.try_into().map_or_else(
126            |_| duration.saturating_mul(cost),
127            |lock_id| {
128                let prepaid = self.gas_tree.unlock_all(id, lock_id).unwrap_or_else(|e| {
129                    let err_msg = format!(
130                        "charge_for_hold: failed unlocking locked gas.
131                        Got error - {e:?}"
132                    );
133
134                    unreachable!("{err_msg}");
135                });
136
137                prepaid.min(duration.saturating_mul(cost))
138            },
139        );
140
141        if !amount.is_zero() {
142            self.spend_gas(id, amount);
143        }
144    }
