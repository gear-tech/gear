50    fn message_dispatched(
51        &mut self,
52        message_id: MessageId,
53        _source: ActorId,
54        outcome: DispatchOutcome,
55    ) {
56        match outcome {
57            DispatchOutcome::MessageTrap { program_id, trap } => {
58                log::debug!("ðŸª¤ Program {program_id} terminated with a trap: {trap}");
59                self.failed.insert(message_id);
60            }
61            DispatchOutcome::NoExecution => {
62                self.not_executed.insert(message_id);
63            }
64            DispatchOutcome::Success | DispatchOutcome::Exit { .. } => {
65                self.succeed.insert(message_id);
66            }
67            DispatchOutcome::InitFailure {
68                program_id, origin, ..
69            } => {
70                self.init_failure(program_id, origin);
71                self.failed.insert(message_id);
72            }
73            DispatchOutcome::InitSuccess { program_id, .. } => {
74                self.init_success(program_id);
75                self.succeed.insert(message_id);
76            }
77        }
78    }

80    fn gas_burned(&mut self, message_id: MessageId, amount: u64) {
81        log::debug!("Burned: {amount:?} from: {message_id:?}");
82
83        self.gas_allowance = self.gas_allowance.saturating_sub(amount);
84        self.spend_burned(message_id, amount);

87    fn exit_dispatch(&mut self, id_exited: ActorId, value_destination: ActorId) {
88        log::debug!(
89            "Exit dispatch: id_exited = {id_exited}, value_destination = {value_destination}"
90        );
91
92        self.clean_waitlist(id_exited);
93        self.remove_gas_reservation_map(id_exited);
94
95        ProgramsStorageManager::modify_program(id_exited, |program| {
96            let program = program
97                .unwrap_or_else(|| panic!("Can't find existing program {id_exited:?}"))
98                .as_primary_program_mut();
99
100            if !program.is_active() {
101                // Guaranteed to be called only on active program
102                unreachable!(
103                    "JournalHandler::exit_dispatch: failed to exit active program. \
104                Program - {id_exited}, actual program - {program:?}"
105                );
106            }
107
108            *program = Program::Exited(value_destination);
109        });
110
111        let value = Accounts::balance(id_exited);
112        if value != 0 {
113            Accounts::transfer(id_exited, value_destination, value, false);
114        }
115    }

121    fn send_dispatch(
122        &mut self,
123        message_id: MessageId,
124        dispatch: Dispatch,
125        delay: u32,
126        reservation: Option<ReservationId>,
127    ) {
128        let to_user = ProgramsStorageManager::is_user(dispatch.destination())
129            && !self.no_code_program.contains(&dispatch.destination());
130        if delay > 0 {
131            log::debug!(
132                "[{message_id}] new delayed dispatch#{} with delay for {delay} blocks",
133                dispatch.id()
134            );
135
136            self.send_delayed_dispatch(message_id, dispatch, delay, to_user, reservation);
137            return;
138        }
139
140        log::debug!("[{message_id}] new dispatch#{}", dispatch.id());
141
142        let source = dispatch.source();
143        let is_program = ProgramsStorageManager::is_program(dispatch.destination())
144            || self.no_code_program.contains(&dispatch.destination());
145
146        if is_program {
147            let gas_limit = dispatch.gas_limit();
148            log::debug!(
149                "Sending message {:?} from {:?} with gas limit {:?}",
150                dispatch.message(),
151                message_id,
152                gas_limit,
153            );
154
155            // It's necessary to deposit value so the source would have enough
156            // balance locked (in gear-bank) for future value processing.
157            //
158            // In case of error replies, we don't need to do it, since original
159            // message value is already on locked balance in gear-bank.
160            if dispatch.value() != 0 && !dispatch.is_error_reply() {
161                self.bank.deposit_value(source, dispatch.value(), false);
162            }
163
164            match (gas_limit, reservation) {
165                (Some(gas_limit), None) => self
166                    .gas_tree
167                    .split_with_value(dispatch.is_reply(), message_id, dispatch.id(), gas_limit)
168                    .unwrap_or_else(|e| unreachable!("GasTree corrupted! {:?}", e)),
169                (None, None) => self
170                    .gas_tree
171                    .split(dispatch.is_reply(), message_id, dispatch.id())
172                    .unwrap_or_else(|e| unreachable!("GasTree corrupted! {:?}", e)),
173                (None, Some(reservation)) => {
174                    self.gas_tree
175                        .split(dispatch.is_reply(), reservation, dispatch.id())
176                        .unwrap_or_else(|e| unreachable!("GasTree corrupted! {:?}", e));
177                    self.remove_gas_reservation_with_task(dispatch.source(), reservation);
178                }
179                (Some(_), Some(_)) => unreachable!(
180                    "Sending dispatch with gas limit from reservation \
181                    is currently unimplemented and there is no way to send such dispatch"
182                ),
183            }
184
185            self.dispatches.push_back(dispatch.into_stored());
186        } else {
187            log::debug!(
188                "Sending user message {:?} from {:?} with gas limit {:?}",
189                dispatch.message(),
190                message_id,
191                dispatch.gas_limit(),
192            );
193            self.send_user_message(message_id, dispatch.into_parts().1, reservation);
194        }
195    }

212    fn wake_message(
213        &mut self,
214        message_id: MessageId,
215        program_id: ActorId,
216        awakening_id: MessageId,
217        delay: u32,
218    ) {
219        log::debug!("[{message_id}] tries to wake message#{awakening_id}");
220
221        if delay.is_zero() {
222            if let Ok(dispatch) = self.wake_dispatch_impl(program_id, awakening_id) {
223                self.dispatches.push_back(dispatch);
224
225                return;
226            }
227        } else if self.waitlist.contains(program_id, awakening_id) {
228            let expected_bn = self.block_height() + delay;
229            let task = ScheduledTask::WakeMessage(program_id, awakening_id);
230
231            // This validation helps us to avoid returning error on insertion into
232            // `TaskPool` in case of duplicate wake.
233            if !self.task_pool.contains(&expected_bn, &task) {
234                self.task_pool.add(expected_bn, task).unwrap_or_else(|e| {
235                    let err_msg = format!(
236                        "JournalHandler::wake_message: failed adding task for waking message. \
237                        Expected bn - {expected_bn:?}, program id - {program_id}, message_id - {awakening_id}.
238                        Got error - {e:?}"
239                    );
240
241                    unreachable!("{err_msg}");
242                });
243                self.on_task_pool_change();
244            }
245
246            return;
247        }
248
249        log::debug!("Failed to wake unknown message {awakening_id:?} from {message_id:?}");
250    }

256    fn update_allocations(&mut self, program_id: ActorId, allocations: IntervalsTree<WasmPage>) {
257        let old_allocations = ProgramsStorageManager::allocations(program_id).unwrap_or_default();
258        old_allocations
259            .difference(&allocations)
260            .flat_map(|page| page.iter())
261            .flat_map(|page| page.to_iter())
262            .for_each(|page| {
263                ProgramsStorageManager::remove_program_page(program_id, page);
264            });
265
266        ProgramsStorageManager::set_allocations(program_id, allocations);
267    }

269    fn send_value(&mut self, from: ActorId, to: ActorId, value: Value, locked: bool) {
270        if value.is_zero() {
271            // Nothing to do
272            return;
273        }
274
275        if locked {
276            self.bank.transfer_locked_value(from, to, value);
277        } else {
278            self.bank.transfer_value(from, to, value);
279        }
280    }

282    fn store_new_programs(
283        &mut self,
284        program_id: ActorId,
285        code_id: CodeId,
286        candidates: Vec<(MessageId, ActorId)>,
287    ) {
288        if self.instrumented_code(code_id).is_some() {
289            for (init_message_id, candidate_id) in candidates {
290                if !ProgramsStorageManager::has_program(candidate_id) {
291                    let expiration_block = self.block_height();
292                    self.store_program(
293                        candidate_id,
294                        GTestProgram::Default(Program::Active(ActiveProgram {
295                            allocations_tree_len: 0,
296                            code_id: code_id.cast(),
297                            state: ProgramState::Uninitialized {
298                                message_id: init_message_id,
299                            },
300                            expiration_block,
301                            memory_infix: Default::default(),
302                            gas_reservation_map: Default::default(),
303                        })),
304                    );
305
306                    // Transfer the ED from the program-creator to the new program
307                    Accounts::transfer(program_id, candidate_id, EXISTENTIAL_DEPOSIT, true);
308                } else {
309                    log::debug!("Program with id {candidate_id:?} already exists");
310                }
311            }
312        } else {
313            log::debug!("No referencing code with code hash {code_id:?} for candidate programs");
314            for (_, invalid_candidate_id) in candidates {
315                self.no_code_program.insert(invalid_candidate_id);
316            }
317        }
318    }

320    fn stop_processing(&mut self, dispatch: StoredDispatch, gas_burned: u64) {
321        log::debug!(
322            "Not enough gas for processing msg id {}, allowance equals {}, gas tried to burn at least {}",
323            dispatch.id(),
324            self.gas_allowance,
325            gas_burned,
326        );
327        self.gas_allowance = self.gas_allowance.saturating_sub(gas_burned);
328        self.messages_processing_enabled = false;
329        self.dispatches.push_front(dispatch);
330    }

332    fn reserve_gas(
333        &mut self,
334        message_id: MessageId,
335        reservation_id: ReservationId,
336        program_id: ActorId,
337        amount: u64,
338        duration: u32,
339    ) {
340        log::debug!(
341            "Reserved: {amount:?} from {message_id:?} with {reservation_id:?} for {duration} blocks"
342        );
343
344        let hold = HoldBoundBuilder::new(StorageType::Reservation).duration(self, duration);
345
346        if hold.expected_duration(self).is_zero() {
347            let err_msg = format!(
348                "JournalHandler::reserve_gas: reservation got zero duration hold bound for storing. \
349                Duration - {duration}, block cost - {cost}, program - {program_id}.",
350                cost = Self::cost_by_storage_type(StorageType::Reservation)
351            );
352
353            unreachable!("{err_msg}");
354        }
355
356        let total_amount = amount.saturating_add(hold.lock_amount(self));
357
358        self.gas_tree
359            .reserve_gas(message_id, reservation_id, total_amount)
360            .unwrap_or_else(|e| unreachable!("GasTree corrupted: {:?}", e));
361
362        let lock_id = hold.lock_id().unwrap_or_else(|| {
363            // Reservation storage is guaranteed to have an associated lock id
364            let err_msg =
365                "JournalHandler::reserve_gas: No associated lock id for the reservation storage";
366
367            unreachable!("{err_msg}");
368        });
369
370        self.gas_tree
371            .lock(reservation_id, lock_id, hold.lock_amount(self))
372            .unwrap_or_else(|e| {
373                let err_msg = format!(
374                    "JournalHandler::reserve_gas: failed locking gas for the reservation hold. \
375                Reseravation - {reservation_id}, lock amount - {lock}. Got error - {e:?}",
376                    lock = hold.lock_amount(self)
377                );
378
379                unreachable!("{err_msg}");
380            });
381
382        self.task_pool.add(
383            hold.expected(),
384            ScheduledTask::RemoveGasReservation(program_id, reservation_id)
385        ).unwrap_or_else(|e| {
386            let err_msg = format!(
387                "JournalHandler::reserve_gas: failed adding task for gas reservation removal. \
388                Expected bn - {bn:?}, program id - {program_id}, reservation id - {reservation_id}. Got error - {e:?}",
389                bn = hold.expected()
390            );
391            unreachable!("{err_msg}");
392        });
393        self.on_task_pool_change();
394    }

396    fn unreserve_gas(
397        &mut self,
398        reservation_id: ReservationId,
399        program_id: ActorId,
400        expiration: u32,
401    ) {
402        <Self as TaskHandler<ActorId, MessageId, bool>>::remove_gas_reservation(
403            self,
404            program_id,
405            reservation_id,
406        );
407
408        let _ = self
409            .task_pool
410            .delete(
411                expiration,
412                ScheduledTask::RemoveGasReservation(program_id, reservation_id),
413            )
414            .map(|_| {
415                self.on_task_pool_change();
416            });
417    }

419    fn update_gas_reservation(&mut self, program_id: ActorId, reserver: GasReserver) {
420        let block_height = self.block_height();
421        self.update_program(program_id, |program| {
422            // TODO #4758 use HoldBoundBuilder here and check all other places:
423            program.gas_reservation_map =
424                reserver.into_map(block_height, |duration| block_height + duration);
425        })
426        .expect("no genuine program was found");
427    }

437    fn system_unreserve_gas(&mut self, message_id: MessageId) {
438        let amount = self
439            .gas_tree
440            .system_unreserve(message_id)
441            .unwrap_or_else(|e| unreachable!("GasTree corrupted: {:?}", e));
442
443        if amount != 0 {
444            log::debug!("Unreserved {amount} gas for system from {message_id}");
445        } else {
446            log::debug!(
447                "Gas for system was not unreserved from {message_id} as there is no supply",
448            );
449        }
450    }

452    fn send_signal(&mut self, message_id: MessageId, destination: ActorId, code: SignalCode) {
453        let reserved = self
454            .gas_tree
455            .system_unreserve(message_id)
456            .unwrap_or_else(|e| unreachable!("GasTree corrupted! {:?}", e));
457
458        if reserved != 0 {
459            log::debug!(
460                "Send signal issued by {message_id} to {destination} with {reserved} supply",
461            );
462
463            let trap_signal = SignalMessage::new(message_id, code)
464                .into_dispatch(message_id, destination)
465                .into_stored();
466
467            self.gas_tree
468                .split_with_value(
469                    trap_signal.is_reply(),
470                    message_id,
471                    trap_signal.id(),
472                    reserved,
473                )
474                .unwrap_or_else(|e| unreachable!("GasTree corrupted! {:?}", e));
475
476            self.dispatches.push_back(trap_signal);
477        } else {
478            log::trace!("Signal wasn't send due to inappropriate supply");
479        }
480    }
