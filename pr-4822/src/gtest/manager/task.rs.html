33pub(crate) fn get_maximum_task_gas(task: &VaraScheduledTask<ActorId>) -> Gas {
34    use ScheduledTask::*;
35    let weights = TaskWeights::default();
36    match task {
37        PauseProgram(_) => 0,
38        #[allow(deprecated)]
39        RemoveResumeSession(_) => 0,
40
41        RemoveFromMailbox(_, _) => weights.remove_from_mailbox.ref_time,
42        RemoveFromWaitlist(_, _) => weights.remove_from_waitlist.ref_time,
43        RemovePausedProgram(_) => todo!("#646"),
44        RemoveCode(_) => todo!("#646"),
45        WakeMessage(_, _) => weights
46            .wake_message
47            .ref_time
48            .max(weights.wake_message_no_wake.ref_time),
49        SendDispatch(_) => weights.send_dispatch.ref_time,
50        SendUserMessage { .. } => weights
51            .send_user_message_to_mailbox
52            .ref_time
53            .max(weights.send_user_message.ref_time),
54        RemoveGasReservation(_, _) => weights.remove_gas_reservation.ref_time,
55    }
56}

69    fn remove_from_mailbox(&mut self, user_id: ActorId, message_id: MessageId) -> GearCommonGas {
70        let message = ReplyMessage::auto(message_id);
71
72        self.gas_tree
73            .create(user_id, message.id(), 0, true)
74            .expect("failed to create gas tree node");
75
76        let mailboxed = self
77            .read_mailbox_message(user_id, message_id)
78            .expect("failed to claim value from mailbox");
79
80        let dispatch = message.into_stored_dispatch(
81            mailboxed.destination(),
82            mailboxed.source(),
83            mailboxed.id(),
84        );
85
86        self.dispatches.push_back(dispatch);
87
88        TaskWeights::default().remove_from_mailbox.ref_time
89    }
90
91    fn remove_from_waitlist(
92        &mut self,
93        program_id: ActorId,
94        message_id: MessageId,
95    ) -> GearCommonGas {
96        let waitlisted = self
97            .wake_dispatch_impl(program_id, message_id)
98            .unwrap_or_else(|e| {
99                let err_msg = format!(
100                    "TaskHandler::remove_from_waitlist: failed waking dispatch. \
101                Program id - {program_id}, waking message - {message_id} \
102                Got error - {e:?}."
103                );
104
105                unreachable!("{err_msg}");
106            });
107
108        self.send_signal(
109            message_id,
110            waitlisted.destination(),
111            SignalCode::RemovedFromWaitlist,
112        );
113
114        if !waitlisted.is_reply() && waitlisted.kind() != DispatchKind::Signal {
115            let err = ErrorReplyReason::RemovedFromWaitlist;
116
117            let err_payload = err
118                .to_string()
119                .into_bytes()
120                .try_into()
121                .expect("internal error: error reply reason bytes size is too big");
122
123            let trap_reply = ReplyMessage::system(message_id, err_payload, 0, err);
124
125            if ProgramsStorageManager::is_program(waitlisted.source()) {
126                let trap_dispatch =
127                    trap_reply.into_stored_dispatch(program_id, waitlisted.source(), message_id);
128
129                self.gas_tree
130                    .split(
131                        trap_dispatch.is_reply(),
132                        waitlisted.id(),
133                        trap_dispatch.id(),
134                    )
135                    .unwrap_or_else(|e| unreachable!("GasTree corrupted: {e:?}"));
136                self.dispatches.push_back(trap_dispatch);
137            } else {
138                let trap_message =
139                    trap_reply.into_stored(program_id, waitlisted.source(), message_id);
140                self.log.push(trap_message);
141            }
142        }
143
144        self.consume_and_retrieve(waitlisted.id());
145
146        if waitlisted.kind() == DispatchKind::Init {
147            let origin = waitlisted.source();
148            self.init_failure(program_id, origin);
149        }
150
151        TaskWeights::default().remove_from_waitlist.ref_time
152    }

154    fn remove_paused_program(&mut self, _program_id: ActorId) -> GearCommonGas {

156    }

158    fn wake_message(&mut self, program_id: ActorId, message_id: MessageId) -> GearCommonGas {
159        if let Ok(dispatch) = self.wake_dispatch_impl(program_id, message_id) {
160            self.dispatches.push_back(dispatch);
161            TaskWeights::default().wake_message.ref_time
162        } else {
163            TaskWeights::default().wake_message_no_wake.ref_time
164        }
165    }
166
167    fn send_dispatch(&mut self, stashed_message_id: MessageId) -> GearCommonGas {
168        let (dispatch, hold_interval) = self
169            .dispatches_stash
170            .remove(&stashed_message_id)
171            .unwrap_or_else(|| unreachable!("TaskPool corrupted"));
172
173        self.charge_for_hold(dispatch.id(), hold_interval, StorageType::DispatchStash);
174
175        self.dispatches.push_back(dispatch.into());
176        TaskWeights::default().send_dispatch.ref_time
177    }
178
179    fn send_user_message(
180        &mut self,
181        stashed_message_id: MessageId,
182        to_mailbox: bool,
183    ) -> GearCommonGas {
184        let (message, hold_interval) = self
185            .dispatches_stash
186            .remove(&stashed_message_id)
187            .map(|(dispatch, interval)| (dispatch.into_parts().1, interval))
188            .unwrap_or_else(|| unreachable!("TaskPool corrupted!"));
189
190        self.charge_for_hold(message.id(), hold_interval, StorageType::DispatchStash);
191
192        let mailbox_message = message.clone().try_into().unwrap_or_else(|e| {
193            unreachable!("invalid message: can't be converted to user message {e:?}")
194        });
195
196        self.send_user_message_after_delay(mailbox_message, to_mailbox);
197        if to_mailbox {
198            TaskWeights::default().send_user_message_to_mailbox.ref_time
199        } else {
200            TaskWeights::default().send_user_message.ref_time
201        }
202    }
203
204    fn remove_gas_reservation(
205        &mut self,
206        program_id: ActorId,
207        reservation_id: ReservationId,
208    ) -> GearCommonGas {
209        let _slot = self.remove_gas_reservation_impl(program_id, reservation_id);
210        TaskWeights::default().remove_gas_reservation.ref_time
211    }
212
213    fn remove_resume_session(&mut self, _session_id: u32) -> GearCommonGas {
214        log::debug!("Program rent logic is disabled");
215        0
216    }
