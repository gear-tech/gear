19use super::*;
20use crate::{constants::BlockNumber, state::waitlist::WaitlistErrorImpl};
21use gear_core::message::StoredDispatch;

23impl ExtManager {
24    pub(crate) fn wait_dispatch_impl(
25        &mut self,
26        dispatch: StoredDispatch,
27        duration: Option<BlockNumber>,
28        reason: MessageWaitedReason,
29    ) {
30        use MessageWaitedRuntimeReason::*;
31
32        let hold_builder = HoldBoundBuilder::new(StorageType::Waitlist);
33
34        let maximal_hold = hold_builder.maximum_for_message(self, dispatch.id());
35
36        let hold = if let Some(duration) = duration {
37            hold_builder.duration(self, duration).min(maximal_hold)
38        } else {
39            maximal_hold
40        };
41
42        let message_id = dispatch.id();
43        let destination = dispatch.destination();
44
45        if hold.expected_duration(self).is_zero() {
46            let gas_limit = self.gas_tree.get_limit(dispatch.id()).unwrap_or_else(|e| {
47                let err_msg = format!(
48                    "wait_dispatch: failed getting message gas limit. Message id - {message_id}. \
49                        Got error - {e:?}",
50                    message_id = dispatch.id()
51                );
52
53                unreachable!("{err_msg}");
54            });
55
56            let err_msg = format!(
57                "wait_dispatch: message got zero duration hold bound for waitlist. \
58                Requested duration - {duration:?}, gas limit - {gas_limit}, \
59                wait reason - {reason:?}, message id - {}.",
60                dispatch.id(),
61            );
62
63            unreachable!("{err_msg}");
64        }
65
66        // Locking funds for holding.
67        let lock_id = hold.lock_id().unwrap_or_else(|| {
68            // Waitlist storage is guaranteed to have an associated lock id
69            let err_msg = "wait_dispatch: No associated lock id for the waitlist storage";
70
71            unreachable!("{err_msg}");
72        });
73        self.gas_tree
74            .lock(message_id, lock_id, hold.lock_amount(self))
75            .unwrap_or_else(|e| {
76                let err_msg = format!(
77                    "wait_dispatch: failed locking gas for the waitlist hold. \
78                    Message id - {message_id}, lock amount - {lock}. Got error - {e:?}",
79                    lock = hold.lock_amount(self)
80                );
81
82                unreachable!("{err_msg}");
83            });
84
85        match reason {
86            MessageWaitedReason::Runtime(WaitForCalled | WaitUpToCalledFull) => {
87                let expected = hold.expected();
88                let task = ScheduledTask::WakeMessage(destination, message_id);
89
90                if !self.task_pool.contains(&expected, &task) {
91                    self.task_pool.add(expected, task).unwrap_or_else(|e| {
92                        let err_msg = format!(
93                            "wait_dispatch: failed adding task for waking message. \
94                            Expected bn - {expected:?}, program id - {destination}, message id - {message_id}. Got error - {e:?}",
95                        );
96
97                        unreachable!("{err_msg}");
98                    });
99                    self.on_task_pool_change();
100                }
101            }
102            MessageWaitedReason::Runtime(WaitCalled | WaitUpToCalled) => {
103                self.task_pool.add(
104                    hold.expected(),
105                    ScheduledTask::RemoveFromWaitlist(dispatch.destination(), dispatch.id()),
106                )
107                .unwrap_or_else(|e| {
108                    let err_msg = format!(
109                        "wait_dispatch: failed adding task for removing message from waitlist. \
110                        Expected bn - {bn:?}, program id - {destination}, message id - {message_id}. Got error - {e:?}",
111                        bn = hold.expected(),
112                    );
113
114                    unreachable!("{err_msg}");
115                });
116                self.on_task_pool_change();
117            }
118            MessageWaitedReason::System(reason) => match reason {},
119        }
120
121        self.waitlist.insert(dispatch, hold.expected())
122            .unwrap_or_else(|e| {
123                let err_msg = format!(
124                    "wait_dispatch: failed inserting message to the wailist. \
125                    Expected bn - {bn:?}, program id - {destination}, message id - {message_id}. Got error - {e:?}",
126                    bn = hold.expected(),
127                );
128
129                unreachable!("{err_msg}");
130            });
131    }
132
133    pub(crate) fn wake_dispatch_impl(
134        &mut self,
135        program_id: ActorId,
136        message_id: MessageId,
137    ) -> Result<StoredDispatch, WaitlistErrorImpl> {
138        self.waitlist
139            .remove(program_id, message_id)
140            .map(|waitlisted_message| self.wake_dispatch_requirements(waitlisted_message))
141    }
142
143    pub(crate) fn wake_dispatch_requirements(
144        &mut self,
145        (waitlisted, hold_interval): (StoredDispatch, Interval<BlockNumber>),
146    ) -> StoredDispatch {
147        let expected = hold_interval.finish;
148
149        self.charge_for_hold(waitlisted.id(), hold_interval, StorageType::Waitlist);
150
151        let _ = self
152            .task_pool
153            .delete(
154                expected,
155                ScheduledTask::RemoveFromWaitlist(waitlisted.destination(), waitlisted.id()),
156            )
157            .map(|_| {
158                self.on_task_pool_change();
159            });
160
161        waitlisted
162    }
