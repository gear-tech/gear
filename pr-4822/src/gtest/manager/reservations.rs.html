19//! Various reservation related methods for ExtManager
20
21use super::ExtManager;
22use crate::state::programs::{GTestProgram, ProgramsStorageManager};
23use gear_common::{ActorId, Program, ReservationId, scheduler::StorageType, storage::Interval};
24use gear_core::{reservation::GasReservationSlot, tasks::ScheduledTask};
25use std::mem;
26
27impl ExtManager {
28    pub(crate) fn remove_gas_reservation_impl(
29        &mut self,
30        program_id: ActorId,
31        reservation: ReservationId,
32    ) -> GasReservationSlot {
33        let slot = self.update_program(program_id, |active_program| {
34            active_program.gas_reservation_map
35                .remove(&reservation)
36                .unwrap_or_else(|| {
37                    let err_msg = format!("ExtManager::remove_gas_reservation_impl: failed removing gas reservation. \
38                    Reservation {reservation} doesn't exist.");
39
40                    unreachable!("{err_msg}")
41                })
42        }).unwrap_or_else(|| {
43            unreachable!("failed to update program {program_id}")
44        });
45
46        self.remove_gas_reservation_slot(reservation, slot)
47    }
48
49    pub(crate) fn remove_gas_reservation_with_task(
50        &mut self,
51        program_id: ActorId,
52        reservation: ReservationId,
53    ) {
54        let slot = self.remove_gas_reservation_impl(program_id, reservation);
55
56        let _ = self
57            .task_pool
58            .delete(
59                slot.finish,
60                ScheduledTask::RemoveGasReservation(program_id, reservation),
61            )
62            .map(|_| {
63                self.on_task_pool_change();
64            });
65    }
66
67    pub(crate) fn remove_gas_reservation_slot(
68        &mut self,
69        reservation: ReservationId,
70        slot: GasReservationSlot,
71    ) -> GasReservationSlot {
72        let interval = Interval {
73            start: slot.start,
74            finish: slot.finish,
75        };
76
77        self.charge_for_hold(reservation, interval, StorageType::Reservation);
78        self.consume_and_retrieve(reservation);
79
80        slot
81    }
82
83    pub(crate) fn remove_gas_reservation_map(&mut self, program_id: ActorId) {
84        ProgramsStorageManager::modify_program(program_id, |program| {
85            if let Some(GTestProgram::Default(Program::Active(active_program))) = program {
86                for (reservation_id, slot) in mem::take(&mut active_program.gas_reservation_map) {
87                    let slot = self.remove_gas_reservation_slot(reservation_id, slot);
88
89                    let result = self.task_pool.delete(
90                        slot.finish,
91                        ScheduledTask::RemoveGasReservation(program_id, reservation_id),
92                    );
93                    log::debug!(
94                        "remove_gas_reservation_map; program_id = {program_id:?}, result = {result:?}"
95                    );
96                }
97            }
98        });
99    }
100}
