19use super::*;
20use crate::state::programs::GTestProgram;
21
22impl ExtManager {
23    /// Call non-void meta function from actor stored in manager.
24    /// Warning! This is a static call that doesn't change actors pages data.
25    pub(crate) fn read_state_bytes(
26        &mut self,
27        payload: Vec<u8>,
28        program_id: ActorId,
29    ) -> Result<Vec<u8>> {
30        ProgramsStorageManager::has_program(program_id)
31            .then_some(())
32            .ok_or(TestError::ActorNotFound(program_id))?;
33
34        if ProgramsStorageManager::is_mock_program(program_id) {
35            ProgramsStorageManager::modify_program(program_id, |program| {
36                let Some(GTestProgram::Mock(mock_program)) = program else {
37                    unreachable!("checked upper, that it's the case for a mock program");
38                };
39
40                mock_program
41                    .handlers_mut()
42                    .state()
43                    .map_err(|e| TestError::ReadStateError(e.to_string()))
44            })
45        } else {
46            let allocations = ProgramsStorageManager::allocations(program_id);
47            let code_id = ProgramsStorageManager::access_primary_program(program_id, |program| {
48                let Some(Program::Active(ActiveProgram { code_id, .. })) = program else {
49                    // Above checked that program exists and it's not a mock program
50                    unreachable!("checked upper, that it's the case for an active program");
51                };
52
53                *code_id
54            });
55            debug_assert!(code_id != CUSTOM_WASM_PROGRAM_CODE_ID);
56
57            let code_metadata = self
58                .code_metadata(code_id)
59                .cloned()
60                .expect("internal error: code metadata not found for existing active program");
61            let instrumented_code = self
62                .instrumented_code(code_id)
63                .cloned()
64                .expect("internal error: instrumented code not found for existing active program");
65            core_processor::informational::execute_for_reply::<Ext<LazyPagesNative>, _>(
66                String::from("state"),
67                instrumented_code,
68                code_metadata,
69                allocations,
70                Some((program_id, Default::default())),
71                payload,
72                MAX_USER_GAS_LIMIT,
73                self.blocks_manager.get(),
74            )
75            .map_err(TestError::ReadStateError)
76        }
77    }
78
79    pub(crate) fn read_memory_pages(&self, program_id: ActorId) -> BTreeMap<GearPage, PageBuf> {
80        ProgramsStorageManager::program_pages(program_id)
81    }
82}
