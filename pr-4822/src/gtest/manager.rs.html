159    pub(crate) fn free_id_nonce(&mut self) -> u64 {
160        let mut id_nonce = self.nonce_manager.id_nonce();
161        while ProgramsStorageManager::has_program(id_nonce.into()) {
162            self.nonce_manager.inc_id_nonce();
163            id_nonce = self.nonce_manager.id_nonce();
164        }
165

187    pub(crate) fn override_balance(&mut self, id: ActorId, balance: Value) {
188        if ProgramsStorageManager::is_user(id) && balance < crate::EXISTENTIAL_DEPOSIT {
189            usage_panic!(
190                "An attempt to override balance with value ({}) less than existential deposit ({}. \
191                Please try to use bigger balance value",
192                balance,
193                crate::EXISTENTIAL_DEPOSIT
194            );
195        }
196        Accounts::override_balance(id, balance);
197    }

204    fn init_success(&mut self, program_id: ActorId) {
205        ProgramsStorageManager::modify_program(program_id, |program| {
206            let Program::Active(active_program) = program
207                .unwrap_or_else(|| panic!("Actor id {program_id:?} not found"))
208                .as_primary_program_mut()
209            else {
210                unreachable!(
211                    "Before init finishes, program must always be active. But {program_id:?} program is not active."
212                );
213            };
214
215            active_program.state = ProgramState::Initialized;
216        });
217    }
218
219    fn init_failure(&mut self, program_id: ActorId, origin: ActorId) {
220        self.clean_waitlist(program_id);
221        self.remove_gas_reservation_map(program_id);
222        ProgramsStorageManager::modify_program(program_id, |program| {
223            if let Some(program) = program.map(GTestProgram::as_primary_program_mut) {
224                if !program.is_active() {
225                    // Guaranteed to be called only on active program
226                    unreachable!(
227                        "ExtManager::init_failure: failed to exit active program. \
228                    Program - {program_id}, actual program - {program:?}"
229                    );
230                }
231
232                *program = Program::Terminated(program_id);
233            } else {
234                // That's a case if no code exists for the program
235                // requested to be created from another program and
236                // there was not enough gas to get program from storage.
237                log::debug!("Failed init is set for non-existing actor");
238            }
239        });
240
241        let value = Accounts::balance(program_id);
242        if value != 0 {
243            Accounts::transfer(program_id, origin, value, false);
244        }
245    }

247    pub(crate) fn update_program<R, F: FnOnce(&mut ActiveProgram<BlockNumber>) -> R>(
248        &mut self,
249        id: ActorId,
250        op: F,
251    ) -> Option<R> {
252        ProgramsStorageManager::modify_program(id, |program| {
253            program.and_then(|program| {
254                if let Program::Active(active_program) = program.as_primary_program_mut() {
255                    Some(op(active_program))
256                } else {
257                    None
258                }
259            })
260        })
261    }

263    pub(crate) fn read_mailbox_message(
264        &mut self,
265        to: ActorId,
266        from_mid: MessageId,
267    ) -> Result<UserStoredMessage, MailboxErrorImpl> {
268        let (message, hold_interval) = self.mailbox.remove(to, from_mid)?;
269
270        let expected = hold_interval.finish;
271
272        let user_id = message.destination();
273        let from = message.source();
274
275        self.charge_for_hold(message.id(), hold_interval, StorageType::Mailbox);
276        self.consume_and_retrieve(message.id());
277
278        self.bank.transfer_value(from, user_id, message.value());
279
280        let _ = self
281            .task_pool
282            .delete(
283                expected,
284                ScheduledTask::RemoveFromMailbox(user_id, message.id()),
285            )
286            .map(|_| {
287                self.on_task_pool_change();
288            });
289
290        Ok(message)
291    }

293    pub(crate) fn clean_waitlist(&mut self, id: ActorId) {
294        self.waitlist.drain_key(id).for_each(|entry| {
295            let message = self.wake_dispatch_requirements(entry);
296
297            self.dispatches.push_back(message);
298        });
299    }
