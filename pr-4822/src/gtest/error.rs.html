19use gear_core::{ids::ActorId, pages::WasmPage};
20use gear_core_errors::ExtError;
21use parity_scale_codec::Error as CodecError;
22
23/// Type alias for the testing functions running result.
24pub type Result<T, E = TestError> = core::result::Result<T, E>;
25
26/// List of general errors.
27#[derive(Clone, Debug, PartialEq, Eq, thiserror::Error)]
28pub enum TestError {
29    /// Invalid return type after execution.
30    #[error("Invalid return type after execution")]
31    InvalidReturnType,
32
33    /// Function not found in executor.
34    #[error("Function not found in executor: `{0}`")]
35    FunctionNotFound(String),
36
37    /// Actor not found.
38    #[error("Actor not found: `{0}`")]
39    ActorNotFound(ActorId),
40
41    /// Actor is not executable.
42    #[error("Actor is not executable: `{0}`")]
43    ActorIsNotExecutable(ActorId),
44
45    /// Meta WASM binary hasn't been provided.
46    #[error("Meta WASM binary hasn't been provided")]
47    MetaBinaryNotProvided,
48
49    /// Insufficient memory.
50    #[error("Insufficient memory: available {0:?} < requested {1:?}")]
51    InsufficientMemory(WasmPage, WasmPage),
52
53    /// Invalid import module.
54    #[error("Invalid import module: `{0}` instead of `env`")]
55    InvalidImportModule(String),
56
57    /// Failed to call unsupported function.
58    #[error("Failed to call unsupported function: `{0}`")]
59    UnsupportedFunction(String),
60
61    /// Wrapper for [`ExtError`].
62    #[error("Execution error: `{0}`")]
63    ExecutionError(#[from] ExtError),
64
65    /// Wrapper for [`wasmi::Error`](https://paritytech.github.io/wasmi/wasmi/enum.Error.html).
66    #[error("Memory error: `{0}`")]
67    MemoryError(#[from] gear_core_errors::MemoryError),
68
69    /// Wrapper for [`parity_scale_codec::Error`](https://docs.rs/parity-scale-codec/latest/parity_scale_codec/struct.Error.html).
70    #[error("Codec error: `{0}`")]
71    ScaleCodecError(#[from] CodecError),
72
73    /// Instrumentation of binary code failed.
74    #[error("Instrumentation of binary code failed.")]
75    Instrumentation,
76
77    /// Reading of program state failed.
78    #[error("Reading of program state failed: `{0}`")]
79    ReadStateError(String),
80
81    /// Searching gbuild artifact failed.
82    #[error("Reading of program state failed: `{0}`")]
83    GbuildArtifactNotFound(String),
84}

86macro_rules! usage_panic {
87    ($($arg:tt)*) => {{
88        use colored::Colorize as _;
89        let panic_msg = format!($($arg)*).red().bold();
90        panic!("{}", panic_msg);
91    }};
92}
93
94pub(crate) use usage_panic;
