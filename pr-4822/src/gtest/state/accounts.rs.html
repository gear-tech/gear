19//! Accounts storage.
20
21use crate::{
22    DEFAULT_USERS_INITIAL_BALANCE, EXISTENTIAL_DEPOSIT, Value, default_users_list,
23    state::WithOverlay,
24};
25use gear_core::ids::ActorId;
26use std::{collections::HashMap, fmt, thread::LocalKey};
27
28fn init_default_accounts(storage: &mut HashMap<ActorId, Balance>) {
29    for &id in default_users_list() {
30        let id = id.into();
31        storage.insert(id, Balance::new(DEFAULT_USERS_INITIAL_BALANCE));
32    }
33}
34
35thread_local! {
36    pub(super) static ACCOUNT_STORAGE: WithOverlay<HashMap<ActorId, Balance>> = WithOverlay::new({
37        let mut storage = HashMap::new();
38        init_default_accounts(&mut storage);
39        storage
40    });
41}
42
43fn storage() -> &'static LocalKey<WithOverlay<HashMap<ActorId, Balance>>> {
44    &ACCOUNT_STORAGE
45}
46
47#[derive(Debug, Clone)]
48pub(super) struct Balance {
49    amount: Value,
50}
51
52impl Balance {
53    fn new(amount: Value) -> Self {
54        if amount < EXISTENTIAL_DEPOSIT {
55            panic!(
56                "Failed to create balance: the amount {amount} cannot be lower than the existential deposit"
57            );
58        }
59
60        Self { amount }
61    }
62
63    fn balance(&self) -> Value {
64        self.amount
65    }
66
67    fn reducible_balance(&self) -> Value {
68        self.amount - EXISTENTIAL_DEPOSIT
69    }
70
71    fn decrease(&mut self, amount: Value) {
72        self.amount -= amount;
73    }
74
75    fn increase(&mut self, amount: Value) {
76        self.amount += amount;
77    }
78}
79
80pub(crate) struct Accounts;
81
82impl Accounts {
83    // Checks if account by program id exists.
84    pub(crate) fn exists(id: ActorId) -> bool {
85        Self::balance(id) != 0
86    }
87
88    // Returns account balance.
89    pub(crate) fn balance(id: ActorId) -> Value {
90        storage().with(|storage| {
91            storage
92                .data()
93                .get(&id)
94                .map(|balance| balance.balance())
95                .unwrap_or_default()
96        })
97    }
98
99    // Returns account reducible balance.
100    pub(crate) fn reducible_balance(id: ActorId) -> Value {
101        storage().with(|storage| {
102            storage
103                .data()
104                .get(&id)
105                .map(|balance| balance.reducible_balance())
106                .unwrap_or_default()
107        })
108    }
109
110    // Decreases account balance.
111    pub(crate) fn decrease(id: ActorId, amount: Value, keep_alive: bool) {
112        storage().with(|storage| {
113            let balance = if let Some(balance) = storage.data_mut().get_mut(&id) {
114                if keep_alive && balance.reducible_balance() < amount {
115                    panic!(
116                        "Not enough balance to decrease, reducible: {}, value: {amount}",
117                        balance.reducible_balance(),
118                    );
119                }
120                if !keep_alive && balance.balance() < amount {
121                    panic!(
122                        "Not enough balance to decrease, total: {}, value: {amount}",
123                        balance.balance(),
124                    );
125                }
126
127                balance.decrease(amount);
128
129                balance.balance()
130            } else {
131                panic!("Failed to decrease balance for account {id:?}, balance is zero");
132            };
133
134            if balance < EXISTENTIAL_DEPOSIT {
135                log::debug!(
136                    "Removing account {id:?} with balance {balance} below the existential deposit",
137                );
138                storage.data_mut().remove(&id);
139            }
140        });
141    }
142
143    // Increases account balance.
144    pub(crate) fn increase(id: ActorId, amount: Value) {
145        storage().with(|storage| {
146            let balance = storage.data().get(&id).map(Balance::balance).unwrap_or_default();
147
148            if balance + amount < EXISTENTIAL_DEPOSIT {
149                panic!(
150                    "Failed to increase balance: the sum ({}) of the total balance ({}) \
151                    and the value ({}) cannot be lower than the existential deposit ({EXISTENTIAL_DEPOSIT})",
152                    balance + amount,
153                    balance,
154                    amount
155                );
156            }
157
158            storage
159                .data_mut()
160                .entry(id)
161                .and_modify(|balance| balance.increase(amount))
162                .or_insert_with(|| Balance::new(amount));
163        });
164    }
165
166    // Transfers value between accounts.
167    pub(crate) fn transfer(from: ActorId, to: ActorId, amount: Value, keep_alive: bool) {
168        Self::decrease(from, amount, keep_alive);
169        Self::increase(to, amount);
170    }
171
172    // Overrides account balance.
173    pub(crate) fn override_balance(id: ActorId, amount: Value) {
174        if amount < EXISTENTIAL_DEPOSIT {
175            panic!(
176                "Failed to override balance: the amount {amount} cannot be lower than the existential deposit"
177            );
178        }
179
180        storage().with(|storage| {
181            storage.data_mut().insert(id, Balance::new(amount));
182        });
183    }
184
185    // Checks if value can be deposited to account.
186    pub(crate) fn can_deposit(id: ActorId, amount: Value) -> bool {
187        Accounts::balance(id) + amount >= EXISTENTIAL_DEPOSIT
188    }
189
190    // Clears accounts storage.
191    pub(crate) fn clear() {
192        storage().with(|storage| {
193            storage.data_mut().clear();
194            init_default_accounts(&mut storage.data_mut());
195        });
196    }
197}
198
199impl fmt::Debug for Accounts {
200    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
201        storage().with(|storage| f.debug_map().entries(storage.data().iter()).finish())
202    }
203}
