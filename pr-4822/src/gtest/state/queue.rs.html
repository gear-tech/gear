19//! Queue storage manager.
20
21use crate::state::WithOverlay;
22use gear_core::message::StoredDispatch;
23use std::{collections::VecDeque, thread::LocalKey};
24
25thread_local! {
26    /// Definition of the storage value storing dispatches queue.
27    pub(super) static DISPATCHES_QUEUE: WithOverlay<VecDeque<StoredDispatch>> = Default::default();
28}
29
30fn storage() -> &'static LocalKey<WithOverlay<VecDeque<StoredDispatch>>> {
31    &DISPATCHES_QUEUE
32}
33
34#[derive(Debug, Clone, Default)]
35pub(crate) struct QueueManager;
36
37impl QueueManager {
38    /// Push dispatch to the queue back.
39    pub(crate) fn push_back(&self, dispatch: StoredDispatch) {
40        storage().with(|queue| queue.data_mut().push_back(dispatch));
41    }
42
43    /// Push dispatch to the queue head.
44    pub(crate) fn push_front(&self, dispatch: StoredDispatch) {
45        storage().with(|queue| queue.data_mut().push_front(dispatch));
46    }
47
48    /// Pop dispatch from the queue head.
49    pub(crate) fn pop_front(&self) -> Option<StoredDispatch> {
50        storage().with(|queue| queue.data_mut().pop_front())
51    }
52
53    /// Clears the queue.
54    pub(crate) fn clear(&self) {
55        storage().with(|queue| queue.data_mut().clear())
56    }
57
58    /// Returns queue length.
59    pub(crate) fn len(&self) -> usize {
60        storage().with(|queue| queue.data().len())
61    }
62}
