19//! Block timestamp and height management.
20
21use crate::{
22    BLOCK_DURATION_IN_MSECS, EPOCH_DURATION_IN_BLOCKS, INITIAL_RANDOM_SEED, constants::BlockNumber,
23    state::WithOverlay,
24};
25use core_processor::configs::BlockInfo;
26use gear_common::storage::GetCallback;
27use rand::{RngCore, SeedableRng, rngs::StdRng};
28use std::{
29    thread::LocalKey,
30    time::{SystemTime, UNIX_EPOCH},
31};
32
33thread_local! {
34    /// Definition of the storage value storing block info (timestamp and height).
35    pub(super) static BLOCK_INFO_STORAGE: WithOverlay<BlockInfo> = WithOverlay::new(BlockInfo {
36        height: 0,
37        timestamp: now(),
38    });
39    pub(super) static CURRENT_EPOCH_RANDOM: WithOverlay<Vec<u8>> = WithOverlay::new(epoch_random(INITIAL_RANDOM_SEED));
40}
41
42fn block_info_storage() -> &'static LocalKey<WithOverlay<BlockInfo>> {
43    &BLOCK_INFO_STORAGE
44}
45
46fn current_epoch_random_storage() -> &'static LocalKey<WithOverlay<Vec<u8>>> {
47    &CURRENT_EPOCH_RANDOM
48}
49
50#[derive(Debug)]
51pub(crate) struct BlocksManager;
52
53impl BlocksManager {
54    /// Get current block info.
55    pub(crate) fn get(&self) -> BlockInfo {
56        block_info_storage().with(|bi| *bi.data())
57    }
58
59    /// Move blocks by one.
60    pub(crate) fn next_block(&self) -> BlockInfo {
61        let bi = self.move_blocks_by(1);
62
63        let block_height = self.get().height;
64        if block_height.is_multiple_of(EPOCH_DURATION_IN_BLOCKS) {
65            let seed = INITIAL_RANDOM_SEED + (block_height / EPOCH_DURATION_IN_BLOCKS) as u64;
66            update_epoch_random(seed);
67        }
68
69        bi
70    }
71
72    /// Adjusts blocks info by moving blocks by `amount`.
73    pub(crate) fn move_blocks_by(&self, amount: u32) -> BlockInfo {
74        block_info_storage().with(|bi| {
75            bi.data_mut().height += amount;
76            let duration = BLOCK_DURATION_IN_MSECS.saturating_mul(amount as u64);
77            bi.data_mut().timestamp += duration;
78
79            *bi.data()
80        })
81    }
82
83    pub(crate) fn reset(&self) {
84        block_info_storage().with(|bi_rc| {
85            *bi_rc.data_mut() = BlockInfo {
86                height: 0,
87                timestamp: now(),
88            };
89        });
90    }
91}
92
93impl Default for BlocksManager {
94    fn default() -> Self {
95        Self
96    }
97}
98
99fn now() -> u64 {
100    SystemTime::now()
101        .duration_since(UNIX_EPOCH)
102        .expect("Time went backwards")
103        .as_millis() as u64
104}

106/// Block number getter.
107///
108/// Used to get block number for auxiliary complex storage managers,
109/// like auxiliary mailbox, waitlist and etc.
110pub(crate) struct GetBlockNumberImpl;
111
112impl GetCallback<BlockNumber> for GetBlockNumberImpl {
113    fn call() -> BlockNumber {
114        BlocksManager.get().height
115    }
116}
117
118pub(crate) fn current_epoch_random() -> Vec<u8> {
119    current_epoch_random_storage().with(|random| random.data().clone())
120}
121
122pub(super) fn update_epoch_random(seed: u64) {
123    current_epoch_random_storage().with(|random| {
124        *random.data_mut() = epoch_random(seed);
125    });
126}
127
128fn epoch_random(seed: u64) -> Vec<u8> {
129    let mut rng = StdRng::seed_from_u64(seed);
130    let mut random = [0u8; 32];
131    rng.fill_bytes(&mut random);
132
133    random.to_vec()
134}
