19//! `gtest` bank
20
21use crate::{
22    GAS_MULTIPLIER, Gas,
23    constants::Value,
24    state::{WithOverlay, accounts::Accounts},
25};
26use gear_common::GasMultiplier;
27use gear_core::ids::ActorId;
28use std::{collections::HashMap, thread::LocalKey};
29
30thread_local! {
31    /// Bank storage.
32    pub(super) static BANK_ACCOUNTS: WithOverlay<HashMap<ActorId, BankBalance>> = Default::default();
33}
34
35fn storage() -> &'static LocalKey<WithOverlay<HashMap<ActorId, BankBalance>>> {
36    &BANK_ACCOUNTS
37}
38
39#[derive(Default, Debug, Clone, Copy)]
40pub(super) struct BankBalance {
41    pub(super) gas: Value,
42    pub(super) value: Value,
43}
44
45/// `gtest` bank.
46#[derive(Default, Debug)]
47pub(crate) struct Bank;
48
49impl Bank {
50    // Create a new bank.
51    pub(crate) fn deposit_value(&self, id: ActorId, value: Value, keep_alive: bool) {
52        Accounts::decrease(id, value, keep_alive);
53        storage().with(|accs| {
54            accs.data_mut()
55                .entry(id)
56                .or_insert(BankBalance { gas: 0, value: 0 })
57                .value += value;
58        });
59    }
60
61    // Deposit gas.
62    pub(crate) fn deposit_gas(&self, id: ActorId, gas: Gas, keep_alive: bool) {
63        let gas_value = GAS_MULTIPLIER.gas_to_value(gas);
64        Accounts::decrease(id, gas_value, keep_alive);
65        storage().with(|accs| {
66            accs.data_mut()
67                .entry(id)
68                .or_insert(BankBalance { gas: 0, value: 0 })
69                .gas += gas_value;
70        });
71    }
72
73    // Withdraw gas.
74    pub(crate) fn spend_gas(&self, id: ActorId, gas: Gas, multiplier: GasMultiplier<Value, Gas>) {
75        let gas_value = multiplier.gas_to_value(gas);
76        storage().with(|accs| {
77            accs.data_mut()
78                .get_mut(&id)
79                .unwrap_or_else(|| panic!("Bank::spend_gas: actor id {id:?} not found in bank"))
80                .gas -= gas_value;
81        });
82    }
83
84    // Withdraw gas.
85    pub(crate) fn withdraw_gas(
86        &self,
87        id: ActorId,
88        gas_left: Gas,
89        multiplier: GasMultiplier<Value, Gas>,
90    ) {
91        let gas_left_value = multiplier.gas_to_value(gas_left);
92        storage().with(|accs| {
93            accs.data_mut()
94                .get_mut(&id)
95                .unwrap_or_else(|| panic!("Bank::spend_gas: actor id {id:?} not found in bank"))
96                .gas -= gas_left_value;
97        });
98
99        if !Accounts::can_deposit(id, gas_left_value) {
100            // Unable to deposit value to account.
101            // In this case unused value will be lost.
102            return;
103        }
104
105        Accounts::increase(id, gas_left_value);
106    }
107
108    // Transfer value.
109    pub(crate) fn transfer_value(&self, from: ActorId, to: ActorId, value: Value) {
110        if value == 0 {
111            return;
112        }
113
114        storage().with(|accs| {
115            accs.data_mut()
116                .get_mut(&from)
117                .unwrap_or_else(|| {
118                    panic!("Bank::transfer_value: actor id {from:?} not found in bank")
119                })
120                .value -= value;
121        });
122
123        if !Accounts::can_deposit(to, value) {
124            // Unable to deposit value to account.
125            // In this case unused value will be lost.
126            return;
127        }
128
129        Accounts::increase(to, value);
130    }
131
132    // Transfer locked value.
133    pub(crate) fn transfer_locked_value(&mut self, from: ActorId, to: ActorId, value: Value) {
134        if value == 0 {
135            return;
136        }
137
138        storage().with(|accs| {
139            accs.data_mut()
140                .get_mut(&from)
141                .unwrap_or_else(|| {
142                    panic!("Bank::transfer_value: actor id {from:?} not found in bank")
143                })
144                .value -= value;
145        });
146
147        storage().with(|accs| {
148            accs.data_mut()
149                .entry(to)
150                .or_insert(BankBalance { gas: 0, value: 0 })
151                .value += value;
152        });
153    }
154
155    pub(crate) fn clear(&self) {
156        storage().with(|accs| {
157            accs.data_mut().clear();
158        });
159    }
160}
