23pub use pallet_gear::{GasInfo, ReplyInfo, manager::HandleKind};
24use sp_core::H256;
25use sp_runtime::traits::Block as BlockT;
26use sp_std::vec::Vec;
27
28sp_api::decl_runtime_apis! {
29    #[api_version(2)]
30    pub trait GearApi {
31        #[allow(clippy::too_many_arguments)]
32        fn calculate_reply_for_handle(origin: H256, destination: H256, payload: Vec<u8>, gas_limit: u64, value: u128, allowance_multiplier: u64) -> Result<ReplyInfo, Vec<u8>>;
33
34        #[allow(clippy::too_many_arguments)]
35        fn calculate_gas_info(source: H256, kind: HandleKind, payload: Vec<u8>, value: u128, allow_other_panics: bool, initial_gas: Option<u64>, allowance_multiplier: Option<u64>) -> Result<GasInfo, Vec<u8>>;
36
37        /// Generate inherent-like extrinsic that runs message queue processing.
38        fn gear_run_extrinsic(max_gas: Option<u64>) -> <Block as BlockT>::Extrinsic;
39
40        fn read_state(program_id: H256, payload: Vec<u8>, allowance_multiplier: Option<u64>) -> Result<Vec<u8>, Vec<u8>>;
41
42        #[allow(clippy::too_many_arguments)]
43        fn read_state_using_wasm(program_id: H256, payload: Vec<u8>, fn_name: Vec<u8>, wasm: Vec<u8>, argument: Option<Vec<u8>>, allowance_multiplier: Option<u64>) -> Result<Vec<u8>, Vec<u8>>;
44
45        fn read_metahash(program_id: H256, allowance_multiplier: Option<u64>) -> Result<H256, Vec<u8>>;
46
47        // DEPRECATED APIS
48
49        #[allow(clippy::too_many_arguments)]
50        #[changed_in(2)]
51        fn calculate_gas_info(source: H256, kind: HandleKind, payload: Vec<u8>, value: u128, allow_other_panics: bool, initial_gas: Option<u64>) -> Result<GasInfo, Vec<u8>>;
52
53        #[changed_in(2)]
54        fn read_state(program_id: H256, payload: Vec<u8>) -> Result<Vec<u8>, Vec<u8>>;
55
56        #[allow(clippy::too_many_arguments)]
57        #[changed_in(2)]
58        fn read_state_using_wasm(program_id: H256, payload: Vec<u8>, fn_name: Vec<u8>, wasm: Vec<u8>, argument: Option<Vec<u8>>) -> Result<Vec<u8>, Vec<u8>>;
59
60        #[changed_in(2)]
61        fn read_metahash(program_id: H256) -> Result<H256, Vec<u8>>;
62    }
63}
