48/// Execute wasm with dispatch and return dispatch result.
49pub(crate) fn execute_wasm<Ext>(
50    balance: u128,
51    dispatch: IncomingDispatch,
52    context: WasmExecutionContext,
53    settings: ExecutionSettings,
54    msg_ctx_settings: ContextSettings,
55) -> Result<DispatchResult, ExecutionError>
56where
57    Ext: ProcessorExternalities + BackendExternalities + Send + 'static,
58    <Ext as Externalities>::AllocError:
59        BackendAllocSyscallError<ExtError = Ext::UnrecoverableError>,
60    RunFallibleError: From<Ext::FallibleError>,
61    <Ext as Externalities>::UnrecoverableError: BackendSyscallError,
62{
63    let WasmExecutionContext {
64        gas_counter,
65        gas_allowance_counter,
66        gas_reserver,
67        program,
68        memory_size,
69    } = context;
70
71    // TODO: consider avoiding cloning here.
72    let initial_gas_reserver = gas_reserver.clone();
73
74    let kind = dispatch.kind();
75
76    log::debug!("Executing program {}", program.id);
77    log::debug!("Executing dispatch {dispatch:?}");
78
79    // Creating allocations context.
80    let allocations_context = AllocationsContext::try_new(
81        memory_size,
82        program.allocations,
83        program.code_metadata.static_pages(),
84        program.code_metadata.stack_end(),
85        settings.max_pages,
86    )
87    .map_err(SystemExecutionError::from)?;
88
89    // Creating value counter.
90    //
91    // NOTE: Value available equals free balance with message value if value
92    // wasn't transferred to program yet.
93    //
94    // In case of second execution (between waits) - message value already
95    // included in free balance or wasted.
96    let value_available = balance.saturating_add(if dispatch.context().is_none() {
97        dispatch.value()
98    } else {
99        Default::default()
100    });
101    let value_counter = ValueCounter::new(value_available);
102
103    // Creating message context.
104    let message_context = MessageContext::new(dispatch, program.id, msg_ctx_settings);
105
106    let context = ProcessorContext {
107        gas_counter,
108        gas_allowance_counter,
109        gas_reserver,
110        system_reservation: None,
111        value_counter,
112        allocations_context,
113        message_context,
114        block_info: settings.block_info,
115        performance_multiplier: settings.performance_multiplier,
116        program_id: program.id,
117        program_candidates_data: Default::default(),
118        forbidden_funcs: settings.forbidden_funcs,
119        reserve_for: settings.reserve_for,
120        random_data: settings.random_data,
121        gas_multiplier: settings.gas_multiplier,
122        existential_deposit: settings.existential_deposit,
123        mailbox_threshold: settings.mailbox_threshold,
124        costs: settings.ext_costs,
125    };
126
127    // Creating externalities.
128    let ext = Ext::new(context);
129
130    // Execute program in backend env.
131    let execute = || {
132        let env = Environment::new(
133            ext,
134            program.instrumented_code.bytes(),
135            kind,
136            program.code_metadata.exports().clone(),
137            memory_size,
138        )?;
139        env.execute(|ctx, memory, globals_config| {
140            Ext::lazy_pages_init_for_program(
141                ctx,
142                memory,
143                program.id,
144                program.memory_infix,
145                program.code_metadata.stack_end(),
146                globals_config,
147                settings.lazy_pages_costs,
148            )
149        })
150    };
151
152    let (termination, mut store, memory, ext) = match execute() {
153        Ok(report) => {
154            let BackendReport {
155                termination_reason,
156                mut store,
157                mut memory,
158                ext,
159            } = report;
160
161            let mut termination = match termination_reason {
162                TerminationReason::Actor(reason) => reason,
163                TerminationReason::System(reason) => {
164                    return Err(ExecutionError::System(reason.into()));
165                }
166            };
167
168            // released pages initial data will be added to `pages_initial_data` after execution.
169            Ext::lazy_pages_post_execution_actions(&mut store, &mut memory);
170
171            if !Ext::lazy_pages_status().is_normal() {
172                termination = ext.current_counter_type().into()
173            }
174
175            (termination, store, memory, ext)
176        }
177        Err(EnvironmentError::System(e)) => {
178            return Err(ExecutionError::System(SystemExecutionError::Environment(e)));
179        }
180        Err(EnvironmentError::Actor(gas_amount, err)) => {
181            log::trace!("ActorExecutionErrorReplyReason::Environment({err}) occurred");
182            return Err(ExecutionError::Actor(ActorExecutionError {
183                gas_amount,
184                reason: ActorExecutionErrorReplyReason::Environment,
185            }));
186        }
187    };
188
189    log::debug!("Termination reason: {termination:?}");
190
191    let info = ext
192        .into_ext_info(&mut store, &memory)
193        .map_err(SystemExecutionError::IntoExtInfo)?;
194
195    // Parsing outcome.
196    let kind = match termination {
197        ActorTerminationReason::Exit(value_dest) => DispatchResultKind::Exit(value_dest),
198        ActorTerminationReason::Leave | ActorTerminationReason::Success => {
199            DispatchResultKind::Success
200        }
201        ActorTerminationReason::Trap(explanation) => {
202            log::debug!(
203                "ðŸ’¥ Trap during execution of {}\nðŸ“” Explanation: {explanation}",
204                program.id
205            );
206            DispatchResultKind::Trap(explanation)
207        }
208        ActorTerminationReason::Wait(duration, waited_type) => {
209            DispatchResultKind::Wait(duration, waited_type)
210        }
211        ActorTerminationReason::GasAllowanceExceeded => DispatchResultKind::GasAllowanceExceed,
212    };
213
214    // With lazy-pages we update some page data in storage,
215    // when it has been write accessed, so no need to compare old and new page data.
216    let page_update = info.pages_data;
217
218    // Getting new programs that are scheduled to be initialized (respected messages are in `generated_dispatches` collection)
219    let program_candidates = info.program_candidates_data;
220
221    // Triggering update of gas reserver only in case of it changed.
222    let gas_reserver = initial_gas_reserver
223        .ne(&info.gas_reserver)
224        .then_some(info.gas_reserver);
225
226    // Output
227    Ok(DispatchResult {
228        kind,
229        program_id: program.id,
230        context_store: info.context_store,
231        generated_dispatches: info.generated_dispatches,
232        awakening: info.awakening,
233        reply_deposits: info.reply_deposits,
234        program_candidates,
235        gas_amount: info.gas_amount,
236        gas_reserver,
237        system_reservation_context: info.system_reservation_context,
238        page_update,
239        allocations: info.allocations,
240        reply_sent: info.reply_sent,
241    })
242}

244/// !!! FOR TESTING / INFORMATIONAL USAGE ONLY
245#[allow(clippy::too_many_arguments)]
246pub fn execute_for_reply<Ext, EP>(
247    function: EP,
248    instrumented_code: InstrumentedCode,
249    code_metadata: CodeMetadata,
250    allocations: Option<IntervalsTree<WasmPage>>,
251    program_info: Option<(ActorId, MemoryInfix)>,
252    payload: Vec<u8>,
253    gas_limit: u64,
254    block_info: BlockInfo,
255) -> Result<Vec<u8>, String>
256where
257    Ext: ProcessorExternalities + BackendExternalities + Send + 'static,
258    <Ext as Externalities>::AllocError:
259        BackendAllocSyscallError<ExtError = Ext::UnrecoverableError>,
260    RunFallibleError: From<Ext::FallibleError>,
261    <Ext as Externalities>::UnrecoverableError: BackendSyscallError,
262    EP: WasmEntryPoint,
263{
264    let (program_id, memory_infix) = program_info.unwrap_or_default();
265    let program = Program {
266        id: program_id,
267        memory_infix,
268        instrumented_code,
269        code_metadata,
270        allocations: allocations.unwrap_or_default(),
271    };
272    let static_pages = program.code_metadata.static_pages();
273    let memory_size = program
274        .allocations
275        .end()
276        .map(|p| p.inc())
277        .unwrap_or(static_pages);
278
279    let incoming_dispatch = IncomingDispatch::new(
280        DispatchKind::Handle,
281        IncomingMessage::new(
282            Default::default(),
283            Default::default(),
284            payload
285                .try_into()
286                .map_err(|e| format!("Failed to create payload: {e:?}"))?,
287            gas_limit,
288            Default::default(),
289            Default::default(),
290        ),
291        None,
292    );
293
294    let gas_reserver = GasReserver::new(&incoming_dispatch, Default::default(), Default::default());
295
296    let message_context = MessageContext::new(incoming_dispatch, program.id, Default::default());
297
298    let context = ProcessorContext {
299        gas_counter: GasCounter::new(gas_limit),
300        gas_allowance_counter: GasAllowanceCounter::new(gas_limit),
301        gas_reserver,
302        value_counter: ValueCounter::new(Default::default()),
303        allocations_context: AllocationsContext::try_new(
304            memory_size,
305            program.allocations,
306            static_pages,
307            program.code_metadata.stack_end(),
308            512.into(),
309        )
310        .map_err(|e| format!("Failed to create alloc ctx: {e:?}"))?,
311        message_context,
312        block_info,
313        performance_multiplier: gsys::Percent::new(100),
314        program_id: program.id,
315        program_candidates_data: Default::default(),
316        forbidden_funcs: Default::default(),
317        reserve_for: Default::default(),
318        random_data: Default::default(),
319        system_reservation: Default::default(),
320        gas_multiplier: gsys::GasMultiplier::from_value_per_gas(100),
321        existential_deposit: Default::default(),
322        mailbox_threshold: Default::default(),
323        costs: Default::default(),
324    };
325
326    // Creating externalities.
327    let ext = Ext::new(context);
328
329    // Execute program in backend env.
330    let execute = || {
331        let env = Environment::new(
332            ext,
333            program.instrumented_code.bytes(),
334            function,
335            program.code_metadata.exports().clone(),
336            memory_size,
337        )?;
338        env.execute(|ctx, memory, globals_config| {
339            Ext::lazy_pages_init_for_program(
340                ctx,
341                memory,
342                program_id,
343                program.memory_infix,
344                program.code_metadata.stack_end(),
345                globals_config,
346                Default::default(),
347            )
348        })
349    };
350
351    let (termination, mut store, memory, ext) = match execute() {
352        Ok(report) => {
353            let BackendReport {
354                termination_reason,
355                store,
356                memory,
357                ext,
358            } = report;
359
360            let termination_reason = match termination_reason {
361                TerminationReason::Actor(reason) => reason,
362                TerminationReason::System(reason) => {
363                    return Err(format!("Backend error: {reason}"));
364                }
365            };
366
367            (termination_reason, store, memory, ext)
368        }
369        Err(e) => return Err(format!("Backend error: {e}")),
370    };
371
372    match termination {
373        ActorTerminationReason::Exit(_)
374        | ActorTerminationReason::Leave
375        | ActorTerminationReason::Wait(_, _) => {
376            return Err("Execution has incorrect termination reason".into());
377        }
378        ActorTerminationReason::Success => (),
379        ActorTerminationReason::Trap(explanation) => {
380            return Err(format!(
381                "Program execution failed with error: {explanation}"
382            ));
383        }
384        ActorTerminationReason::GasAllowanceExceeded => return Err("Unreachable".into()),
385    };
386
387    let info = ext
388        .into_ext_info(&mut store, &memory)
389        .map_err(|e| format!("Backend postprocessing error: {e:?}"))?;
390
391    log::debug!(
392        "[execute_for_reply] Gas burned: {}",
393        info.gas_amount.burned()
394    );
395
396    for (dispatch, _, _) in info.generated_dispatches {
397        if matches!(dispatch.kind(), DispatchKind::Reply) {
398            return Ok(dispatch.payload_bytes().to_vec());
399        }
400    }
401
402    Err("Reply not found".into())
403}
