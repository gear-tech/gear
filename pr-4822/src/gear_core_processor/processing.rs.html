46/// Process program & dispatch for it and return journal for updates.
47pub fn process<Ext>(
48    block_config: &BlockConfig,
49    execution_context: ProcessExecutionContext,
50    random_data: (Vec<u8>, u32),
51) -> Result<Vec<JournalNote>, SystemExecutionError>
52where
53    Ext: ProcessorExternalities + BackendExternalities + Send + 'static,
54    <Ext as Externalities>::AllocError:
55        BackendAllocSyscallError<ExtError = Ext::UnrecoverableError>,
56    RunFallibleError: From<Ext::FallibleError>,
57    <Ext as Externalities>::UnrecoverableError: BackendSyscallError,
58{
59    use crate::common::SuccessfulDispatchResultKind::*;
60
61    let BlockConfig {
62        block_info,
63        performance_multiplier,
64        forbidden_funcs,
65        reserve_for,
66        gas_multiplier,
67        costs,
68        existential_deposit,
69        mailbox_threshold,
70        max_pages,
71        outgoing_limit,
72        outgoing_bytes_limit,
73        ..
74    } = block_config.clone();
75
76    let execution_settings = ExecutionSettings {
77        block_info,
78        performance_multiplier,
79        existential_deposit,
80        mailbox_threshold,
81        max_pages,
82        ext_costs: costs.ext,
83        lazy_pages_costs: costs.lazy_pages,
84        forbidden_funcs,
85        reserve_for,
86        random_data,
87        gas_multiplier,
88    };
89
90    let dispatch = execution_context.dispatch;
91    let balance = execution_context.balance;
92    let program_id = execution_context.program.id;
93    let initial_reservations_amount = execution_context.gas_reserver.states().len();
94
95    let execution_context = WasmExecutionContext {
96        gas_counter: execution_context.gas_counter,
97        gas_allowance_counter: execution_context.gas_allowance_counter,
98        gas_reserver: execution_context.gas_reserver,
99        program: execution_context.program,
100        memory_size: execution_context.memory_size,
101    };
102
103    // Sending fee: double write cost for addition and removal some time soon
104    // from queue.
105    //
106    // Scheduled sending fee: double write cost for addition and removal some time soon
107    // from queue and double write cost (addition and removal) for dispatch stash.
108    //
109    // Waiting fee: triple write cost for addition and removal some time soon
110    // from waitlist and enqueuing / sending error reply afterward.
111    //
112    // Waking fee: double write cost for removal from waitlist
113    // and further enqueueing.
114    let msg_ctx_settings = ContextSettings {
115        sending_fee: costs.db.write.cost_for(2.into()),
116        scheduled_sending_fee: costs.db.write.cost_for(4.into()),
117        waiting_fee: costs.db.write.cost_for(3.into()),
118        waking_fee: costs.db.write.cost_for(2.into()),
119        reservation_fee: costs.db.write.cost_for(2.into()),
120        outgoing_limit,
121        outgoing_bytes_limit,
122    };
123
124    // TODO: add tests that system reservation is successfully unreserved after
125    // actor execution error #3756.
126
127    // Get system reservation context in order to use it if actor execution error occurs.
128    let system_reservation_ctx = SystemReservationContext::from_dispatch(&dispatch);
129
130    let exec_result = executor::execute_wasm::<Ext>(
131        balance,
132        dispatch.clone(),
133        execution_context,
134        execution_settings,
135        msg_ctx_settings,
136    )
137    .map_err(|err| {
138        log::debug!("Wasm execution error: {err}");
139        err
140    });
141
142    match exec_result {
143        Ok(res) => {
144            match res.kind {
145                DispatchResultKind::Success
146                | DispatchResultKind::Wait(_, _)
147                | DispatchResultKind::Exit(_) => {
148                    // assert that after processing the initial reservation is less or equal to the current one.
149                    // during execution reservation amount might increase due to `system_reserve_gas` calls
150                    // thus making initial reservation less than current one.
151                    debug_assert!(
152                        res.context_store.system_reservation()
153                            >= system_reservation_ctx.previous_reservation
154                    );
155                    debug_assert!(
156                        system_reservation_ctx.previous_reservation
157                            == res.system_reservation_context.previous_reservation
158                    );
159                    debug_assert!(
160                        res.gas_reserver
161                            .as_ref()
162                            .map(|reserver| initial_reservations_amount <= reserver.states().len())
163                            .unwrap_or(true)
164                    );
165                }
166                // reservation does not change in case of failure
167                _ => (),
168            }
169            Ok(match res.kind {
170                DispatchResultKind::Trap(reason) => process_execution_error(
171                    dispatch,
172                    program_id,
173                    res.gas_amount.burned(),
174                    res.system_reservation_context,
175                    ActorExecutionErrorReplyReason::Trap(reason),
176                ),
177
178                DispatchResultKind::Success => process_success(Success, res, dispatch),
179                DispatchResultKind::Wait(duration, ref waited_type) => {
180                    process_success(Wait(duration, waited_type.clone()), res, dispatch)
181                }
182                DispatchResultKind::Exit(value_destination) => {
183                    process_success(Exit(value_destination), res, dispatch)
184                }
185                DispatchResultKind::GasAllowanceExceed => {
186                    process_allowance_exceed(dispatch, program_id, res.gas_amount.burned())
187                }
188            })
189        }
190        Err(ExecutionError::Actor(e)) => Ok(process_execution_error(
191            dispatch,
192            program_id,
193            e.gas_amount.burned(),
194            system_reservation_ctx,
195            e.reason,
196        )),
197        Err(ExecutionError::System(e)) => Err(e),
198    }
199}

201enum ProcessErrorCase {
202    /// Program exited.
203    ProgramExited {
204        /// Inheritor of an exited program.
205        inheritor: ActorId,
206    },
207    /// Program failed during init.
208    FailedInit,
209    /// Program is not initialized yet.
210    Uninitialized,
211    /// Given code id for program creation doesn't exist.
212    CodeNotExists,
213    /// Message is executable, but its execution failed due to re-instrumentation.
214    ReinstrumentationFailed,
215    /// Error is considered as an execution failure.
216    ExecutionFailed(ActorExecutionErrorReplyReason),
217}
218
219impl fmt::Display for ProcessErrorCase {
220    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
221        match self {
222            ProcessErrorCase::ExecutionFailed(reason) => fmt::Display::fmt(reason, f),
223            this => fmt::Display::fmt(&this.to_reason(), f),
224        }
225    }
226}
227
228impl ProcessErrorCase {
229    fn to_reason(&self) -> ErrorReplyReason {
230        match self {
231            ProcessErrorCase::ProgramExited { .. } => {
232                ErrorReplyReason::UnavailableActor(SimpleUnavailableActorError::ProgramExited)
233            }
234            ProcessErrorCase::FailedInit => ErrorReplyReason::UnavailableActor(
235                SimpleUnavailableActorError::InitializationFailure,
236            ),
237            ProcessErrorCase::Uninitialized => {
238                ErrorReplyReason::UnavailableActor(SimpleUnavailableActorError::Uninitialized)
239            }
240            ProcessErrorCase::CodeNotExists => {
241                ErrorReplyReason::UnavailableActor(SimpleUnavailableActorError::ProgramNotCreated)
242            }
243            ProcessErrorCase::ReinstrumentationFailed => ErrorReplyReason::UnavailableActor(
244                SimpleUnavailableActorError::ReinstrumentationFailure,
245            ),
246            ProcessErrorCase::ExecutionFailed(reason) => reason.as_simple().into(),
247        }
248    }
249
250    // TODO: consider to convert `self` into `Payload` to avoid `PanicBuffer` cloning (#4594)
251    fn to_payload(&self) -> Payload {
252        match self {
253            ProcessErrorCase::ProgramExited { inheritor } => {
254                const _: () = assert!(size_of::<ActorId>() <= Payload::MAX_LEN);
255                inheritor
256                    .into_bytes()
257                    .to_vec()
258                    .try_into()
259                    .unwrap_or_else(|PayloadSizeError| {
260                        unreachable!("`ActorId` is always smaller than maximum payload size")
261                    })
262            }
263            ProcessErrorCase::ExecutionFailed(ActorExecutionErrorReplyReason::Trap(
264                TrapExplanation::Panic(buf),
265            )) => buf.inner().clone(),
266            _ => Payload::default(),
267        }
268    }
269}

271fn process_error(
272    dispatch: IncomingDispatch,
273    program_id: ActorId,
274    gas_burned: u64,
275    system_reservation_ctx: SystemReservationContext,
276    case: ProcessErrorCase,
277) -> Vec<JournalNote> {
278    let mut journal = Vec::new();
279
280    let message_id = dispatch.id();
281    let origin = dispatch.source();
282    let value = dispatch.value();
283
284    journal.push(JournalNote::GasBurned {
285        message_id,
286        amount: gas_burned,
287    });
288
289    let to_send_reply = !matches!(dispatch.kind(), DispatchKind::Reply | DispatchKind::Signal);
290
291    // We check if value is greater than zero to don't provide
292    // no-op journal note.
293    //
294    // We also check if dispatch had context of previous executions:
295    // it's existence shows that we have processed message after
296    // being waken, so the value were already transferred in
297    // execution, where `gr_wait` was called.
298    if dispatch.context().is_none() && value != 0 {
299        // Value on error is always delivered to the program, but may return with error reply.
300        journal.push(JournalNote::SendValue {
301            from: origin,
302            to: program_id,
303            value,
304            // in case of upcoming error reply, we want to send locked value,
305            // instead of deposit, to avoid ED manipulations.
306            locked: to_send_reply,
307        });
308    }
309
310    if let Some(amount) = system_reservation_ctx.current_reservation {
311        journal.push(JournalNote::SystemReserveGas { message_id, amount });
312    }
313
314    if let ProcessErrorCase::ExecutionFailed(reason) = &case {
315        // TODO: consider to handle error reply and init #3701
316        if system_reservation_ctx.has_any()
317            && !dispatch.is_error_reply()
318            && !matches!(dispatch.kind(), DispatchKind::Signal | DispatchKind::Init)
319        {
320            journal.push(JournalNote::SendSignal {
321                message_id,
322                destination: program_id,
323                code: SignalCode::Execution(reason.as_simple()),
324            });
325        }
326    }
327
328    if system_reservation_ctx.has_any() {
329        journal.push(JournalNote::SystemUnreserveGas { message_id });
330    }
331
332    if to_send_reply {
333        let err = case.to_reason();
334        let err_payload = case.to_payload();
335
336        let value = if dispatch.context().is_none() {
337            value
338        } else {
339            0
340        };
341
342        // # Safety
343        //
344        // 1. The dispatch.id() has already been checked
345        // 2. This reply message is generated by our system
346        //
347        // So, the message id of this reply message will not be duplicated.
348        let dispatch = ReplyMessage::system(dispatch.id(), err_payload, value, err).into_dispatch(
349            program_id,
350            dispatch.source(),
351            dispatch.id(),
352        );
353
354        journal.push(JournalNote::SendDispatch {
355            message_id,
356            dispatch,
357            delay: 0,
358            reservation: None,
359        });
360    }
361
362    let outcome = match case {
363        ProcessErrorCase::ExecutionFailed { .. } | ProcessErrorCase::ReinstrumentationFailed => {
364            let err_msg = case.to_string();
365            match dispatch.kind() {
366                DispatchKind::Init => DispatchOutcome::InitFailure {
367                    program_id,
368                    origin,
369                    reason: err_msg,
370                },
371                _ => DispatchOutcome::MessageTrap {
372                    program_id,
373                    trap: err_msg,
374                },
375            }
376        }
377        ProcessErrorCase::ProgramExited { .. }
378        | ProcessErrorCase::FailedInit
379        | ProcessErrorCase::Uninitialized
380        | ProcessErrorCase::CodeNotExists => DispatchOutcome::NoExecution,
381    };
382
383    journal.push(JournalNote::MessageDispatched {
384        message_id,
385        source: origin,
386        outcome,
387    });
388    journal.push(JournalNote::MessageConsumed(message_id));
389
390    journal
391}

393/// Helper function for journal creation in trap/error case.
394pub fn process_execution_error(
395    dispatch: IncomingDispatch,
396    program_id: ActorId,
397    gas_burned: u64,
398    system_reservation_ctx: SystemReservationContext,
399    err: impl Into<ActorExecutionErrorReplyReason>,
400) -> Vec<JournalNote> {

402        dispatch,
403        program_id,
404        gas_burned,
405        system_reservation_ctx,
406        ProcessErrorCase::ExecutionFailed(err.into()),

408}

509/// Helper function for journal creation in success case
510pub fn process_success(
511    kind: SuccessfulDispatchResultKind,
512    dispatch_result: DispatchResult,
513    dispatch: IncomingDispatch,
514) -> Vec<JournalNote> {
515    use crate::common::SuccessfulDispatchResultKind::*;
516
517    let DispatchResult {
518        generated_dispatches,
519        awakening,
520        program_candidates,
521        gas_amount,
522        gas_reserver,
523        system_reservation_context,
524        page_update,
525        program_id,
526        context_store,
527        allocations,
528        reply_deposits,
529        reply_sent,
530        ..
531    } = dispatch_result;
532
533    let mut journal = Vec::new();
534
535    let message_id = dispatch.id();
536    let origin = dispatch.source();
537    let value = dispatch.value();
538
539    journal.push(JournalNote::GasBurned {
540        message_id,
541        amount: gas_amount.burned(),
542    });
543
544    if let Some(gas_reserver) = gas_reserver {
545        journal.extend(gas_reserver.states().iter().flat_map(
546            |(&reservation_id, &state)| match state {
547                GasReservationState::Exists { .. } => None,
548                GasReservationState::Created {
549                    amount, duration, ..
550                } => Some(JournalNote::ReserveGas {
551                    message_id,
552                    reservation_id,
553                    program_id,
554                    amount,
555                    duration,
556                }),
557                GasReservationState::Removed { expiration } => Some(JournalNote::UnreserveGas {
558                    reservation_id,
559                    program_id,
560                    expiration,
561                }),
562            },
563        ));
564
565        journal.push(JournalNote::UpdateGasReservations {
566            program_id,
567            reserver: gas_reserver,
568        });
569    }
570
571    if let Some(amount) = system_reservation_context.current_reservation {
572        journal.push(JournalNote::SystemReserveGas { message_id, amount });
573    }
574
575    // We check if value is greater than zero to don't provide
576    // no-op journal note.
577    //
578    // We also check if dispatch had context of previous executions:
579    // it's existence shows that we have processed message after
580    // being waken, so the value were already transferred in
581    // execution, where `gr_wait` was called.
582    if dispatch.context().is_none() && value != 0 {
583        // Send value further
584        journal.push(JournalNote::SendValue {
585            from: origin,
586            to: program_id,
587            value,
588            locked: false,
589        });
590    }
591
592    // Must be handled before handling generated dispatches.
593    for (code_id, candidates) in program_candidates {
594        journal.push(JournalNote::StoreNewPrograms {
595            program_id,
596            code_id,
597            candidates,
598        });
599    }
600
601    // Sending auto-generated reply about success execution.
602    if !matches!(kind, SuccessfulDispatchResultKind::Wait(_, _))
603        && !matches!(dispatch.kind(), DispatchKind::Reply | DispatchKind::Signal)
604        && !reply_sent
605    {
606        let auto_reply = ReplyMessage::auto(dispatch.id()).into_dispatch(
607            program_id,
608            dispatch.source(),
609            dispatch.id(),
610        );
611
612        journal.push(JournalNote::SendDispatch {
613            message_id,
614            dispatch: auto_reply,
615            delay: 0,
616            reservation: None,
617        });
618    }
619
620    for (message_id_sent, amount) in reply_deposits {
621        journal.push(JournalNote::ReplyDeposit {
622            message_id,
623            future_reply_id: MessageId::generate_reply(message_id_sent),
624            amount,
625        });
626    }
627
628    for (dispatch, delay, reservation) in generated_dispatches {
629        journal.push(JournalNote::SendDispatch {
630            message_id,
631            dispatch,
632            delay,
633            reservation,
634        });
635    }
636
637    for (awakening_id, delay) in awakening {
638        journal.push(JournalNote::WakeMessage {
639            message_id,
640            program_id,
641            awakening_id,
642            delay,
643        });
644    }
645
646    for (page_number, data) in page_update {
647        journal.push(JournalNote::UpdatePage {
648            program_id,
649            page_number,
650            data,
651        })
652    }
653
654    if let Some(allocations) = allocations {
655        journal.push(JournalNote::UpdateAllocations {
656            program_id,
657            allocations,
658        });
659    }
660
661    let outcome = match kind {
662        Wait(duration, waited_type) => {
663            journal.push(JournalNote::WaitDispatch {
664                dispatch: dispatch.into_stored(program_id, context_store),
665                duration,
666                waited_type,
667            });
668
669            return journal;
670        }
671        Success => match dispatch.kind() {
672            DispatchKind::Init => DispatchOutcome::InitSuccess { program_id },
673            _ => DispatchOutcome::Success,
674        },
675        Exit(value_destination) => {
676            journal.push(JournalNote::ExitDispatch {
677                id_exited: program_id,
678                value_destination,
679            });
680
681            DispatchOutcome::Exit { program_id }
682        }
683    };
684
685    if system_reservation_context.has_any() {
686        journal.push(JournalNote::SystemUnreserveGas { message_id });
687    }
688
689    journal.push(JournalNote::MessageDispatched {
690        message_id,
691        source: origin,
692        outcome,
693    });
694    journal.push(JournalNote::MessageConsumed(message_id));
695    journal
696}

698/// Helper function for journal creation if the block gas allowance has been exceeded.
699pub fn process_allowance_exceed(
700    dispatch: IncomingDispatch,
701    program_id: ActorId,
702    gas_burned: u64,
703) -> Vec<JournalNote> {
704    let mut journal = Vec::with_capacity(1);
705
706    let (kind, message, opt_context) = dispatch.into_parts();
707
708    let dispatch = StoredDispatch::new(kind, message.into_stored(program_id), opt_context);
709
710    journal.push(JournalNote::StopProcessing {
711        dispatch,
712        gas_burned,
713    });
714
715    journal
716}
