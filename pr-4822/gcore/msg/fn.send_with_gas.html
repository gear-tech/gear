Same as send, but with an explicit gas limit.

use gcore::msg;

#[unsafe(no_mangle)]
extern "C" fn handle() {
    let id: [u8; 32] = core::array::from_fn(|i| i as u8);
    msg::send_with_gas(id.into(), b"HELLO", 5_000_000, 42).expect("Unable to send");
}
