48/// Execute wasm with dispatch and return dispatch result.
49pub(crate) fn execute_wasm<Ext>(
50    balance: u128,
51    dispatch: &IncomingDispatch,
52    context: WasmExecutionContext,
53    settings: ExecutionSettings,
54    msg_ctx_settings: ContextSettings,
55) -> Result<DispatchResult, ExecutionError>
56where
57    Ext: ProcessorExternalities + BackendExternalities + Send + 'static,
58    <Ext as Externalities>::AllocError:
59        BackendAllocSyscallError<ExtError = Ext::UnrecoverableError>,
60    RunFallibleError: From<Ext::FallibleError>,
61    <Ext as Externalities>::UnrecoverableError: BackendSyscallError,
62{
63    let WasmExecutionContext {
64        gas_counter,
65        gas_allowance_counter,
66        gas_reserver,
67        program,
68        memory_size,
69    } = context;
70
71    // TODO: consider avoiding cloning here.
72    let initial_gas_reserver = gas_reserver.clone();
73
74    let kind = dispatch.kind();
75
76    log::debug!("Executing program {}", program.id);
77    log::debug!("Executing dispatch {:?}", dispatch);
78
79    // Creating allocations context.
80    let allocations_context = AllocationsContext::try_new(
81        memory_size,
82        program.allocations,
83        program.code.static_pages(),
84        program.code.stack_end(),
85        settings.max_pages,
86    )
87    .map_err(SystemExecutionError::from)?;
88
89    // Creating value counter.
90    //
91    // NOTE: Value available equals free balance with message value if value
92    // wasn't transferred to program yet.
93    //
94    // In case of second execution (between waits) - message value already
95    // included in free balance or wasted.
96    let value_available = balance.saturating_add(if dispatch.context().is_none() {
97        dispatch.value()
98    } else {
99        Default::default()
100    });
101    let value_counter = ValueCounter::new(value_available);
102
103    // Creating message context.
104    let message_context = MessageContext::new(
105        // dispatch clone is cheap because is uses `Arc` for payload
106        dispatch.clone(),
107        program.id,
108        msg_ctx_settings,
109    );
110
111    let context = ProcessorContext {
112        gas_counter,
113        gas_allowance_counter,
114        gas_reserver,
115        system_reservation: None,
116        value_counter,
117        allocations_context,
118        message_context,
119        block_info: settings.block_info,
120        performance_multiplier: settings.performance_multiplier,
121        program_id: program.id,
122        program_candidates_data: Default::default(),
123        forbidden_funcs: settings.forbidden_funcs,
124        reserve_for: settings.reserve_for,
125        random_data: settings.random_data,
126        gas_multiplier: settings.gas_multiplier,
127        existential_deposit: settings.existential_deposit,
128        mailbox_threshold: settings.mailbox_threshold,
129        costs: settings.ext_costs,
130    };
131
132    // Creating externalities.
133    let ext = Ext::new(context);
134
135    // Execute program in backend env.
136    let execute = || {
137        let env = Environment::new(
138            ext,
139            program.code.code(),
140            kind,
141            program.code.exports().clone(),
142            memory_size,
143        )?;
144        env.execute(|ctx, memory, globals_config| {
145            Ext::lazy_pages_init_for_program(
146                ctx,
147                memory,
148                program.id,
149                program.memory_infix,
150                program.code.stack_end(),
151                globals_config,
152                settings.lazy_pages_costs,
153            )
154        })
155    };
156
157    let (termination, mut store, memory, ext) = match execute() {
158        Ok(report) => {
159            let BackendReport {
160                termination_reason,
161                mut store,
162                mut memory,
163                ext,
164            } = report;
165
166            let mut termination = match termination_reason {
167                TerminationReason::Actor(reason) => reason,
168                TerminationReason::System(reason) => {
169                    return Err(ExecutionError::System(reason.into()));
170                }
171            };
172
173            // released pages initial data will be added to `pages_initial_data` after execution.
174            Ext::lazy_pages_post_execution_actions(&mut store, &mut memory);
175
176            if !Ext::lazy_pages_status().is_normal() {
177                termination = ext.current_counter_type().into()
178            }
179
180            (termination, store, memory, ext)
181        }
182        Err(EnvironmentError::System(e)) => {
183            return Err(ExecutionError::System(SystemExecutionError::Environment(e)));
184        }
185        Err(EnvironmentError::Actor(gas_amount, err)) => {
186            log::trace!("ActorExecutionErrorReplyReason::Environment({err}) occurred");
187            return Err(ExecutionError::Actor(ActorExecutionError {
188                gas_amount,
189                reason: ActorExecutionErrorReplyReason::Environment,
190            }));
191        }
192    };
193
194    log::debug!("Termination reason: {:?}", termination);
195
196    let info = ext
197        .into_ext_info(&mut store, &memory)
198        .map_err(SystemExecutionError::IntoExtInfo)?;
199
200    // Parsing outcome.
201    let kind = match termination {
202        ActorTerminationReason::Exit(value_dest) => DispatchResultKind::Exit(value_dest),
203        ActorTerminationReason::Leave | ActorTerminationReason::Success => {
204            DispatchResultKind::Success
205        }
206        ActorTerminationReason::Trap(explanation) => {
207            log::debug!(
208                "💥 Trap during execution of {}\n📔 Explanation: {explanation}",
209                program.id
210            );
211            DispatchResultKind::Trap(explanation)
212        }
213        ActorTerminationReason::Wait(duration, waited_type) => {
214            DispatchResultKind::Wait(duration, waited_type)
215        }
216        ActorTerminationReason::GasAllowanceExceeded => DispatchResultKind::GasAllowanceExceed,
217    };
218
219    // With lazy-pages we update some page data in storage,
220    // when it has been write accessed, so no need to compare old and new page data.
221    let page_update = info.pages_data;
222
223    // Getting new programs that are scheduled to be initialized (respected messages are in `generated_dispatches` collection)
224    let program_candidates = info.program_candidates_data;
225
226    // Triggering update of gas reserver only in case of it changed.
227    let gas_reserver = initial_gas_reserver
228        .ne(&info.gas_reserver)
229        .then_some(info.gas_reserver);
230
231    // Output
232    Ok(DispatchResult {
233        kind,
234        // dispatch clone is cheap because is uses `Arc` for payload
235        dispatch: dispatch.clone(),
236        program_id: program.id,
237        context_store: info.context_store,
238        generated_dispatches: info.generated_dispatches,
239        awakening: info.awakening,
240        reply_deposits: info.reply_deposits,
241        program_candidates,
242        gas_amount: info.gas_amount,
243        gas_reserver,
244        system_reservation_context: info.system_reservation_context,
245        page_update,
246        allocations: info.allocations,
247        reply_sent: info.reply_sent,
248    })
249}

251/// !!! FOR TESTING / INFORMATIONAL USAGE ONLY
252#[allow(clippy::too_many_arguments)]
253pub fn execute_for_reply<Ext, EP>(
254    function: EP,
255    instrumented_code: InstrumentedCode,
256    allocations: Option<IntervalsTree<WasmPage>>,
257    program_info: Option<(ActorId, MemoryInfix)>,
258    payload: Vec<u8>,
259    gas_limit: u64,
260    block_info: BlockInfo,
261) -> Result<Vec<u8>, String>
262where
263    Ext: ProcessorExternalities + BackendExternalities + Send + 'static,
264    <Ext as Externalities>::AllocError:
265        BackendAllocSyscallError<ExtError = Ext::UnrecoverableError>,
266    RunFallibleError: From<Ext::FallibleError>,
267    <Ext as Externalities>::UnrecoverableError: BackendSyscallError,
268    EP: WasmEntryPoint,
269{
270    let (program_id, memory_infix) = program_info.unwrap_or_default();
271    let program = Program {
272        id: program_id,
273        memory_infix,
274        code: instrumented_code,
275        allocations: allocations.unwrap_or_default(),
276    };
277    let static_pages = program.code.static_pages();
278    let memory_size = program
279        .allocations
280        .end()
281        .map(|p| p.inc())
282        .unwrap_or(static_pages);
283
284    let message_context = MessageContext::new(
285        IncomingDispatch::new(
286            DispatchKind::Handle,
287            IncomingMessage::new(
288                Default::default(),
289                Default::default(),
290                payload
291                    .try_into()
292                    .map_err(|e| format!("Failed to create payload: {e:?}"))?,
293                gas_limit,
294                Default::default(),
295                Default::default(),
296            ),
297            None,
298        ),
299        program.id,
300        Default::default(),
301    );
302
303    let context = ProcessorContext {
304        gas_counter: GasCounter::new(gas_limit),
305        gas_allowance_counter: GasAllowanceCounter::new(gas_limit),
306        gas_reserver: GasReserver::new(&Default::default(), Default::default(), Default::default()),
307        value_counter: ValueCounter::new(Default::default()),
308        allocations_context: AllocationsContext::try_new(
309            memory_size,
310            program.allocations,
311            static_pages,
312            program.code.stack_end(),
313            512.into(),
314        )
315        .map_err(|e| format!("Failed to create alloc ctx: {e:?}"))?,
316        message_context,
317        block_info,
318        performance_multiplier: gsys::Percent::new(100),
319        program_id: program.id,
320        program_candidates_data: Default::default(),
321        forbidden_funcs: Default::default(),
322        reserve_for: Default::default(),
323        random_data: Default::default(),
324        system_reservation: Default::default(),
325        gas_multiplier: gsys::GasMultiplier::from_value_per_gas(100),
326        existential_deposit: Default::default(),
327        mailbox_threshold: Default::default(),
328        costs: Default::default(),
329    };
330
331    // Creating externalities.
332    let ext = Ext::new(context);
333
334    // Execute program in backend env.
335    let execute = || {
336        let env = Environment::new(
337            ext,
338            program.code.code(),
339            function,
340            program.code.exports().clone(),
341            memory_size,
342        )?;
343        env.execute(|ctx, memory, globals_config| {
344            Ext::lazy_pages_init_for_program(
345                ctx,
346                memory,
347                program_id,
348                program.memory_infix,
349                program.code.stack_end(),
350                globals_config,
351                Default::default(),
352            )
353        })
354    };
355
356    let (termination, mut store, memory, ext) = match execute() {
357        Ok(report) => {
358            let BackendReport {
359                termination_reason,
360                store,
361                memory,
362                ext,
363            } = report;
364
365            let termination_reason = match termination_reason {
366                TerminationReason::Actor(reason) => reason,
367                TerminationReason::System(reason) => {
368                    return Err(format!("Backend error: {reason}"));
369                }
370            };
371
372            (termination_reason, store, memory, ext)
373        }
374        Err(e) => return Err(format!("Backend error: {e}")),
375    };
376
377    match termination {
378        ActorTerminationReason::Exit(_)
379        | ActorTerminationReason::Leave
380        | ActorTerminationReason::Wait(_, _) => {
381            return Err("Execution has incorrect termination reason".into());
382        }
383        ActorTerminationReason::Success => (),
384        ActorTerminationReason::Trap(explanation) => {
385            return Err(format!(
386                "Program execution failed with error: {explanation}"
387            ));
388        }
389        ActorTerminationReason::GasAllowanceExceeded => return Err("Unreachable".into()),
390    };
391
392    let info = ext
393        .into_ext_info(&mut store, &memory)
394        .map_err(|e| format!("Backend postprocessing error: {e:?}"))?;
395
396    log::debug!(
397        "[execute_for_reply] Gas burned: {}",
398        info.gas_amount.burned()
399    );
400
401    for (dispatch, _, _) in info.generated_dispatches {
402        if matches!(dispatch.kind(), DispatchKind::Reply) {
403            return Ok(dispatch.payload_bytes().to_vec());
404        }
405    }
406
407    Err("Reply not found".into())
408}
