19//! Actors storage.
20
21use std::{cell::RefCell, collections::BTreeMap, fmt};
22
23use core_processor::common::ExecutableActorData;
24use gear_common::{ActorId, CodeId, GearPage, MessageId, PageBuf};
25use gear_core::{
26    code::InstrumentedCode,
27    pages::{numerated::tree::IntervalsTree, WasmPage},
28    reservation::GasReservationMap,
29};
30
31thread_local! {
32    static ACTORS_STORAGE: RefCell<BTreeMap<ActorId, TestActor>> = RefCell::new(Default::default());
33}
34
35pub(crate) struct Actors;
36
37impl Actors {
38    // Accesses actor by program id.
39
40    pub(crate) fn access<R>(
41        program_id: ActorId,
42        access: impl FnOnce(Option<&TestActor>) -> R,
43    ) -> R {
44        ACTORS_STORAGE.with_borrow(|storage| access(storage.get(&program_id)))
45    }
46
47    // Modifies actor by program id.
48    pub(crate) fn modify<R>(
49        program_id: ActorId,
50        modify: impl FnOnce(Option<&mut TestActor>) -> R,
51    ) -> R {
52        ACTORS_STORAGE.with_borrow_mut(|storage| modify(storage.get_mut(&program_id)))
53    }
54
55    // Inserts actor by program id.
56    pub(crate) fn insert(program_id: ActorId, actor: TestActor) -> Option<TestActor> {
57        ACTORS_STORAGE.with_borrow_mut(|storage| storage.insert(program_id, actor))
58    }
59
60    // Checks if actor by program id exists.
61    pub(crate) fn contains_key(program_id: ActorId) -> bool {
62        ACTORS_STORAGE.with_borrow(|storage| storage.contains_key(&program_id))
63    }
64
65    // Checks if actor by program id is a user.
66    pub(crate) fn is_user(id: ActorId) -> bool {
67        // Non-existent program is a user
68        ACTORS_STORAGE.with_borrow(|storage| storage.get(&id).is_none())
69    }
70
71    // Checks if actor by program id is active.
72    pub(crate) fn is_active_program(id: ActorId) -> bool {
73        ACTORS_STORAGE.with_borrow(|storage| {
74            matches!(
75                storage.get(&id),
76                Some(TestActor::Initialized(_) | TestActor::Uninitialized(_, _))
77            )
78        })
79    }
80
81    // Checks if actor by program id is a program.
82    pub(crate) fn is_program(id: ActorId) -> bool {
83        // if it's not a user, then it's a program
84        !Self::is_user(id)
85    }
86
87    // Returns all program ids.
88    pub(crate) fn program_ids() -> Vec<ActorId> {
89        ACTORS_STORAGE.with_borrow(|storage| storage.keys().copied().collect())
90    }
91
92    // Clears actors storage.
93    pub(crate) fn clear() {
94        ACTORS_STORAGE.with_borrow_mut(|storage| storage.clear())
95    }
96}
97
98impl fmt::Debug for Actors {
99    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
100        ACTORS_STORAGE.with_borrow(|storage| f.debug_map().entries(storage.iter()).finish())
101    }
102}

104#[derive(Debug)]
105pub(crate) enum TestActor {
106    Initialized(Program),
107    // Contract: program is always `Some`, option is used to take ownership
108    Uninitialized(Option<MessageId>, Option<Program>),
109    FailedInit,
110    Exited(ActorId),
111}
112
113impl TestActor {
114    // Creates a new uninitialized actor.
115    pub(crate) fn new(init_message_id: Option<MessageId>, program: Program) -> Self {
116        TestActor::Uninitialized(init_message_id, Some(program))
117    }
118
119    // # Panics
120    // If actor is initialized or dormant
121    pub(crate) fn set_initialized(&mut self) {
122        assert!(
123            self.is_uninitialized(),
124            "can't transmute actor, which isn't uninitialized"
125        );
126
127        if let TestActor::Uninitialized(_, maybe_prog) = self {
128            *self = TestActor::Initialized(
129                maybe_prog
130                    .take()
131                    .expect("actor storage contains only `Some` values by contract"),
132            );
133        }
134    }
135
136    // Checks if actor is uninitialized.
137    pub(crate) fn is_uninitialized(&self) -> bool {
138        matches!(self, TestActor::Uninitialized(..))
139    }
140
141    // Checks if actor is initialized.
142    pub(crate) fn is_initialized(&self) -> bool {
143        matches!(self, TestActor::Initialized(..))
144    }
145
146    // Returns `Some` if actor contains genuine program.
147    pub(crate) fn program(&self) -> Option<&Program> {
148        match self {
149            TestActor::Initialized(program) | TestActor::Uninitialized(_, Some(program)) => {
150                Some(program)
151            }
152            _ => None,
153        }
154    }
155
156    // Returns `Some` if actor contains genuine program but mutable.
157    pub(crate) fn program_mut(&mut self) -> Option<&mut Program> {
158        match self {
159            TestActor::Initialized(program) | TestActor::Uninitialized(_, Some(program)) => {
160                Some(program)
161            }
162            _ => None,
163        }
164    }
165
166    // Returns pages data of genuine program.
167    pub(crate) fn pages(&self) -> Option<&BTreeMap<GearPage, PageBuf>> {
168        self.program().map(|program| &program.pages_data)
169    }
170
171    // Returns pages data of genuine program but mutable.
172    pub(crate) fn pages_mut(&mut self) -> Option<&mut BTreeMap<GearPage, PageBuf>> {
173        self.program_mut().map(|program| &mut program.pages_data)
174    }
175
176    // Gets a new executable actor derived from the inner program.
177    pub(crate) fn executable_actor_data(&self) -> Option<(ExecutableActorData, InstrumentedCode)> {
178        self.program().map(Program::executable_actor_data)
179    }
180}
181
182#[derive(Debug, Clone)]
183pub(crate) struct Program {
184    pub code_id: CodeId,
185    pub code: InstrumentedCode,
186    pub allocations: IntervalsTree<WasmPage>,
187    pub pages_data: BTreeMap<GearPage, PageBuf>,
188    pub gas_reservation_map: GasReservationMap,
189}
190
191impl Program {
192    pub(crate) fn executable_actor_data(&self) -> (ExecutableActorData, InstrumentedCode) {
193        (
194            ExecutableActorData {
195                allocations: self.allocations.clone(),
196                code_id: self.code_id,
197                code_exports: self.code.exports().clone(),
198                static_pages: self.code.static_pages(),
199                gas_reservation_map: self.gas_reservation_map.clone(),
200                memory_infix: Default::default(),
201            },
202            self.code.clone(),
203        )
204    }
205}
