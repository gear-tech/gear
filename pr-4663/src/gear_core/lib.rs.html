51pub mod utils {
52    //! The purpose of this module is to make it easier to import `gprimitives` extensions.
53    use gprimitives::{
54        hashing::hash_array as hash_of_array, ActorId, CodeId, MessageId, ReservationId,
55    };
56
57    /// TO BE REFACTORED IN THE NEXT COMMIT.
58    pub fn generate_pid_from_user(code_id: CodeId, salt: &[u8]) -> ActorId {
59        const SALT: &[u8] = b"program_from_user";
60        hash_of_array([SALT, code_id.as_ref(), salt]).into()
61    }
62
63    /// TO BE REFACTORED IN THE NEXT COMMIT.
64    pub fn generate_pid_from_program(
65        message_id: MessageId,
66        code_id: CodeId,
67        salt: &[u8],
68    ) -> ActorId {
69        const SALT: &[u8] = b"program_from_wasm";
70        hash_of_array([SALT, message_id.as_ref(), code_id.as_ref(), salt]).into()
71    }
72
73    /// TO BE REFACTORED IN THE NEXT COMMIT.
74    pub fn generate_mid_from_user(
75        block_number: u32,
76        user_id: ActorId,
77        local_nonce: u128,
78    ) -> MessageId {
79        const SALT: &[u8] = b"external";
80        hash_of_array([
81            SALT,
82            &block_number.to_le_bytes(),
83            user_id.as_ref(),
84            &local_nonce.to_le_bytes(),
85        ])
86        .into()
87    }
88
89    /// TO BE REFACTORED IN THE NEXT COMMIT.
90    pub fn generate_mid_outgoing(origin_msg_id: MessageId, local_nonce: u32) -> MessageId {
91        const SALT: &[u8] = b"outgoing";
92        hash_of_array([SALT, origin_msg_id.as_ref(), &local_nonce.to_le_bytes()]).into()
93    }
94
95    /// TO BE REFACTORED IN THE NEXT COMMIT.
96    pub fn generate_mid_reply(origin_msg_id: MessageId) -> MessageId {
97        const SALT: &[u8] = b"reply";
98        hash_of_array([SALT, origin_msg_id.as_ref()]).into()
99    }
100
101    /// TO BE REFACTORED IN THE NEXT COMMIT.
102    pub fn generate_mid_signal(origin_msg_id: MessageId) -> MessageId {
103        const SALT: &[u8] = b"signal";
104        hash_of_array([SALT, origin_msg_id.as_ref()]).into()
105    }
106
107    /// TO BE REFACTORED IN THE NEXT COMMIT.
108    pub fn generate_rid(msg_id: MessageId, nonce: u64) -> ReservationId {
109        const SALT: &[u8] = b"reservation";
110        hash_of_array([SALT, msg_id.as_ref(), &nonce.to_le_bytes()]).into()
111    }
112}
