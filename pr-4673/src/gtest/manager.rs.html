115impl ExtManager {
116    pub(crate) fn new() -> Self {
117        Self {
118            msg_nonce: 1,
119            id_nonce: 1,
120            blocks_manager: BlocksManager::new(),
121            messages_processing_enabled: true,
122            random_data: (
123                {
124                    let mut rng = StdRng::seed_from_u64(INITIAL_RANDOM_SEED);
125                    let mut random = [0u8; 32];
126                    rng.fill_bytes(&mut random);
127
128                    random.to_vec()
129                },
130                0,
131            ),
132            ..Default::default()
133        }
134    }
135
136    pub fn block_height(&self) -> u32 {
137        self.blocks_manager.get().height
138    }
139
140    pub(crate) fn store_new_actor(
141        &mut self,
142        program_id: ActorId,
143        program: Program,
144        init_message_id: Option<MessageId>,
145    ) -> Option<TestActor> {
146        Actors::insert(program_id, TestActor::new(init_message_id, program))
147    }
148
149    pub(crate) fn store_new_code(&mut self, code_id: CodeId, code: Vec<u8>) {
150        self.opt_binaries.insert(code_id, code);
151    }
152
153    pub(crate) fn read_code(&self, code_id: CodeId) -> Option<&[u8]> {
154        self.opt_binaries.get(&code_id).map(|code| code.as_ref())
155    }
156
157    pub(crate) fn fetch_inc_message_nonce(&mut self) -> u64 {
158        let nonce = self.msg_nonce;
159        self.msg_nonce += 1;
160        nonce
161    }
162
163    pub(crate) fn free_id_nonce(&mut self) -> u64 {
164        while Actors::contains_key(self.id_nonce.into()) {
165            self.id_nonce += 1;
166        }
167        self.id_nonce
168    }
169
170    /// Check if the current block number should trigger new epoch and reset
171    /// the provided random data.
172    pub(crate) fn check_epoch(&mut self) {
173        let block_height = self.block_height();
174        if block_height % EPOCH_DURATION_IN_BLOCKS == 0 {
175            let mut rng = StdRng::seed_from_u64(
176                INITIAL_RANDOM_SEED + (block_height / EPOCH_DURATION_IN_BLOCKS) as u64,
177            );
178            let mut random = [0u8; 32];
179            rng.fill_bytes(&mut random);
180
181            self.random_data = (random.to_vec(), block_height + 1);
182        }
183    }
184
185    pub(crate) fn update_storage_pages(
186        &mut self,
187        program_id: &ActorId,
188        memory_pages: BTreeMap<GearPage, PageBuf>,
189    ) {
190        Actors::modify(*program_id, |actor| {
191            let pages_data = actor
192                .unwrap_or_else(|| panic!("Actor id {program_id:?} not found"))
193                .pages_mut()
194                .expect("No pages data found for program");
195
196            for (page, buf) in memory_pages {
197                pages_data.insert(page, buf);
198            }
199        });
200    }
201
202    pub(crate) fn mint_to(&mut self, id: &ActorId, value: Value) {
203        Accounts::increase(*id, value);
204    }
205
206    pub(crate) fn balance_of(&self, id: &ActorId) -> Value {
207        Accounts::balance(*id)
208    }
209
210    pub(crate) fn override_balance(&mut self, &id: &ActorId, balance: Value) {
211        if Actors::is_user(id) && balance < crate::EXISTENTIAL_DEPOSIT {
212            usage_panic!(
213                "An attempt to override balance with value ({}) less than existential deposit ({}. \
214                Please try to use bigger balance value",
215                balance,
216                crate::EXISTENTIAL_DEPOSIT
217            );
218        }
219        Accounts::override_balance(id, balance);
220    }
221
222    pub(crate) fn on_task_pool_change(&mut self) {
223        let write = DbWeights::default().write.ref_time;
224        self.gas_allowance = self.gas_allowance.saturating_sub(write);
225    }
226
227    fn init_success(&mut self, program_id: ActorId) {
228        Actors::modify(program_id, |actor| {
229            actor
230                .unwrap_or_else(|| panic!("Actor id {program_id:?} not found"))
231                .set_initialized()
232        });
233    }
234
235    fn init_failure(&mut self, program_id: ActorId, origin: ActorId) {
236        Actors::modify(program_id, |actor| {
237            if let Some(actor) = actor {
238                *actor = TestActor::FailedInit;
239            } else {
240                // That's a case if no code exists for the program
241                // requested to be created from another program and
242                // there was not enough to get program from storage.
243                log::debug!("Failed init is set for non-existing actor");
244            }
245        });
246
247        let value = Accounts::balance(program_id);
248        if value != 0 {
249            Accounts::transfer(program_id, origin, value, false);
250        }
251    }
252
253    pub(crate) fn update_program<R, F: FnOnce(&mut Program) -> R>(
254        &mut self,
255        id: ActorId,
256        op: F,
257    ) -> Option<R> {
258        Actors::modify(id, |actor| {
259            actor.and_then(|actor| actor.program_mut().map(op))
260        })
261    }
262
263    pub(crate) fn read_mailbox_message(
264        &mut self,
265        to: ActorId,
266        from_mid: MessageId,
267    ) -> Result<UserStoredMessage, MailboxErrorImpl> {
268        let (message, hold_interval) = self.mailbox.remove(to, from_mid)?;
269
270        let expected = hold_interval.finish;
271
272        let user_id = message.destination();
273        let from = message.source();
274
275        self.charge_for_hold(message.id(), hold_interval, StorageType::Mailbox);
276        self.consume_and_retrieve(message.id());
277
278        self.bank.transfer_value(from, user_id, message.value());
279
280        let _ = self
281            .task_pool
282            .delete(
283                expected,
284                ScheduledTask::RemoveFromMailbox(user_id, message.id()),
285            )
286            .map(|_| {
287                self.on_task_pool_change();
288            });
289
290        Ok(message)
291    }
