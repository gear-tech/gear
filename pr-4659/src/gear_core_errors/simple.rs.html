32/// Enum representing reply code with reason of its creation.
33pub enum ReplyCode {
34    /// Success reply.
35    #[error("Success reply sent due to {0}")]
36    Success(#[from] SuccessReplyReason) = 0,
37
38    /// Error reply.
39    #[error("Error reply sent due to {0}")]
40    Error(#[from] ErrorReplyReason) = 1,
41
42    /// Unsupported code.
43    /// Variant exists for backward compatibility.
44    #[default]
45    #[error("<unsupported reply code>")]
46    Unsupported = 255,
47}
48
49impl ReplyCode {
50    fn discriminant(&self) -> u8 {
51        // SAFETY: Because `Self` is marked `repr(u8)`, its layout is a `repr(C)` `union`
52        // between `repr(C)` structs, each of which has the `u8` discriminant as its first
53        // field, so we can read the discriminant without offsetting the pointer.
54        unsafe { *<*const _>::from(self).cast::<u8>() }
55    }
56
57    /// Converts `ReplyCode` to 4 bytes array.
58    pub fn to_bytes(self) -> [u8; 4] {
59        let mut bytes = [self.discriminant(), 0, 0, 0];
60
61        match self {
62            Self::Success(reason) => bytes[1..].copy_from_slice(&reason.to_bytes()),
63            Self::Error(reason) => bytes[1..].copy_from_slice(&reason.to_bytes()),
64            Self::Unsupported => {}
65        }
66
67        bytes
68    }
69
70    /// Parses 4 bytes array to `ReplyCode`.
71    pub fn from_bytes(bytes: [u8; 4]) -> Self {
72        match bytes[0] {
73            b if Self::Success(Default::default()).discriminant() == b => {
74                let reason_bytes = bytes[1..].try_into().unwrap_or_else(|_| unreachable!());
75                Self::Success(SuccessReplyReason::from_bytes(reason_bytes))
76            }
77            b if Self::Error(Default::default()).discriminant() == b => {
78                let reason_bytes = bytes[1..].try_into().unwrap_or_else(|_| unreachable!());
79                Self::Error(ErrorReplyReason::from_bytes(reason_bytes))
80            }
81            _ => Self::Unsupported,
82        }
83    }
84
85    /// Constructs `ReplyCode::Error(_)` variant from underlying reason.
86    pub fn error(reason: impl Into<ErrorReplyReason>) -> Self {
87        Self::Error(reason.into())
88    }
89
90    /// Returns bool, defining if `ReplyCode` represents success reply.
91    pub fn is_success(&self) -> bool {
92        matches!(self, Self::Success(_))
93    }
94
95    /// Returns bool, defining if `ReplyCode` represents error reply.
96    pub fn is_error(&self) -> bool {
97        matches!(self, Self::Error(_))
98    }
99
100    /// Returns bool, defining if `ReplyCode` represents unsupported reason.
101    pub fn is_unsupported(&self) -> bool {
102        matches!(self, Self::Unsupported)
103    }
104}

112/// Reason of success reply creation.
113pub enum SuccessReplyReason {
114    /// Success reply was created by system automatically.
115    #[error("automatic sending")]
116    Auto = 0,
117
118    /// Success reply was created by actor manually.
119    #[error("manual sending")]
120    Manual = 1,
121
122    /// Unsupported reason of success reply.
123    /// Variant exists for backward compatibility.
124    #[default]
125    #[error("<unsupported reason>")]
126    Unsupported = 255,
127}
128
129impl SuccessReplyReason {
130    fn to_bytes(self) -> [u8; 3] {
131        [self as u8, 0, 0]
132    }
133
134    fn from_bytes(bytes: [u8; 3]) -> Self {
135        match bytes[0] {
136            b if Self::Auto as u8 == b => Self::Auto,
137            b if Self::Manual as u8 == b => Self::Manual,
138            _ => Self::Unsupported,
139        }
140    }
141}

149/// Reason of error reply creation.
150// NOTE: Adding new variants to this enum you must also update `ErrorReplyReason::to_bytes` and
151// `ErrorReplyReason::from_bytes` methods.
152pub enum ErrorReplyReason {
153    /// Error reply was created due to underlying execution error.
154    #[error("execution error ({0})")]
155    Execution(#[from] SimpleExecutionError) = 0,
156
157    /// Destination actor is unavailable, so it can't process the message.
158    #[error("destination actor is unavailable ({0})")]
159    UnavailableActor(#[from] SimpleUnavailableActorError) = 2,
160
161    /// Message has died in Waitlist as out of rent one.
162    #[error("removal from waitlist")]
163    RemovedFromWaitlist = 3,
164
165    /// Unsupported reason of error reply.
166    /// Variant exists for backward compatibility.
167    #[default]
168    #[error("<unsupported reason>")]
169    Unsupported = 255,
170}
171
172impl ErrorReplyReason {
173    /// Returns bool indicating if self is UnavailableActor::ProgramExited variant.
174    pub fn is_exited(&self) -> bool {
175        matches!(
176            self,
177            Self::UnavailableActor(SimpleUnavailableActorError::ProgramExited)
178        )
179    }
180
181    /// Returns bool indicating if self is Execution::UserspacePanic variant.
182    pub fn is_userspace_panic(&self) -> bool {
183        matches!(self, Self::Execution(SimpleExecutionError::UserspacePanic))
184    }
185
186    fn discriminant(&self) -> u8 {
187        // SAFETY: Because `Self` is marked `repr(u8)`, its layout is a `repr(C)` `union`
188        // between `repr(C)` structs, each of which has the `u8` discriminant as its first
189        // field, so we can read the discriminant without offsetting the pointer.
190        unsafe { *<*const _>::from(self).cast::<u8>() }
191    }
192
193    fn to_bytes(self) -> [u8; 3] {
194        let mut bytes = [self.discriminant(), 0, 0];
195
196        match self {
197            Self::Execution(error) => bytes[1..].copy_from_slice(&error.to_bytes()),
198            Self::UnavailableActor(error) => bytes[1..].copy_from_slice(&error.to_bytes()),
199            Self::RemovedFromWaitlist | Self::Unsupported => {}
200        }
201
202        bytes
203    }
204
205    fn from_bytes(bytes: [u8; 3]) -> Self {
206        match bytes[0] {
207            1 /* removed `FailedToCreateProgram` variant */ |
208            4 /* moved `ReinstrumentationFailure` variant */ => Self::Unsupported,
209            b if Self::Execution(Default::default()).discriminant() == b => {
210                let err_bytes = bytes[1..].try_into().unwrap_or_else(|_| unreachable!());
211                Self::Execution(SimpleExecutionError::from_bytes(err_bytes))
212            }
213            b if Self::UnavailableActor(Default::default()).discriminant() == b => {
214                let err_bytes = bytes[1..].try_into().unwrap_or_else(|_| unreachable!());
215                Self::UnavailableActor(SimpleUnavailableActorError::from_bytes(err_bytes))
216            }
217            b if Self::RemovedFromWaitlist.discriminant() == b => Self::RemovedFromWaitlist,
218            _ => Self::Unsupported,
219        }
220    }
221}

229/// Simplified error occurred during execution.
230pub enum SimpleExecutionError {
231    /// Message ran out of gas while executing.
232    #[error("Message ran out of gas")]
233    RanOutOfGas = 0,
234
235    /// Program has reached memory limit while executing.
236    #[error("Program reached memory limit")]
237    MemoryOverflow = 1,
238
239    /// Execution failed with backend error that couldn't been caught.
240    #[error("Message ran into uncatchable error")]
241    BackendError = 2,
242
243    /// Execution failed with userspace panic.
244    ///
245    /// **PAYLOAD**: Arbitrary payload given by the program as `gr_panic` argument.
246    #[error("Message panicked")]
247    UserspacePanic = 3,
248
249    /// Execution failed with `unreachable` instruction call.
250    #[error("Program called WASM `unreachable` instruction")]
251    UnreachableInstruction = 4,
252
253    /// Program has reached stack limit while executing.
254    #[error("Program reached stack limit")]
255    StackLimitExceeded = 5,
256
257    /// Unsupported reason of execution error.
258    /// Variant exists for backward compatibility.
259    #[default]
260    #[error("<unsupported error>")]
261    Unsupported = 255,
262}
263
264impl SimpleExecutionError {
265    fn to_bytes(self) -> [u8; 2] {
266        [self as u8, 0]
267    }
268
269    fn from_bytes(bytes: [u8; 2]) -> Self {
270        match bytes[0] {
271            b if Self::RanOutOfGas as u8 == b => Self::RanOutOfGas,
272            b if Self::MemoryOverflow as u8 == b => Self::MemoryOverflow,
273            b if Self::BackendError as u8 == b => Self::BackendError,
274            b if Self::UserspacePanic as u8 == b => Self::UserspacePanic,
275            b if Self::UnreachableInstruction as u8 == b => Self::UnreachableInstruction,
276            b if Self::StackLimitExceeded as u8 == b => Self::StackLimitExceeded,
277            _ => Self::Unsupported,
278        }
279    }
280}

288/// Simplified error occurred because of actor unavailability.

302    Uninitialized = 2,
303

307

309    #[error("Program re-instrumentation failed")]
310    ReinstrumentationFailure = 4,
311
312    /// Unsupported reason of inactive actor error.

314    #[default]
315    #[error("<unsupported error>")]

341/// Enum representing signal code and reason of its creation.
342///
343/// # Testing
344/// See [this document](../signal-code-testing.md).
345pub enum SignalCode {
346    /// Signal was sent due to some execution errors.
347    #[error("Signal message sent due to execution error ({0})")]
348    Execution(#[from] SimpleExecutionError),
349
350    /// Signal was sent due to removal from waitlist as out of rent.
351    #[default]
352    #[error("Signal message sent due to removal from waitlist")]
353    RemovedFromWaitlist,
354}
355
356impl SignalCode {
357    /// Converts `SignalCode` into `u32`.
358    pub const fn to_u32(self) -> u32 {
359        match self {
360            Self::Execution(SimpleExecutionError::UserspacePanic) => 100,
361            Self::Execution(SimpleExecutionError::RanOutOfGas) => 101,
362            Self::Execution(SimpleExecutionError::BackendError) => 102,
363            Self::Execution(SimpleExecutionError::MemoryOverflow) => 103,
364            Self::Execution(SimpleExecutionError::UnreachableInstruction) => 104,
365            Self::Execution(SimpleExecutionError::StackLimitExceeded) => 105,
366            Self::RemovedFromWaitlist => 200,
367            // Must be unreachable.
368            Self::Execution(SimpleExecutionError::Unsupported) => u32::MAX,
369        }
370    }
371
372    /// Parses `SignalCode` from `u32` if possible.
373    pub const fn from_u32(num: u32) -> Option<Self> {
374        let res = match num {
375            v if Self::Execution(SimpleExecutionError::UserspacePanic).to_u32() == v => {
376                Self::Execution(SimpleExecutionError::UserspacePanic)
377            }
378            v if Self::Execution(SimpleExecutionError::RanOutOfGas).to_u32() == v => {
379                Self::Execution(SimpleExecutionError::RanOutOfGas)
380            }
381            v if Self::Execution(SimpleExecutionError::BackendError).to_u32() == v => {
382                Self::Execution(SimpleExecutionError::BackendError)
383            }
384            v if Self::Execution(SimpleExecutionError::MemoryOverflow).to_u32() == v => {
385                Self::Execution(SimpleExecutionError::MemoryOverflow)
386            }
387            v if Self::Execution(SimpleExecutionError::UnreachableInstruction).to_u32() == v => {
388                Self::Execution(SimpleExecutionError::UnreachableInstruction)
389            }
390            v if Self::Execution(SimpleExecutionError::StackLimitExceeded).to_u32() == v => {
391                Self::Execution(SimpleExecutionError::StackLimitExceeded)
392            }
393            v if Self::Execution(SimpleExecutionError::Unsupported).to_u32() == v => {
394                Self::Execution(SimpleExecutionError::Unsupported)
395            }
396            v if Self::RemovedFromWaitlist.to_u32() == v => Self::RemovedFromWaitlist,
397            _ => return None,
398        };
399
400        Some(res)
401    }
402}

404#[cfg(test)]
405mod tests {
406    use super::*;
407
408    #[test]
409    fn test_forbidden_codes() {
410        let codes = [
411            1, // `FailedToCreateProgram` variant
412            4, // `ReinstrumentationFailure` variant
413        ];
414
415        // check forbidden code is `Unsupported` variant now
416        for code in codes {
417            let err = ErrorReplyReason::from_bytes([code, 0, 0]);
418            assert_eq!(err, ErrorReplyReason::Unsupported);
419        }
420
421        // check forbidden code is never produced
422        for code in enum_iterator::all::<ErrorReplyReason>() {
423            let bytes = code.to_bytes();
424            assert!(!codes.contains(&bytes[0]));
425        }
426    }
427
428    #[test]
429    fn test_reply_code_encode_decode() {
430        for code in enum_iterator::all::<ReplyCode>() {
431            let bytes = code.to_bytes();
432            assert_eq!(code, ReplyCode::from_bytes(bytes));
433        }
434    }
435
436    #[test]
437    fn test_signal_code_encode_decode() {
438        for signal in enum_iterator::all::<SignalCode>() {
439            let code = signal.to_u32();
440            assert_eq!(signal, SignalCode::from_u32(code).unwrap());
441        }
442    }
