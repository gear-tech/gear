33/// Limited len vector.
34/// `T` is data type.
35/// `E` is overflow error type.
36/// `N` is max len which a vector can have.
37#[derive(Clone, Default, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo)]
38#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
39pub struct LimitedVec<T, E, const N: usize>(Vec<T>, PhantomData<E>);
40
41/// Formatter for [`LimitedVec`] will print to precision of 8 by default, to print the whole data, use `{:+}`.
42impl<T: Clone + Default, E: Default, const N: usize> Display for LimitedVec<T, E, N>
43where
44    [T]: AsRef<[u8]>,
45{
46    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
47        let len = self.0.len();
48        let median = len.div_ceil(2);
49
50        let mut e1 = median;
51        let mut s2 = median;
52
53        if let Some(precision) = f.precision() {
54            if precision < median {
55                e1 = precision;
56                s2 = len - precision;
57            }
58        } else if !f.sign_plus() && median > 8 {
59            e1 = 8;
60            s2 = len - 8;
61        }
62
63        let p1 = hex::encode(&self.0[..e1]);
64        let p2 = hex::encode(&self.0[s2..]);
65        let sep = if e1.ne(&s2) { ".." } else { Default::default() };
66
67        if f.alternate() {
68            write!(f, "LimitedVec(0x{p1}{sep}{p2})")
69        } else {
70            write!(f, "0x{p1}{sep}{p2}")
71        }
72    }
73}

75impl<T: Clone + Default, E: Default, const N: usize> Debug for LimitedVec<T, E, N>
76where
77    [T]: AsRef<[u8]>,
78{
79    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
80        core::fmt::Display::fmt(self, f)
81    }

84impl<T: Clone, E: Default, const N: usize> TryFrom<&[T]> for LimitedVec<T, E, N> {
85    type Error = E;
86    fn try_from(x: &[T]) -> Result<Self, Self::Error> {
87        (x.len() <= N).then_some(()).ok_or_else(E::default)?;
88        Ok(Self(Vec::from(x), PhantomData))
89    }
90}
91
92impl<T, E: Default, const N: usize> TryFrom<Vec<T>> for LimitedVec<T, E, N> {
93    type Error = E;
94    fn try_from(x: Vec<T>) -> Result<Self, Self::Error> {
95        (x.len() <= N).then_some(()).ok_or_else(E::default)?;
96        Ok(Self(x, PhantomData))
97    }
98}

100impl<T: Clone + Default, E: Default, const N: usize> LimitedVec<T, E, N> {
101    /// Maximum length of the vector.
102    pub const MAX_LEN: usize = N;
103
104    /// Constructs a new, empty `LimitedVec<T>`.
105    pub const fn new() -> Self {
106        Self(Vec::new(), PhantomData)
107    }
108
109    /// Tries to create new limited vector of length `len`
110    /// with default initialized elements.
111    pub fn try_new_default(len: usize) -> Result<Self, E> {
112        (len <= N).then_some(()).ok_or_else(E::default)?;
113        Ok(Self(vec![T::default(); len], PhantomData))
114    }
115
116    /// Creates new limited vector with default initialized elements.
117    pub fn new_default() -> Self {
118        Self(vec![T::default(); N], PhantomData)
119    }
120
121    /// Creates limited vector filled with the specified `value`.
122    pub fn filled_with(value: T) -> Self {
123        Self(vec![value; N], PhantomData)
124    }
125
126    /// Extends the array to its limit and fills with the specified `value`.
127    pub fn extend_with(&mut self, value: T) {
128        self.0.resize(N, value);
129    }
130
131    /// Append `value` to the end of vector.
132    pub fn try_push(&mut self, value: T) -> Result<(), E> {
133        (self.0.len() != N).then_some(()).ok_or_else(E::default)?;
134        self.0.push(value);
135        Ok(())
136    }
137
138    /// Append `values` to the end of vector.
139    pub fn try_extend_from_slice(&mut self, values: &[T]) -> Result<(), E> {
140        self.0
141            .len()
142            .checked_add(values.len())
143            .and_then(|len| (len <= N).then_some(()))
144            .ok_or_else(E::default)?;
145
146        self.0.extend_from_slice(values);
147
148        Ok(())
149    }
150
151    /// Append `values` to the begin of vector.
152    pub fn try_prepend(&mut self, values: Self) -> Result<(), E> {
153        self.0
154            .len()
155            .checked_add(values.0.len())
156            .and_then(|len| (len <= N).then_some(()))
157            .ok_or_else(E::default)?;
158
159        self.0.splice(0..0, values.0);
160
161        Ok(())
162    }
163
164    /// Returns ref to the internal data.
165    pub fn inner(&self) -> &[T] {
166        &self.0
167    }
168
169    /// Returns mut ref to the internal data slice.
170    pub fn inner_mut(&mut self) -> &mut [T] {
171        &mut self.0
172    }
173
174    /// Clones self into vector.
175    pub fn to_vec(&self) -> Vec<T> {
176        self.0.clone()
177    }
178
179    /// Destruct limited vector and returns inner vector.
180    pub fn into_vec(self) -> Vec<T> {
181        self.0
182    }
183
184    /// Returns max len which this type of limited vector can have.
185    pub const fn max_len() -> usize {
186        N
187    }
188}

190/// Max memory size, which runtime can allocate at once.
191/// Substrate allocator restrict allocations bigger then 512 wasm pages at once.
192/// See more information about:
193/// https://github.com/paritytech/substrate/blob/cc4d5cc8654d280f03a13421669ba03632e14aa7/client/allocator/src/freeing_bump.rs#L136-L149
194/// https://github.com/paritytech/substrate/blob/cc4d5cc8654d280f03a13421669ba03632e14aa7/primitives/core/src/lib.rs#L385-L388
195const RUNTIME_MAX_ALLOC_SIZE: usize = 512 * 0x10000;
196
197/// Take half from [RUNTIME_MAX_ALLOC_SIZE] in order to avoid problems with capacity overflow.
198const RUNTIME_MAX_BUFF_SIZE: usize = RUNTIME_MAX_ALLOC_SIZE / 2;
199
200/// Runtime buffer size exceed error
201#[derive(
202    Clone, Copy, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo,
203)]
204pub struct RuntimeBufferSizeError;
205
206impl From<RuntimeBufferSizeError> for &str {
207    fn from(_: RuntimeBufferSizeError) -> Self {
208        "Runtime buffer size exceed"
209    }
210}
211
212impl Display for RuntimeBufferSizeError {
213    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
214        f.write_str((*self).into())
215    }
216}
217
218/// Buffer which size cannot be bigger then max allowed allocation size in runtime.
219pub type RuntimeBuffer = LimitedVec<u8, RuntimeBufferSizeError, RUNTIME_MAX_BUFF_SIZE>;

221#[cfg(test)]
222mod test {
223    use super::{LimitedVec, RuntimeBufferSizeError};
224    use alloc::{string::String, vec, vec::Vec};
225    use core::convert::{TryFrom, TryInto};
226
227    const N: usize = 1000;
228    type TestBuffer = LimitedVec<u8, RuntimeBufferSizeError, N>;
229    const M: usize = 64;
230    type SmallTestBuffer = LimitedVec<u8, RuntimeBufferSizeError, M>;

232    #[test]
233    fn test_try_from() {
234        let v1 = vec![1; N];
235        let v2 = vec![1; N + 1];
236        let v3 = vec![1; N - 1];
237
238        let x = TestBuffer::try_from(v1).unwrap();
239        let _ = TestBuffer::try_from(v2).expect_err("Must be err because of size overflow");
240        let z = TestBuffer::try_from(v3).unwrap();
241
242        assert_eq!(x.inner().len(), N);
243        assert_eq!(z.inner().len(), N - 1);
244        assert_eq!(x.inner()[N / 2], 1);
245        assert_eq!(z.inner()[N / 2], 1);
246    }
247
248    #[test]
249    fn test_new_default() {
250        let x = LimitedVec::<String, RuntimeBufferSizeError, N>::try_new_default(N).unwrap();
251        assert!(
252            LimitedVec::<u64, RuntimeBufferSizeError, N>::try_new_default(N + 1).is_err(),
253            "Must be error because of size overflow"
254        );
255        let z = LimitedVec::<Vec<u8>, RuntimeBufferSizeError, N>::try_new_default(0).unwrap();
256
257        assert_eq!(x.inner().len(), N);
258        assert_eq!(z.inner().len(), 0);
259        assert_eq!(x.inner()[N / 2], "");
260    }
261
262    #[test]
263    fn test_prepend_works() {
264        let mut buf = TestBuffer::try_from(vec![1, 2, 3, 4, 5]).unwrap();
265        let prepend_buf = TestBuffer::try_from(vec![6, 7, 8]).unwrap();
266        buf.try_prepend(prepend_buf).unwrap();
267
268        assert_eq!(buf.inner(), &[6, 7, 8, 1, 2, 3, 4, 5]);
269    }
270
271    #[test]
272    fn test_full() {
273        let mut x = TestBuffer::try_from(vec![1; N]).unwrap();
274        let mut y = TestBuffer::try_from(vec![2; N / 2]).unwrap();
275        let mut z = TestBuffer::try_from(vec![3; 0]).unwrap();
276
277        x.try_push(42).unwrap_err();
278        y.try_push(42).unwrap();
279        z.try_push(42).unwrap();
280
281        x.try_extend_from_slice(&[1, 2, 3]).unwrap_err();
282        y.try_extend_from_slice(&[1, 2, 3]).unwrap();
283        z.try_extend_from_slice(&[1, 2, 3]).unwrap();
284
285        x.try_prepend(vec![1, 2, 3].try_into().unwrap())
286            .unwrap_err();
287        y.try_prepend(vec![1, 2, 3].try_into().unwrap()).unwrap();
288        z.try_prepend(vec![1, 2, 3].try_into().unwrap()).unwrap();
289
290        z.inner_mut()[0] = 0;
291
292        assert_eq!(&z.into_vec(), &[0, 2, 3, 42, 1, 2, 3]);
293        assert_eq!(TestBuffer::max_len(), N);
294    }

296    #[test]
297    fn formatting_test() {
298        use alloc::format;
299
300        let buffer = SmallTestBuffer::try_from(b"abcdefghijklmnopqrstuvwxyz012345".to_vec())
301            .expect("String is 64 bytes");
302
303        // `Debug`/`Display`.
304        assert_eq!(
305            format!("{buffer:+?}"),
306            "0x6162636465666768696a6b6c6d6e6f707172737475767778797a303132333435"
307        );
308        // `Debug`/`Display` with default precision.
309        assert_eq!(
310            format!("{buffer:?}"),
311            "0x6162636465666768..797a303132333435"
312        );
313        // `Debug`/`Display` with precision 0.
314        assert_eq!(format!("{buffer:.0?}"), "0x..");
315        // `Debug`/`Display` with precision 1.
316        assert_eq!(format!("{buffer:.1?}"), "0x61..35");
317        // `Debug`/`Display` with precision 2.
318        assert_eq!(format!("{buffer:.2?}"), "0x6162..3435");
319        // `Debug`/`Display` with precision 4.
320        assert_eq!(format!("{buffer:.4?}"), "0x61626364..32333435");
321        // `Debug`/`Display` with precision 15.
322        assert_eq!(
323            format!("{buffer:.15?}"),
324            "0x6162636465666768696a6b6c6d6e6f..72737475767778797a303132333435"
325        );
326        // `Debug`/`Display` with precision 30.
327        assert_eq!(
328            format!("{buffer:.30?}"),
329            "0x6162636465666768696a6b6c6d6e6f707172737475767778797a303132333435"
330        );
331        // Alternate formatter with default precision.
332        assert_eq!(
333            format!("{buffer:#}"),
334            "LimitedVec(0x6162636465666768..797a303132333435)"
335        );
336        // Alternate formatter with max precision.
337        assert_eq!(
338            format!("{buffer:+#}"),
339            "LimitedVec(0x6162636465666768696a6b6c6d6e6f707172737475767778797a303132333435)"
340        );
341        // Alternate formatter with precision 2.
342        assert_eq!(format!("{buffer:#.2}"), "LimitedVec(0x6162..3435)");
343    }
